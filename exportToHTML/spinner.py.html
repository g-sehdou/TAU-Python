17   from __future__ import unicode_literals
18   from . import Infinite
19   
20   
21   class Spinner(Infinite):
22       phases = ('-', '\\', '|', '/')
23       hide_cursor = True
24   
25       def update(self):
26           i = self.index % len(self.phases)
27           message = self.message % self
28           line = ''.join([message, self.phases[i]])
29           self.writeln(line)
30   
31   
32   class PieSpinner(Spinner):
33       phases = ['◷', '◶', '◵', '◴']
34   
35   
36   class MoonSpinner(Spinner):
37       phases = ['◑', '◒', '◐', '◓']
