67   def on_rm_rf_error(func, path: str, exc, *, start_path: Path) -> bool:
68       """Handle known read-only errors during rmtree. 
69    
70       The returned value is used only by our own tests. 
71       """
72       exctype, excvalue = exc[:2]
73   
74       # Another process removed the file in the middle of the "rm_rf" (xdist for example).
75       # More context: https://github.com/pytest-dev/pytest/issues/5974#issuecomment-543799018
76       if isinstance(excvalue, FileNotFoundError):
77           return False
78   
79       if not isinstance(excvalue, PermissionError):
80           warnings.warn(
81               PytestWarning(f"(rm_rf) error removing {path}\n{exctype}: {excvalue}")
82           )
83           return False
84   
85       if func not in (os.rmdir, os.remove, os.unlink):
86           if func not in (os.open,):
87               warnings.warn(
88                   PytestWarning(
89                       "(rm_rf) unknown function {} when removing {}:\n{}: {}".format(
90                           func, path, exctype, excvalue
91                       )
92                   )
93               )
94           return False
95   
96       # Chmod + retry.
97       import stat
98   
99       def chmod_rw(p: str) -> None:
100          mode = os.stat(p).st_mode
101          os.chmod(p, mode | stat.S_IRUSR | stat.S_IWUSR)
102  
103      # For files, we need to recursively go upwards in the directories to
104      # ensure they all are also writable.
105      p = Path(path)
106      if p.is_file():
107          for parent in p.parents:
108              chmod_rw(str(parent))
109              # Stop when we reach the original path passed to rm_rf.
110              if parent == start_path:
111                  break
112      chmod_rw(str(path))
113  
114      func(path)
115      return True

118  def ensure_extended_length_path(path: Path) -> Path:
119      """Get the extended-length version of a path (Windows). 
120   
121      On Windows, by default, the maximum length of a path (MAX_PATH) is 260 
122      characters, and operations on paths longer than that fail. But it is possible 
123      to overcome this by converting the path to "extended-length" form before 
124      performing the operation: 
125      https://docs.microsoft.com/en-us/windows/win32/fileio/naming-a-file#maximum-path-length-limitation 
126   
127      On Windows, this function returns the extended-length absolute version of path. 
128      On other platforms it returns path unchanged. 
129      """
130      if sys.platform.startswith("win32"):
131          path = path.resolve()
132          path = Path(get_extended_length_path_str(str(path)))
133      return path
134

136  def get_extended_length_path_str(path: str) -> str:
137      """Convert a path to a Windows extended length path."""
138      long_path_prefix = "\\\\?\\"
139      unc_long_path_prefix = "\\\\?\\UNC\\"
140      if path.startswith((long_path_prefix, unc_long_path_prefix)):
141          return path
142      # UNC
143      if path.startswith("\\\\"):
144          return unc_long_path_prefix + path[2:]
145      return long_path_prefix + path

148  def rm_rf(path: Path) -> None:
149      """Remove the path contents recursively, even if some elements 
150      are read-only."""
151      path = ensure_extended_length_path(path)
152      onerror = partial(on_rm_rf_error, start_path=path)
153      shutil.rmtree(str(path), onerror=onerror)

156  def find_prefixed(root: Path, prefix: str) -> Iterator[Path]:
157      """Find all elements in root that begin with the prefix, case insensitive."""
158      l_prefix = prefix.lower()
159      for x in root.iterdir():
160          if x.name.lower().startswith(l_prefix):
161              yield x

164  def extract_suffixes(iter: Iterable[PurePath], prefix: str) -> Iterator[str]:
165      """Return the parts of the paths following the prefix. 
166   
167      :param iter: Iterator over path names. 
168      :param prefix: Expected prefix of the path names. 
169      """
170      p_len = len(prefix)
171      for p in iter:
172          yield p.name[p_len:]

188  def _force_symlink(
189      root: Path, target: Union[str, PurePath], link_to: Union[str, Path]
190  ) -> None:
191      """Helper to create the current symlink. 
192   
193      It's full of race conditions that are reasonably OK to ignore 
194      for the context of best effort linking to the latest test run. 
195   
196      The presumption being that in case of much parallelism 
197      the inaccuracy is going to be acceptable. 
198      """
199      current_symlink = root.joinpath(target)
200      try:
201          current_symlink.unlink()
202      except OSError:
203          pass
204      try:
205          current_symlink.symlink_to(link_to)
206      except Exception:
207          pass

210  def make_numbered_dir(root: Path, prefix: str, mode: int = 0o700) -> Path:
211      """Create a directory with an increased number as suffix for the given prefix."""
212      for i in range(10):
213          # try up to 10 times to create the folder
214          max_existing = max(map(parse_num, find_suffixes(root, prefix)), default=-1)
215          new_number = max_existing + 1
216          new_path = root.joinpath(f"{prefix}{new_number}")
217          try:
218              new_path.mkdir(mode=mode)
219          except Exception:
220              pass
221          else:
222              _force_symlink(root, prefix + "current", new_path)
223              return new_path
224      else:
225          raise OSError(
226              "could not create numbered dir with prefix "
227              "{prefix} in {root} after 10 tries".format(prefix=prefix, root=root)
228          )

231  def create_cleanup_lock(p: Path) -> Path:
232      """Create a lock to prevent premature folder cleanup."""
233      lock_path = get_lock_path(p)
234      try:
235          fd = os.open(str(lock_path), os.O_WRONLY | os.O_CREAT | os.O_EXCL, 0o644)
236      except FileExistsError as e:
237          raise OSError(f"cannot create lockfile in {p}") from e
238      else:
239          pid = os.getpid()
240          spid = str(pid).encode()
241          os.write(fd, spid)
242          os.close(fd)
243          if not lock_path.is_file():
244              raise OSError("lock path got renamed after successful creation")
245          return lock_path

248  def register_cleanup_lock_removal(lock_path: Path, register=atexit.register):
249      """Register a cleanup function for removing a lock, by default on atexit."""
250      pid = os.getpid()
251  
252      def cleanup_on_exit(lock_path: Path = lock_path, original_pid: int = pid) -> None:
253          current_pid = os.getpid()
254          if current_pid != original_pid:
255              # fork
256              return
257          try:
258              lock_path.unlink()
259          except OSError:
260              pass
261  
262      return register(cleanup_on_exit)

265  def maybe_delete_a_numbered_dir(path: Path) -> None:
266      """Remove a numbered directory if its lock can be obtained and it does 
267      not seem to be in use."""
268      path = ensure_extended_length_path(path)
269      lock_path = None
270      try:
271          lock_path = create_cleanup_lock(path)
272          parent = path.parent
273  
274          garbage = parent.joinpath(f"garbage-{uuid.uuid4()}")
275          path.rename(garbage)
276          rm_rf(garbage)
277      except OSError:
278          #  known races:
279          #  * other process did a cleanup at the same time
280          #  * deletable folder was found
281          #  * process cwd (Windows)
282          return
283      finally:
284          # If we created the lock, ensure we remove it even if we failed
285          # to properly remove the numbered dir.
286          if lock_path is not None:
287              try:
288                  lock_path.unlink()
289              except OSError:
290                  pass

293  def ensure_deletable(path: Path, consider_lock_dead_if_created_before: float) -> bool:
294      """Check if `path` is deletable based on whether the lock file is expired."""
295      if path.is_symlink():
296          return False
297      lock = get_lock_path(path)
298      try:
299          if not lock.is_file():
300              return True
301      except OSError:
302          # we might not have access to the lock file at all, in this case assume
303          # we don't have access to the entire directory (#7491).
304          return False
305      try:
306          lock_time = lock.stat().st_mtime
307      except Exception:
308          return False
309      else:
310          if lock_time < consider_lock_dead_if_created_before:
311              # We want to ignore any errors while trying to remove the lock such as:
312              # - PermissionDenied, like the file permissions have changed since the lock creation;
313              # - FileNotFoundError, in case another pytest process got here first;
314              # and any other cause of failure.
315              with contextlib.suppress(OSError):
316                  lock.unlink()
317                  return True
318          return False

321  def try_cleanup(path: Path, consider_lock_dead_if_created_before: float) -> None:
322      """Try to cleanup a folder if we can ensure it's deletable."""
323      if ensure_deletable(path, consider_lock_dead_if_created_before):
324          maybe_delete_a_numbered_dir(path)

327  def cleanup_candidates(root: Path, prefix: str, keep: int) -> Iterator[Path]:
328      """List candidates for numbered directories to be removed - follows py.path."""
329      max_existing = max(map(parse_num, find_suffixes(root, prefix)), default=-1)
330      max_delete = max_existing - keep
331      paths = find_prefixed(root, prefix)
332      paths, paths2 = itertools.tee(paths)
333      numbers = map(parse_num, extract_suffixes(paths2, prefix))
334      for path, number in zip(paths, numbers):
335          if number <= max_delete:
336              yield path

339  def cleanup_numbered_dir(
340      root: Path, prefix: str, keep: int, consider_lock_dead_if_created_before: float
341  ) -> None:
342      """Cleanup for lock driven numbered directories."""
343      for path in cleanup_candidates(root, prefix, keep):
344          try_cleanup(path, consider_lock_dead_if_created_before)
345      for path in root.glob("garbage-*"):
346          try_cleanup(path, consider_lock_dead_if_created_before)

349  def make_numbered_dir_with_cleanup(
350      root: Path, prefix: str, keep: int, lock_timeout: float, mode: int,
351  ) -> Path:
352      """Create a numbered dir with a cleanup lock and remove old ones."""
353      e = None
354      for i in range(10):
355          try:
356              p = make_numbered_dir(root, prefix, mode)
357              lock_path = create_cleanup_lock(p)
358              register_cleanup_lock_removal(lock_path)
359          except Exception as exc:
360              e = exc
361          else:
362              consider_lock_dead_if_created_before = p.stat().st_mtime - lock_timeout
363              # Register a cleanup for program exit
364              atexit.register(
365                  cleanup_numbered_dir,
366                  root,
367                  prefix,
368                  keep,
369                  consider_lock_dead_if_created_before,
370              )
371              return p
372      assert e is not None
373      raise e

385  def fnmatch_ex(pattern: str, path) -> bool:
386      """A port of FNMatcher from py.path.common which works with PurePath() instances. 
387   
388      The difference between this algorithm and PurePath.match() is that the 
389      latter matches "**" glob expressions for each part of the path, while 
390      this algorithm uses the whole path instead. 
391   
392      For example: 
393          "tests/foo/bar/doc/test_foo.py" matches pattern "tests/**/doc/test*.py" 
394          with this algorithm, but not with PurePath.match(). 
395   
396      This algorithm was ported to keep backward-compatibility with existing 
397      settings which assume paths match according this logic. 
398   
399      References: 
400      * https://bugs.python.org/issue29249 
401      * https://bugs.python.org/issue34731 
402      """
403      path = PurePath(path)
404      iswin32 = sys.platform.startswith("win")
405  
406      if iswin32 and sep not in pattern and posix_sep in pattern:
407          # Running on Windows, the pattern has no Windows path separators,
408          # and the pattern has one or more Posix path separators. Replace
409          # the Posix path separators with the Windows path separator.
410          pattern = pattern.replace(posix_sep, sep)
411  
412      if sep not in pattern:
413          name = path.name
414      else:
415          name = str(path)
416          if path.is_absolute() and not os.path.isabs(pattern):
417              pattern = f"*{os.sep}{pattern}"
418      return fnmatch.fnmatch(name, pattern)

442  class ImportPathMismatchError(ImportError):
443      """Raised on import_path() if there is a mismatch of __file__'s. 
444   
445      This can happen when `import_path` is called multiple times with different filenames that has 
446      the same basename but reside in packages 
447      (for example "/tests1/test_foo.py" and "/tests2/test_foo.py"). 
448      """

451  def import_path(
452      p: Union[str, py.path.local, Path],
453      *,
454      mode: Union[str, ImportMode] = ImportMode.prepend,
455  ) -> ModuleType:
456      """Import and return a module from the given path, which can be a file (a module) or 
457      a directory (a package). 
458   
459      The import mechanism used is controlled by the `mode` parameter: 
460   
461      * `mode == ImportMode.prepend`: the directory containing the module (or package, taking 
462        `__init__.py` files into account) will be put at the *start* of `sys.path` before 
463        being imported with `__import__. 
464   
465      * `mode == ImportMode.append`: same as `prepend`, but the directory will be appended 
466        to the end of `sys.path`, if not already in `sys.path`. 
467   
468      * `mode == ImportMode.importlib`: uses more fine control mechanisms provided by `importlib` 
469        to import the module, which avoids having to use `__import__` and muck with `sys.path` 
470        at all. It effectively allows having same-named test modules in different places. 
471   
472      :raises ImportPathMismatchError: 
473          If after importing the given `path` and the module `__file__` 
474          are different. Only raised in `prepend` and `append` modes. 
475      """
476      mode = ImportMode(mode)
477  
478      path = Path(str(p))
479  
480      if not path.exists():
481          raise ImportError(path)
482  
483      if mode is ImportMode.importlib:
484          module_name = path.stem
485  
486          for meta_importer in sys.meta_path:
487              spec = meta_importer.find_spec(module_name, [str(path.parent)])
488              if spec is not None:
489                  break
490          else:
491              spec = importlib.util.spec_from_file_location(module_name, str(path))
492  
493          if spec is None:
494              raise ImportError(
495                  "Can't find module {} at location {}".format(module_name, str(path))
496              )
497          mod = importlib.util.module_from_spec(spec)
498          spec.loader.exec_module(mod)  # type: ignore[union-attr]
499          return mod
500  
501      pkg_path = resolve_package_path(path)
502      if pkg_path is not None:
503          pkg_root = pkg_path.parent
504          names = list(path.with_suffix("").relative_to(pkg_root).parts)
505          if names[-1] == "__init__":
506              names.pop()
507          module_name = ".".join(names)
508      else:
509          pkg_root = path.parent
510          module_name = path.stem
511  
512      # Change sys.path permanently: restoring it at the end of this function would cause surprising
513      # problems because of delayed imports: for example, a conftest.py file imported by this function
514      # might have local imports, which would fail at runtime if we restored sys.path.
515      if mode is ImportMode.append:
516          if str(pkg_root) not in sys.path:
517              sys.path.append(str(pkg_root))
518      elif mode is ImportMode.prepend:
519          if str(pkg_root) != sys.path[0]:
520              sys.path.insert(0, str(pkg_root))
521      else:
522          assert_never(mode)
523  
524      importlib.import_module(module_name)
525  
526      mod = sys.modules[module_name]
527      if path.name == "__init__.py":
528          return mod
529  
530      ignore = os.environ.get("PY_IGNORE_IMPORTMISMATCH", "")
531      if ignore != "1":
532          module_file = mod.__file__
533          if module_file.endswith((".pyc", ".pyo")):
534              module_file = module_file[:-1]
535          if module_file.endswith(os.path.sep + "__init__.py"):
536              module_file = module_file[: -(len(os.path.sep + "__init__.py"))]
537  
538          try:
539              is_same = _is_same(str(path), module_file)
540          except FileNotFoundError:
541              is_same = False
542  
543          if not is_same:
544              raise ImportPathMismatchError(module_name, module_file, path)
545  
546      return mod
547  
548  
549  # Implement a special _is_same function on Windows which returns True if the two filenames
550  # compare equal, to circumvent os.path.samefile returning False for mounts in UNC (#7678).
551  if sys.platform.startswith("win"):
552  
553      def _is_same(f1: str, f2: str) -> bool:
554          return Path(f1) == Path(f2) or os.path.samefile(f1, f2)
555  
556  
557  else:
558  
559      def _is_same(f1: str, f2: str) -> bool:
560          return os.path.samefile(f1, f2)

563  def resolve_package_path(path: Path) -> Optional[Path]:
564      """Return the Python package path by looking for the last 
565      directory upwards which still contains an __init__.py. 
566   
567      Returns None if it can not be determined. 
568      """
569      result = None
570      for parent in itertools.chain((path,), path.parents):
571          if parent.is_dir():
572              if not parent.joinpath("__init__.py").is_file():
573                  break
574              if not parent.name.isidentifier():
575                  break
576              result = parent
577      return result

580  def visit(
581      path: str, recurse: Callable[["os.DirEntry[str]"], bool]
582  ) -> Iterator["os.DirEntry[str]"]:
583      """Walk a directory recursively, in breadth-first order. 
584   
585      Entries at each directory level are sorted. 
586      """
587  
588      # Skip entries with symlink loops and other brokenness, so the caller doesn't
589      # have to deal with it.
590      entries = []
591      for entry in os.scandir(path):
592          try:
593              entry.is_file()
594          except OSError as err:
595              if _ignore_error(err):
596                  continue
597              raise
598          entries.append(entry)
599  
600      entries.sort(key=lambda entry: entry.name)
601  
602      yield from entries
603  
604      for entry in entries:
605          if entry.is_dir() and recurse(entry):
606              yield from visit(entry.path, recurse)

630  def bestrelpath(directory: Path, dest: Path) -> str:
631      """Return a string which is a relative path from directory to dest such 
632      that directory/bestrelpath == dest. 
633   
634      The paths must be either both absolute or both relative. 
635   
636      If no such path can be determined, returns dest. 
637      """
638      if dest == directory:
639          return os.curdir
640      # Find the longest common directory.
641      base = commonpath(directory, dest)
642      # Can be the case on Windows for two absolute paths on different drives.
643      # Can be the case for two relative paths without common prefix.
644      # Can be the case for a relative path and an absolute path.
645      if not base:
646          return str(dest)
647      reldirectory = directory.relative_to(base)
648      reldest = dest.relative_to(base)
649      return os.path.join(
650          # Back from directory to base.
651          *([os.pardir] * len(reldirectory.parts)),
652          # Forward from base to dest.
653          *reldest.parts,
654      )
655
