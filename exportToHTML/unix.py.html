17   class Unix(PlatformDirsABC):
18       """ 
19       On Unix/Linux, we follow the 
20       `XDG Basedir Spec <https://specifications.freedesktop.org/basedir-spec/basedir-spec-latest.html>`_. The spec allows 
21       overriding directories with environment variables. The examples show are the default values, alongside the name of 
22       the environment variable that overrides them. Makes use of the 
23       `appname <platformdirs.api.PlatformDirsABC.appname>`, 
24       `version <platformdirs.api.PlatformDirsABC.version>`, 
25       `multipath <platformdirs.api.PlatformDirsABC.multipath>`, 
26       `opinion <platformdirs.api.PlatformDirsABC.opinion>`. 
27       """
28   
29       @property
30       def user_data_dir(self) -> str:
31           """ 
32           :return: data directory tied to the user, e.g. ``~/.local/share/$appname/$version`` or 
33            ``$XDG_DATA_HOME/$appname/$version`` 
34           """
35           path = os.environ.get("XDG_DATA_HOME", "")
36           if not path.strip():
37               path = os.path.expanduser("~/.local/share")
38           return self._append_app_name_and_version(path)
39   
40       @property
41       def site_data_dir(self) -> str:
42           """ 
43           :return: data directories shared by users (if `multipath <platformdirs.api.PlatformDirsABC.multipath>` is 
44            enabled and ``XDG_DATA_DIR`` is set and a multi path the response is also a multi path separated by the OS 
45            path separator), e.g. ``/usr/local/share/$appname/$version`` or ``/usr/share/$appname/$version`` 
46           """
47           # XDG default for $XDG_DATA_DIRS; only first, if multipath is False
48           path = os.environ.get("XDG_DATA_DIRS", "")
49           if not path.strip():
50               path = f"/usr/local/share{os.pathsep}/usr/share"
51           return self._with_multi_path(path)
52   
53       def _with_multi_path(self, path: str) -> str:
54           path_list = path.split(os.pathsep)
55           if not self.multipath:
56               path_list = path_list[0:1]
57           path_list = [self._append_app_name_and_version(os.path.expanduser(p)) for p in path_list]
58           return os.pathsep.join(path_list)
59   
60       @property
61       def user_config_dir(self) -> str:
62           """ 
63           :return: config directory tied to the user, e.g. ``~/.config/$appname/$version`` or 
64            ``$XDG_CONFIG_HOME/$appname/$version`` 
65           """
66           path = os.environ.get("XDG_CONFIG_HOME", "")
67           if not path.strip():
68               path = os.path.expanduser("~/.config")
69           return self._append_app_name_and_version(path)
70   
71       @property
72       def site_config_dir(self) -> str:
73           """ 
74           :return: config directories shared by users (if `multipath <platformdirs.api.PlatformDirsABC.multipath>` 
75            is enabled and ``XDG_DATA_DIR`` is set and a multi path the response is also a multi path separated by the OS 
76            path separator), e.g. ``/etc/xdg/$appname/$version`` 
77           """
78           # XDG default for $XDG_CONFIG_DIRS only first, if multipath is False
79           path = os.environ.get("XDG_CONFIG_DIRS", "")
80           if not path.strip():
81               path = "/etc/xdg"
82           return self._with_multi_path(path)

106      @property
107      def user_log_dir(self) -> str:
108          """ 
109          :return: log directory tied to the user, same as `user_data_dir` if not opinionated else ``log`` in it 
110          """
111          path = self.user_cache_dir
112          if self.opinion:
113              path = os.path.join(path, "log")
114          return path
115  
116      @property
117      def user_documents_dir(self) -> str:
118          """ 
119          :return: documents directory tied to the user, e.g. ``~/Documents`` 
120          """
121          documents_dir = _get_user_dirs_folder("XDG_DOCUMENTS_DIR")
122          if documents_dir is None:
123              documents_dir = os.environ.get("XDG_DOCUMENTS_DIR", "").strip()
124              if not documents_dir:
125                  documents_dir = os.path.expanduser("~/Documents")
126  
127          return documents_dir
128  
129      @property
130      def user_runtime_dir(self) -> str:
131          """ 
132          :return: runtime directory tied to the user, e.g. ``/run/user/$(id -u)/$appname/$version`` or 
133           ``$XDG_RUNTIME_DIR/$appname/$version`` 
134          """
135          path = os.environ.get("XDG_RUNTIME_DIR", "")
136          if not path.strip():
137              path = f"/run/user/{getuid()}"
138          return self._append_app_name_and_version(path)
139  
140      @property
141      def site_data_path(self) -> Path:
142          """:return: data path shared by users. Only return first item, even if ``multipath`` is set to ``True``"""
143          return self._first_item_as_path_if_multipath(self.site_data_dir)
144  
145      @property
146      def site_config_path(self) -> Path:
147          """:return: config path shared by the users. Only return first item, even if ``multipath`` is set to ``True``"""
148          return self._first_item_as_path_if_multipath(self.site_config_dir)
149  
150      def _first_item_as_path_if_multipath(self, directory: str) -> Path:
151          if self.multipath:
152              # If multipath is True, the first path is returned.
153              directory = directory.split(os.pathsep)[0]
154          return Path(directory)

157  def _get_user_dirs_folder(key: str) -> Optional[str]:
158      """Return directory from user-dirs.dirs config file. See https://freedesktop.org/wiki/Software/xdg-user-dirs/"""
159      user_dirs_config_path = os.path.join(Unix().user_config_dir, "user-dirs.dirs")
160      if os.path.exists(user_dirs_config_path):
161          parser = ConfigParser()
162  
163          with open(user_dirs_config_path) as stream:
164              # Add fake section header, so ConfigParser doesn't complain
165              parser.read_string(f"[top]\n{stream.read()}")
166  
167          if key not in parser["top"]:
168              return None
169  
170          path = parser["top"][key].strip('"')
171          # Handle relative home paths
172          path = path.replace("$HOME", os.path.expanduser("~"))
173          return path
174  
175      return None
