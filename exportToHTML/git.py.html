29   GIT_VERSION_REGEX = re.compile(
30       r"^git version "  # Prefix.
31       r"(\d+)"  # Major.
32       r"\.(\d+)"  # Dot, minor.
33       r"(?:\.(\d+))?"  # Optional dot, patch.
34       r".*$"  # Suffix, including any pre- and post-release segments we don't care about.
35   )
36   
37   HASH_REGEX = re.compile("^[a-fA-F0-9]{40}$")
38   
39   # SCP (Secure copy protocol) shorthand. e.g. 'git@example.com:foo/bar.git'
40   SCP_REGEX = re.compile(
41       r"""^ 
42       # Optional user, e.g. 'git@' 
43       (\w+@)? 
44       # Server, e.g. 'github.com'. 
45       ([^/:]+): 
46       # The server-side path. e.g. 'user/project.git'. Must start with an 
47       # alphanumeric character so as not to be confusable with a Windows paths 
48       # like 'C:/foo/bar' or 'C:\foo\bar'. 
49       (\w[^:]*) 
50       $""",
51       re.VERBOSE,
52   )
53

55   def looks_like_hash(sha: str) -> bool:
56       return bool(HASH_REGEX.match(sha))

79       def is_immutable_rev_checkout(self, url: str, dest: str) -> bool:
80           _, rev_options = self.get_url_rev_options(hide_url(url))
81           if not rev_options.rev:
82               return False
83           if not self.is_commit_id_equal(dest, rev_options.rev):
84               # the current commit is different from rev,
85               # which means rev was something else than a commit hash
86               return False
87           # return False in the rare case rev is both a commit hash
88           # and a tag or a branch; we don't want to cache in that case
89           # because that branch/tag could point to something else in the future
90           is_tag_or_branch = bool(self.get_revision_sha(dest, rev_options.rev)[0])
91           return not is_tag_or_branch

93       def get_git_version(self) -> Tuple[int, ...]:
94           version = self.run_command(["version"], show_stdout=False, stdout_only=True)
95           match = GIT_VERSION_REGEX.match(version)
96           if not match:
97               logger.warning("Can't parse git version: %s", version)
98               return ()
99           return tuple(int(c) for c in match.groups())

101      @classmethod
102      def get_current_branch(cls, location: str) -> Optional[str]:
103          """ 
104          Return the current branch, or None if HEAD isn't at a branch 
105          (e.g. detached HEAD). 
106          """
107          # git-symbolic-ref exits with empty stdout if "HEAD" is a detached
108          # HEAD rather than a symbolic ref.  In addition, the -q causes the
109          # command to exit with status code 1 instead of 128 in this case
110          # and to suppress the message to stderr.
111          args = ["symbolic-ref", "-q", "HEAD"]
112          output = cls.run_command(
113              args,
114              extra_ok_returncodes=(1,),
115              show_stdout=False,
116              stdout_only=True,
117              cwd=location,
118          )
119          ref = output.strip()
120  
121          if ref.startswith("refs/heads/"):
122              return ref[len("refs/heads/") :]
123  
124          return None

126      @classmethod
127      def get_revision_sha(cls, dest: str, rev: str) -> Tuple[Optional[str], bool]:
128          """ 
129          Return (sha_or_none, is_branch), where sha_or_none is a commit hash 
130          if the revision names a remote branch or tag, otherwise None. 
131   
132          Args: 
133            dest: the repository directory. 
134            rev: the revision name. 
135          """
136          # Pass rev to pre-filter the list.
137          output = cls.run_command(
138              ["show-ref", rev],
139              cwd=dest,
140              show_stdout=False,
141              stdout_only=True,
142              on_returncode="ignore",
143          )
144          refs = {}
145          # NOTE: We do not use splitlines here since that would split on other
146          #       unicode separators, which can be maliciously used to install a
147          #       different revision.
148          for line in output.strip().split("\n"):
149              line = line.rstrip("\r")
150              if not line:
151                  continue
152              try:
153                  ref_sha, ref_name = line.split(" ", maxsplit=2)
154              except ValueError:
155                  # Include the offending line to simplify troubleshooting if
156                  # this error ever occurs.
157                  raise ValueError(f"unexpected show-ref line: {line!r}")
158  
159              refs[ref_name] = ref_sha
160  
161          branch_ref = f"refs/remotes/origin/{rev}"
162          tag_ref = f"refs/tags/{rev}"
163  
164          sha = refs.get(branch_ref)
165          if sha is not None:
166              return (sha, True)
167  
168          sha = refs.get(tag_ref)
169  
170          return (sha, False)

172      @classmethod
173      def _should_fetch(cls, dest: str, rev: str) -> bool:
174          """ 
175          Return true if rev is a ref or is a commit that we don't have locally. 
176   
177          Branches and tags are not considered in this method because they are 
178          assumed to be always available locally (which is a normal outcome of 
179          ``git clone`` and ``git fetch --tags``). 
180          """
181          if rev.startswith("refs/"):
182              # Always fetch remote refs.
183              return True
184  
185          if not looks_like_hash(rev):
186              # Git fetch would fail with abbreviated commits.
187              return False
188  
189          if cls.has_commit(dest, rev):
190              # Don't fetch if we have the commit locally.
191              return False
192  
193          return True

195      @classmethod
196      def resolve_revision(
197          cls, dest: str, url: HiddenText, rev_options: RevOptions
198      ) -> RevOptions:
199          """ 
200          Resolve a revision to a new RevOptions object with the SHA1 of the 
201          branch, tag, or ref if found. 
202   
203          Args: 
204            rev_options: a RevOptions object. 
205          """
206          rev = rev_options.arg_rev
207          # The arg_rev property's implementation for Git ensures that the
208          # rev return value is always non-None.
209          assert rev is not None
210  
211          sha, is_branch = cls.get_revision_sha(dest, rev)
212  
213          if sha is not None:
214              rev_options = rev_options.make_new(sha)
215              rev_options.branch_name = rev if is_branch else None
216  
217              return rev_options
218  
219          # Do not show a warning for the common case of something that has
220          # the form of a Git commit hash.
221          if not looks_like_hash(rev):
222              logger.warning(
223                  "Did not find branch or tag '%s', assuming revision or ref.",
224                  rev,
225              )
226  
227          if not cls._should_fetch(dest, rev):
228              return rev_options
229  
230          # fetch the requested revision
231          cls.run_command(
232              make_command("fetch", "-q", url, rev_options.to_args()),
233              cwd=dest,
234          )
235          # Change the revision to the SHA of the ref we fetched
236          sha = cls.get_revision(dest, rev="FETCH_HEAD")
237          rev_options = rev_options.make_new(sha)
238  
239          return rev_options

241      @classmethod
242      def is_commit_id_equal(cls, dest: str, name: Optional[str]) -> bool:
243          """ 
244          Return whether the current commit hash equals the given name. 
245   
246          Args: 
247            dest: the repository directory. 
248            name: a string name. 
249          """
250          if not name:
251              # Then avoid an unnecessary subprocess call.
252              return False
253  
254          return cls.get_revision(dest) == name

256      def fetch_new(self, dest: str, url: HiddenText, rev_options: RevOptions) -> None:
257          rev_display = rev_options.to_display()
258          logger.info("Cloning %s%s to %s", url, rev_display, display_path(dest))
259          if self.get_git_version() >= (2, 17):
260              # Git added support for partial clone in 2.17
261              # https://git-scm.com/docs/partial-clone
262              # Speeds up cloning by functioning without a complete copy of repository
263              self.run_command(
264                  make_command(
265                      "clone",
266                      "--filter=blob:none",
267                      "-q",
268                      url,
269                      dest,
270                  )
271              )
272          else:
273              self.run_command(make_command("clone", "-q", url, dest))
274  
275          if rev_options.rev:
276              # Then a specific revision was requested.
277              rev_options = self.resolve_revision(dest, url, rev_options)
278              branch_name = getattr(rev_options, "branch_name", None)
279              logger.debug("Rev options %s, branch_name %s", rev_options, branch_name)
280              if branch_name is None:
281                  # Only do a checkout if the current commit id doesn't match
282                  # the requested revision.
283                  if not self.is_commit_id_equal(dest, rev_options.rev):
284                      cmd_args = make_command(
285                          "checkout",
286                          "-q",
287                          rev_options.to_args(),
288                      )
289                      self.run_command(cmd_args, cwd=dest)
290              elif self.get_current_branch(dest) != branch_name:
291                  # Then a specific branch was requested, and that branch
292                  # is not yet checked out.
293                  track_branch = f"origin/{branch_name}"
294                  cmd_args = [
295                      "checkout",
296                      "-b",
297                      branch_name,
298                      "--track",
299                      track_branch,
300                  ]
301                  self.run_command(cmd_args, cwd=dest)
302          else:
303              sha = self.get_revision(dest)
304              rev_options = rev_options.make_new(sha)
305  
306          logger.info("Resolved %s to commit %s", url, rev_options.rev)
307  
308          #: repo may contain submodules
309          self.update_submodules(dest)

311      def switch(self, dest: str, url: HiddenText, rev_options: RevOptions) -> None:
312          self.run_command(
313              make_command("config", "remote.origin.url", url),
314              cwd=dest,
315          )
316          cmd_args = make_command("checkout", "-q", rev_options.to_args())
317          self.run_command(cmd_args, cwd=dest)
318  
319          self.update_submodules(dest)

321      def update(self, dest: str, url: HiddenText, rev_options: RevOptions) -> None:
322          # First fetch changes from the default remote
323          if self.get_git_version() >= (1, 9):
324              # fetch tags in addition to everything else
325              self.run_command(["fetch", "-q", "--tags"], cwd=dest)
326          else:
327              self.run_command(["fetch", "-q"], cwd=dest)
328          # Then reset to wanted revision (maybe even origin/master)
329          rev_options = self.resolve_revision(dest, url, rev_options)
330          cmd_args = make_command("reset", "--hard", "-q", rev_options.to_args())
331          self.run_command(cmd_args, cwd=dest)
332          #: update submodules
333          self.update_submodules(dest)

335      @classmethod
336      def get_remote_url(cls, location: str) -> str:
337          """ 
338          Return URL of the first remote encountered. 
339   
340          Raises RemoteNotFoundError if the repository does not have a remote 
341          url configured. 
342          """
343          # We need to pass 1 for extra_ok_returncodes since the command
344          # exits with return code 1 if there are no matching lines.
345          stdout = cls.run_command(
346              ["config", "--get-regexp", r"remote\..*\.url"],
347              extra_ok_returncodes=(1,),
348              show_stdout=False,
349              stdout_only=True,
350              cwd=location,
351          )
352          remotes = stdout.splitlines()
353          try:
354              found_remote = remotes[0]
355          except IndexError:
356              raise RemoteNotFoundError
357  
358          for remote in remotes:
359              if remote.startswith("remote.origin.url "):
360                  found_remote = remote
361                  break
362          url = found_remote.split(" ")[1]
363          return cls._git_remote_to_pip_url(url.strip())

365      @staticmethod
366      def _git_remote_to_pip_url(url: str) -> str:
367          """ 
368          Convert a remote url from what git uses to what pip accepts. 
369   
370          There are 3 legal forms **url** may take: 
371   
372              1. A fully qualified url: ssh://git@example.com/foo/bar.git 
373              2. A local project.git folder: /path/to/bare/repository.git 
374              3. SCP shorthand for form 1: git@example.com:foo/bar.git 
375   
376          Form 1 is output as-is. Form 2 must be converted to URI and form 3 must 
377          be converted to form 1. 
378   
379          See the corresponding test test_git_remote_url_to_pip() for examples of 
380          sample inputs/outputs. 
381          """
382          if re.match(r"\w+://", url):
383              # This is already valid. Pass it though as-is.
384              return url
385          if os.path.exists(url):
386              # A local bare remote (git clone --mirror).
387              # Needs a file:// prefix.
388              return pathlib.PurePath(url).as_uri()
389          scp_match = SCP_REGEX.match(url)
390          if scp_match:
391              # Add an ssh:// prefix and replace the ':' with a '/'.
392              return scp_match.expand(r"ssh://\1\2/\3")
393          # Otherwise, bail out.
394          raise RemoteNotValidError(url)

396      @classmethod
397      def has_commit(cls, location: str, rev: str) -> bool:
398          """ 
399          Check if rev is a commit that is available in the local repository. 
400          """
401          try:
402              cls.run_command(
403                  ["rev-parse", "-q", "--verify", "sha^" + rev],
404                  cwd=location,
405                  log_failed_cmd=False,
406              )
407          except InstallationError:
408              return False
409          else:
410              return True

412      @classmethod
413      def get_revision(cls, location: str, rev: Optional[str] = None) -> str:
414          if rev is None:
415              rev = "HEAD"
416          current_rev = cls.run_command(
417              ["rev-parse", rev],
418              show_stdout=False,
419              stdout_only=True,
420              cwd=location,
421          )
422          return current_rev.strip()

424      @classmethod
425      def get_subdirectory(cls, location: str) -> Optional[str]:
426          """ 
427          Return the path to Python project root, relative to the repo root. 
428          Return None if the project root is in the repo root. 
429          """
430          # find the repo root
431          git_dir = cls.run_command(
432              ["rev-parse", "--git-dir"],
433              show_stdout=False,
434              stdout_only=True,
435              cwd=location,
436          ).strip()
437          if not os.path.isabs(git_dir):
438              git_dir = os.path.join(location, git_dir)
439          repo_root = os.path.abspath(os.path.join(git_dir, ".."))
440          return find_path_to_project_root_from_repo_root(location, repo_root)

442      @classmethod
443      def get_url_rev_and_auth(cls, url: str) -> Tuple[str, Optional[str], AuthInfo]:
444          """ 
445          Prefixes stub URLs like 'user@hostname:user/repo.git' with 'ssh://'. 
446          That's required because although they use SSH they sometimes don't 
447          work with a ssh:// scheme (e.g. GitHub). But we need a scheme for 
448          parsing. Hence we remove it again afterwards and return it as a stub. 
449          """
450          # Works around an apparent Git bug
451          # (see https://article.gmane.org/gmane.comp.version-control.git/146500)
452          scheme, netloc, path, query, fragment = urlsplit(url)
453          if scheme.endswith("file"):
454              initial_slashes = path[: -len(path.lstrip("/"))]
455              newpath = initial_slashes + urllib.request.url2pathname(path).replace(
456                  "\\", "/"
457              ).lstrip("/")
458              after_plus = scheme.find("+") + 1
459              url = scheme[:after_plus] + urlunsplit(
460                  (scheme[after_plus:], netloc, newpath, query, fragment),
461              )
462  
463          if "://" not in url:
464              assert "file:" not in url
465              url = url.replace("git+", "git+ssh://")
466              url, rev, user_pass = super().get_url_rev_and_auth(url)
467              url = url.replace("ssh://", "")
468          else:
469              url, rev, user_pass = super().get_url_rev_and_auth(url)
470  
471          return url, rev, user_pass

473      @classmethod
474      def update_submodules(cls, location: str) -> None:
475          if not os.path.exists(os.path.join(location, ".gitmodules")):
476              return
477          cls.run_command(
478              ["submodule", "update", "--init", "--recursive", "-q"],
479              cwd=location,
480          )

482      @classmethod
483      def get_repository_root(cls, location: str) -> Optional[str]:
484          loc = super().get_repository_root(location)
485          if loc:
486              return loc
487          try:
488              r = cls.run_command(
489                  ["rev-parse", "--show-toplevel"],
490                  cwd=location,
491                  show_stdout=False,
492                  stdout_only=True,
493                  on_returncode="raise",
494                  log_failed_cmd=False,
495              )
496          except BadCommand:
497              logger.debug(
498                  "could not determine if %s is under git control "
499                  "because git is not available",
500                  location,
501              )
502              return None
503          except InstallationError:
504              return None
505          return os.path.normpath(r.rstrip("\r\n"))
