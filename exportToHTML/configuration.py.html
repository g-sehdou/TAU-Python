34   class LogLevel(object):
35       names = [
36           "NOTSET", "CRITICAL", "FATAL", "ERROR",
37           "WARNING", "WARN", "INFO", "DEBUG",
38       ]
39   
40       @staticmethod
41       def parse(levelname, unknown_level=None):
42           """ 
43           Convert levelname into a numeric log level. 
44    
45           :param levelname: Logging levelname (as string) 
46           :param unknown_level: Used if levelname is unknown (optional). 
47           :return: Numeric log-level or unknown_level, if levelname is unknown. 
48           """
49           return getattr(logging, levelname.upper(), unknown_level)
50   
51       @classmethod
52       def parse_type(cls, levelname):
53           level = cls.parse(levelname, Unknown)
54           if level is Unknown:
55               message = "%s is unknown, use: %s" % \
56                         (levelname, ", ".join(cls.names[1:]))
57               raise argparse.ArgumentTypeError(message)
58           return level
59   
60       @staticmethod
61       def to_string(level):
62           return logging.getLevelName(level)

368  def read_configuration(path):
369      # pylint: disable=too-many-locals, too-many-branches
370      config = ConfigParser()
371      config.optionxform = str    # -- SUPPORT: case-sensitive keys
372      config.read(path)
373      config_dir = os.path.dirname(path)
374      result = {}
375      for fixed, keywords in options:
376          if "dest" in keywords:
377              dest = keywords["dest"]
378          else:
379              for opt in fixed:
380                  if opt.startswith("--"):
381                      dest = opt[2:].replace("-", "_")
382                  else:
383                      assert len(opt) == 2
384                      dest = opt[1:]
385          if dest in "tags_help lang_list lang_help version".split():
386              continue
387          if not config.has_option("behave", dest):
388              continue
389          action = keywords.get("action", "store")
390          if action == "store":
391              use_raw_value = dest in raw_value_options
392              result[dest] = config.get("behave", dest, raw=use_raw_value)
393          elif action in ("store_true", "store_false"):
394              result[dest] = config.getboolean("behave", dest)
395          elif action == "append":
396              if dest == "userdata_defines":
397                  continue    # -- SKIP-CONFIGFILE: Command-line only option.
398              result[dest] = \
399                  [s.strip() for s in config.get("behave", dest).splitlines()]
400          else:
401              raise ValueError('action "%s" not implemented' % action)
402  
403      # -- STEP: format/outfiles coupling
404      if "format" in result:
405          # -- OPTIONS: format/outfiles are coupled in configuration file.
406          formatters = result["format"]
407          formatter_size = len(formatters)
408          outfiles = result.get("outfiles", [])
409          outfiles_size = len(outfiles)
410          if outfiles_size < formatter_size:
411              for formatter_name in formatters[outfiles_size:]:
412                  outfile = "%s.output" % formatter_name
413                  outfiles.append(outfile)
414              result["outfiles"] = outfiles
415          elif len(outfiles) > formatter_size:
416              print("CONFIG-ERROR: Too many outfiles (%d) provided." %
417                    outfiles_size)
418              result["outfiles"] = outfiles[:formatter_size]
419  
420      for paths_name in ("paths", "outfiles"):
421          if paths_name in result:
422              # -- Evaluate relative paths relative to location.
423              # NOTE: Absolute paths are preserved by os.path.join().
424              paths = result[paths_name]
425              result[paths_name] = \
426                  [os.path.normpath(os.path.join(config_dir, p)) for p in paths]
427  
428      # -- STEP: Special additional configuration sections.
429      # SCHEMA: config_section: data_name
430      special_config_section_map = {
431          "behave.formatters": "more_formatters",
432          "behave.userdata":   "userdata",
433      }
434      for section_name, data_name in special_config_section_map.items():
435          result[data_name] = {}
436          if config.has_section(section_name):
437              result[data_name].update(config.items(section_name))
438  
439      return result

442  def config_filenames():
443      paths = ["./", os.path.expanduser("~")]
444      if sys.platform in ("cygwin", "win32") and "APPDATA" in os.environ:
445          paths.append(os.path.join(os.environ["APPDATA"]))
446  
447      for path in reversed(paths):
448          for filename in reversed(
449                  ("behave.ini", ".behaverc", "setup.cfg", "tox.ini")):
450              filename = os.path.join(path, filename)
451              if os.path.isfile(filename):
452                  yield filename
453

455  def load_configuration(defaults, verbose=False):
456      for filename in config_filenames():
457          if verbose:
458              print('Loading config defaults from "%s"' % filename)
459          defaults.update(read_configuration(filename))
460  
461      if verbose:
462          print("Using defaults:")
463          for k, v in six.iteritems(defaults):
464              print("%15s %s" % (k, v))

467  def setup_parser():
468      # construct the parser
469      # usage = "%(prog)s [options] [ [FILE|DIR|URL][:LINE[:LINE]*] ]+"
470      usage = "%(prog)s [options] [ [DIR|FILE|FILE:LINE] ]+"
471      description = """\ 
472      Run a number of feature tests with behave."""
473      more = """ 
474      EXAMPLES: 
475      behave features/ 
476      behave features/one.feature features/two.feature 
477      behave features/one.feature:10 
478      behave @features.txt 
479      """
480      parser = argparse.ArgumentParser(usage=usage, description=description)
481      for fixed, keywords in options:
482          if not fixed:
483              continue    # -- CONFIGFILE only.
484          if "config_help" in keywords:
485              keywords = dict(keywords)
486              del keywords["config_help"]
487          parser.add_argument(*fixed, **keywords)
488      parser.add_argument("paths", nargs="*",
489                          help="Feature directory, file or file location (FILE:LINE).")
490      return parser

493  class Configuration(object):
494      """Configuration object for behave and behave runners."""
495      # pylint: disable=too-many-instance-attributes
496      defaults = dict(
497          color=sys.platform != "win32",
498          show_snippets=True,
499          show_skipped=True,
500          dry_run=False,
501          show_source=True,
502          show_timings=True,
503          stdout_capture=True,
504          stderr_capture=True,
505          log_capture=True,
506          logging_format="%(levelname)s:%(name)s:%(message)s",
507          logging_level=logging.INFO,
508          steps_catalog=False,
509          summary=True,
510          junit=False,
511          stage=None,
512          userdata={},
513          # -- SPECIAL:
514          default_format="pretty",    # -- Used when no formatters are configured.
515          default_tags="",            # -- Used when no tags are defined.
516          scenario_outline_annotation_schema=u"{name} -- @{row.id} {examples.name}"
517      )
518      cmdline_only_options = set("userdata_defines")
519  
520      def __init__(self, command_args=None, load_config=True, verbose=None,
521                   **kwargs):
522          """ 
523          Constructs a behave configuration object. 
524            * loads the configuration defaults (if needed). 
525            * process the command-line args 
526            * store the configuration results 
527   
528          :param command_args: Provide command args (as sys.argv). 
529              If command_args is None, sys.argv[1:] is used. 
530          :type command_args: list<str>, str 
531          :param load_config: Indicate if configfile should be loaded (=true) 
532          :param verbose: Indicate if diagnostic output is enabled 
533          :param kwargs:  Used to hand-over/overwrite default values. 
534          """
535          # pylint: disable=too-many-branches, too-many-statements
536          if command_args is None:
537              command_args = sys.argv[1:]
538          elif isinstance(command_args, six.string_types):
539              encoding = select_best_encoding() or "utf-8"
540              if six.PY2 and isinstance(command_args, six.text_type):
541                  command_args = command_args.encode(encoding)
542              elif six.PY3 and isinstance(command_args, six.binary_type):
543                  command_args = command_args.decode(encoding)
544              command_args = shlex.split(command_args)
545          elif isinstance(command_args, (list, tuple)):
546              command_args = to_texts(command_args)
547  
548          if verbose is None:
549              # -- AUTO-DISCOVER: Verbose mode from command-line args.
550              verbose = ("-v" in command_args) or ("--verbose" in command_args)
551  
552          self.version = None
553          self.tags_help = None
554          self.lang_list = None
555          self.lang_help = None
556          self.default_tags = None
557          self.junit = None
558          self.logging_format = None
559          self.logging_datefmt = None
560          self.name = None
561          self.scope = None
562          self.steps_catalog = None
563          self.userdata = None
564          self.wip = None
565  
566          defaults = self.defaults.copy()
567          for name, value in six.iteritems(kwargs):
568              defaults[name] = value
569          self.defaults = defaults
570          self.formatters = []
571          self.reporters = []
572          self.name_re = None
573          self.outputs = []
574          self.include_re = None
575          self.exclude_re = None
576          self.scenario_outline_annotation_schema = None  # pylint: disable=invalid-name
577          self.steps_dir = "steps"
578          self.environment_file = "environment.py"
579          self.userdata_defines = None
580          self.more_formatters = None
581          if load_config:
582              load_configuration(self.defaults, verbose=verbose)
583          parser = setup_parser()
584          parser.set_defaults(**self.defaults)
585          args = parser.parse_args(command_args)
586          for key, value in six.iteritems(args.__dict__):
587              if key.startswith("_") and key not in self.cmdline_only_options:
588                  continue
589              setattr(self, key, value)
590  
591          self.paths = [os.path.normpath(path) for path in self.paths]
592          self.setup_outputs(args.outfiles)
593  
594          if self.steps_catalog:
595              # -- SHOW STEP-CATALOG: As step summary.
596              self.default_format = "steps.catalog"
597              self.format = ["steps.catalog"]
598              self.dry_run = True
599              self.summary = False
600              self.show_skipped = False
601              self.quiet = True
602  
603          if self.wip:
604              # Only run scenarios tagged with "wip".
605              # Additionally:
606              #  * use the "plain" formatter (per default)
607              #  * do not capture stdout or logging output and
608              #  * stop at the first failure.
609              self.default_format = "plain"
610              self.tags = ["wip"] + self.default_tags.split()
611              self.color = False
612              self.stop = True
613              self.log_capture = False
614              self.stdout_capture = False
615  
616          self.tags = TagExpression(self.tags or self.default_tags.split())
617  
618          if self.quiet:
619              self.show_source = False
620              self.show_snippets = False
621  
622          if self.exclude_re:
623              self.exclude_re = re.compile(self.exclude_re)
624  
625          if self.include_re:
626              self.include_re = re.compile(self.include_re)
627          if self.name:
628              # -- SELECT: Scenario-by-name, build regular expression.
629              self.name_re = self.build_name_re(self.name)
630  
631          if self.stage is None:  # pylint: disable=access-member-before-definition
632              # -- USE ENVIRONMENT-VARIABLE, if stage is undefined.
633              self.stage = os.environ.get("BEHAVE_STAGE", None)
634          self.setup_stage(self.stage)
635          self.setup_model()
636          self.setup_userdata()
637  
638          # -- FINALLY: Setup Reporters and Formatters
639          # NOTE: Reporters and Formatters can now use userdata information.
640          if self.junit:
641              # Buffer the output (it will be put into Junit report)
642              self.stdout_capture = True
643              self.stderr_capture = True
644              self.log_capture = True
645              self.reporters.append(JUnitReporter(self))
646          if self.summary:
647              self.reporters.append(SummaryReporter(self))
648  
649          self.setup_formats()
650          unknown_formats = self.collect_unknown_formats()
651          if unknown_formats:
652              parser.error("format=%s is unknown" % ", ".join(unknown_formats))

686      @staticmethod
687      def build_name_re(names):
688          """ 
689          Build regular expression for scenario selection by name 
690          by using a list of name parts or name regular expressions. 
691   
692          :param names: List of name parts or regular expressions (as text). 
693          :return: Compiled regular expression to use. 
694          """
695          # -- NOTE: re.LOCALE is removed in Python 3.6 (deprecated in Python 3.5)
696          # flags = (re.UNICODE | re.LOCALE)
697          # -- ENSURE: Names are all unicode/text values (for issue #606).
698          names = to_texts(names)
699          pattern = u"|".join(names)
700          return re.compile(pattern, flags=re.UNICODE)

702      def exclude(self, filename):
703          if isinstance(filename, FileLocation):
704              filename = six.text_type(filename)
705  
706          if self.include_re and self.include_re.search(filename) is None:
707              return True
708          if self.exclude_re and self.exclude_re.search(filename) is not None:
709              return True

712      def setup_logging(self, level=None, configfile=None, **kwargs):
713          """ 
714          Support simple setup of logging subsystem. 
715          Ensures that the logging level is set. 
716          But note that the logging setup can only occur once. 
717   
718          SETUP MODES: 
719            * :func:`logging.config.fileConfig()`, if ``configfile`` is provided. 
720            * :func:`logging.basicConfig()`, otherwise. 
721   
722          .. code-block: python 
723              # -- FILE: features/environment.py 
724              def before_all(context): 
725                  context.config.setup_logging() 
726   
727          :param level:       Logging level of root logger. 
728                              If None, use :attr:`logging_level` value. 
729          :param configfile:  Configuration filename for fileConfig() setup. 
730          :param kwargs:      Passed to :func:`logging.basicConfig()` 
731          """
732          if level is None:
733              level = self.logging_level      # pylint: disable=no-member
734  
735          if configfile:
736              from logging.config import fileConfig
737              fileConfig(configfile)
738          else:
739              # pylint: disable=no-member
740              format_ = kwargs.pop("format", self.logging_format)
741              datefmt = kwargs.pop("datefmt", self.logging_datefmt)
742              logging.basicConfig(format=format_, datefmt=datefmt, **kwargs)
743          # -- ENSURE: Default log level is set
744          #    (even if logging subsystem is already configured).
745          logging.getLogger().setLevel(level)

752      def setup_stage(self, stage=None):
753          """Setup the test stage that selects a different set of 
754          steps and environment implementations. 
755   
756          :param stage:   Name of current test stage (as string or None). 
757   
758          EXAMPLE:: 
759   
760              # -- SETUP DEFAULT TEST STAGE (unnamed): 
761              config = Configuration() 
762              config.setup_stage() 
763              assert config.steps_dir == "steps" 
764              assert config.environment_file == "environment.py" 
765   
766              # -- SETUP PRODUCT TEST STAGE: 
767              config.setup_stage("product") 
768              assert config.steps_dir == "product_steps" 
769              assert config.environment_file == "product_environment.py" 
770          """
771          steps_dir = "steps"
772          environment_file = "environment.py"
773          if stage:
774              # -- USE A TEST STAGE: Select different set of implementations.
775              prefix = stage + "_"
776              steps_dir = prefix + steps_dir
777              environment_file = prefix + environment_file
778          self.steps_dir = steps_dir
779          self.environment_file = environment_file
