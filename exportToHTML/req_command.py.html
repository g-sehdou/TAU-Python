48   class SessionCommandMixin(CommandContextMixIn):
49   
50       """ 
51       A class mixin for command classes needing _build_session(). 
52       """
53   
54       def __init__(self) -> None:
55           super().__init__()
56           self._session: Optional[PipSession] = None
57   
58       @classmethod
59       def _get_index_urls(cls, options: Values) -> Optional[List[str]]:
60           """Return a list of index urls from user-provided options."""
61           index_urls = []
62           if not getattr(options, "no_index", False):
63               url = getattr(options, "index_url", None)
64               if url:
65                   index_urls.append(url)
66           urls = getattr(options, "extra_index_urls", None)
67           if urls:
68               index_urls.extend(urls)
69           # Return None rather than an empty list
70           return index_urls or None
71   
72       def get_default_session(self, options: Values) -> PipSession:
73           """Get a default-managed session."""
74           if self._session is None:
75               self._session = self.enter_context(self._build_session(options))
76               # there's no type annotation on requests.Session, so it's
77               # automatically ContextManager[Any] and self._session becomes Any,
78               # then https://github.com/python/mypy/issues/7696 kicks in
79               assert self._session is not None
80           return self._session
81   
82       def _build_session(
83           self,
84           options: Values,
85           retries: Optional[int] = None,
86           timeout: Optional[int] = None,
87       ) -> PipSession:
88           assert not options.cache_dir or os.path.isabs(options.cache_dir)
89           session = PipSession(
90               cache=(
91                   os.path.join(options.cache_dir, "http") if options.cache_dir else None
92               ),
93               retries=retries if retries is not None else options.retries,
94               trusted_hosts=options.trusted_hosts,
95               index_urls=self._get_index_urls(options),
96           )
97   
98           # Handle custom ca-bundles from the user
99           if options.cert:
100              session.verify = options.cert
101  
102          # Handle SSL client certificate
103          if options.client_cert:
104              session.cert = options.client_cert
105  
106          # Handle timeouts
107          if options.timeout or timeout:
108              session.timeout = timeout if timeout is not None else options.timeout
109  
110          # Handle configured proxies
111          if options.proxy:
112              session.proxies = {
113                  "http": options.proxy,
114                  "https": options.proxy,
115              }
116  
117          # Determine if we can prompt the user for authentication or not
118          session.auth.prompting = not options.no_input
119  
120          return session

124  
125      """ 
126      Abstract base class for commands with the index_group options. 
127   
128      This also corresponds to the commands that permit the pip version check. 
129      """

131      def handle_pip_version_check(self, options: Values) -> None:
132          """ 
133          Do the pip version check if not disabled. 
134   
135          This overrides the default behavior of not doing the check. 
136          """
137          # Make sure the index_group options are present.
138          assert hasattr(options, "no_index")
139  
140          if options.disable_pip_version_check or options.no_index:
141              return
142  
143          # Otherwise, check if we're using the latest version of pip available.
144          session = self._build_session(
145              options, retries=0, timeout=min(5, options.timeout)
146          )
147          with session:
148              pip_self_version_check(session, options)

158  def warn_if_run_as_root() -> None:
159      """Output a warning for sudo users on Unix. 
160   
161      In a virtual environment, sudo pip still writes to virtualenv. 
162      On Windows, users may run pip as Administrator without issues. 
163      This warning only applies to Unix root users outside of virtualenv. 
164      """
165      if running_under_virtualenv():
166          return
167      if not hasattr(os, "getuid"):
168          return
169      # On Windows, there are no "system managed" Python packages. Installing as
170      # Administrator via pip is the correct way of updating system environments.
171      #
172      # We choose sys.platform over utils.compat.WINDOWS here to enable Mypy platform
173      # checks: https://mypy.readthedocs.io/en/stable/common_issues.html
174      if sys.platform == "win32" or sys.platform == "cygwin":
175          return
176  
177      if os.getuid() != 0:
178          return
179  
180      logger.warning(
181          "Running pip as the 'root' user can result in broken permissions and "
182          "conflicting behaviour with the system package manager. "
183          "It is recommended to use a virtual environment instead: "
184          "https://pip.pypa.io/warnings/venv"
185      )

188  def with_cleanup(func: Any) -> Any:
189      """Decorator for common logic related to managing temporary 
190      directories. 
191      """
192  
193      def configure_tempdir_registry(registry: TempDirectoryTypeRegistry) -> None:
194          for t in KEEPABLE_TEMPDIR_TYPES:
195              registry.set_delete(t, False)
196  
197      def wrapper(
198          self: RequirementCommand, options: Values, args: List[Any]
199      ) -> Optional[int]:
200          assert self.tempdir_registry is not None
201          if options.no_clean:
202              configure_tempdir_registry(self.tempdir_registry)
203  
204          try:
205              return func(self, options, args)
206          except PreviousBuildDirError:
207              # This kind of conflict can occur when the user passes an explicit
208              # build directory with a pre-existing folder. In that case we do
209              # not want to accidentally remove it.
210              configure_tempdir_registry(self.tempdir_registry)
211              raise
212  
213      return wrapper

222      @staticmethod
223      def determine_resolver_variant(options: Values) -> str:
224          """Determines which resolver should be used, based on the given options."""
225          if "legacy-resolver" in options.deprecated_features_enabled:
226              return "legacy"
227  
228          return "2020-resolver"

230      @classmethod
231      def make_requirement_preparer(
232          cls,
233          temp_build_dir: TempDirectory,
234          options: Values,
235          req_tracker: RequirementTracker,
236          session: PipSession,
237          finder: PackageFinder,
238          use_user_site: bool,
239          download_dir: Optional[str] = None,
240      ) -> RequirementPreparer:
241          """ 
242          Create a RequirementPreparer instance for the given parameters. 
243          """
244          temp_build_dir_path = temp_build_dir.path
245          assert temp_build_dir_path is not None
246  
247          resolver_variant = cls.determine_resolver_variant(options)
248          if resolver_variant == "2020-resolver":
249              lazy_wheel = "fast-deps" in options.features_enabled
250              if lazy_wheel:
251                  logger.warning(
252                      "pip is using lazily downloaded wheels using HTTP "
253                      "range requests to obtain dependency information. "
254                      "This experimental feature is enabled through "
255                      "--use-feature=fast-deps and it is not ready for "
256                      "production."
257                  )
258          else:
259              lazy_wheel = False
260              if "fast-deps" in options.features_enabled:
261                  logger.warning(
262                      "fast-deps has no effect when used with the legacy resolver."
263                  )
264  
265          in_tree_build = "out-of-tree-build" not in options.deprecated_features_enabled
266          if "in-tree-build" in options.features_enabled:
267              deprecated(
268                  reason="In-tree builds are now the default.",
269                  replacement="to remove the --use-feature=in-tree-build flag",
270                  gone_in="22.1",
271              )
272          if "out-of-tree-build" in options.deprecated_features_enabled:
273              deprecated(
274                  reason="Out-of-tree builds are deprecated.",
275                  replacement=None,
276                  gone_in="22.1",
277              )
278  
279          return RequirementPreparer(
280              build_dir=temp_build_dir_path,
281              src_dir=options.src_dir,
282              download_dir=download_dir,
283              build_isolation=options.build_isolation,
284              req_tracker=req_tracker,
285              session=session,
286              progress_bar=options.progress_bar,
287              finder=finder,
288              require_hashes=options.require_hashes,
289              use_user_site=use_user_site,
290              lazy_wheel=lazy_wheel,
291              in_tree_build=in_tree_build,
292          )

294      @classmethod
295      def make_resolver(
296          cls,
297          preparer: RequirementPreparer,
298          finder: PackageFinder,
299          options: Values,
300          wheel_cache: Optional[WheelCache] = None,
301          use_user_site: bool = False,
302          ignore_installed: bool = True,
303          ignore_requires_python: bool = False,
304          force_reinstall: bool = False,
305          upgrade_strategy: str = "to-satisfy-only",
306          use_pep517: Optional[bool] = None,
307          py_version_info: Optional[Tuple[int, ...]] = None,
308      ) -> BaseResolver:
309          """ 
310          Create a Resolver instance for the given parameters. 
311          """
312          make_install_req = partial(
313              install_req_from_req_string,
314              isolated=options.isolated_mode,
315              use_pep517=use_pep517,
316          )
317          resolver_variant = cls.determine_resolver_variant(options)
318          # The long import name and duplicated invocation is needed to convince
319          # Mypy into correctly typechecking. Otherwise it would complain the
320          # "Resolver" class being redefined.
321          if resolver_variant == "2020-resolver":
322              import pip._internal.resolution.resolvelib.resolver
323  
324              return pip._internal.resolution.resolvelib.resolver.Resolver(
325                  preparer=preparer,
326                  finder=finder,
327                  wheel_cache=wheel_cache,
328                  make_install_req=make_install_req,
329                  use_user_site=use_user_site,
330                  ignore_dependencies=options.ignore_dependencies,
331                  ignore_installed=ignore_installed,
332                  ignore_requires_python=ignore_requires_python,
333                  force_reinstall=force_reinstall,
334                  upgrade_strategy=upgrade_strategy,
335                  py_version_info=py_version_info,
336              )
337          import pip._internal.resolution.legacy.resolver
338  
339          return pip._internal.resolution.legacy.resolver.Resolver(
340              preparer=preparer,
341              finder=finder,
342              wheel_cache=wheel_cache,
343              make_install_req=make_install_req,
344              use_user_site=use_user_site,
345              ignore_dependencies=options.ignore_dependencies,
346              ignore_installed=ignore_installed,
347              ignore_requires_python=ignore_requires_python,
348              force_reinstall=force_reinstall,
349              upgrade_strategy=upgrade_strategy,
350              py_version_info=py_version_info,
351          )

353      def get_requirements(
354          self,
355          args: List[str],
356          options: Values,
357          finder: PackageFinder,
358          session: PipSession,
359      ) -> List[InstallRequirement]:
360          """ 
361          Parse command-line arguments into the corresponding requirements. 
362          """
363          requirements: List[InstallRequirement] = []
364          for filename in options.constraints:
365              for parsed_req in parse_requirements(
366                  filename,
367                  constraint=True,
368                  finder=finder,
369                  options=options,
370                  session=session,
371              ):
372                  req_to_add = install_req_from_parsed_requirement(
373                      parsed_req,
374                      isolated=options.isolated_mode,
375                      user_supplied=False,
376                  )
377                  requirements.append(req_to_add)
378  
379          for req in args:
380              req_to_add = install_req_from_line(
381                  req,
382                  None,
383                  isolated=options.isolated_mode,
384                  use_pep517=options.use_pep517,
385                  user_supplied=True,
386              )
387              requirements.append(req_to_add)
388  
389          for req in options.editables:
390              req_to_add = install_req_from_editable(
391                  req,
392                  user_supplied=True,
393                  isolated=options.isolated_mode,
394                  use_pep517=options.use_pep517,
395              )
396              requirements.append(req_to_add)
397  
398          # NOTE: options.require_hashes may be set if --require-hashes is True
399          for filename in options.requirements:
400              for parsed_req in parse_requirements(
401                  filename, finder=finder, options=options, session=session
402              ):
403                  req_to_add = install_req_from_parsed_requirement(
404                      parsed_req,
405                      isolated=options.isolated_mode,
406                      use_pep517=options.use_pep517,
407                      user_supplied=True,
408                  )
409                  requirements.append(req_to_add)
410  
411          # If any requirement has hash options, enable hash checking.
412          if any(req.has_hash_options for req in requirements):
413              options.require_hashes = True
414  
415          if not (args or options.editables or options.requirements):
416              opts = {"name": self.name}
417              if options.find_links:
418                  raise CommandError(
419                      "You must give at least one requirement to {name} "
420                      '(maybe you meant "pip {name} {links}"?)'.format(
421                          **dict(opts, links=" ".join(options.find_links))
422                      )
423                  )
424              else:
425                  raise CommandError(
426                      "You must give at least one requirement to {name} "
427                      '(see "pip help {name}")'.format(**opts)
428                  )
429  
430          return requirements

432      @staticmethod
433      def trace_basic_info(finder: PackageFinder) -> None:
434          """ 
435          Trace basic information about the provided objects. 
436          """
437          # Display where finder is looking for packages
438          search_scope = finder.search_scope
439          locations = search_scope.get_formatted_locations()
440          if locations:
441              logger.info(locations)

443      def _build_package_finder(
444          self,
445          options: Values,
446          session: PipSession,
447          target_python: Optional[TargetPython] = None,
448          ignore_requires_python: Optional[bool] = None,
449      ) -> PackageFinder:
450          """ 
451          Create a package finder appropriate to this requirement command. 
452   
453          :param ignore_requires_python: Whether to ignore incompatible 
454              "Requires-Python" values in links. Defaults to False. 
455          """
456          link_collector = LinkCollector.create(session, options=options)
457          selection_prefs = SelectionPreferences(
458              allow_yanked=True,
459              format_control=options.format_control,
460              allow_all_prereleases=options.pre,
461              prefer_binary=options.prefer_binary,
462              ignore_requires_python=ignore_requires_python,
463          )
464  
465          return PackageFinder.create(
466              link_collector=link_collector,
467              selection_prefs=selection_prefs,
468              target_python=target_python,
469          )
