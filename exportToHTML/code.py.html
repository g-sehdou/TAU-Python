54   class Code:
55       """Wrapper around Python code objects."""
56   
57       __slots__ = ("raw",)

59       def __init__(self, obj: CodeType) -> None:

62       @classmethod
63       def from_function(cls, obj: object) -> "Code":
64           return cls(getrawcode(obj))
65   
66       def __eq__(self, other):
67           return self.raw == other.raw

69       # Ignore type because of https://github.com/python/mypy/issues/4266.
70       __hash__ = None  # type: ignore

72       @property
73       def firstlineno(self) -> int:
74           return self.raw.co_firstlineno - 1

76       @property
77       def name(self) -> str:
78           return self.raw.co_name
79   
80       @property
81       def path(self) -> Union[py.path.local, str]:
82           """Return a path object pointing to source code, or an ``str`` in 
83           case of ``OSError`` / non-existing file."""
84           if not self.raw.co_filename:
85               return ""
86           try:
87               p = py.path.local(self.raw.co_filename)
88               # maybe don't try this checking
89               if not p.check():
90                   raise OSError("py.path check failed.")
91               return p
92           except OSError:
93               # XXX maybe try harder like the weird logic
94               # in the standard lib [linecache.updatecache] does?
95               return self.raw.co_filename

97       @property
98       def fullsource(self) -> Optional["Source"]:
99           """Return a _pytest._code.Source object for the full source file of the code."""
100          full, _ = findsource(self.raw)
101          return full

103      def source(self) -> "Source":
104          """Return a _pytest._code.Source object for the code object's source only."""
105          # return source only for that part of code
106          return Source(self.raw)

108      def getargs(self, var: bool = False) -> Tuple[str, ...]:
109          """Return a tuple with the argument names for the code object. 
110   
111          If 'var' is set True also return the names of the variable and 
112          keyword arguments when present. 
113          """
114          # Handy shortcut for getting args.
115          raw = self.raw
116          argcount = raw.co_argcount
117          if var:
118              argcount += raw.co_flags & CO_VARARGS
119              argcount += raw.co_flags & CO_VARKEYWORDS
120          return raw.co_varnames[:argcount]

123  class Frame:
124      """Wrapper around a Python frame holding f_locals and f_globals 
125      in which expressions can be evaluated."""
126  
127      __slots__ = ("raw",)
128  
129      def __init__(self, frame: FrameType) -> None:
130          self.raw = frame

132      @property
133      def lineno(self) -> int:
134          return self.raw.f_lineno - 1
135  
136      @property
137      def f_globals(self) -> Dict[str, Any]:
138          return self.raw.f_globals
139  
140      @property
141      def f_locals(self) -> Dict[str, Any]:
142          return self.raw.f_locals
143  
144      @property
145      def code(self) -> Code:
146          return Code(self.raw.f_code)
147  
148      @property
149      def statement(self) -> "Source":
150          """Statement this frame is at."""
151          if self.code.fullsource is None:
152              return Source("")

155      def eval(self, code, **vars):
156          """Evaluate 'code' in the frame. 
157   
158          'vars' are optional additional local variables. 
159   
160          Returns the result of the evaluation. 
161          """
162          f_locals = self.f_locals.copy()
163          f_locals.update(vars)
164          return eval(code, self.f_globals, f_locals)
165  
166      def repr(self, object: object) -> str:
167          """Return a 'safe' (non-recursive, one-line) string repr for 'object'."""
168          return saferepr(object)
169  
170      def getargs(self, var: bool = False):
171          """Return a list of tuples (name, value) for all arguments. 
172   
173          If 'var' is set True, also include the variable and keyword arguments 
174          when present. 
175          """
176          retval = []
177          for arg in self.code.getargs(var):
178              try:
179                  retval.append((arg, self.f_locals[arg]))
180              except KeyError:
181                  pass  # this can occur when using Psyco
182          return retval

238      def getsource(self, astcache=None) -> Optional["Source"]:
239          """Return failing source code."""
240          # we use the passed in astcache to not reparse asttrees
241          # within exception info printing
242          source = self.frame.code.fullsource
243          if source is None:
244              return None
245          key = astnode = None
246          if astcache is not None:
247              key = self.frame.code.path
248              if key is not None:
249                  astnode = astcache.get(key, None)
250          start = self.getfirstlinesource()
251          try:
252              astnode, _, end = getstatementrange_ast(
253                  self.lineno, source, astnode=astnode
254              )
255          except SyntaxError:
256              end = self.lineno + 1
257          else:
258              if key is not None:
259                  astcache[key] = astnode
260          return source[start:end]

264      def ishidden(self) -> bool:
265          """Return True if the current frame has a var __tracebackhide__ 
266          resolving to True. 
267   
268          If __tracebackhide__ is a callable, it gets called with the 
269          ExceptionInfo instance and can decide whether to hide the traceback. 
270   
271          Mostly for internal use. 
272          """
273          tbh: Union[bool, Callable[[Optional[ExceptionInfo[BaseException]]], bool]] = (
274              False
275          )
276          for maybe_ns_dct in (self.frame.f_locals, self.frame.f_globals):
277              # in normal cases, f_locals and f_globals are dictionaries
278              # however via `exec(...)` / `eval(...)` they can be other types
279              # (even incorrect types!).
280              # as such, we suppress all exceptions while accessing __tracebackhide__
281              try:
282                  tbh = maybe_ns_dct["__tracebackhide__"]
283              except Exception:
284                  pass
285              else:
286                  break
287          if tbh and callable(tbh):
288              return tbh(None if self._excinfo is None else self._excinfo())
289          return tbh

291      def __str__(self) -> str:
292          name = self.frame.code.name
293          try:
294              line = str(self.statement).lstrip()
295          except KeyboardInterrupt:
296              raise
297          except BaseException:
298              line = "???"
299          # This output does not quite match Python's repr for traceback entries,
300          # but changing it to do so would break certain plugins.  See
301          # https://github.com/pytest-dev/pytest/pull/7535/ for details.
302          return "  File %r:%d in %s\n  %s\n" % (
303              str(self.path),
304              self.lineno + 1,
305              name,
306              line,
307          )

316      """Traceback objects encapsulate and offer higher level access to Traceback entries."""

318      def __init__(
319          self,
320          tb: Union[TracebackType, Iterable[TracebackEntry]],
321          excinfo: Optional["ReferenceType[ExceptionInfo[BaseException]]"] = None,
322      ) -> None:
323          """Initialize from given python traceback object and ExceptionInfo."""
324          self._excinfo = excinfo
325          if isinstance(tb, TracebackType):
326  
327              def f(cur: TracebackType) -> Iterable[TracebackEntry]:
328                  cur_: Optional[TracebackType] = cur
329                  while cur_ is not None:
330                      yield TracebackEntry(cur_, excinfo=excinfo)
331                      cur_ = cur_.tb_next
332  
333              super().__init__(f(tb))
334          else:
335              super().__init__(tb)

337      def cut(
338          self,
339          path=None,
340          lineno: Optional[int] = None,
341          firstlineno: Optional[int] = None,
342          excludepath: Optional[py.path.local] = None,
343      ) -> "Traceback":
344          """Return a Traceback instance wrapping part of this Traceback. 
345   
346          By providing any combination of path, lineno and firstlineno, the 
347          first frame to start the to-be-returned traceback is determined. 
348   
349          This allows cutting the first part of a Traceback instance e.g. 
350          for formatting reasons (removing some uninteresting bits that deal 
351          with handling of the exception/traceback). 
352          """
353          for x in self:
354              code = x.frame.code
355              codepath = code.path
356              if (
357                  (path is None or codepath == path)
358                  and (
359                      excludepath is None
360                      or not isinstance(codepath, py.path.local)
361                      or not codepath.relto(excludepath)
362                  )
363                  and (lineno is None or x.lineno == lineno)
364                  and (firstlineno is None or x.frame.code.firstlineno == firstlineno)
365              ):
366                  return Traceback(x._rawentry, self._excinfo)
367          return self

384          self, fn: Callable[[TracebackEntry], bool] = lambda x: not x.ishidden()
385      ) -> "Traceback":
386          """Return a Traceback instance with certain items removed 
387   
388          fn is a function that gets a single argument, a TracebackEntry 
389          instance, and should return True when the item should be added 
390          to the Traceback, False when not. 
391   
392          By default this removes all the TracebackEntries which are hidden 
393          (see ishidden() above).

395          return Traceback(filter(fn, self), self._excinfo)

397      def getcrashentry(self) -> TracebackEntry:
398          """Return last non-hidden traceback entry that lead to the exception of a traceback."""
399          for i in range(-1, -len(self) - 1, -1):
400              entry = self[i]
401              if not entry.ishidden():
402                  return entry
403          return self[-1]

405      def recursionindex(self) -> Optional[int]:
406          """Return the index of the frame/TracebackEntry where recursion originates if 
407          appropriate, None if no recursion occurred."""
408          cache: Dict[Tuple[Any, int, int], List[Dict[str, Any]]] = {}
409          for i, entry in enumerate(self):
410              # id for the code.raw is needed to work around
411              # the strange metaprogramming in the decorator lib from pypi
412              # which generates code objects that have hash/value equality
413              # XXX needs a test
414              key = entry.frame.code.path, id(entry.frame.code.raw), entry.lineno
415              # print "checking for recursion at", key
416              values = cache.setdefault(key, [])
417              if values:
418                  f = entry.frame
419                  loc = f.f_locals
420                  for otherloc in values:
421                      if f.eval(
422                          co_equal,
423                          __recursioncache_locals_1=loc,
424                          __recursioncache_locals_2=otherloc,
425                      ):
426                          return i
427              values.append(entry.frame.f_locals)
428          return None

450      @classmethod
451      def from_exc_info(
452          cls,
453          exc_info: Tuple[Type[_E], _E, TracebackType],
454          exprinfo: Optional[str] = None,
455      ) -> "ExceptionInfo[_E]":
456          """Return an ExceptionInfo for an existing exc_info tuple. 
457   
458          .. warning:: 
459   
460              Experimental API 
461   
462          :param exprinfo: 
463              A text string helping to determine if we should strip 
464              ``AssertionError`` from the output. Defaults to the exception 
465              message/``__str__()``. 
466          """
467          _striptext = ""
468          if exprinfo is None and isinstance(exc_info[1], AssertionError):
469              exprinfo = getattr(exc_info[1], "msg", None)
470              if exprinfo is None:
471                  exprinfo = saferepr(exc_info[1])
472              if exprinfo and exprinfo.startswith(cls._assert_start_repr):
473                  _striptext = "AssertionError: "
474  
475          return cls(exc_info, _striptext)

477      @classmethod
478      def from_current(
479          cls, exprinfo: Optional[str] = None
480      ) -> "ExceptionInfo[BaseException]":
481          """Return an ExceptionInfo matching the current traceback. 
482   
483          .. warning:: 
484   
485              Experimental API 
486   
487          :param exprinfo: 
488              A text string helping to determine if we should strip 
489              ``AssertionError`` from the output. Defaults to the exception 
490              message/``__str__()``. 
491          """
492          tup = sys.exc_info()

494          assert tup[1] is not None, "no current exception"
495          assert tup[2] is not None, "no current exception"

650      def match(self, regexp: Union[str, Pattern[str]]) -> "Literal[True]":
651          """Check whether the regular expression `regexp` matches the string 
652          representation of the exception using :func:`python:re.search`. 
653   
654          If it matches `True` is returned, otherwise an `AssertionError` is raised. 
655          """
656          __tracebackhide__ = True
657          msg = "Regex pattern {!r} does not match {!r}."
658          if regexp == str(self.value):
659              msg += " Did you mean to `re.escape()` the regex?"
660          assert re.search(regexp, str(self.value)), msg.format(regexp, str(self.value))
661          # Return True to allow for "assert excinfo.match()".
662          return True

682      def _getindent(self, source: "Source") -> int:

684          try:

711      def get_source(
712          self,
713          source: Optional["Source"],
714          line_index: int = -1,
715          excinfo: Optional[ExceptionInfo[BaseException]] = None,
716          short: bool = False,
717      ) -> List[str]:
718          """Return formatted and marked up source lines."""
719          lines = []
720          if source is None or line_index >= len(source.lines):
721              source = Source("???")
722              line_index = 0
723          if line_index < 0:
724              line_index += len(source)
725          space_prefix = "    "
726          if short:
727              lines.append(space_prefix + source.lines[line_index].strip())
728          else:
729              for line in source.lines[:line_index]:
730                  lines.append(space_prefix + line)
731              lines.append(self.flow_marker + "   " + source.lines[line_index])
732              for line in source.lines[line_index + 1 :]:
733                  lines.append(space_prefix + line)
734          if excinfo is not None:
735              indent = 4 if short else self._getindent(source)
736              lines.extend(self.get_exconly(excinfo, indent=indent, markall=True))
737          return lines

739      def get_exconly(
740          self,
741          excinfo: ExceptionInfo[BaseException],
742          indent: int = 4,
743          markall: bool = False,
744      ) -> List[str]:
745          lines = []
746          indentstr = " " * indent
747          # Get the real exception information out.
748          exlines = excinfo.exconly(tryshort=True).split("\n")
749          failindent = self.fail_marker + indentstr[1:]

751              lines.append(failindent + line)
752              if not markall:

782      def repr_traceback_entry(
783          self,
784          entry: TracebackEntry,
785          excinfo: Optional[ExceptionInfo[BaseException]] = None,
786      ) -> "ReprEntry":
787          lines: List[str] = []
788          style = entry._repr_style if entry._repr_style is not None else self.style
789          if style in ("short", "long"):
790              source = self._getentrysource(entry)
791              if source is None:
792                  source = Source("???")
793                  line_index = 0
794              else:
795                  line_index = entry.lineno - entry.getfirstlinesource()
796              short = style == "short"
797              reprargs = self.repr_args(entry) if not short else None
798              s = self.get_source(source, line_index, excinfo, short=short)
799              lines.extend(s)
800              if short:
801                  message = "in %s" % (entry.name)
802              else:
803                  message = excinfo and excinfo.typename or ""
804              path = self._makepath(entry.path)
805              reprfileloc = ReprFileLocation(path, entry.lineno + 1, message)
806              localsrepr = self.repr_locals(entry.locals)
807              return ReprEntry(lines, reprargs, localsrepr, reprfileloc, style)
808          elif style == "value":
809              if excinfo:
810                  lines.extend(str(excinfo.value).split("\n"))
811              return ReprEntry(lines, None, None, None, style)
812          else:
813              if excinfo:

827      def repr_traceback(self, excinfo: ExceptionInfo[BaseException]) -> "ReprTraceback":
828          traceback = excinfo.traceback
829          if self.tbfilter:
830              traceback = traceback.filter()
831  
832          if isinstance(excinfo.value, RecursionError):
833              traceback, extraline = self._truncate_recursive_traceback(traceback)
834          else:
835              extraline = None
836  
837          last = traceback[-1]
838          entries = []
839          if self.style == "value":
840              reprentry = self.repr_traceback_entry(last, excinfo)
841              entries.append(reprentry)
842              return ReprTraceback(entries, None, style=self.style)
843  
844          for index, entry in enumerate(traceback):
845              einfo = (last == entry) and excinfo or None
846              reprentry = self.repr_traceback_entry(entry, einfo)
847              entries.append(reprentry)
848          return ReprTraceback(entries, extraline, style=self.style)

850      def _truncate_recursive_traceback(
851          self, traceback: Traceback
852      ) -> Tuple[Traceback, Optional[str]]:
853          """Truncate the given recursive traceback trying to find the starting 
854          point of the recursion. 
855   
856          The detection is done by going through each traceback entry and 
857          finding the point in which the locals of the frame are equal to the 
858          locals of a previous frame (see ``recursionindex()``). 
859   
860          Handle the situation where the recursion process might raise an 
861          exception (for example comparing numpy arrays using equality raises a 
862          TypeError), in which case we do our best to warn the user of the 
863          error and show a limited traceback. 
864          """
865          try:
866              recursionindex = traceback.recursionindex()
867          except Exception as e:
868              max_frames = 10
869              extraline: Optional[str] = (
870                  "!!! Recursion error detected, but an error occurred locating the origin of recursion.\n"
871                  "  The following exception happened when comparing locals in the stack frame:\n"
872                  "    {exc_type}: {exc_msg}\n"
873                  "  Displaying first and last {max_frames} stack frames out of {total}."
874              ).format(
875                  exc_type=type(e).__name__,
876                  exc_msg=str(e),
877                  max_frames=max_frames,
878                  total=len(traceback),
879              )
880              # Type ignored because adding two instaces of a List subtype
881              # currently incorrectly has type List instead of the subtype.
882              traceback = traceback[:max_frames] + traceback[-max_frames:]  # type: ignore
883          else:
884              if recursionindex is not None:
885                  extraline = "!!! Recursion detected (same locals & position)"
886                  traceback = traceback[: recursionindex + 1]
887              else:
888                  extraline = None
889  
890          return traceback, extraline

892      def repr_excinfo(
893          self, excinfo: ExceptionInfo[BaseException]
894      ) -> "ExceptionChainRepr":
895          repr_chain: List[
896              Tuple[ReprTraceback, Optional[ReprFileLocation], Optional[str]]
897          ] = []
898          e: Optional[BaseException] = excinfo.value
899          excinfo_: Optional[ExceptionInfo[BaseException]] = excinfo
900          descr = None
901          seen: Set[int] = set()
902          while e is not None and id(e) not in seen:
903              seen.add(id(e))
904              if excinfo_:
905                  reprtraceback = self.repr_traceback(excinfo_)
906                  reprcrash: Optional[ReprFileLocation] = (
907                      excinfo_._getreprcrash() if self.style != "value" else None
908                  )
909              else:
910                  # Fallback to native repr if the exception doesn't have a traceback:
911                  # ExceptionInfo objects require a full traceback to work.
912                  reprtraceback = ReprTracebackNative(
913                      traceback.format_exception(type(e), e, None)
914                  )
915                  reprcrash = None
916  
917              repr_chain += [(reprtraceback, reprcrash, descr)]
918              if e.__cause__ is not None and self.chain:
919                  e = e.__cause__
920                  excinfo_ = (
921                      ExceptionInfo((type(e), e, e.__traceback__))
922                      if e.__traceback__
923                      else None
924                  )
925                  descr = "The above exception was the direct cause of the following exception:"
926              elif (
927                  e.__context__ is not None and not e.__suppress_context__ and self.chain
928              ):
929                  e = e.__context__
930                  excinfo_ = (
931                      ExceptionInfo((type(e), e, e.__traceback__))
932                      if e.__traceback__
933                      else None
934                  )
935                  descr = "During handling of the above exception, another exception occurred:"
936              else:
937                  e = None
938          repr_chain.reverse()
939          return ExceptionChainRepr(repr_chain)

1063     def _write_entry_lines(self, tw: TerminalWriter) -> None:
1064         """Write the source code portions of a list of traceback entries with syntax highlighting. 
1065  
1066         Usually entries are lines like these: 
1067  
1068             "     x = 1" 
1069             ">    assert x == 2" 
1070             "E    assert 1 == 2" 
1071  
1072         This function takes care of rendering the "source" portions of it (the lines without 
1073         the "E" prefix) using syntax highlighting, taking care to not highlighting the ">" 
1074         character, as doing so might break line continuations. 
1075         """
1076 
1077         if not self.lines:
1078             return
1079 
1080         # separate indents and source lines that are not failures: we want to
1081         # highlight the code but not the indentation, which may contain markers
1082         # such as ">   assert 0"
1083         fail_marker = f"{FormattedExcinfo.fail_marker}   "
1084         indent_size = len(fail_marker)
1085         indents: List[str] = []
1086         source_lines: List[str] = []
1087         failure_lines: List[str] = []
1088         for index, line in enumerate(self.lines):
1089             is_failure_line = line.startswith(fail_marker)
1090             if is_failure_line:
1091                 # from this point on all lines are considered part of the failure
1092                 failure_lines.extend(self.lines[index:])
1093                 break
1094             else:
1095                 if self.style == "value":
1096                     source_lines.append(line)
1097                 else:
1098                     indents.append(line[:indent_size])
1099                     source_lines.append(line[indent_size:])
1100 
1101         tw._write_source(source_lines, indents)
1102 
1103         # failure lines are always completely red and bold
1104         for line in failure_lines:
1105             tw.line(line, bold=True, red=True)

1231 def filter_traceback(entry: TracebackEntry) -> bool:
1232     """Return True if a TracebackEntry instance should be included in tracebacks. 
1233  
1234     We hide traceback entries of: 
1235  
1236     * dynamically generated code (no code to show up for it); 
1237     * internal traceback from pytest or its internal libraries, py and pluggy. 
1238     """
1239     # entry.path might sometimes return a str object when the entry
1240     # points to dynamically generated code.
1241     # See https://bitbucket.org/pytest-dev/py/issues/71.
1242     raw_filename = entry.frame.code.raw.co_filename
1243     is_generated = "<" in raw_filename and ">" in raw_filename
1244     if is_generated:
1245         return False
1246 
1247     # entry.path might point to a non-existing file, in which case it will
1248     # also return a str object. See #1133.
1249     p = Path(entry.path)
1250 
1251     parents = p.parents
1252     if _PLUGGY_DIR in parents:
1253         return False
1254     if _PYTEST_DIR in parents:
1255         return False
1256     if _PY_DIR in parents:
1257         return False
1258 
1259     return True
