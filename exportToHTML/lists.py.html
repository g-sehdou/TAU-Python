1    """

4     
5    Lists are mutable, which means that the item inside of a list can be retrieved, changed, rearranged, and acted upon in multiple different ways.

7    Python lists maintain their order, which means that you can access something by its position or index using some of the list methods available in Python.

33   # extend method is allowed us to extend a list with another list
34   
35   euro_countries = ["Germany", "Italy", "Spain"]
36   countries = ["The Netherlands", "Belgium", "Greece"]
37   countries.extend(euro_countries)
38   print(countries)

40   # remove method allows us to move an item from a list
41   
42   mobile = ["apple", "samsung", "nokia", "ericsson", "Blackberry"]
43   mobile.remove("nokia")
44   print(mobile)
45   
46   # pop method allows us to move an item from a list by index
47   
48   soft_drinks = ["orange", "apple", "cola", "redbull"]
49   print(soft_drinks)
50   # using 0 will remove the first item from the list
51   soft_drinks.pop(0)
52   print(soft_drinks)
53   # using negative number e.g. -1  will remove the last item from the list
54   soft_drinks.pop(-1)
55   print(soft_drinks)
56   
57   # sort method will allows us to sort items (not string items)
58   
59   salaris = [200, 494, 34.39, 3948937, 3.0003]
60   print(salaris)
61   salaris.sort()
62   print(salaris)
63   
64   # Python has a function called "in" which allows us to check membership in a list. it returns False Or True
65   
66   workers = ["George", "Hans", "Jan", "Greg", "Mark"]
67   # it will return False because "jan" is not a member of the list workers
68   print("jan" in workers)
69   # it will return True because "George" is a member of the list workers
70   print("George" in workers)
71   
72   # Python has function called count which allows us to count the same items in a list
73   
74   tickets = ["fly", "fly", "car", "plan", "film", "car", "film", "fly"]
75   print(tickets.count("film"))
76   # this will show you in which position appears the item in the list
77   print(tickets.index("car"))
