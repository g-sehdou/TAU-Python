48   def _match_vcs_scheme(url: str) -> Optional[str]:
49       """Look for VCS schemes in the URL. 
50    
51       Returns the matched VCS scheme, or None if there's no match. 
52       """
53       for scheme in vcs.schemes:
54           if url.lower().startswith(scheme) and url[len(scheme)] in "+:":
55               return scheme
56       return None

59   class _NotHTML(Exception):
60       def __init__(self, content_type: str, request_desc: str) -> None:
61           super().__init__(content_type, request_desc)
62           self.content_type = content_type
63           self.request_desc = request_desc

66   def _ensure_html_header(response: Response) -> None:
67       """Check the Content-Type header to ensure the response contains HTML. 
68    
69       Raises `_NotHTML` if the content type is not text/html. 
70       """
71       content_type = response.headers.get("Content-Type", "")
72       if not content_type.lower().startswith("text/html"):
73           raise _NotHTML(content_type, response.request.method)

80   def _ensure_html_response(url: str, session: PipSession) -> None:
81       """Send a HEAD request to the URL, and ensure the response contains HTML. 
82    
83       Raises `_NotHTTP` if the URL is not available for a HEAD request, or 
84       `_NotHTML` if the content type is not text/html. 
85       """
86       scheme, netloc, path, query, fragment = urllib.parse.urlsplit(url)
87       if scheme not in {"http", "https"}:
88           raise _NotHTTP()
89   
90       resp = session.head(url, allow_redirects=True)
91       raise_for_status(resp)
92   
93       _ensure_html_header(resp)

96   def _get_html_response(url: str, session: PipSession) -> Response:
97       """Access an HTML page with GET, and return the response. 
98    
99       This consists of three parts: 
100   
101      1. If the URL looks suspiciously like an archive, send a HEAD first to 
102         check the Content-Type is HTML, to avoid downloading a large file. 
103         Raise `_NotHTTP` if the content type cannot be determined, or 
104         `_NotHTML` if it is not HTML. 
105      2. Actually perform the request. Raise HTTP exceptions on network failures. 
106      3. Check the Content-Type header to make sure we got HTML, and raise 
107         `_NotHTML` otherwise. 
108      """
109      if is_archive_file(Link(url).filename):
110          _ensure_html_response(url, session=session)
111  
112      logger.debug("Getting page %s", redact_auth_from_url(url))
113  
114      resp = session.get(
115          url,
116          headers={
117              "Accept": "text/html",
118              # We don't want to blindly returned cached data for
119              # /simple/, because authors generally expecting that
120              # twine upload && pip install will function, but if
121              # they've done a pip install in the last ~10 minutes
122              # it won't. Thus by setting this to zero we will not
123              # blindly use any cached data, however the benefit of
124              # using max-age=0 instead of no-cache, is that we will
125              # still support conditional requests, so we will still
126              # minimize traffic sent in cases where the page hasn't
127              # changed at all, we will just always incur the round
128              # trip for the conditional GET now instead of only
129              # once per 10 minutes.
130              # For more information, please see pypa/pip#5670.
131              "Cache-Control": "max-age=0",
132          },
133      )
134      raise_for_status(resp)
135  
136      # The check for archives above only works if the url ends with
137      # something that looks like an archive. However that is not a
138      # requirement of an url. Unless we issue a HEAD request on every
139      # url we cannot know ahead of time for sure if something is HTML
140      # or not. However we can check after we've downloaded it.
141      _ensure_html_header(resp)
142  
143      return resp

146  def _get_encoding_from_headers(headers: ResponseHeaders) -> Optional[str]:
147      """Determine if we have any encoding information in our headers."""
148      if headers and "Content-Type" in headers:
149          content_type, params = cgi.parse_header(headers["Content-Type"])
150          if "charset" in params:
151              return params["charset"]
152      return None

155  def _determine_base_url(document: HTMLElement, page_url: str) -> str:
156      """Determine the HTML document's base URL. 
157   
158      This looks for a ``<base>`` tag in the HTML document. If present, its href 
159      attribute denotes the base URL of anchor tags in the document. If there is 
160      no such tag (or if it does not have a valid href attribute), the HTML 
161      file's URL is used as the base URL. 
162   
163      :param document: An HTML document representation. The current 
164          implementation expects the result of ``html5lib.parse()``. 
165      :param page_url: The URL of the HTML document. 
166      """
167      for base in document.findall(".//base"):
168          href = base.get("href")
169          if href is not None:
170              return href
171      return page_url

174  def _clean_url_path_part(part: str) -> str:
175      """ 
176      Clean a "part" of a URL path (i.e. after splitting on "@" characters). 
177      """
178      # We unquote prior to quoting to make sure nothing is double quoted.
179      return urllib.parse.quote(urllib.parse.unquote(part))
180  
181  
182  def _clean_file_url_path(part: str) -> str:
183      """ 
184      Clean the first part of a URL path that corresponds to a local 
185      filesystem path (i.e. the first part after splitting on "@" characters). 
186      """
187      # We unquote prior to quoting to make sure nothing is double quoted.
188      # Also, on Windows the path part might contain a drive letter which
189      # should not be quoted. On Linux where drive letters do not
190      # exist, the colon should be quoted. We rely on urllib.request
191      # to do the right thing here.
192      return urllib.request.pathname2url(urllib.request.url2pathname(part))

195  # percent-encoded:                   /
196  _reserved_chars_re = re.compile("(@|%2F)", re.IGNORECASE)
197  
198  
199  def _clean_url_path(path: str, is_local_path: bool) -> str:
200      """ 
201      Clean the path portion of a URL. 
202      """
203      if is_local_path:
204          clean_func = _clean_file_url_path
205      else:
206          clean_func = _clean_url_path_part
207  
208      # Split on the reserved characters prior to cleaning so that
209      # revision strings in VCS URLs are properly preserved.
210      parts = _reserved_chars_re.split(path)
211  
212      cleaned_parts = []
213      for to_clean, reserved in pairwise(itertools.chain(parts, [""])):
214          cleaned_parts.append(clean_func(to_clean))
215          # Normalize %xx escapes (e.g. %2f -> %2F)
216          cleaned_parts.append(reserved.upper())
217  
218      return "".join(cleaned_parts)

221  def _clean_link(url: str) -> str:
222      """ 
223      Make sure a link is fully quoted. 
224      For example, if ' ' occurs in the URL, it will be replaced with "%20", 
225      and without double-quoting other characters. 
226      """
227      # Split the URL into parts according to the general structure
228      # `scheme://netloc/path;parameters?query#fragment`.
229      result = urllib.parse.urlparse(url)
230      # If the netloc is empty, then the URL refers to a local filesystem path.
231      is_local_path = not result.netloc
232      path = _clean_url_path(result.path, is_local_path=is_local_path)
233      return urllib.parse.urlunparse(result._replace(path=path))

236  def _create_link_from_element(
237      anchor: HTMLElement,
238      page_url: str,
239      base_url: str,
240  ) -> Optional[Link]:
241      """ 
242      Convert an anchor element in a simple repository page to a Link. 
243      """
244      href = anchor.get("href")
245      if not href:
246          return None
247  
248      url = _clean_link(urllib.parse.urljoin(base_url, href))
249      pyrequire = anchor.get("data-requires-python")
250      yanked_reason = anchor.get("data-yanked")
251  
252      link = Link(
253          url,
254          comes_from=page_url,
255          requires_python=pyrequire,
256          yanked_reason=yanked_reason,
257      )
258  
259      return link

262  class CacheablePageContent:
263      def __init__(self, page: "HTMLPage") -> None:
264          assert page.cache_link_parsing
265          self.page = page
266  
267      def __eq__(self, other: object) -> bool:
268          return isinstance(other, type(self)) and self.page.url == other.page.url
269  
270      def __hash__(self) -> int:
271          return hash(self.page.url)

274  def with_cached_html_pages(
275      fn: Callable[["HTMLPage"], Iterable[Link]],
276  ) -> Callable[["HTMLPage"], List[Link]]:
277      """ 
278      Given a function that parses an Iterable[Link] from an HTMLPage, cache the 
279      function's result (keyed by CacheablePageContent), unless the HTMLPage 
280      `page` has `page.cache_link_parsing == False`. 
281      """
282  
283      @functools.lru_cache(maxsize=None)
284      def wrapper(cacheable_page: CacheablePageContent) -> List[Link]:
285          return list(fn(cacheable_page.page))
286  
287      @functools.wraps(fn)
288      def wrapper_wrapper(page: "HTMLPage") -> List[Link]:
289          if page.cache_link_parsing:
290              return wrapper(CacheablePageContent(page))
291          return list(fn(page))
292  
293      return wrapper_wrapper

296  @with_cached_html_pages
297  def parse_links(page: "HTMLPage") -> Iterable[Link]:
298      """ 
299      Parse an HTML document, and yield its anchor elements as Link objects. 
300      """
301      document = html5lib.parse(
302          page.content,
303          transport_encoding=page.encoding,
304          namespaceHTMLElements=False,
305      )
306  
307      url = page.url
308      base_url = _determine_base_url(document, url)
309      for anchor in document.findall(".//a"):
310          link = _create_link_from_element(
311              anchor,
312              page_url=url,
313              base_url=base_url,
314          )
315          if link is None:
316              continue
317          yield link

356  def _make_html_page(response: Response, cache_link_parsing: bool = True) -> HTMLPage:
357      encoding = _get_encoding_from_headers(response.headers)
358      return HTMLPage(
359          response.content,
360          encoding=encoding,
361          url=response.url,
362          cache_link_parsing=cache_link_parsing,

366  def _get_html_page(
367      link: Link, session: Optional[PipSession] = None
368  ) -> Optional["HTMLPage"]:
369      if session is None:
370          raise TypeError(
371              "_get_html_page() missing 1 required keyword argument: 'session'"
372          )
373  
374      url = link.url.split("#", 1)[0]
375  
376      # Check for VCS schemes that do not support lookup as web pages.
377      vcs_scheme = _match_vcs_scheme(url)
378      if vcs_scheme:
379          logger.warning(
380              "Cannot look at %s URL %s because it does not support lookup as web pages.",
381              vcs_scheme,
382              link,
383          )
384          return None
385  
386      # Tack index.html onto file:// URLs that point to directories
387      scheme, _, path, _, _, _ = urllib.parse.urlparse(url)
388      if scheme == "file" and os.path.isdir(urllib.request.url2pathname(path)):
389          # add trailing slash if not present so urljoin doesn't trim
390          # final segment
391          if not url.endswith("/"):
392              url += "/"
393          url = urllib.parse.urljoin(url, "index.html")
394          logger.debug(" file: URL is directory, getting %s", url)
395  
396      try:
397          resp = _get_html_response(url, session=session)
398      except _NotHTTP:
399          logger.warning(
400              "Skipping page %s because it looks like an archive, and cannot "
401              "be checked by a HTTP HEAD request.",
402              link,
403          )
404      except _NotHTML as exc:
405          logger.warning(
406              "Skipping page %s because the %s request got Content-Type: %s."
407              "The only supported Content-Type is text/html",
408              link,
409              exc.request_desc,
410              exc.content_type,
411          )
412      except NetworkConnectionError as exc:
413          _handle_get_page_fail(link, exc)
414      except RetryError as exc:
415          _handle_get_page_fail(link, exc)
416      except SSLError as exc:
417          reason = "There was a problem confirming the ssl certificate: "
418          reason += str(exc)
419          _handle_get_page_fail(link, reason, meth=logger.info)
420      except requests.ConnectionError as exc:
421          _handle_get_page_fail(link, f"connection error: {exc}")
422      except requests.Timeout:
423          _handle_get_page_fail(link, "timed out")
424      else:
425          return _make_html_page(resp, cache_link_parsing=link.cache_link_parsing)
426      return None

434  class LinkCollector:
435  
436      """ 
437      Responsible for collecting Link objects from all configured locations, 
438      making network requests as needed. 
439   
440      The class's main method is its collect_sources() method. 
441      """
442  
443      def __init__(
444          self,
445          session: PipSession,
446          search_scope: SearchScope,
447      ) -> None:
448          self.search_scope = search_scope
449          self.session = session
450  
451      @classmethod
452      def create(
453          cls,
454          session: PipSession,
455          options: Values,
456          suppress_no_index: bool = False,
457      ) -> "LinkCollector":
458          """ 
459          :param session: The Session to use to make requests. 
460          :param suppress_no_index: Whether to ignore the --no-index option 
461              when constructing the SearchScope object. 
462          """
463          index_urls = [options.index_url] + options.extra_index_urls
464          if options.no_index and not suppress_no_index:
465              logger.debug(
466                  "Ignoring indexes: %s",
467                  ",".join(redact_auth_from_url(url) for url in index_urls),
468              )
469              index_urls = []
470  
471          # Make sure find_links is a list before passing to create().
472          find_links = options.find_links or []
473  
474          search_scope = SearchScope.create(
475              find_links=find_links,
476              index_urls=index_urls,
477          )
478          link_collector = LinkCollector(
479              session=session,
480              search_scope=search_scope,
481          )
482          return link_collector
483  
484      @property
485      def find_links(self) -> List[str]:
486          return self.search_scope.find_links
487  
488      def fetch_page(self, location: Link) -> Optional[HTMLPage]:
489          """ 
490          Fetch an HTML page containing package links. 
491          """
492          return _get_html_page(location, session=self.session)
493  
494      def collect_sources(
495          self,
496          project_name: str,
497          candidates_from_page: CandidatesFromPage,
498      ) -> CollectedSources:
499          # The OrderedDict calls deduplicate sources by URL.
500          index_url_sources = collections.OrderedDict(
501              build_source(
502                  loc,
503                  candidates_from_page=candidates_from_page,
504                  page_validator=self.session.is_secure_origin,
505                  expand_dir=False,
506                  cache_link_parsing=False,
507              )
508              for loc in self.search_scope.get_index_urls_locations(project_name)
509          ).values()
510          find_links_sources = collections.OrderedDict(
511              build_source(
512                  loc,
513                  candidates_from_page=candidates_from_page,
514                  page_validator=self.session.is_secure_origin,
515                  expand_dir=True,
516                  cache_link_parsing=True,
517              )
518              for loc in self.find_links
519          ).values()
520  
521          if logger.isEnabledFor(logging.DEBUG):
522              lines = [
523                  f"* {s.link}"
524                  for s in itertools.chain(find_links_sources, index_url_sources)
525                  if s is not None and s.link is not None
526              ]
527              lines = [
528                  f"{len(lines)} location(s) to search "
529                  f"for versions of {project_name}:"
530              ] + lines
531              logger.debug("\n".join(lines))
532  
533          return CollectedSources(
534              find_links=list(find_links_sources),
535              index_urls=list(index_url_sources),
536          )
