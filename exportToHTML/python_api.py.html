44   class ApproxBase:
45       """Provide shared utilities for making approximate comparisons between 
46       numbers or sequences of numbers."""
47   
48       # Tell numpy to use our `__eq__` operator instead of its.
49       __array_ufunc__ = None
50       __array_priority__ = 100
51   
52       def __init__(self, expected, rel=None, abs=None, nan_ok: bool = False) -> None:
53           __tracebackhide__ = True
54           self.expected = expected
55           self.abs = abs
56           self.rel = rel
57           self.nan_ok = nan_ok
58           self._check_type()
59   
60       def __repr__(self) -> str:
61           raise NotImplementedError
62   
63       def __eq__(self, actual) -> bool:
64           return all(
65               a == self._approx_scalar(x) for a, x in self._yield_comparisons(actual)
66           )
67   
68       # Ignore type because of https://github.com/python/mypy/issues/4266.
69       __hash__ = None  # type: ignore
70   
71       def __ne__(self, actual) -> bool:
72           return not (actual == self)
73   
74       def _approx_scalar(self, x) -> "ApproxScalar":
75           return ApproxScalar(x, rel=self.rel, abs=self.abs, nan_ok=self.nan_ok)
76   
77       def _yield_comparisons(self, actual):
78           """Yield all the pairs of numbers to be compared. 
79    
80           This is used to implement the `__eq__` method. 
81           """
82           raise NotImplementedError
83   
84       def _check_type(self) -> None:
85           """Raise a TypeError if the expected value is not a valid type."""
86           # This is only a concern if the expected value is a sequence.  In every
87           # other case, the approx() function ensures that the expected value has
88           # a numeric type.  For this reason, the default is to do nothing.  The
89           # classes that deal with sequences should reimplement this method to
90           # raise if there are any non-numeric elements in the sequence.
91           pass

95       if isinstance(x, list):
96           return list(_recursive_list_map(f, xi) for xi in x)
97       else:
98           return f(x)

101  class ApproxNumpy(ApproxBase):
102      """Perform approximate comparisons where the expected value is numpy array."""
103  
104      def __repr__(self) -> str:
105          list_scalars = _recursive_list_map(self._approx_scalar, self.expected.tolist())
106          return f"approx({list_scalars!r})"
107  
108      def __eq__(self, actual) -> bool:
109          import numpy as np
110  
111          # self.expected is supposed to always be an array here.
112  
113          if not np.isscalar(actual):
114              try:
115                  actual = np.asarray(actual)
116              except Exception as e:
117                  raise TypeError(f"cannot compare '{actual}' to numpy.ndarray") from e
118  
119          if not np.isscalar(actual) and actual.shape != self.expected.shape:
120              return False
121  
122          return ApproxBase.__eq__(self, actual)
123  
124      def _yield_comparisons(self, actual):
125          import numpy as np
126  
127          # `actual` can either be a numpy array or a scalar, it is treated in
128          # `__eq__` before being passed to `ApproxBase.__eq__`, which is the
129          # only method that calls this one.
130  
131          if np.isscalar(actual):
132              for i in np.ndindex(self.expected.shape):
133                  yield actual, self.expected[i].item()
134          else:
135              for i in np.ndindex(self.expected.shape):
136                  yield actual[i].item(), self.expected[i].item()

139  class ApproxMapping(ApproxBase):
140      """Perform approximate comparisons where the expected value is a mapping 
141      with numeric values (the keys can be anything)."""
142  
143      def __repr__(self) -> str:
144          return "approx({!r})".format(
145              {k: self._approx_scalar(v) for k, v in self.expected.items()}
146          )
147  
148      def __eq__(self, actual) -> bool:
149          try:
150              if set(actual.keys()) != set(self.expected.keys()):
151                  return False
152          except AttributeError:
153              return False
154  
155          return ApproxBase.__eq__(self, actual)
156  
157      def _yield_comparisons(self, actual):
158          for k in self.expected.keys():
159              yield actual[k], self.expected[k]
160  
161      def _check_type(self) -> None:
162          __tracebackhide__ = True
163          for key, value in self.expected.items():
164              if isinstance(value, type(self.expected)):
165                  msg = "pytest.approx() does not support nested dictionaries: key={!r} value={!r}\n  full mapping={}"
166                  raise TypeError(msg.format(key, value, pprint.pformat(self.expected)))

169  class ApproxSequencelike(ApproxBase):
170      """Perform approximate comparisons where the expected value is a sequence of numbers."""
171  
172      def __repr__(self) -> str:
173          seq_type = type(self.expected)
174          if seq_type not in (tuple, list, set):
175              seq_type = list
176          return "approx({!r})".format(
177              seq_type(self._approx_scalar(x) for x in self.expected)
178          )
179  
180      def __eq__(self, actual) -> bool:
181          try:
182              if len(actual) != len(self.expected):
183                  return False
184          except TypeError:
185              return False
186          return ApproxBase.__eq__(self, actual)
187  
188      def _yield_comparisons(self, actual):
189          return zip(actual, self.expected)
190  
191      def _check_type(self) -> None:
192          __tracebackhide__ = True
193          for index, x in enumerate(self.expected):
194              if isinstance(x, type(self.expected)):
195                  msg = "pytest.approx() does not support nested data structures: {!r} at index {}\n  full sequence: {}"
196                  raise TypeError(msg.format(x, index, pprint.pformat(self.expected)))

199  class ApproxScalar(ApproxBase):
200      """Perform approximate comparisons where the expected value is a single number."""
201  
202      # Using Real should be better than this Union, but not possible yet:
203      # https://github.com/python/typeshed/pull/3108
204      DEFAULT_ABSOLUTE_TOLERANCE: Union[float, Decimal] = 1e-12
205      DEFAULT_RELATIVE_TOLERANCE: Union[float, Decimal] = 1e-6
206  
207      def __repr__(self) -> str:
208          """Return a string communicating both the expected value and the 
209          tolerance for the comparison being made. 
210   
211          For example, ``1.0 ± 1e-6``, ``(3+4j) ± 5e-6 ∠ ±180°``. 
212          """
213  
214          # Don't show a tolerance for values that aren't compared using
215          # tolerances, i.e. non-numerics and infinities. Need to call abs to
216          # handle complex numbers, e.g. (inf + 1j).
217          if (not isinstance(self.expected, (Complex, Decimal))) or math.isinf(
218              abs(self.expected)  # type: ignore[arg-type]
219          ):
220              return str(self.expected)
221  
222          # If a sensible tolerance can't be calculated, self.tolerance will
223          # raise a ValueError.  In this case, display '???'.
224          try:
225              vetted_tolerance = f"{self.tolerance:.1e}"
226              if (
227                  isinstance(self.expected, Complex)
228                  and self.expected.imag
229                  and not math.isinf(self.tolerance)
230              ):
231                  vetted_tolerance += " ∠ ±180°"
232          except ValueError:
233              vetted_tolerance = "???"
234  
235          return f"{self.expected} ± {vetted_tolerance}"
236  
237      def __eq__(self, actual) -> bool:
238          """Return whether the given value is equal to the expected value 
239          within the pre-specified tolerance."""
240          asarray = _as_numpy_array(actual)
241          if asarray is not None:
242              # Call ``__eq__()`` manually to prevent infinite-recursion with
243              # numpy<1.13.  See #3748.
244              return all(self.__eq__(a) for a in asarray.flat)
245  
246          # Short-circuit exact equality.
247          if actual == self.expected:
248              return True
249  
250          # If either type is non-numeric, fall back to strict equality.
251          # NB: we need Complex, rather than just Number, to ensure that __abs__,
252          # __sub__, and __float__ are defined.
253          if not (
254              isinstance(self.expected, (Complex, Decimal))
255              and isinstance(actual, (Complex, Decimal))
256          ):
257              return False
258  
259          # Allow the user to control whether NaNs are considered equal to each
260          # other or not.  The abs() calls are for compatibility with complex
261          # numbers.
262          if math.isnan(abs(self.expected)):  # type: ignore[arg-type]
263              return self.nan_ok and math.isnan(abs(actual))  # type: ignore[arg-type]
264  
265          # Infinity shouldn't be approximately equal to anything but itself, but
266          # if there's a relative tolerance, it will be infinite and infinity
267          # will seem approximately equal to everything.  The equal-to-itself
268          # case would have been short circuited above, so here we can just
269          # return false if the expected value is infinite.  The abs() call is
270          # for compatibility with complex numbers.
271          if math.isinf(abs(self.expected)):  # type: ignore[arg-type]
272              return False
273  
274          # Return true if the two numbers are within the tolerance.
275          result: bool = abs(self.expected - actual) <= self.tolerance
276          return result
277  
278      # Ignore type because of https://github.com/python/mypy/issues/4266.
279      __hash__ = None  # type: ignore
280  
281      @property
282      def tolerance(self):
283          """Return the tolerance for the comparison. 
284   
285          This could be either an absolute tolerance or a relative tolerance, 
286          depending on what the user specified or which would be larger. 
287          """
288  
289          def set_default(x, default):
290              return x if x is not None else default
291  
292          # Figure out what the absolute tolerance should be.  ``self.abs`` is
293          # either None or a value specified by the user.
294          absolute_tolerance = set_default(self.abs, self.DEFAULT_ABSOLUTE_TOLERANCE)
295  
296          if absolute_tolerance < 0:
297              raise ValueError(
298                  f"absolute tolerance can't be negative: {absolute_tolerance}"
299              )
300          if math.isnan(absolute_tolerance):
301              raise ValueError("absolute tolerance can't be NaN.")
302  
303          # If the user specified an absolute tolerance but not a relative one,
304          # just return the absolute tolerance.
305          if self.rel is None:
306              if self.abs is not None:
307                  return absolute_tolerance
308  
309          # Figure out what the relative tolerance should be.  ``self.rel`` is
310          # either None or a value specified by the user.  This is done after
311          # we've made sure the user didn't ask for an absolute tolerance only,
312          # because we don't want to raise errors about the relative tolerance if
313          # we aren't even going to use it.
314          relative_tolerance = set_default(
315              self.rel, self.DEFAULT_RELATIVE_TOLERANCE
316          ) * abs(self.expected)
317  
318          if relative_tolerance < 0:
319              raise ValueError(
320                  f"relative tolerance can't be negative: {absolute_tolerance}"
321              )
322          if math.isnan(relative_tolerance):
323              raise ValueError("relative tolerance can't be NaN.")
324  
325          # Return the larger of the relative and absolute tolerances.
326          return max(relative_tolerance, absolute_tolerance)

336  def approx(expected, rel=None, abs=None, nan_ok: bool = False) -> ApproxBase:
337      """Assert that two numbers (or two sets of numbers) are equal to each other 
338      within some tolerance. 
339   
340      Due to the `intricacies of floating-point arithmetic`__, numbers that we 
341      would intuitively expect to be equal are not always so:: 
342   
343          >>> 0.1 + 0.2 == 0.3 
344          False 
345   
346      __ https://docs.python.org/3/tutorial/floatingpoint.html 
347   
348      This problem is commonly encountered when writing tests, e.g. when making 
349      sure that floating-point values are what you expect them to be.  One way to 
350      deal with this problem is to assert that two floating-point numbers are 
351      equal to within some appropriate tolerance:: 
352   
353          >>> abs((0.1 + 0.2) - 0.3) < 1e-6 
354          True 
355   
356      However, comparisons like this are tedious to write and difficult to 
357      understand.  Furthermore, absolute comparisons like the one above are 
358      usually discouraged because there's no tolerance that works well for all 
359      situations.  ``1e-6`` is good for numbers around ``1``, but too small for 
360      very big numbers and too big for very small ones.  It's better to express 
361      the tolerance as a fraction of the expected value, but relative comparisons 
362      like that are even more difficult to write correctly and concisely. 
363   
364      The ``approx`` class performs floating-point comparisons using a syntax 
365      that's as intuitive as possible:: 
366   
367          >>> from pytest import approx 
368          >>> 0.1 + 0.2 == approx(0.3) 
369          True 
370   
371      The same syntax also works for sequences of numbers:: 
372   
373          >>> (0.1 + 0.2, 0.2 + 0.4) == approx((0.3, 0.6)) 
374          True 
375   
376      Dictionary *values*:: 
377   
378          >>> {'a': 0.1 + 0.2, 'b': 0.2 + 0.4} == approx({'a': 0.3, 'b': 0.6}) 
379          True 
380   
381      ``numpy`` arrays:: 
382   
383          >>> import numpy as np                                                          # doctest: +SKIP 
384          >>> np.array([0.1, 0.2]) + np.array([0.2, 0.4]) == approx(np.array([0.3, 0.6])) # doctest: +SKIP 
385          True 
386   
387      And for a ``numpy`` array against a scalar:: 
388   
389          >>> import numpy as np                                         # doctest: +SKIP 
390          >>> np.array([0.1, 0.2]) + np.array([0.2, 0.1]) == approx(0.3) # doctest: +SKIP 
391          True 
392   
393      By default, ``approx`` considers numbers within a relative tolerance of 
394      ``1e-6`` (i.e. one part in a million) of its expected value to be equal. 
395      This treatment would lead to surprising results if the expected value was 
396      ``0.0``, because nothing but ``0.0`` itself is relatively close to ``0.0``. 
397      To handle this case less surprisingly, ``approx`` also considers numbers 
398      within an absolute tolerance of ``1e-12`` of its expected value to be 
399      equal.  Infinity and NaN are special cases.  Infinity is only considered 
400      equal to itself, regardless of the relative tolerance.  NaN is not 
401      considered equal to anything by default, but you can make it be equal to 
402      itself by setting the ``nan_ok`` argument to True.  (This is meant to 
403      facilitate comparing arrays that use NaN to mean "no data".) 
404   
405      Both the relative and absolute tolerances can be changed by passing 
406      arguments to the ``approx`` constructor:: 
407   
408          >>> 1.0001 == approx(1) 
409          False 
410          >>> 1.0001 == approx(1, rel=1e-3) 
411          True 
412          >>> 1.0001 == approx(1, abs=1e-3) 
413          True 
414   
415      If you specify ``abs`` but not ``rel``, the comparison will not consider 
416      the relative tolerance at all.  In other words, two numbers that are within 
417      the default relative tolerance of ``1e-6`` will still be considered unequal 
418      if they exceed the specified absolute tolerance.  If you specify both 
419      ``abs`` and ``rel``, the numbers will be considered equal if either 
420      tolerance is met:: 
421   
422          >>> 1 + 1e-8 == approx(1) 
423          True 
424          >>> 1 + 1e-8 == approx(1, abs=1e-12) 
425          False 
426          >>> 1 + 1e-8 == approx(1, rel=1e-6, abs=1e-12) 
427          True 
428   
429      You can also use ``approx`` to compare nonnumeric types, or dicts and 
430      sequences containing nonnumeric types, in which case it falls back to 
431      strict equality. This can be useful for comparing dicts and sequences that 
432      can contain optional values:: 
433   
434          >>> {"required": 1.0000005, "optional": None} == approx({"required": 1, "optional": None}) 
435          True 
436          >>> [None, 1.0000005] == approx([None,1]) 
437          True 
438          >>> ["foo", 1.0000005] == approx([None,1]) 
439          False 
440   
441      If you're thinking about using ``approx``, then you might want to know how 
442      it compares to other good ways of comparing floating-point numbers.  All of 
443      these algorithms are based on relative and absolute tolerances and should 
444      agree for the most part, but they do have meaningful differences: 
445   
446      - ``math.isclose(a, b, rel_tol=1e-9, abs_tol=0.0)``:  True if the relative 
447        tolerance is met w.r.t. either ``a`` or ``b`` or if the absolute 
448        tolerance is met.  Because the relative tolerance is calculated w.r.t. 
449        both ``a`` and ``b``, this test is symmetric (i.e.  neither ``a`` nor 
450        ``b`` is a "reference value").  You have to specify an absolute tolerance 
451        if you want to compare to ``0.0`` because there is no tolerance by 
452        default.  `More information...`__ 
453   
454        __ https://docs.python.org/3/library/math.html#math.isclose 
455   
456      - ``numpy.isclose(a, b, rtol=1e-5, atol=1e-8)``: True if the difference 
457        between ``a`` and ``b`` is less that the sum of the relative tolerance 
458        w.r.t. ``b`` and the absolute tolerance.  Because the relative tolerance 
459        is only calculated w.r.t. ``b``, this test is asymmetric and you can 
460        think of ``b`` as the reference value.  Support for comparing sequences 
461        is provided by ``numpy.allclose``.  `More information...`__ 
462   
463        __ https://numpy.org/doc/stable/reference/generated/numpy.isclose.html 
464   
465      - ``unittest.TestCase.assertAlmostEqual(a, b)``: True if ``a`` and ``b`` 
466        are within an absolute tolerance of ``1e-7``.  No relative tolerance is 
467        considered and the absolute tolerance cannot be changed, so this function 
468        is not appropriate for very large or very small numbers.  Also, it's only 
469        available in subclasses of ``unittest.TestCase`` and it's ugly because it 
470        doesn't follow PEP8.  `More information...`__ 
471   
472        __ https://docs.python.org/3/library/unittest.html#unittest.TestCase.assertAlmostEqual 
473   
474      - ``a == pytest.approx(b, rel=1e-6, abs=1e-12)``: True if the relative 
475        tolerance is met w.r.t. ``b`` or if the absolute tolerance is met. 
476        Because the relative tolerance is only calculated w.r.t. ``b``, this test 
477        is asymmetric and you can think of ``b`` as the reference value.  In the 
478        special case that you explicitly specify an absolute tolerance but not a 
479        relative tolerance, only the absolute tolerance is considered. 
480   
481      .. warning:: 
482   
483         .. versionchanged:: 3.2 
484   
485         In order to avoid inconsistent behavior, ``TypeError`` is 
486         raised for ``>``, ``>=``, ``<`` and ``<=`` comparisons. 
487         The example below illustrates the problem:: 
488   
489             assert approx(0.1) > 0.1 + 1e-10  # calls approx(0.1).__gt__(0.1 + 1e-10) 
490             assert 0.1 + 1e-10 > approx(0.1)  # calls approx(0.1).__lt__(0.1 + 1e-10) 
491   
492         In the second example one expects ``approx(0.1).__le__(0.1 + 1e-10)`` 
493         to be called. But instead, ``approx(0.1).__lt__(0.1 + 1e-10)`` is used to 
494         comparison. This is because the call hierarchy of rich comparisons 
495         follows a fixed behavior. `More information...`__ 
496   
497         __ https://docs.python.org/3/reference/datamodel.html#object.__ge__ 
498   
499      .. versionchanged:: 3.7.1 
500         ``approx`` raises ``TypeError`` when it encounters a dict value or 
501         sequence element of nonnumeric type. 
502   
503      .. versionchanged:: 6.1.0 
504         ``approx`` falls back to strict equality for nonnumeric types instead 
505         of raising ``TypeError``. 
506      """
507  
508      # Delegate the comparison to a class that knows how to deal with the type
509      # of the expected value (e.g. int, float, list, dict, numpy.array, etc).
510      #
511      # The primary responsibility of these classes is to implement ``__eq__()``
512      # and ``__repr__()``.  The former is used to actually check if some
513      # "actual" value is equivalent to the given expected value within the
514      # allowed tolerance.  The latter is used to show the user the expected
515      # value and tolerance, in the case that a test failed.
516      #
517      # The actual logic for making approximate comparisons can be found in
518      # ApproxScalar, which is used to compare individual numbers.  All of the
519      # other Approx classes eventually delegate to this class.  The ApproxBase
520      # class provides some convenient methods and overloads, but isn't really
521      # essential.
522  
523      __tracebackhide__ = True
524  
525      if isinstance(expected, Decimal):
526          cls: Type[ApproxBase] = ApproxDecimal
527      elif isinstance(expected, Mapping):
528          cls = ApproxMapping
529      elif _is_numpy_array(expected):
530          expected = _as_numpy_array(expected)
531          cls = ApproxNumpy
532      elif (
533          isinstance(expected, Iterable)
534          and isinstance(expected, Sized)
535          # Type ignored because the error is wrong -- not unreachable.
536          and not isinstance(expected, STRING_TYPES)  # type: ignore[unreachable]
537      ):
538          cls = ApproxSequencelike
539      else:
540          cls = ApproxScalar
541  
542      return cls(expected, rel, abs, nan_ok)
543

545  def _is_numpy_array(obj: object) -> bool:
546      """ 
547      Return true if the given object is implicitly convertible to ndarray, 
548      and numpy is already imported. 
549      """
550      return _as_numpy_array(obj) is not None

553  def _as_numpy_array(obj: object) -> Optional["ndarray"]:
554      """ 
555      Return an ndarray if the given object is implicitly convertible to ndarray, 
556      and numpy is already imported, otherwise None. 
557      """
558      import sys
559  
560      np: Any = sys.modules.get("numpy")
561      if np is not None:
562          # avoid infinite recursion on numpy scalars, which have __array__
563          if np.isscalar(obj):
564              return None
565          elif isinstance(obj, np.ndarray):
566              return obj
567          elif hasattr(obj, "__array__") or hasattr("obj", "__array_interface__"):
568              return np.asarray(obj)
569      return None

596  def raises(
597      expected_exception: Union[Type[_E], Tuple[Type[_E], ...]], *args: Any, **kwargs: Any
598  ) -> Union["RaisesContext[_E]", _pytest._code.ExceptionInfo[_E]]:
599      r"""Assert that a code block/function call raises ``expected_exception`` 
600      or raise a failure exception otherwise. 
601   
602      :kwparam match: 
603          If specified, a string containing a regular expression, 
604          or a regular expression object, that is tested against the string 
605          representation of the exception using ``re.search``. To match a literal 
606          string that may contain `special characters`__, the pattern can 
607          first be escaped with ``re.escape``. 
608   
609          (This is only used when ``pytest.raises`` is used as a context manager, 
610          and passed through to the function otherwise. 
611          When using ``pytest.raises`` as a function, you can use: 
612          ``pytest.raises(Exc, func, match="passed on").match("my pattern")``.) 
613   
614          __ https://docs.python.org/3/library/re.html#regular-expression-syntax 
615   
616      .. currentmodule:: _pytest._code 
617   
618      Use ``pytest.raises`` as a context manager, which will capture the exception of the given 
619      type:: 
620   
621          >>> import pytest 
622          >>> with pytest.raises(ZeroDivisionError): 
623          ...    1/0 
624   
625      If the code block does not raise the expected exception (``ZeroDivisionError`` in the example 
626      above), or no exception at all, the check will fail instead. 
627   
628      You can also use the keyword argument ``match`` to assert that the 
629      exception matches a text or regex:: 
630   
631          >>> with pytest.raises(ValueError, match='must be 0 or None'): 
632          ...     raise ValueError("value must be 0 or None") 
633   
634          >>> with pytest.raises(ValueError, match=r'must be \d+$'): 
635          ...     raise ValueError("value must be 42") 
636   
637      The context manager produces an :class:`ExceptionInfo` object which can be used to inspect the 
638      details of the captured exception:: 
639   
640          >>> with pytest.raises(ValueError) as exc_info: 
641          ...     raise ValueError("value must be 42") 
642          >>> assert exc_info.type is ValueError 
643          >>> assert exc_info.value.args[0] == "value must be 42" 
644   
645      .. note:: 
646   
647         When using ``pytest.raises`` as a context manager, it's worthwhile to 
648         note that normal context manager rules apply and that the exception 
649         raised *must* be the final line in the scope of the context manager. 
650         Lines of code after that, within the scope of the context manager will 
651         not be executed. For example:: 
652   
653             >>> value = 15 
654             >>> with pytest.raises(ValueError) as exc_info: 
655             ...     if value > 10: 
656             ...         raise ValueError("value must be <= 10") 
657             ...     assert exc_info.type is ValueError  # this will not execute 
658   
659         Instead, the following approach must be taken (note the difference in 
660         scope):: 
661   
662             >>> with pytest.raises(ValueError) as exc_info: 
663             ...     if value > 10: 
664             ...         raise ValueError("value must be <= 10") 
665             ... 
666             >>> assert exc_info.type is ValueError 
667   
668      **Using with** ``pytest.mark.parametrize`` 
669   
670      When using :ref:`pytest.mark.parametrize ref` 
671      it is possible to parametrize tests such that 
672      some runs raise an exception and others do not. 
673   
674      See :ref:`parametrizing_conditional_raising` for an example. 
675   
676      **Legacy form** 
677   
678      It is possible to specify a callable by passing a to-be-called lambda:: 
679   
680          >>> raises(ZeroDivisionError, lambda: 1/0) 
681          <ExceptionInfo ...> 
682   
683      or you can specify an arbitrary callable with arguments:: 
684   
685          >>> def f(x): return 1/x 
686          ... 
687          >>> raises(ZeroDivisionError, f, 0) 
688          <ExceptionInfo ...> 
689          >>> raises(ZeroDivisionError, f, x=0) 
690          <ExceptionInfo ...> 
691   
692      The form above is fully supported but discouraged for new code because the 
693      context manager form is regarded as more readable and less error-prone. 
694   
695      .. note:: 
696          Similar to caught exception objects in Python, explicitly clearing 
697          local references to returned ``ExceptionInfo`` objects can 
698          help the Python interpreter speed up its garbage collection. 
699   
700          Clearing those references breaks a reference cycle 
701          (``ExceptionInfo`` --> caught exception --> frame stack raising 
702          the exception --> current frame stack --> local variables --> 
703          ``ExceptionInfo``) which makes Python keep all objects referenced 
704          from that cycle (including all local variables in the current 
705          frame) alive until the next cyclic garbage collection run. 
706          More detailed information can be found in the official Python 
707          documentation for :ref:`the try statement <python:try>`. 
708      """
709      __tracebackhide__ = True

711      if isinstance(expected_exception, type):
712          excepted_exceptions: Tuple[Type[_E], ...] = (expected_exception,)
713      else:
714          excepted_exceptions = expected_exception
715      for exc in excepted_exceptions:
716          if not isinstance(exc, type) or not issubclass(exc, BaseException):  # type: ignore[unreachable]
717              msg = "expected exception must be a BaseException type, not {}"  # type: ignore[unreachable]
718              not_a = exc.__name__ if isinstance(exc, type) else type(exc).__name__
719              raise TypeError(msg.format(not_a))
720  
721      message = f"DID NOT RAISE {expected_exception}"
722  
723      if not args:
724          match: Optional[Union[str, Pattern[str]]] = kwargs.pop("match", None)
725          if kwargs:
726              msg = "Unexpected keyword arguments passed to pytest.raises: "
727              msg += ", ".join(sorted(kwargs))
728              msg += "\nUse context-manager form instead?"
729              raise TypeError(msg)
730          return RaisesContext(expected_exception, message, match)
731      else:
732          func = args[0]
733          if not callable(func):
734              raise TypeError(
735                  "{!r} object (type: {}) must be callable".format(func, type(func))
736              )
737          try:
738              func(*args[1:], **kwargs)
739          except expected_exception as e:
740              # We just caught the exception - there is a traceback.
741              assert e.__traceback__ is not None
742              return _pytest._code.ExceptionInfo.from_exc_info(
743                  (type(e), e, e.__traceback__)
744              )
745      fail(message)

752  @final
753  class RaisesContext(Generic[_E]):
754      def __init__(
755          self,
756          expected_exception: Union[Type[_E], Tuple[Type[_E], ...]],
757          message: str,
758          match_expr: Optional[Union[str, Pattern[str]]] = None,
759      ) -> None:
760          self.expected_exception = expected_exception
761          self.message = message
762          self.match_expr = match_expr
763          self.excinfo: Optional[_pytest._code.ExceptionInfo[_E]] = None
764  
765      def __enter__(self) -> _pytest._code.ExceptionInfo[_E]:
766          self.excinfo = _pytest._code.ExceptionInfo.for_later()
767          return self.excinfo
768  
769      def __exit__(
770          self,
771          exc_type: Optional[Type[BaseException]],
772          exc_val: Optional[BaseException],
773          exc_tb: Optional[TracebackType],
774      ) -> bool:
775          __tracebackhide__ = True
776          if exc_type is None:
777              fail(self.message)
778          assert self.excinfo is not None
779          if not issubclass(exc_type, self.expected_exception):
780              return False
781          # Cast to narrow the exception type now that it's verified.
782          exc_info = cast(Tuple[Type[_E], _E, TracebackType], (exc_type, exc_val, exc_tb))
783          self.excinfo.fill_unfilled(exc_info)
784          if self.match_expr is not None:
785              self.excinfo.match(self.match_expr)
786          return True
