1    r"""Evaluate match expressions, as used by `-k` and `-m`. 
2     
3    The grammar is: 
4     
5    expression: expr? EOF 
6    expr:       and_expr ('or' and_expr)* 
7    and_expr:   not_expr ('and' not_expr)* 
8    not_expr:   'not' not_expr | '(' expr ')' | ident 
9    ident:      (\w|:|\+|-|\.|\[|\])+ 
10    
11   The semantics are: 
12    
13   - Empty expression evaluates to False. 
14   - ident evaluates to True of False according to a provided matcher function. 
15   - or/and/not evaluate according to the usual boolean semantics. 
16   """
17   import ast
18   import enum
19   import re

21   from typing import Callable
22   from typing import Iterator

24   from typing import Optional

40   class TokenType(enum.Enum):
41       LPAREN = "left parenthesis"
42       RPAREN = "right parenthesis"
43       OR = "or"
44       AND = "and"
45       NOT = "not"
46       IDENT = "identifier"
47       EOF = "end of input"
48

50   @attr.s(frozen=True, slots=True)
51   class Token:
52       type = attr.ib(type=TokenType)
53       value = attr.ib(type=str)
54       pos = attr.ib(type=int)

57   class ParseError(Exception):
58       """The expression contains invalid syntax. 
59    
60       :param column: The column in the line where the error occurred (1-based). 
61       :param message: A description of the error. 
62       """
63   
64       def __init__(self, column: int, message: str) -> None:
65           self.column = column
66           self.message = message
67   
68       def __str__(self) -> str:
69           return f"at column {self.column}: {self.message}"

72   class Scanner:
73       __slots__ = ("tokens", "current")
74   
75       def __init__(self, input: str) -> None:
76           self.tokens = self.lex(input)
77           self.current = next(self.tokens)
78   
79       def lex(self, input: str) -> Iterator[Token]:
80           pos = 0
81           while pos < len(input):
82               if input[pos] in (" ", "\t"):
83                   pos += 1
84               elif input[pos] == "(":
85                   yield Token(TokenType.LPAREN, "(", pos)
86                   pos += 1
87               elif input[pos] == ")":
88                   yield Token(TokenType.RPAREN, ")", pos)
89                   pos += 1
90               else:
91                   match = re.match(r"(:?\w|:|\+|-|\.|\[|\])+", input[pos:])
92                   if match:
93                       value = match.group(0)
94                       if value == "or":
95                           yield Token(TokenType.OR, value, pos)
96                       elif value == "and":
97                           yield Token(TokenType.AND, value, pos)
98                       elif value == "not":
99                           yield Token(TokenType.NOT, value, pos)
100                      else:
101                          yield Token(TokenType.IDENT, value, pos)
102                      pos += len(value)
103                  else:
104                      raise ParseError(
105                          pos + 1, 'unexpected character "{}"'.format(input[pos]),
106                      )
107          yield Token(TokenType.EOF, "", pos)
108  
109      def accept(self, type: TokenType, *, reject: bool = False) -> Optional[Token]:
110          if self.current.type is type:
111              token = self.current
112              if token.type is not TokenType.EOF:
113                  self.current = next(self.tokens)
114              return token
115          if reject:
116              self.reject((type,))
117          return None
118  
119      def reject(self, expected: Sequence[TokenType]) -> "NoReturn":
120          raise ParseError(
121              self.current.pos + 1,
122              "expected {}; got {}".format(
123                  " OR ".join(type.value for type in expected), self.current.type.value,
124              ),
125          )

128  # True, False and None are legal match expression identifiers,
129  # but illegal as Python identifiers. To fix this, this prefix
130  # is added to identifiers in the conversion to Python AST.
131  IDENT_PREFIX = "$"
132  
133  
134  def expression(s: Scanner) -> ast.Expression:
135      if s.accept(TokenType.EOF):
136          ret: ast.expr = ast.NameConstant(False)
137      else:
138          ret = expr(s)
139          s.accept(TokenType.EOF, reject=True)
140      return ast.fix_missing_locations(ast.Expression(ret))
141  
142  
143  def expr(s: Scanner) -> ast.expr:
144      ret = and_expr(s)
145      while s.accept(TokenType.OR):
146          rhs = and_expr(s)
147          ret = ast.BoolOp(ast.Or(), [ret, rhs])
148      return ret
149  
150  
151  def and_expr(s: Scanner) -> ast.expr:
152      ret = not_expr(s)
153      while s.accept(TokenType.AND):
154          rhs = not_expr(s)
155          ret = ast.BoolOp(ast.And(), [ret, rhs])
156      return ret
157  
158  
159  def not_expr(s: Scanner) -> ast.expr:
160      if s.accept(TokenType.NOT):
161          return ast.UnaryOp(ast.Not(), not_expr(s))
162      if s.accept(TokenType.LPAREN):
163          ret = expr(s)
164          s.accept(TokenType.RPAREN, reject=True)
165          return ret
166      ident = s.accept(TokenType.IDENT)
167      if ident:
168          return ast.Name(IDENT_PREFIX + ident.value, ast.Load())
169      s.reject((TokenType.NOT, TokenType.LPAREN, TokenType.IDENT))

172  class MatcherAdapter(Mapping[str, bool]):
173      """Adapts a matcher function to a locals mapping as required by eval()."""
174  
175      def __init__(self, matcher: Callable[[str], bool]) -> None:
176          self.matcher = matcher
177  
178      def __getitem__(self, key: str) -> bool:
179          return self.matcher(key[len(IDENT_PREFIX) :])
180  
181      def __iter__(self) -> Iterator[str]:
182          raise NotImplementedError()
183  
184      def __len__(self) -> int:
185          raise NotImplementedError()

188  class Expression:
189      """A compiled match expression as used by -k and -m. 
190   
191      The expression can be evaulated against different matchers. 
192      """
193  
194      __slots__ = ("code",)
195  
196      def __init__(self, code: types.CodeType) -> None:
197          self.code = code
198  
199      @classmethod
200      def compile(self, input: str) -> "Expression":
201          """Compile a match expression. 
202   
203          :param input: The input expression - one line. 
204          """
205          astexpr = expression(Scanner(input))
206          code: types.CodeType = compile(
207              astexpr, filename="<pytest match expression>", mode="eval",
208          )
209          return Expression(code)
210  
211      def evaluate(self, matcher: Callable[[str], bool]) -> bool:
212          """Evaluate the match expression. 
213   
214          :param matcher: 
215              Given an identifier, should return whether it matches or not. 
216              Should be prepared to handle arbitrary strings as input. 
217   
218          :returns: Whether the expression matches or not. 
219          """
220          ret: bool = eval(self.code, {"__builtins__": {}}, MatcherAdapter(matcher))
221          return ret
