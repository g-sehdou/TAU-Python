118  @final
119  @attr.s
120  class UnformattedWarning(Generic[_W]):
121      """A warning meant to be formatted during runtime. 
122   
123      This is used to hold warnings that need to format their message at runtime, 
124      as opposed to a direct message. 
125      """
126  
127      category = attr.ib(type=Type["_W"])
128      template = attr.ib(type=str)
129  
130      def format(self, **kwargs: Any) -> _W:
131          """Return an instance of the warning category, formatted with given kwargs."""
132          return self.category(self.template.format(**kwargs))
