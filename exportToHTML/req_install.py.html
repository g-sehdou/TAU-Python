65   def _get_dist(metadata_directory: str) -> Distribution:
66       """Return a pkg_resources.Distribution for the provided 
67       metadata directory. 
68       """
69       dist_dir = metadata_directory.rstrip(os.sep)
70   
71       # Build a PathMetadata object, from path to metadata. :wink:
72       base_dir, dist_dir_name = os.path.split(dist_dir)
73       metadata = pkg_resources.PathMetadata(base_dir, dist_dir)
74   
75       # Determine the correct Distribution object type.
76       if dist_dir.endswith(".egg-info"):
77           dist_cls = pkg_resources.Distribution
78           dist_name = os.path.splitext(dist_dir_name)[0]
79       else:
80           assert dist_dir.endswith(".dist-info")
81           dist_cls = pkg_resources.DistInfoDistribution
82           dist_name = os.path.splitext(dist_dir_name)[0].split("-")[0]
83   
84       return dist_cls(
85           base_dir,
86           project_name=dist_name,
87           metadata=metadata,
88       )

98       def __init__(
99           self,
100          req: Optional[Requirement],
101          comes_from: Optional[Union[str, "InstallRequirement"]],
102          editable: bool = False,
103          link: Optional[Link] = None,
104          markers: Optional[Marker] = None,
105          use_pep517: Optional[bool] = None,
106          isolated: bool = False,
107          install_options: Optional[List[str]] = None,
108          global_options: Optional[List[str]] = None,
109          hash_options: Optional[Dict[str, List[str]]] = None,
110          constraint: bool = False,
111          extras: Collection[str] = (),
112          user_supplied: bool = False,
113          permit_editable_wheels: bool = False,
114      ) -> None:
115          assert req is None or isinstance(req, Requirement), req
116          self.req = req
117          self.comes_from = comes_from
118          self.constraint = constraint
119          self.editable = editable
120          self.permit_editable_wheels = permit_editable_wheels
121          self.legacy_install_reason: Optional[int] = None
122  
123          # source_dir is the local directory where the linked requirement is
124          # located, or unpacked. In case unpacking is needed, creating and
125          # populating source_dir is done by the RequirementPreparer. Note this
126          # is not necessarily the directory where pyproject.toml or setup.py is
127          # located - that one is obtained via unpacked_source_directory.
128          self.source_dir: Optional[str] = None
129          if self.editable:
130              assert link
131              if link.is_file:
132                  self.source_dir = os.path.normpath(os.path.abspath(link.file_path))
133  
134          if link is None and req and req.url:
135              # PEP 508 URL requirement
136              link = Link(req.url)
137          self.link = self.original_link = link
138          self.original_link_is_in_wheel_cache = False
139  
140          # Path to any downloaded or already-existing package.
141          self.local_file_path: Optional[str] = None
142          if self.link and self.link.is_file:
143              self.local_file_path = self.link.file_path
144  
145          if extras:
146              self.extras = extras
147          elif req:
148              self.extras = {pkg_resources.safe_extra(extra) for extra in req.extras}
149          else:
150              self.extras = set()
151          if markers is None and req:
152              markers = req.marker
153          self.markers = markers
154  
155          # This holds the pkg_resources.Distribution object if this requirement
156          # is already available:
157          self.satisfied_by: Optional[Distribution] = None
158          # Whether the installation process should try to uninstall an existing
159          # distribution before installing this requirement.
160          self.should_reinstall = False
161          # Temporary build location
162          self._temp_build_dir: Optional[TempDirectory] = None
163          # Set to True after successful installation
164          self.install_succeeded: Optional[bool] = None
165          # Supplied options
166          self.install_options = install_options if install_options else []
167          self.global_options = global_options if global_options else []
168          self.hash_options = hash_options if hash_options else {}
169          # Set to True after successful preparation of this requirement
170          self.prepared = False
171          # User supplied requirement are explicitly requested for installation
172          # by the user via CLI arguments or requirements files, as opposed to,
173          # e.g. dependencies, extras or constraints.
174          self.user_supplied = user_supplied
175  
176          self.isolated = isolated
177          self.build_env: BuildEnvironment = NoOpBuildEnvironment()
178  
179          # For PEP 517, the directory where we request the project metadata
180          # gets stored. We need this to pass to build_wheel, so the backend
181          # can ensure that the wheel matches the metadata (see the PEP for
182          # details).
183          self.metadata_directory: Optional[str] = None
184  
185          # The static build requirements (from pyproject.toml)
186          self.pyproject_requires: Optional[List[str]] = None
187  
188          # Build requirements that we will check are available
189          self.requirements_to_check: List[str] = []
190  
191          # The PEP 517 backend we should use to build the project
192          self.pep517_backend: Optional[Pep517HookCaller] = None
193  
194          # Are we using PEP 517 for this requirement?
195          # After pyproject.toml has been loaded, the only valid values are True
196          # and False. Before loading, None is valid (meaning "use the default").
197          # Setting an explicit value before loading pyproject.toml is supported,
198          # but after loading this flag should be treated as read only.
199          self.use_pep517 = use_pep517
200  
201          # This requirement needs more preparation before it can be built
202          self.needs_more_preparation = False

204      def __str__(self) -> str:
205          if self.req:
206              s = str(self.req)
207              if self.link:
208                  s += " from {}".format(redact_auth_from_url(self.link.url))
209          elif self.link:
210              s = redact_auth_from_url(self.link.url)
211          else:
212              s = "<InstallRequirement>"
213          if self.satisfied_by is not None:
214              s += " in {}".format(display_path(self.satisfied_by.location))
215          if self.comes_from:
216              if isinstance(self.comes_from, str):
217                  comes_from: Optional[str] = self.comes_from
218              else:
219                  comes_from = self.comes_from.from_path()
220              if comes_from:
221                  s += f" (from {comes_from})"
222          return s

272      def match_markers(self, extras_requested: Optional[Iterable[str]] = None) -> bool:
273          if not extras_requested:
274              # Provide an extra to safely evaluate the markers
275              # without matching any extra
276              extras_requested = ("",)

278              return any(
279                  self.markers.evaluate({"extra": extra}) for extra in extras_requested

294      def hashes(self, trust_internet: bool = True) -> Hashes:
295          """Return a hash-comparer that considers my option- and URL-based 
296          hashes to be known-good. 
297   
298          Hashes in URLs--ones embedded in the requirements file, not ones 
299          downloaded from an index server--are almost peers with ones from 
300          flags. They satisfy --require-hashes (whether it was implicitly or 
301          explicitly activated) but do not activate it. md5 and sha224 are not 
302          allowed in flags, which should nudge people toward good algos. We 
303          always OR all hashes together, even ones from URLs. 
304   
305          :param trust_internet: Whether to trust URL-based (#md5=...) hashes 
306              downloaded from the internet, as by populate_link() 
307   
308          """
309          good_hashes = self.hash_options.copy()
310          link = self.link if trust_internet else self.original_link
311          if link and link.hash:
312              good_hashes.setdefault(link.hash_name, []).append(link.hash)
313          return Hashes(good_hashes)

315      def from_path(self) -> Optional[str]:
316          """Format a nice indicator to show where this "comes from" """
317          if self.req is None:
318              return None
319          s = str(self.req)
320          if self.comes_from:
321              if isinstance(self.comes_from, str):
322                  comes_from = self.comes_from
323              else:
324                  comes_from = self.comes_from.from_path()
325              if comes_from:
326                  s += "->" + comes_from
327          return s

329      def ensure_build_location(
330          self, build_dir: str, autodelete: bool, parallel_builds: bool
331      ) -> str:
332          assert build_dir is not None
333          if self._temp_build_dir is not None:
334              assert self._temp_build_dir.path
335              return self._temp_build_dir.path
336          if self.req is None:
337              # Some systems have /tmp as a symlink which confuses custom
338              # builds (such as numpy). Thus, we ensure that the real path
339              # is returned.
340              self._temp_build_dir = TempDirectory(
341                  kind=tempdir_kinds.REQ_BUILD, globally_managed=True
342              )
343  
344              return self._temp_build_dir.path
345  
346          # This is the only remaining place where we manually determine the path
347          # for the temporary directory. It is only needed for editables where
348          # it is the value of the --src option.
349  
350          # When parallel builds are enabled, add a UUID to the build directory
351          # name so multiple builds do not interfere with each other.
352          dir_name: str = canonicalize_name(self.name)
353          if parallel_builds:
354              dir_name = f"{dir_name}_{uuid.uuid4().hex}"
355  
356          # FIXME: Is there a better place to create the build_dir? (hg and bzr
357          # need this)
358          if not os.path.exists(build_dir):
359              logger.debug("Creating directory %s", build_dir)
360              os.makedirs(build_dir)
361          actual_build_dir = os.path.join(build_dir, dir_name)
362          # `None` indicates that we respect the globally-configured deletion
363          # settings, which is what we actually want when auto-deleting.
364          delete_arg = None if autodelete else False
365          return TempDirectory(
366              path=actual_build_dir,
367              delete=delete_arg,
368              kind=tempdir_kinds.REQ_BUILD,
369              globally_managed=True,
370          ).path

372      def _set_requirement(self) -> None:
373          """Set requirement after generating metadata."""
374          assert self.req is None
375          assert self.metadata is not None
376          assert self.source_dir is not None
377  
378          # Construct a Requirement object from the generated metadata
379          if isinstance(parse_version(self.metadata["Version"]), Version):
380              op = "=="
381          else:
382              op = "==="
383  
384          self.req = Requirement(
385              "".join(
386                  [
387                      self.metadata["Name"],
388                      op,
389                      self.metadata["Version"],
390                  ]
391              )
392          )

394      def warn_on_mismatching_name(self) -> None:
395          metadata_name = canonicalize_name(self.metadata["Name"])
396          if canonicalize_name(self.req.name) == metadata_name:
397              # Everything is fine.
398              return
399  
400          # If we're here, there's a mismatch. Log a warning about it.
401          logger.warning(
402              "Generating metadata for package %s "
403              "produced metadata for project name %s. Fix your "
404              "#egg=%s fragments.",
405              self.name,
406              metadata_name,
407              self.name,
408          )
409          self.req = Requirement(metadata_name)

411      def check_if_exists(self, use_user_site: bool) -> None:
412          """Find an installed distribution that satisfies or conflicts 
413          with this requirement, and set self.satisfied_by or 
414          self.should_reinstall appropriately. 
415          """
416          if self.req is None:
417              return
418          existing_dist = get_distribution(self.req.name)
419          if not existing_dist:
420              return
421  
422          # pkg_resouces may contain a different copy of packaging.version from
423          # pip in if the downstream distributor does a poor job debundling pip.
424          # We avoid existing_dist.parsed_version and let SpecifierSet.contains
425          # parses the version instead.
426          existing_version = existing_dist.version
427          version_compatible = (
428              existing_version is not None
429              and self.req.specifier.contains(existing_version, prereleases=True)
430          )
431          if not version_compatible:
432              self.satisfied_by = None
433              if use_user_site:
434                  if dist_in_usersite(existing_dist):
435                      self.should_reinstall = True
436                  elif running_under_virtualenv() and dist_in_site_packages(
437                      existing_dist
438                  ):
439                      raise InstallationError(
440                          "Will not install to the user site because it will "
441                          "lack sys.path precedence to {} in {}".format(
442                              existing_dist.project_name, existing_dist.location
443                          )
444                      )
445              else:
446                  self.should_reinstall = True
447          else:
448              if self.editable:
449                  self.should_reinstall = True
450                  # when installing editables, nothing pre-existing should ever
451                  # satisfy
452                  self.satisfied_by = None
453              else:
454                  self.satisfied_by = existing_dist

489      def load_pyproject_toml(self) -> None:
490          """Load the pyproject.toml file. 
491   
492          After calling this routine, all of the attributes related to PEP 517 
493          processing for this requirement have been set. In particular, the 
494          use_pep517 attribute can be used to determine whether we should 
495          follow the PEP 517 or legacy (setup.py) code path. 
496          """
497          pyproject_toml_data = load_pyproject_toml(
498              self.use_pep517, self.pyproject_toml_path, self.setup_py_path, str(self)
499          )
500  
501          if pyproject_toml_data is None:
502              self.use_pep517 = False
503              return
504  
505          self.use_pep517 = True
506          requires, backend, check, backend_path = pyproject_toml_data
507          self.requirements_to_check = check
508          self.pyproject_requires = requires
509          self.pep517_backend = Pep517HookCaller(
510              self.unpacked_source_directory,
511              backend,
512              backend_path=backend_path,
513          )

515      def isolated_editable_sanity_check(self) -> None:
516          """Check that an editable requirement if valid for use with PEP 517/518. 
517   
518          This verifies that an editable that has a pyproject.toml either supports PEP 660 
519          or as a setup.py or a setup.cfg 
520          """
521          if (
522              self.editable
523              and self.use_pep517
524              and not self.supports_pyproject_editable()
525              and not os.path.isfile(self.setup_py_path)
526              and not os.path.isfile(self.setup_cfg_path)
527          ):
528              raise InstallationError(
529                  f"Project {self} has a 'pyproject.toml' and its build "
530                  f"backend is missing the 'build_editable' hook. Since it does not "
531                  f"have a 'setup.py' nor a 'setup.cfg', "
532                  f"it cannot be installed in editable mode. "
533                  f"Consider using a build backend that supports PEP 660."
534              )

536      def prepare_metadata(self) -> None:
537          """Ensure that project metadata is available. 
538   
539          Under PEP 517 and PEP 660, call the backend hook to prepare the metadata. 
540          Under legacy processing, call setup.py egg-info. 
541          """
542          assert self.source_dir
543  
544          if self.use_pep517:
545              assert self.pep517_backend is not None
546              if (
547                  self.editable
548                  and self.permit_editable_wheels
549                  and self.supports_pyproject_editable()
550              ):
551                  self.metadata_directory = generate_editable_metadata(
552                      build_env=self.build_env,
553                      backend=self.pep517_backend,
554                  )
555              else:
556                  self.metadata_directory = generate_metadata(
557                      build_env=self.build_env,
558                      backend=self.pep517_backend,
559                  )
560          else:
561              self.metadata_directory = generate_metadata_legacy(
562                  build_env=self.build_env,
563                  setup_py_path=self.setup_py_path,
564                  source_dir=self.unpacked_source_directory,
565                  isolated=self.isolated,
566                  details=self.name or f"from {self.link}",
567              )
568  
569          # Act on the newly generated metadata, based on the name and version.
570          if not self.name:
571              self._set_requirement()
572          else:
573              self.warn_on_mismatching_name()
574  
575          self.assert_source_matches_version()

587      def assert_source_matches_version(self) -> None:
588          assert self.source_dir
589          version = self.metadata["version"]
590          if self.req.specifier and version not in self.req.specifier:
591              logger.warning(
592                  "Requested %s, but installing version %s",
593                  self,
594                  version,
595              )
596          else:
597              logger.debug(
598                  "Source in %s has version %s, which satisfies requirement %s",
599                  display_path(self.source_dir),
600                  version,
601                  self,
602              )

604      # For both source distributions and editables
605      def ensure_has_source_dir(
606          self,
607          parent_dir: str,
608          autodelete: bool = False,
609          parallel_builds: bool = False,
610      ) -> None:
611          """Ensure that a source_dir is set. 
612   
613          This will create a temporary build dir if the name of the requirement 
614          isn't known yet. 
615   
616          :param parent_dir: The ideal pip parent_dir for the source_dir. 
617              Generally src_dir for editables and build_dir for sdists. 
618          :return: self.source_dir 
619          """
620          if self.source_dir is None:
621              self.source_dir = self.ensure_build_location(
622                  parent_dir,
623                  autodelete=autodelete,
624                  parallel_builds=parallel_builds,

627      # For editable installations
628      def update_editable(self) -> None:
629          if not self.link:
630              logger.debug(
631                  "Cannot update repository at %s; repository location is unknown",
632                  self.source_dir,
633              )
634              return
635          assert self.editable
636          assert self.source_dir
637          if self.link.scheme == "file":
638              # Static paths don't get updated
639              return
640          vcs_backend = vcs.get_backend_for_scheme(self.link.scheme)
641          # Editable requirements are validated in Requirement constructors.
642          # So here, if it's neither a path nor a valid VCS URL, it's a bug.
643          assert vcs_backend, f"Unsupported VCS URL {self.link.url}"
644          hidden_url = hide_url(self.link.url)
645          vcs_backend.obtain(self.source_dir, url=hidden_url)

647      # Top-level Actions
648      def uninstall(
649          self, auto_confirm: bool = False, verbose: bool = False
650      ) -> Optional[UninstallPathSet]:
651          """ 
652          Uninstall the distribution currently satisfying this requirement. 
653   
654          Prompts before removing or modifying files unless 
655          ``auto_confirm`` is True. 
656   
657          Refuses to delete or modify files outside of ``sys.prefix`` - 
658          thus uninstallation within a virtual environment can only 
659          modify that virtual environment, even if the virtualenv is 
660          linked to global site-packages. 
661   
662          """
663          assert self.req
664          dist = get_distribution(self.req.name)
665          if not dist:
666              logger.warning("Skipping %s as it is not installed.", self.name)
667              return None
668          logger.info("Found existing installation: %s", dist)
669  
670          uninstalled_pathset = UninstallPathSet.from_dist(dist)
671          uninstalled_pathset.remove(auto_confirm, verbose)
672          return uninstalled_pathset
673  
674      def _get_archive_name(self, path: str, parentdir: str, rootdir: str) -> str:
675          def _clean_zip_name(name: str, prefix: str) -> str:
676              assert name.startswith(
677                  prefix + os.path.sep
678              ), f"name {name!r} doesn't start with prefix {prefix!r}"
679              name = name[len(prefix) + 1 :]
680              name = name.replace(os.path.sep, "/")
681              return name
682  
683          path = os.path.join(parentdir, path)
684          name = _clean_zip_name(path, rootdir)
685          return self.name + "/" + name

687      def archive(self, build_dir: Optional[str]) -> None:
688          """Saves archive to provided build_dir. 
689   
690          Used for saving downloaded VCS requirements as part of `pip download`. 
691          """
692          assert self.source_dir
693          if build_dir is None:
694              return
695  
696          create_archive = True
697          archive_name = "{}-{}.zip".format(self.name, self.metadata["version"])
698          archive_path = os.path.join(build_dir, archive_name)
699  
700          if os.path.exists(archive_path):
701              response = ask_path_exists(
702                  "The file {} exists. (i)gnore, (w)ipe, "
703                  "(b)ackup, (a)bort ".format(display_path(archive_path)),
704                  ("i", "w", "b", "a"),
705              )
706              if response == "i":
707                  create_archive = False
708              elif response == "w":
709                  logger.warning("Deleting %s", display_path(archive_path))
710                  os.remove(archive_path)
711              elif response == "b":
712                  dest_file = backup_dir(archive_path)
713                  logger.warning(
714                      "Backing up %s to %s",
715                      display_path(archive_path),
716                      display_path(dest_file),
717                  )
718                  shutil.move(archive_path, dest_file)
719              elif response == "a":
720                  sys.exit(-1)
721  
722          if not create_archive:
723              return
724  
725          zip_output = zipfile.ZipFile(
726              archive_path,
727              "w",
728              zipfile.ZIP_DEFLATED,
729              allowZip64=True,
730          )
731          with zip_output:
732              dir = os.path.normcase(os.path.abspath(self.unpacked_source_directory))
733              for dirpath, dirnames, filenames in os.walk(dir):
734                  for dirname in dirnames:
735                      dir_arcname = self._get_archive_name(
736                          dirname,
737                          parentdir=dirpath,
738                          rootdir=dir,
739                      )
740                      zipdir = zipfile.ZipInfo(dir_arcname + "/")
741                      zipdir.external_attr = 0x1ED << 16  # 0o755
742                      zip_output.writestr(zipdir, "")
743                  for filename in filenames:
744                      file_arcname = self._get_archive_name(
745                          filename,
746                          parentdir=dirpath,
747                          rootdir=dir,
748                      )
749                      filename = os.path.join(dirpath, filename)
750                      zip_output.write(filename, file_arcname)
751  
752          logger.info("Saved %s", display_path(archive_path))

754      def install(
755          self,
756          install_options: List[str],
757          global_options: Optional[Sequence[str]] = None,
758          root: Optional[str] = None,
759          home: Optional[str] = None,
760          prefix: Optional[str] = None,
761          warn_script_location: bool = True,
762          use_user_site: bool = False,
763          pycompile: bool = True,
764      ) -> None:
765          scheme = get_scheme(
766              self.name,
767              user=use_user_site,
768              home=home,
769              root=root,
770              isolated=self.isolated,
771              prefix=prefix,
772          )
773  
774          global_options = global_options if global_options is not None else []
775          if self.editable and not self.is_wheel:
776              install_editable_legacy(
777                  install_options,
778                  global_options,
779                  prefix=prefix,
780                  home=home,
781                  use_user_site=use_user_site,
782                  name=self.name,
783                  setup_py_path=self.setup_py_path,
784                  isolated=self.isolated,
785                  build_env=self.build_env,
786                  unpacked_source_directory=self.unpacked_source_directory,
787              )
788              self.install_succeeded = True
789              return
790  
791          if self.is_wheel:
792              assert self.local_file_path
793              direct_url = None
794              if self.editable:
795                  direct_url = direct_url_for_editable(self.unpacked_source_directory)
796              elif self.original_link:
797                  direct_url = direct_url_from_link(
798                      self.original_link,
799                      self.source_dir,
800                      self.original_link_is_in_wheel_cache,
801                  )
802              install_wheel(
803                  self.name,
804                  self.local_file_path,
805                  scheme=scheme,
806                  req_description=str(self.req),
807                  pycompile=pycompile,
808                  warn_script_location=warn_script_location,
809                  direct_url=direct_url,
810                  requested=self.user_supplied,
811              )
812              self.install_succeeded = True
813              return
814  
815          # TODO: Why don't we do this for editable installs?
816  
817          # Extend the list of global and install options passed on to
818          # the setup.py call with the ones from the requirements file.
819          # Options specified in requirements file override those
820          # specified on the command line, since the last option given
821          # to setup.py is the one that is used.
822          global_options = list(global_options) + self.global_options
823          install_options = list(install_options) + self.install_options
824  
825          try:
826              success = install_legacy(
827                  install_options=install_options,
828                  global_options=global_options,
829                  root=root,
830                  home=home,
831                  prefix=prefix,
832                  use_user_site=use_user_site,
833                  pycompile=pycompile,
834                  scheme=scheme,
835                  setup_py_path=self.setup_py_path,
836                  isolated=self.isolated,
837                  req_name=self.name,
838                  build_env=self.build_env,
839                  unpacked_source_directory=self.unpacked_source_directory,
840                  req_description=str(self.req),
841              )
842          except LegacyInstallFailure as exc:
843              self.install_succeeded = False
844              raise exc.__cause__
845          except Exception:
846              self.install_succeeded = True
847              raise
848  
849          self.install_succeeded = success
850  
851          if success and self.legacy_install_reason == 8368:
852              deprecated(
853                  reason=(
854                      "{} was installed using the legacy 'setup.py install' "
855                      "method, because a wheel could not be built for it.".format(
856                          self.name
857                      )
858                  ),
859                  replacement="to fix the wheel build issue reported above",
860                  gone_in=None,
861                  issue=8368,
862              )
863

865  def check_invalid_constraint_type(req: InstallRequirement) -> str:
866  
867      # Check for unsupported forms
868      problem = ""
869      if not req.name:
870          problem = "Unnamed requirements are not allowed as constraints"
871      elif req.editable:
872          problem = "Editable requirements are not allowed as constraints"
873      elif req.extras:
874          problem = "Constraints cannot have extras"
875  
876      if problem:
877          deprecated(
878              reason=(
879                  "Constraints are only allowed to take the form of a package "
880                  "name and a version specifier. Other forms were originally "
881                  "permitted as an accident of the implementation, but were "
882                  "undocumented. The new implementation of the resolver no "
883                  "longer supports these forms."
884              ),
885              replacement="replacing the constraint with a requirement",
886              # No plan yet for when the new resolver becomes default
887              gone_in=None,
888              issue=8210,
889          )
890  
891      return problem
