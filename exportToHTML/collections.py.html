9    try:

11       from collections import OrderedDict

13       try:
14           # -- BACK-PORTED FOR: Python 2.4 .. 2.6
15           from ordereddict import OrderedDict
16       except ImportError:
17           message = "collections.OrderedDict is missing: Install 'ordereddict'."
18           warnings.warn(message)
19           # -- BACKWARD-COMPATIBLE: Better than nothing (for behave use case).
20           OrderedDict = dict
