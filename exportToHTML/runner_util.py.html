37   class FileLocationParser(object):
38       pattern = re.compile(r"^\s*(?P<filename>.*):(?P<line>\d+)\s*$", re.UNICODE)
39   
40       @classmethod
41       def parse(cls, text):
42           match = cls.pattern.match(text)
43           if match:
44               filename = match.group("filename").strip()
45               line = int(match.group("line"))
46               return FileLocation(filename, line)
47           # -- NORMAL PATH/FILENAME:
48           filename = text.strip()
49           return FileLocation(filename)
50

52       # def compare(cls, location1, location2):
53       #     loc1 = cls.parse(location1)
54       #     loc2 = cls.parse(location2)
55       #     return cmp(loc1, loc2)

61   class FeatureScenarioLocationCollector(object):
62       """ 
63       Collects FileLocation objects for a feature. 
64       This is used to select a subset of scenarios in a feature that should run. 
65    
66       USE CASE: 
67           behave feature/foo.feature:10 
68           behave @selected_features.txt 
69           behave @rerun_failed_scenarios.txt 
70    
71       With features configuration files, like: 
72    
73           # -- file:rerun_failed_scenarios.txt 
74           feature/foo.feature:10 
75           feature/foo.feature:25 
76           feature/bar.feature 
77           # -- EOF 
78    
79       """
80       def __init__(self, feature=None, location=None, filename=None):
81           if not filename and location:
82               filename = location.filename
83           self.feature = feature
84           self.filename = filename
85           self.use_all_scenarios = False
86           self.scenario_lines = set()
87           self.all_scenarios = set()
88           self.selected_scenarios = set()
89           if location:
90               self.add_location(location)
91   
92       def clear(self):
93           self.feature = None
94           self.filename = None
95           self.use_all_scenarios = False
96           self.scenario_lines = set()
97           self.all_scenarios = set()
98           self.selected_scenarios = set()
99   
100      def add_location(self, location):
101          if not self.filename:
102              self.filename = location.filename
103              # if self.feature and False:
104              #     self.filename = self.feature.filename
105          # -- NORMAL CASE:
106          assert self.filename == location.filename, \
107              "%s <=> %s" % (self.filename, location.filename)
108          if location.line:
109              self.scenario_lines.add(location.line)
110          else:
111              # -- LOCATION WITHOUT LINE NUMBER:
112              # Selects all scenarios in a feature.
113              self.use_all_scenarios = True
114  
115      @staticmethod
116      def select_scenario_line_for(line, scenario_lines):
117          """ 
118          Select scenario line for any given line. 
119   
120          ALGORITHM: scenario.line <= line < next_scenario.line 
121   
122          :param line:  A line number in the file (as number). 
123          :param scenario_lines: Sorted list of scenario lines. 
124          :return: Scenario.line (first line) for the given line. 
125          """
126          if not scenario_lines:
127              return 0    # -- Select all scenarios.
128          pos = bisect(scenario_lines, line) - 1
129          if pos < 0:
130              pos = 0
131          return scenario_lines[pos]
132  
133      def discover_selected_scenarios(self, strict=False):
134          """ 
135          Discovers selected scenarios based on the provided file locations. 
136          In addition: 
137            * discover all scenarios 
138            * auto-correct BAD LINE-NUMBERS 
139   
140          :param strict:  If true, raises exception if file location is invalid. 
141          :return: List of selected scenarios of this feature (as set). 
142          :raises InvalidFileLocationError: 
143              If file location is no exactly correct and strict is true. 
144          """
145          assert self.feature
146          if not self.all_scenarios:
147              self.all_scenarios = self.feature.walk_scenarios()
148  
149          # -- STEP: Check if lines are correct.
150          existing_lines = [scenario.line for scenario in self.all_scenarios]
151          selected_lines = list(self.scenario_lines)
152          for line in selected_lines:
153              new_line = self.select_scenario_line_for(line, existing_lines)
154              if new_line != line:
155                  # -- AUTO-CORRECT BAD-LINE:
156                  self.scenario_lines.remove(line)
157                  self.scenario_lines.add(new_line)
158                  if strict:
159                      msg = "Scenario location '...:%d' should be: '%s:%d'" % \
160                            (line, self.filename, new_line)
161                      raise InvalidFileLocationError(msg)
162  
163          # -- STEP: Determine selected scenarios and store them.
164          scenario_lines = set(self.scenario_lines)
165          selected_scenarios = set()
166          for scenario in self.all_scenarios:
167              if scenario.line in scenario_lines:
168                  selected_scenarios.add(scenario)
169                  scenario_lines.remove(scenario.line)
170          # -- CHECK ALL ARE RESOLVED:
171          assert not scenario_lines
172          return selected_scenarios
173  
174      def build_feature(self):
175          """ 
176          Determines which scenarios in the feature are selected and marks the 
177          remaining scenarios as skipped. Scenarios with the following tags 
178          are excluded from skipped-marking: 
179   
180            * @setup 
181            * @teardown 
182   
183          If no file locations are stored, the unmodified feature is returned. 
184   
185          :return: Feature object to use. 
186          """
187          use_all_scenarios = not self.scenario_lines or self.use_all_scenarios
188          if not self.feature or use_all_scenarios:
189              return self.feature
190  
191          # -- CASE: Select subset of all scenarios of this feature.
192          #    Mark other scenarios as skipped (except in a few cases).
193          self.all_scenarios = self.feature.walk_scenarios()
194          self.selected_scenarios = self.discover_selected_scenarios()
195          unselected_scenarios = set(self.all_scenarios) - self.selected_scenarios
196          for scenario in unselected_scenarios:
197              if "setup" in scenario.tags or "teardown" in scenario.tags:
198                  continue
199              scenario.mark_skipped()
200          return self.feature

203  class FeatureListParser(object):
204      """ 
205      Read textual file, ala '@features.txt'. This file contains: 
206   
207        * a feature filename or FileLocation on each line 
208        * empty lines (skipped) 
209        * comment lines (skipped) 
210        * wildcards are expanded to select 0..N filenames or directories 
211   
212      Relative path names are evaluated relative to the listfile directory. 
213      A leading '@' (AT) character is removed from the listfile name. 
214      """
215  
216      @staticmethod
217      def parse(text, here=None):
218          """ 
219          Parse contents of a features list file as text. 
220   
221          :param text: Contents of a features list(file). 
222          :param here: Current working directory to use (optional). 
223          :return: List of FileLocation objects 
224          """
225          locations = []
226          for line in text.splitlines():
227              filename = line.strip()
228              if not filename:
229                  continue    # SKIP: Over empty line(s).
230              elif filename.startswith('#'):
231                  continue    # SKIP: Over comment line(s).
232  
233              if here and not os.path.isabs(filename):
234                  filename = os.path.join(here, line)
235              filename = os.path.normpath(filename)
236              if glob.has_magic(filename):
237                  # -- WITH WILDCARDS:
238                  for filename2 in glob.iglob(filename):
239                      location = FileLocationParser.parse(filename2)
240                      locations.append(location)
241              else:
242                  location = FileLocationParser.parse(filename)
243                  locations.append(location)
244          return locations
245  
246      @classmethod
247      def parse_file(cls, filename):
248          """ 
249          Read textual file, ala '@features.txt'. 
250   
251          :param filename:  Name of feature list file. 
252          :return: List of feature file locations. 
253          """
254          if filename.startswith('@'):
255              filename = filename[1:]
256          if not os.path.isfile(filename):
257              raise FileNotFoundError(filename)
258          here = os.path.dirname(filename) or "."
259          # -- MAYBE BETTER:
260          # contents = codecs.open(filename, "utf-8").read()
261          contents = open(filename).read()
262          return cls.parse(contents, here)

265  class PathManager(object):
266      """Context manager to add paths to sys.path (python search path) 
267      within a scope. 
268      """
269  
270      def __init__(self, paths=None):
271          self.initial_paths = paths or []
272          self.paths = None
273  
274      def __enter__(self):
275          self.paths = list(self.initial_paths)
276          sys.path = self.paths + sys.path
277  
278      def __exit__(self, *crap):
279          for path in self.paths:
280              sys.path.remove(path)
281          self.paths = None
282  
283      def add(self, path):
284          if self.paths is None:
285              # -- CALLED OUTSIDE OF CONTEXT:
286              self.initial_paths.append(path)
287          else:
288              sys.path.insert(0, path)
289              self.paths.append(path)

295  def parse_features(feature_files, language=None):
296      """ 
297      Parse feature files and return list of Feature model objects. 
298      Handles: 
299   
300        * feature file names, ala "alice.feature" 
301        * feature file locations, ala: "alice.feature:10" 
302   
303      :param feature_files: List of feature file names to parse. 
304      :param language:      Default language to use. 
305      :return: List of feature objects. 
306      """
307      scenario_collector = FeatureScenarioLocationCollector()
308      features = []
309      for location in feature_files:
310          if not isinstance(location, FileLocation):
311              assert isinstance(location, string_types)
312              location = FileLocation(os.path.normpath(location))
313  
314          if location.filename == scenario_collector.filename:
315              scenario_collector.add_location(location)
316              continue
317          elif scenario_collector.feature:
318              # -- ADD CURRENT FEATURE: As collection of scenarios.
319              current_feature = scenario_collector.build_feature()
320              features.append(current_feature)
321              scenario_collector.clear()
322  
323          # -- NEW FEATURE:
324          assert isinstance(location, FileLocation)
325          filename = os.path.abspath(location.filename)
326          feature = parser.parse_file(filename, language=language)
327          if feature:
328              # -- VALID FEATURE:
329              # SKIP CORNER-CASE: Feature file without any feature(s).
330              scenario_collector.feature = feature
331              scenario_collector.add_location(location)
332      # -- FINALLY:
333      if scenario_collector.feature:
334          current_feature = scenario_collector.build_feature()
335          features.append(current_feature)
336      return features

339  def collect_feature_locations(paths, strict=True):
340      """ 
341      Collect feature file names by processing list of paths (from command line). 
342      A path can be a: 
343   
344        * filename (ending with ".feature") 
345        * location, ala "{filename}:{line_number}" 
346        * features configuration filename, ala "@features.txt" 
347        * directory, to discover and collect all "*.feature" files below. 
348   
349      :param paths:  Paths to process. 
350      :return: Feature file locations to use (as list of FileLocations). 
351      """
352      locations = []
353      for path in paths:
354          if os.path.isdir(path):
355              for dirpath, dirnames, filenames in os.walk(path):
356                  dirnames.sort()
357                  for filename in sorted(filenames):
358                      if filename.endswith(".feature"):
359                          location = FileLocation(os.path.join(dirpath, filename))
360                          locations.append(location)
361          elif path.startswith('@'):
362              # -- USE: behave @list_of_features.txt
363              locations.extend(FeatureListParser.parse_file(path[1:]))
364          else:
365              # -- OTHERWISE: Normal filename or location (schema: filename:line)
366              location = FileLocationParser.parse(path)
367              if not location.filename.endswith(".feature"):
368                  raise InvalidFilenameError(location.filename)
369              elif location.exists():
370                  locations.append(location)
371              elif strict:
372                  raise FileNotFoundError(path)
373      return locations

376  def exec_file(filename, globals_=None, locals_=None):
377      if globals_ is None:
378          globals_ = {}
379      if locals_ is None:
380          locals_ = globals_
381      locals_["__file__"] = filename
382      with open(filename, "rb") as f:
383          # pylint: disable=exec-used
384          filename2 = os.path.relpath(filename, os.getcwd())
385          code = compile(f.read(), filename2, "exec", dont_inherit=True)
386          exec(code, globals_, locals_)

389  def load_step_modules(step_paths):
390      """Load step modules with step definitions from step_paths directories."""
391      from behave import matchers
392      from behave.step_registry import setup_step_decorators
393      step_globals = {
394          "use_step_matcher": matchers.use_step_matcher,
395          "step_matcher":     matchers.step_matcher, # -- DEPRECATING
396      }
397      setup_step_decorators(step_globals)
398  
399      # -- Allow steps to import other stuff from the steps dir
400      # NOTE: Default matcher can be overridden in "environment.py" hook.
401      with PathManager(step_paths):
402          default_matcher = matchers.current_matcher
403          for path in step_paths:
404              for name in sorted(os.listdir(path)):
405                  if name.endswith(".py"):
406                      # -- LOAD STEP DEFINITION:
407                      # Reset to default matcher after each step-definition.
408                      # A step-definition may change the matcher 0..N times.
409                      # ENSURE: Each step definition has clean globals.
410                      # try:
411                      step_module_globals = step_globals.copy()
412                      exec_file(os.path.join(path, name), step_module_globals)
413                      matchers.current_matcher = default_matcher

416  def make_undefined_step_snippet(step, language=None):
417      """Helper function to create an undefined-step snippet for a step. 
418   
419      :param step: Step to use (as Step object or string). 
420      :param language: i18n language, optionally needed for step text parsing. 
421      :return: Undefined-step snippet (as string). 
422      """
423      if isinstance(step, string_types):
424          step_text = step
425          steps = parser.parse_steps(step_text, language=language)
426          step = steps[0]
427          assert step, "ParseError: %s" % step_text
428  
429      prefix = u"u"
430      single_quote = "'"
431      if single_quote in step.name:
432          step.name = step.name.replace(single_quote, r"\'")
433  
434      schema = u"@%s(%s'%s')\ndef step_impl(context):\n"
435      schema += u"    raise NotImplementedError(%s'STEP: %s %s')\n\n"
436      snippet = schema % (step.step_type, prefix, step.name,
437                          prefix, step.step_type.title(), step.name)
438      return snippet

441  def make_undefined_step_snippets(undefined_steps, make_snippet=None):
442      """Creates a list of undefined step snippets. 
443      Note that duplicated steps are removed internally. 
444   
445      :param undefined_steps: List of undefined steps (as Step object or string). 
446      :param make_snippet:    Function that generates snippet (optional) 
447      :return: List of undefined step snippets (as list of strings) 
448      """
449      if make_snippet is None:
450          make_snippet = make_undefined_step_snippet
451  
452      # -- NOTE: Remove any duplicated undefined steps.
453      step_snippets = []
454      collected_steps = set()
455      for undefined_step in undefined_steps:
456          if undefined_step in collected_steps:
457              continue
458          collected_steps.add(undefined_step)
459          step_snippet = make_snippet(undefined_step)
460          step_snippets.append(step_snippet)
461      return step_snippets

464  def print_undefined_step_snippets(undefined_steps, stream=None, colored=True):
465      """ 
466      Print snippets for the undefined steps that were discovered. 
467   
468      :param undefined_steps:  List of undefined steps (as list<string>). 
469      :param stream:      Output stream to use (default: sys.stderr). 
470      :param colored:     Indicates if coloring should be used (default: True) 
471      """
472      if not undefined_steps:
473          return
474      if not stream:
475          stream = sys.stderr
476  
477      msg = u"\nYou can implement step definitions for undefined steps with "
478      msg += u"these snippets:\n\n"
479      msg += u"\n".join(make_undefined_step_snippets(undefined_steps))
480  
481      if colored:
482          # -- OOPS: Unclear if stream supports ANSI coloring.
483          from behave.formatter.ansi_escapes import escapes
484          msg = escapes['undefined'] + msg + escapes['reset']
485  
486      stream = ensure_stream_with_encoder(stream)
487      stream.write(msg)
488      stream.flush()
