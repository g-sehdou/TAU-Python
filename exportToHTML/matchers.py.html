21   class StepParseError(ValueError):
22       """Exception class, used when step matching fails before a step is run. 
23       This is normally the case when an error occurs during the type conversion 
24       of step parameters. 
25       """
26   
27       def __init__(self, text=None, exc_cause=None):
28           if not text and exc_cause:
29               text = six.text_type(exc_cause)
30           if exc_cause and six.PY2:
31               # -- NOTE: Python2 does not show chained-exception causes.
32               #    Therefore, provide some hint (see also: PEP-3134).
33               cause_text = ExceptionUtil.describe(exc_cause,
34                                                   use_traceback=True,
35                                                   prefix="CAUSED-BY: ")
36               text += u"\n" + cause_text
37   
38           ValueError.__init__(self, text)
39           if exc_cause:
40               # -- CHAINED EXCEPTION (see: PEP 3134)
41               ChainedExceptionUtil.set_cause(self, exc_cause)

48   class Match(Replayable):
49       """An parameter-matched *feature file* step name extracted using 
50       step decorator `parameters`_. 
51    
52       .. attribute:: func 
53    
54          The step function that this match will be applied to. 
55    
56       .. attribute:: arguments 
57    
58          A list of :class:`~behave.model_core.Argument` instances containing the 
59          matched parameters from the step name. 
60       """
61       type = "match"
62   
63       def __init__(self, func, arguments=None):
64           super(Match, self).__init__()
65           self.func = func
66           self.arguments = arguments
67           self.location = None
68           if func:
69               self.location = self.make_location(func)
70   
71       def __repr__(self):
72           if self.func:
73               func_name = self.func.__name__
74           else:
75               func_name = '<no function>'
76           return '<Match %s, %s>' % (func_name, self.location)
77   
78       def __eq__(self, other):
79           if not isinstance(other, Match):
80               return False
81           return (self.func, self.location) == (other.func, other.location)
82   
83       def with_arguments(self, arguments):
84           match = copy.copy(self)
85           match.arguments = arguments
86           return match
87   
88       def run(self, context):
89           args = []
90           kwargs = {}
91           for arg in self.arguments:
92               if arg.name is not None:
93                   kwargs[arg.name] = arg.value
94               else:
95                   args.append(arg.value)
96   
97           with context.use_with_user_mode():
98               self.func(context, *args, **kwargs)
99   
100      @staticmethod
101      def make_location(step_function):
102          """Extracts the location information from the step function and 
103          builds a FileLocation object with (filename, line_number) info. 
104   
105          :param step_function: Function whose location should be determined. 
106          :return: FileLocation object for step function. 
107          """
108          return FileLocation.for_function(step_function)

123  class MatchWithError(Match):
124      """Match class when error occur during step-matching 
125   
126      REASON: 
127        * Type conversion error occured. 
128        * ... 
129      """
130      def __init__(self, func, error):
131          if not ExceptionUtil.has_traceback(error):
132              ExceptionUtil.set_traceback(error)
133          Match.__init__(self, func=func)
134          self.stored_error = error
135  
136      def run(self, context):
137          """Raises stored error from step matching phase (type conversion)."""
138          raise StepParseError(exc_cause=self.stored_error)

146  class Matcher(object):
147      """Pull parameters out of step names. 
148   
149      .. attribute:: pattern 
150   
151         The match pattern attached to the step function. 
152   
153      .. attribute:: func 
154   
155         The step function the pattern is being attached to. 
156      """
157      schema = u"@%s('%s')"   # Schema used to describe step definition (matcher)
158  
159      def __init__(self, func, pattern, step_type=None):
160          self.func = func
161          self.pattern = pattern
162          self.step_type = step_type
163          self._location = None
164  
165      # -- BACKWARD-COMPATIBILITY:
166      @property
167      def string(self):
168          warnings.warn("deprecated: Use 'pattern' instead", DeprecationWarning)
169          return self.pattern
170  
171      @property
172      def location(self):
173          if self._location is None:
174              self._location = Match.make_location(self.func)
175          return self._location
176  
177      @property
178      def regex_pattern(self):
179          """Return the used textual regex pattern."""
180          # -- ASSUMPTION: pattern attribute provides regex-pattern
181          # NOTE: Method must be overridden if assumption is not met.
182          return self.pattern
183  
184      def describe(self, schema=None):
185          """Provide a textual description of the step function/matcher object. 
186   
187          :param schema:  Text schema to use. 
188          :return: Textual description of this step definition (matcher). 
189          """
190          step_type = self.step_type or "step"
191          if not schema:
192              schema = self.schema
193          return schema % (step_type, self.pattern)
194  
195  
196      def check_match(self, step):
197          """Match me against the "step" name supplied. 
198   
199          Return None, if I don't match otherwise return a list of matches as 
200          :class:`~behave.model_core.Argument` instances. 
201   
202          The return value from this function will be converted into a 
203          :class:`~behave.matchers.Match` instance by *behave*. 
204          """
205          raise NotImplementedError
206  
207      def match(self, step):
208          # -- PROTECT AGAINST: Type conversion errors (with ParseMatcher).
209          try:
210              result = self.check_match(step)
211          except Exception as e:  # pylint: disable=broad-except
212              return MatchWithError(self.func, e)
213  
214          if result is None:
215              return None     # -- NO-MATCH
216          return Match(self.func, result)
217  
218      def __repr__(self):
219          return u"<%s: %r>" % (self.__class__.__name__, self.pattern)

222  class ParseMatcher(Matcher):
223      """Uses :class:`~parse.Parser` class to be able to use simpler 
224      parse expressions compared to normal regular expressions. 
225      """
226      custom_types = {}
227      parser_class = parse.Parser
228  
229      def __init__(self, func, pattern, step_type=None):
230          super(ParseMatcher, self).__init__(func, pattern, step_type)
231          self.parser = self.parser_class(pattern, self.custom_types)
232  
233      @property
234      def regex_pattern(self):
235          # -- OVERWRITTEN: Pattern as regex text.
236          return self.parser._expression  # pylint: disable=protected-access
237  
238      def check_match(self, step):
239          # -- FAILURE-POINT: Type conversion of parameters may fail here.
240          #    NOTE: Type converter should raise ValueError in case of PARSE ERRORS.
241          result = self.parser.parse(step)
242          if not result:
243              return None
244  
245          args = []
246          for index, value in enumerate(result.fixed):
247              start, end = result.spans[index]
248              args.append(Argument(start, end, step[start:end], value))
249          for name, value in result.named.items():
250              start, end = result.spans[name]
251              args.append(Argument(start, end, step[start:end], value, name))
252          args.sort(key=lambda x: x.start)
253          return args

264  def register_type(**kw):
265      # pylint: disable=anomalous-backslash-in-string
266      # REQUIRED-BY: code example
267      """Registers a custom type that will be available to "parse" 
268      for type conversion during step matching. 
269   
270      Converters should be supplied as ``name=callable`` arguments (or as dict). 
271   
272      A type converter should follow :pypi:`parse` module rules. 
273      In general, a type converter is a function that converts text (as string) 
274      into a value-type (type converted value). 
275   
276      EXAMPLE: 
277   
278      .. code-block:: python 
279   
280          from behave import register_type, given 
281          import parse 
282   
283          # -- TYPE CONVERTER: For a simple, positive integer number. 
284          @parse.with_pattern(r"\d+") 
285          def parse_number(text): 
286              return int(text) 
287   
288          # -- REGISTER TYPE-CONVERTER: With behave 
289          register_type(Number=parse_number) 
290   
291          # -- STEP DEFINITIONS: Use type converter. 
292          @given('{amount:Number} vehicles') 
293          def step_impl(context, amount): 
294              assert isinstance(amount, int) 
295      """
296      ParseMatcher.custom_types.update(kw)

299  class RegexMatcher(Matcher):
300      def __init__(self, func, pattern, step_type=None):
301          super(RegexMatcher, self).__init__(func, pattern, step_type)
302          self.regex = re.compile(self.pattern)
303  
304      def check_match(self, step):
305          m = self.regex.match(step)
306          if not m:
307              return None
308  
309          groupindex = dict((y, x) for x, y in self.regex.groupindex.items())
310          args = []
311          for index, group in enumerate(m.groups()):
312              index += 1
313              name = groupindex.get(index, None)
314              args.append(Argument(m.start(index), m.end(index), group,
315                                   group, name))
316  
317          return args

319  class SimplifiedRegexMatcher(RegexMatcher):
320      """Simplified regular expression step-matcher that automatically adds 
321      start-of-line/end-of-line matcher symbols to string: 
322   
323      .. code-block:: python 
324   
325          @when(u'a step passes')     # re.pattern = "^a step passes$" 
326          def step_impl(context): pass 
327      """
328  
329      def __init__(self, func, pattern, step_type=None):
330          assert not (pattern.startswith("^") or pattern.endswith("$")), \
331              "Regular expression should not use begin/end-markers: "+ pattern
332          expression = "^%s$" % pattern
333          super(SimplifiedRegexMatcher, self).__init__(func, expression, step_type)
334          self.pattern = pattern

363  def use_step_matcher(name):
364      """Change the parameter matcher used in parsing step text. 
365   
366      The change is immediate and may be performed between step definitions in 
367      your step implementation modules - allowing adjacent steps to use different 
368      matchers if necessary. 
369   
370      There are several parsers available in *behave* (by default): 
371   
372      **parse** (the default, based on: :pypi:`parse`) 
373          Provides a simple parser that replaces regular expressions for 
374          step parameters with a readable syntax like ``{param:Type}``. 
375          The syntax is inspired by the Python builtin ``string.format()`` 
376          function. 
377          Step parameters must use the named fields syntax of :pypi:`parse` 
378          in step definitions. The named fields are extracted, 
379          optionally type converted and then used as step function arguments. 
380   
381          Supports type conversions by using type converters 
382          (see :func:`~behave.register_type()`). 
383   
384      **cfparse** (extends: :pypi:`parse`, requires: :pypi:`parse_type`) 
385          Provides an extended parser with "Cardinality Field" (CF) support. 
386          Automatically creates missing type converters for related cardinality 
387          as long as a type converter for cardinality=1 is provided. 
388          Supports parse expressions like: 
389   
390          * ``{values:Type+}`` (cardinality=1..N, many) 
391          * ``{values:Type*}`` (cardinality=0..N, many0) 
392          * ``{value:Type?}``  (cardinality=0..1, optional) 
393   
394          Supports type conversions (as above). 
395   
396      **re** 
397          This uses full regular expressions to parse the clause text. You will 
398          need to use named groups "(?P<name>...)" to define the variables pulled 
399          from the text and passed to your ``step()`` function. 
400   
401          Type conversion is **not supported**. 
402          A step function writer may implement type conversion 
403          inside the step function (implementation). 
404   
405      You may `define your own matcher`_. 
406   
407      .. _`define your own matcher`: api.html#step-parameters 
408      """
409      global current_matcher  # pylint: disable=global-statement
410      current_matcher = matcher_mapping[name]
411  
412  def step_matcher(name):
