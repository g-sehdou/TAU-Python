239  def getreportopt(config: Config) -> str:
240      reportchars: str = config.option.reportchars
241  
242      old_aliases = {"F", "S"}
243      reportopts = ""
244      for char in reportchars:
245          if char in old_aliases:
246              char = char.lower()
247          if char == "a":
248              reportopts = "sxXEf"
249          elif char == "A":
250              reportopts = "PpsxXEf"
251          elif char == "N":
252              reportopts = ""
253          elif char not in reportopts:
254              reportopts += char
255  
256      if not config.option.disable_warnings and "w" not in reportopts:
257          reportopts = "w" + reportopts
258      elif config.option.disable_warnings and "w" in reportopts:
259          reportopts = reportopts.replace("w", "")
260  
261      return reportopts

264  @hookimpl(trylast=True)  # after _pytest.runner
265  def pytest_report_teststatus(report: BaseReport) -> Tuple[str, str, str]:
266      letter = "F"
267      if report.passed:
268          letter = "."
269      elif report.skipped:
270          letter = "s"
271  
272      outcome: str = report.outcome
273      if report.when in ("collect", "setup", "teardown") and outcome == "failed":
274          outcome = "error"
275          letter = "E"
276  
277      return outcome, letter, outcome.upper()

280  @attr.s
281  class WarningReport:
282      """Simple structure to hold warnings information captured by ``pytest_warning_recorded``. 
283   
284      :ivar str message: 
285          User friendly message about the warning. 
286      :ivar str|None nodeid: 
287          nodeid that generated the warning (see ``get_location``). 
288      :ivar tuple|py.path.local fslocation: 
289          File system location of the source of the warning (see ``get_location``). 
290      """
291  
292      message = attr.ib(type=str)
293      nodeid = attr.ib(type=Optional[str], default=None)
294      fslocation = attr.ib(
295          type=Optional[Union[Tuple[str, int], py.path.local]], default=None
296      )
297      count_towards_summary = True
298  
299      def get_location(self, config: Config) -> Optional[str]:
300          """Return the more user-friendly information about the location of a warning, or None."""
301          if self.nodeid:
302              return self.nodeid
303          if self.fslocation:
304              if isinstance(self.fslocation, tuple) and len(self.fslocation) >= 2:
305                  filename, linenum = self.fslocation[:2]
306                  relpath = bestrelpath(
307                      config.invocation_params.dir, absolutepath(filename)
308                  )
309                  return f"{relpath}:{linenum}"
310              else:
311                  return str(self.fslocation)
312          return None

344      def _determine_show_progress_info(self) -> "Literal['progress', 'count', False]":
345          """Return whether we should display progress information based on the current config."""
346          # do not show progress if we are not capturing output (#3038)
347          if self.config.getoption("capture", "no") == "no":
348              return False
349          # do not show progress if we are showing fixture setup/teardown
350          if self.config.getoption("setupshow", False):
351              return False
352          cfg: str = self.config.getini("console_output_style")
353          if cfg == "progress":
354              return "progress"
355          elif cfg == "count":
356              return "count"
357          else:
358              return False

395      def write_fspath_result(self, nodeid: str, res, **markup: bool) -> None:
396          fspath = self.config.rootpath / nodeid.split("::")[0]
397          if self.currentfspath is None or fspath != self.currentfspath:
398              if self.currentfspath is not None and self._show_progress_info:
399                  self._write_progress_information_filling_space()
400              self.currentfspath = fspath
401              relfspath = bestrelpath(self.startpath, fspath)
402              self._tw.line()
403              self._tw.write(relfspath + " ")
404          self._tw.write(res, flush=True, **markup)

432      def rewrite(self, line: str, **markup: bool) -> None:
433          """Rewinds the terminal cursor to the beginning and writes the given line. 
434   
435          :param erase: 
436              If True, will also add spaces until the full terminal width to ensure 
437              previous lines are properly erased. 
438   
439          The rest of the keyword arguments are markup instructions. 
440          """
441          erase = markup.pop("erase", False)
442          if erase:
443              fill_count = self._tw.fullwidth - len(line) - 1
444              fill = " " * fill_count
445          else:
446              fill = ""
447          line = str(line)
448          self._tw.write("\r" + line + fill, **markup)

514      def pytest_runtest_logreport(self, report: TestReport) -> None:
515          self._tests_ran = True
516          rep = report
517          res: Tuple[
518              str, str, Union[str, Tuple[str, Mapping[str, bool]]]
519          ] = self.config.hook.pytest_report_teststatus(report=rep, config=self.config)
520          category, letter, word = res
521          if not isinstance(word, tuple):
522              markup = None
523          else:
524              word, markup = word
525          self._add_stats(category, [rep])
526          if not letter and not word:
527              # Probably passed setup/teardown.
528              return
529          running_xdist = hasattr(rep, "node")
530          if markup is None:
531              was_xfail = hasattr(report, "wasxfail")
532              if rep.passed and not was_xfail:
533                  markup = {"green": True}
534              elif rep.passed and was_xfail:
535                  markup = {"yellow": True}
536              elif rep.failed:
537                  markup = {"red": True}
538              elif rep.skipped:
539                  markup = {"yellow": True}
540              else:
541                  markup = {}
542          if self.verbosity <= 0:
543              self._tw.write(letter, **markup)
544          else:
545              self._progress_nodeids_reported.add(rep.nodeid)
546              line = self._locationline(rep.nodeid, *rep.location)
547              if not running_xdist:
548                  self.write_ensure_prefix(line, word, **markup)
549                  if rep.skipped or hasattr(report, "wasxfail"):
550                      available_width = (
551                          (self._tw.fullwidth - self._tw.width_of_current_line)
552                          - len(" [100%]")
553                          - 1
554                      )
555                      reason = _get_raw_skip_reason(rep)
556                      reason_ = _format_trimmed(" ({})", reason, available_width)
557                      if reason and reason_ is not None:
558                          self._tw.write(reason_)
559                  if self._show_progress_info:
560                      self._write_progress_information_filling_space()
561              else:
562                  self.ensure_newline()
563                  self._tw.write("[%s]" % rep.node.gateway.id)
564                  if self._show_progress_info:
565                      self._tw.write(
566                          self._get_progress_information_message() + " ", cyan=True
567                      )
568                  else:
569                      self._tw.write(" ")
570                  self._tw.write(word, **markup)
571                  self._tw.write(" " + line)
572                  self.currentfspath = -2
573          self.flush()

575      @property
576      def _is_last_item(self) -> bool:
577          assert self._session is not None
578          return len(self._progress_nodeids_reported) == self._session.testscollected

580      def pytest_runtest_logfinish(self, nodeid: str) -> None:
581          assert self._session
582          if self.verbosity <= 0 and self._show_progress_info:
583              if self._show_progress_info == "count":
584                  num_tests = self._session.testscollected
585                  progress_length = len(" [{}/{}]".format(str(num_tests), str(num_tests)))
586              else:
587                  progress_length = len(" [100%]")
588  
589              self._progress_nodeids_reported.add(nodeid)
590  
591              if self._is_last_item:
592                  self._write_progress_information_filling_space()
593              else:
594                  main_color, _ = self._get_main_color()
595                  w = self._width_of_current_line
596                  past_edge = w + progress_length + 1 >= self._screen_width
597                  if past_edge:
598                      msg = self._get_progress_information_message()
599                      self._tw.write(msg + "\n", **{main_color: True})
600  
601      def _get_progress_information_message(self) -> str:
602          assert self._session
603          collected = self._session.testscollected
604          if self._show_progress_info == "count":
605              if collected:
606                  progress = self._progress_nodeids_reported
607                  counter_format = "{{:{}d}}".format(len(str(collected)))
608                  format_string = f" [{counter_format}/{{}}]"
609                  return format_string.format(len(progress), collected)
610              return f" [ {collected} / {collected} ]"
611          else:
612              if collected:
613                  return " [{:3d}%]".format(
614                      len(self._progress_nodeids_reported) * 100 // collected
615                  )
616              return " [100%]"
617  
618      def _write_progress_information_filling_space(self) -> None:
619          color, _ = self._get_main_color()
620          msg = self._get_progress_information_message()
621          w = self._width_of_current_line
622          fill = self._tw.fullwidth - w - 1
623          self.write(msg.rjust(fill), flush=True, **{color: True})

630      def pytest_collection(self) -> None:
631          if self.isatty:
632              if self.config.option.verbose >= 0:
633                  self.write("collecting ... ", flush=True, bold=True)
634                  self._collect_report_last_write = timing.time()
635          elif self.config.option.verbose >= 1:

638      def pytest_collectreport(self, report: CollectReport) -> None:
639          if report.failed:
640              self._add_stats("error", [report])
641          elif report.skipped:
642              self._add_stats("skipped", [report])
643          items = [x for x in report.result if isinstance(x, Item)]
644          self._numcollected += len(items)
645          if self.isatty:
646              self.report_collect()
647  
648      def report_collect(self, final: bool = False) -> None:
649          if self.config.option.verbose < 0:
650              return
651  
652          if not final:
653              # Only write "collecting" report every 0.5s.
654              t = timing.time()
655              if (
656                  self._collect_report_last_write is not None
657                  and self._collect_report_last_write > t - REPORT_COLLECTING_RESOLUTION
658              ):
659                  return
660              self._collect_report_last_write = t
661  
662          errors = len(self.stats.get("error", []))
663          skipped = len(self.stats.get("skipped", []))
664          deselected = len(self.stats.get("deselected", []))
665          selected = self._numcollected - errors - skipped - deselected
666          if final:
667              line = "collected "
668          else:
669              line = "collecting "
670          line += (
671              str(self._numcollected) + " item" + ("" if self._numcollected == 1 else "s")
672          )
673          if errors:
674              line += " / %d error%s" % (errors, "s" if errors != 1 else "")
675          if deselected:
676              line += " / %d deselected" % deselected
677          if skipped:
678              line += " / %d skipped" % skipped
679          if self._numcollected > selected > 0:
680              line += " / %d selected" % selected
681          if self.isatty:

764      def _printcollecteditems(self, items: Sequence[Item]) -> None:
765          # To print out items and their parent collectors
766          # we take care to leave out Instances aka ()
767          # because later versions are going to get rid of them anyway.
768          if self.config.option.verbose < 0:
769              if self.config.option.verbose < -1:
770                  counts = Counter(item.nodeid.split("::", 1)[0] for item in items)
771                  for name, count in sorted(counts.items()):
772                      self._tw.line("%s: %d" % (name, count))
773              else:
774                  for item in items:
775                      self._tw.line(item.nodeid)
776              return
777          stack: List[Node] = []
778          indent = ""
779          for item in items:
780              needed_collectors = item.listchain()[1:]  # strip root node
781              while stack:
782                  if stack == needed_collectors[: len(stack)]:
783                      break
784                  stack.pop()
785              for col in needed_collectors[len(stack) :]:
786                  stack.append(col)
787                  if col.name == "()":  # Skip Instances.
788                      continue
789                  indent = (len(stack) - 1) * "  "
790                  self._tw.line(f"{indent}{col}")
791                  if self.config.option.verbose >= 1:
792                      obj = getattr(col, "obj", None)
793                      doc = inspect.getdoc(obj) if obj else None
794                      if doc:
795                          for line in doc.splitlines():
796                              self._tw.line("{}{}".format(indent + "  ", line))

859      def _locationline(self, nodeid, fspath, lineno, domain):
860          def mkrel(nodeid):
861              line = self.config.cwd_relative_nodeid(nodeid)

863                  line = line[: -len(domain)]
864                  values = domain.split("[")
865                  values[0] = values[0].replace(".", "::")  # don't replace '.' in params
866                  line += "[".join(values)
867              return line
868  
869          # collect_fspath comes from testid which has a "/"-normalized path.
870  
871          if fspath:

874                  "\\", nodes.SEP

906      def summary_warnings(self) -> None:
907          if self.hasopt("w"):
908              all_warnings: Optional[List[WarningReport]] = self.stats.get("warnings")
909              if not all_warnings:
910                  return
911  
912              final = self._already_displayed_warnings is not None
913              if final:
914                  warning_reports = all_warnings[self._already_displayed_warnings :]
915              else:
916                  warning_reports = all_warnings
917              self._already_displayed_warnings = len(warning_reports)
918              if not warning_reports:
919                  return
920  
921              reports_grouped_by_message: Dict[str, List[WarningReport]] = {}
922              for wr in warning_reports:
923                  reports_grouped_by_message.setdefault(wr.message, []).append(wr)
924  
925              def collapsed_location_report(reports: List[WarningReport]) -> str:
926                  locations = []
927                  for w in reports:
928                      location = w.get_location(self.config)
929                      if location:
930                          locations.append(location)
931  
932                  if len(locations) < 10:
933                      return "\n".join(map(str, locations))
934  
935                  counts_by_filename = Counter(
936                      str(loc).split("::", 1)[0] for loc in locations
937                  )
938                  return "\n".join(
939                      "{}: {} warning{}".format(k, v, "s" if v > 1 else "")
940                      for k, v in counts_by_filename.items()
941                  )
942  
943              title = "warnings summary (final)" if final else "warnings summary"
944              self.write_sep("=", title, yellow=True, bold=False)
945              for message, message_reports in reports_grouped_by_message.items():
946                  maybe_location = collapsed_location_report(message_reports)
947                  if maybe_location:
948                      self._tw.line(maybe_location)
949                      lines = message.splitlines()
950                      indented = "\n".join("  " + x for x in lines)
951                      message = indented.rstrip()
952                  else:
953                      message = message.rstrip()
954                  self._tw.line(message)
955                  self._tw.line()
956              self._tw.line("-- Docs: https://docs.pytest.org/en/stable/warnings.html")

1042     def summary_stats(self) -> None:
1043         if self.verbosity < -1:
1044             return
1045 
1046         session_duration = timing.time() - self._sessionstarttime
1047         (parts, main_color) = self.build_summary_stats_line()
1048         line_parts = []
1049 
1050         display_sep = self.verbosity >= 0
1051         if display_sep:
1052             fullwidth = self._tw.fullwidth
1053         for text, markup in parts:
1054             with_markup = self._tw.markup(text, **markup)
1055             if display_sep:
1056                 fullwidth += len(with_markup) - len(text)
1057             line_parts.append(with_markup)
1058         msg = ", ".join(line_parts)
1059 
1060         main_markup = {main_color: True}
1061         duration = " in {}".format(format_session_duration(session_duration))
1062         duration_with_markup = self._tw.markup(duration, **main_markup)
1063         if display_sep:
1064             fullwidth += len(duration_with_markup) - len(duration)
1065         msg += duration_with_markup
1066 
1067         if display_sep:
1068             markup_for_end_sep = self._tw.markup("", **main_markup)
1069             if markup_for_end_sep.endswith("\x1b[0m"):
1070                 markup_for_end_sep = markup_for_end_sep[:-4]
1071             fullwidth += len(markup_for_end_sep)
1072             msg += markup_for_end_sep
1073 
1074         if display_sep:
1075             self.write_sep("=", msg, fullwidth=fullwidth, **main_markup)
1076         else:
1077             self.write_line(msg, **main_markup)

1079     def short_test_summary(self) -> None:
1080         if not self.reportchars:
1081             return
1082 
1083         def show_simple(stat, lines: List[str]) -> None:
1084             failed = self.stats.get(stat, [])

1121                     lines.append(

1123                         % (verbose_word, num, fspath, lineno, reason)
1124                     )
1125                 else:
1126                     lines.append("%s [%d] %s: %s" % (verbose_word, num, fspath, reason))

1176     def build_summary_stats_line(self) -> Tuple[List[Tuple[str, Dict[str, bool]]], str]:
1177         """ 
1178         Build the parts used in the last summary stats line. 
1179  
1180         The summary stats line is the line shown at the end, "=== 12 passed, 2 errors in Xs===". 
1181  
1182         This function builds a list of the "parts" that make up for the text in that line, in 
1183         the example above it would be: 
1184  
1185             [ 
1186                 ("12 passed", {"green": True}), 
1187                 ("2 errors", {"red": True} 
1188             ] 
1189  
1190         That last dict for each line is a "markup dictionary", used by TerminalWriter to 
1191         color output. 
1192  
1193         The final color of the line is also determined by this function, and is the second 
1194         element of the returned tuple. 
1195         """
1196         if self.config.getoption("collectonly"):
1197             return self._build_collect_only_summary_stats_line()
1198         else:
1199             return self._build_normal_summary_stats_line()

1206     def _build_normal_summary_stats_line(
1207         self,
1208     ) -> Tuple[List[Tuple[str, Dict[str, bool]]], str]:
1209         main_color, known_types = self._get_main_color()
1210         parts = []
1211 
1212         for key in known_types:
1213             reports = self._get_reports_to_display(key)
1214             if reports:
1215                 count = len(reports)
1216                 color = _color_for_type.get(key, _color_for_type_default)
1217                 markup = {color: True, "bold": color == main_color}
1218                 parts.append(("%d %s" % pluralize(count, key), markup))
1219 
1220         if not parts:
1221             parts = [("no tests ran", {_color_for_type_default: True})]
1222 
1223         return parts, main_color

1225     def _build_collect_only_summary_stats_line(
1226         self,

1228         deselected = len(self._get_reports_to_display("deselected"))
1229         errors = len(self._get_reports_to_display("error"))
1230 
1231         if self._numcollected == 0:
1232             parts = [("no tests collected", {"yellow": True})]
1233             main_color = "yellow"
1234 
1235         elif deselected == 0:

1238             parts = [(collected_output, {main_color: True})]

1241             if all_tests_were_deselected:

1244             else:
1245                 main_color = "green"
1246                 selected = self._numcollected - deselected
1247                 collected_output = f"{selected}/{self._numcollected} tests collected ({deselected} deselected)"
1248 
1249             parts = [(collected_output, {main_color: True})]
1250 
1251         if errors:
1252             main_color = _color_for_type["error"]
1253             parts += [("%d %s" % pluralize(errors, "error"), {main_color: True})]
1254 
1255         return parts, main_color

1263 def _format_trimmed(format: str, msg: str, available_width: int) -> Optional[str]:
1264     """Format msg into format, ellipsizing it if doesn't fit in available_width. 
1265  
1266     Returns None if even the ellipsis can't fit.

1275     if format_width + len(ellipsis) > available_width:
1276         return None
1277 
1278     if format_width + wcswidth(msg) > available_width:
1279         available_width -= len(ellipsis)
1280         msg = msg[:available_width]
1281         while format_width + wcswidth(msg) > available_width:
1282             msg = msg[:-1]
1283         msg += ellipsis
1284 
1285     return format.format(msg)

1312 def _folded_skips(
1313     startpath: Path, skipped: Sequence[CollectReport],
1314 ) -> List[Tuple[int, str, Optional[int], str]]:
1315     d: Dict[Tuple[str, Optional[int], str], List[CollectReport]] = {}
1316     for event in skipped:
1317         assert event.longrepr is not None
1318         assert isinstance(event.longrepr, tuple), (event, event.longrepr)
1319         assert len(event.longrepr) == 3, (event, event.longrepr)
1320         fspath, lineno, reason = event.longrepr
1321         # For consistency, report all fspaths in relative form.
1322         fspath = bestrelpath(startpath, Path(fspath))
1323         keywords = getattr(event, "keywords", {})
1324         # Folding reports with global pytestmark variable.
1325         # This is a workaround, because for now we cannot identify the scope of a skip marker
1326         # TODO: Revisit after marks scope would be fixed.
1327         if (
1328             event.when == "setup"
1329             and "skip" in keywords
1330             and "pytestmark" not in keywords
1331         ):
1332             key: Tuple[str, Optional[int], str] = (fspath, None, reason)
1333         else:
1334             key = (fspath, lineno, reason)
1335         d.setdefault(key, []).append(event)
1336     values: List[Tuple[int, str, Optional[int], str]] = []
1337     for key, events in d.items():
1338         values.append((len(events), *key))
1339     return values

1351 def pluralize(count: int, noun: str) -> Tuple[int, str]:
1352     # No need to pluralize words such as `failed` or `passed`.
1353     if noun not in ["error", "warnings", "test"]:
1354         return count, noun
1355 
1356     # The `warnings` key is plural. To avoid API breakage, we keep it that way but
1357     # set it to singular here so we can determine plurality in the same way as we do
1358     # for `error`.
1359     noun = noun.replace("warnings", "warning")
1360 
1361     return count, noun + "s" if count != 1 else noun

1387 def _get_raw_skip_reason(report: TestReport) -> str:
1388     """Get the reason string of a skip/xfail/xpass test report. 
1389  
1390     The string is just the part given by the user. 
1391     """

1395             reason = reason[len("reason: ") :]
1396         return reason
1397     else:
1398         assert report.skipped
1399         assert isinstance(report.longrepr, tuple)
1400         _, _, reason = report.longrepr

1402             reason = reason[len("Skipped: ") :]
1403         elif reason == "Skipped":
