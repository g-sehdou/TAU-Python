9    import random

11   """ 
12   initialize atrributes to be used in/available for all/class methods in this class, and for class objects created by this class

16   class Person:
17       def __init__(self, firstname, lastname, health, status):
18           self.firstname = firstname
19           self.lastname = lastname
20           self.health = health
21           self.status = status
22   
23       def intro(self):
24           print("Hallo, my name is {} {}".format(self.firstname, self.lastname))
25   
26       def emo(self):
27           emotion = random.randrange(1,4)
28           if emotion == 1:
29               print("{} {} is happy today".format(self.firstname, self.lastname))
30           elif emotion == 2:
31               print("{} {} is sad today".format(self.firstname, self.lastname))
32           elif emotion == 3:
33               print("{} {} is having nice time".format(self.firstname, self.lastname))
34   
35       def change_status(self):
36           if self.health >= 100:
37               print("{} {} is totally healthy !".format(self.firstname, self.lastname))
38           elif self.health <= 65:
39               print("{} {} is feeling ok ".format(self.firstname, self.lastname))
40           elif self.health <= 30:
41               print("{} {} is not feeling ok and need a doctor".format(self.firstname, self.lastname))
42           else:
43               print("{} {} is unconscious. Please call 112")
44   
45   
46   Maria = Person("Maria", "Marco", 37, status=True)
47   John = Person("John", "Janssen", 80, status=False)
48   Patrick = Person("Patrick", "Longer", 19, status=True)
49   
50   print("{} {} is a friend? {}".format(Maria.firstname, Maria.lastname, Maria.status))
51   print("{} {} is a friend? {}".format(John.firstname, John.lastname, John.status))
52   print("{} {} is my friend? {}".format(Patrick.firstname, Patrick.lastname, Patrick.status))
