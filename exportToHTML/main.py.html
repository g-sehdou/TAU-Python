228  def validate_basetemp(path: str) -> str:
229      # GH 7119
230      msg = "basetemp must not be empty, the current working directory or any parent directory of it"
231  
232      # empty path
233      if not path:
234          raise argparse.ArgumentTypeError(msg)
235  
236      def is_ancestor(base: Path, query: Path) -> bool:
237          """Return whether query is an ancestor of base."""
238          if base == query:
239              return True
240          for parent in base.parents:
241              if parent == query:
242                  return True
243          return False
244  
245      # check if path is an ancestor of cwd
246      if is_ancestor(Path.cwd(), Path(path).absolute()):
247          raise argparse.ArgumentTypeError(msg)
248  
249      # check symlinks for ancestors
250      if is_ancestor(Path.cwd().resolve(), Path(path).resolve()):
251          raise argparse.ArgumentTypeError(msg)
252  
253      return path

256  def wrap_session(
257      config: Config, doit: Callable[[Config, "Session"], Optional[Union[int, ExitCode]]]
258  ) -> Union[int, ExitCode]:
259      """Skeleton command line program."""
260      session = Session.from_config(config)
261      session.exitstatus = ExitCode.OK
262      initstate = 0
263      try:
264          try:
265              config._do_configure()
266              initstate = 1
267              config.hook.pytest_sessionstart(session=session)
268              initstate = 2
269              session.exitstatus = doit(config, session) or 0
270          except UsageError:
271              session.exitstatus = ExitCode.USAGE_ERROR
272              raise
273          except Failed:
274              session.exitstatus = ExitCode.TESTS_FAILED
275          except (KeyboardInterrupt, exit.Exception):
276              excinfo = _pytest._code.ExceptionInfo.from_current()
277              exitstatus: Union[int, ExitCode] = ExitCode.INTERRUPTED
278              if isinstance(excinfo.value, exit.Exception):
279                  if excinfo.value.returncode is not None:
280                      exitstatus = excinfo.value.returncode
281                  if initstate < 2:
282                      sys.stderr.write(f"{excinfo.typename}: {excinfo.value.msg}\n")
283              config.hook.pytest_keyboard_interrupt(excinfo=excinfo)
284              session.exitstatus = exitstatus
285          except BaseException:
286              session.exitstatus = ExitCode.INTERNAL_ERROR
287              excinfo = _pytest._code.ExceptionInfo.from_current()
288              try:
289                  config.notify_exception(excinfo, config.option)
290              except exit.Exception as exc:
291                  if exc.returncode is not None:
292                      session.exitstatus = exc.returncode
293                  sys.stderr.write("{}: {}\n".format(type(exc).__name__, exc))
294              else:
295                  if isinstance(excinfo.value, SystemExit):
296                      sys.stderr.write("mainloop: caught unexpected SystemExit!\n")
297  
298      finally:
299          # Explicitly break reference cycle.
300          excinfo = None  # type: ignore
301          session.startdir.chdir()
302          if initstate >= 2:
303              try:
304                  config.hook.pytest_sessionfinish(
305                      session=session, exitstatus=session.exitstatus
306                  )
307              except exit.Exception as exc:
308                  if exc.returncode is not None:
309                      session.exitstatus = exc.returncode
310                  sys.stderr.write("{}: {}\n".format(type(exc).__name__, exc))
311          config._ensure_unconfigure()
312      return session.exitstatus

319  def _main(config: Config, session: "Session") -> Optional[Union[int, ExitCode]]:
320      """Default command line protocol for initialization, session, 
321      running tests and reporting."""
322      config.hook.pytest_collection(session=session)
323      config.hook.pytest_runtestloop(session=session)
324  
325      if session.testsfailed:
326          return ExitCode.TESTS_FAILED
327      elif session.testscollected == 0:
328          return ExitCode.NO_TESTS_COLLECTED
329      return None

336  def pytest_runtestloop(session: "Session") -> bool:
337      if session.testsfailed and not session.config.option.continue_on_collection_errors:
338          raise session.Interrupted(
339              "%d error%s during collection"
340              % (session.testsfailed, "s" if session.testsfailed != 1 else "")
341          )
342  
343      if session.config.option.collectonly:
344          return True
345  
346      for i, item in enumerate(session.items):
347          nextitem = session.items[i + 1] if i + 1 < len(session.items) else None
348          item.config.hook.pytest_runtest_protocol(item=item, nextitem=nextitem)
349          if session.shouldfail:
350              raise session.Failed(session.shouldfail)
351          if session.shouldstop:
352              raise session.Interrupted(session.shouldstop)
353      return True

356  def _in_venv(path: py.path.local) -> bool:
357      """Attempt to detect if ``path`` is the root of a Virtual Environment by 
358      checking for the existence of the appropriate activate script."""
359      bindir = path.join("Scripts" if sys.platform.startswith("win") else "bin")
360      if not bindir.isdir():
361          return False
362      activates = (
363          "activate",
364          "activate.csh",
365          "activate.fish",
366          "Activate",
367          "Activate.bat",
368          "Activate.ps1",
369      )
370      return any([fname.basename in activates for fname in bindir.listdir()])

373  def pytest_ignore_collect(path: py.path.local, config: Config) -> Optional[bool]:
374      ignore_paths = config._getconftest_pathlist("collect_ignore", path=path.dirpath())
375      ignore_paths = ignore_paths or []
376      excludeopt = config.getoption("ignore")
377      if excludeopt:
378          ignore_paths.extend([py.path.local(x) for x in excludeopt])
379  
380      if py.path.local(path) in ignore_paths:
381          return True
382  
383      ignore_globs = config._getconftest_pathlist(
384          "collect_ignore_glob", path=path.dirpath()
385      )
386      ignore_globs = ignore_globs or []
387      excludeglobopt = config.getoption("ignore_glob")
388      if excludeglobopt:
389          ignore_globs.extend([py.path.local(x) for x in excludeglobopt])
390  
391      if any(fnmatch.fnmatch(str(path), str(glob)) for glob in ignore_globs):
392          return True
393  
394      allow_in_venv = config.getoption("collect_in_virtualenv")
395      if not allow_in_venv and _in_venv(path):
396          return True
397      return None

400  def pytest_collection_modifyitems(items: List[nodes.Item], config: Config) -> None:
401      deselect_prefixes = tuple(config.getoption("deselect") or [])
402      if not deselect_prefixes:
403          return
404  
405      remaining = []
406      deselected = []
407      for colitem in items:
408          if colitem.nodeid.startswith(deselect_prefixes):
409              deselected.append(colitem)
410          else:
411              remaining.append(colitem)
412  
413      if deselected:
414          config.hook.pytest_deselected(items=deselected)
415          items[:] = remaining

515      def gethookproxy(self, fspath: py.path.local):
516          # Check if we have the common case of running
517          # hooks with all conftest.py files.
518          pm = self.config.pluginmanager
519          my_conftestmodules = pm._getconftestmodules(
520              fspath, self.config.getoption("importmode")
521          )
522          remove_mods = pm._conftest_plugins.difference(my_conftestmodules)
523          if remove_mods:
524              # One or more conftests are not in use at this fspath.
525              proxy = FSHookProxy(pm, remove_mods)
526          else:
527              # All plugins are active for this fspath.
528              proxy = self.config.hook
529          return proxy

531      def _recurse(self, direntry: "os.DirEntry[str]") -> bool:
532          if direntry.name == "__pycache__":
533              return False
534          path = py.path.local(direntry.path)
535          ihook = self.gethookproxy(path.dirpath())
536          if ihook.pytest_ignore_collect(path=path, config=self.config):
537              return False
538          norecursepatterns = self.config.getini("norecursedirs")
539          if any(path.check(fnmatch=pat) for pat in norecursepatterns):
540              return False
541          return True

543      def _collectfile(
544          self, path: py.path.local, handle_dupes: bool = True
545      ) -> Sequence[nodes.Collector]:
546          assert (
547              path.isfile()
548          ), "{!r} is not a file (isdir={!r}, exists={!r}, islink={!r})".format(
549              path, path.isdir(), path.exists(), path.islink()
550          )
551          ihook = self.gethookproxy(path)
552          if not self.isinitpath(path):
553              if ihook.pytest_ignore_collect(path=path, config=self.config):
554                  return ()
555  
556          if handle_dupes:
557              keepduplicates = self.config.getoption("keepduplicates")
558              if not keepduplicates:
559                  duplicate_paths = self.config.pluginmanager._duplicatepaths
560                  if path in duplicate_paths:
561                      return ()
562                  else:
563                      duplicate_paths.add(path)
564  
565          return ihook.pytest_collect_file(path=path, parent=self)  # type: ignore[no-any-return]

579      def perform_collect(
580          self, args: Optional[Sequence[str]] = None, genitems: bool = True
581      ) -> Sequence[Union[nodes.Item, nodes.Collector]]:
582          """Perform the collection phase for this session. 
583   
584          This is called by the default 
585          :func:`pytest_collection <_pytest.hookspec.pytest_collection>` hook 
586          implementation; see the documentation of this hook for more details. 
587          For testing purposes, it may also be called directly on a fresh 
588          ``Session``. 
589   
590          This function normally recursively expands any collectors collected 
591          from the session to their items, and only items are returned. For 
592          testing purposes, this may be suppressed by passing ``genitems=False``, 
593          in which case the return value contains these collectors unexpanded, 
594          and ``session.items`` is empty. 
595          """
596          if args is None:
597              args = self.config.args
598  
599          self.trace("perform_collect", self, args)
600          self.trace.root.indent += 1
601  
602          self._notfound: List[Tuple[str, Sequence[nodes.Collector]]] = []
603          self._initial_parts: List[Tuple[py.path.local, List[str]]] = []
604          self.items: List[nodes.Item] = []
605  
606          hook = self.config.hook
607  
608          items: Sequence[Union[nodes.Item, nodes.Collector]] = self.items
609          try:
610              initialpaths: List[py.path.local] = []
611              for arg in args:
612                  fspath, parts = resolve_collection_argument(
613                      self.config.invocation_params.dir,
614                      arg,
615                      as_pypath=self.config.option.pyargs,
616                  )
617                  self._initial_parts.append((fspath, parts))
618                  initialpaths.append(fspath)
619              self._initialpaths = frozenset(initialpaths)
620              rep = collect_one_node(self)
621              self.ihook.pytest_collectreport(report=rep)
622              self.trace.root.indent -= 1
623              if self._notfound:
624                  errors = []
625                  for arg, cols in self._notfound:
626                      line = f"(no name {arg!r} in any of {cols!r})"
627                      errors.append(f"not found: {arg}\n{line}")
628                  raise UsageError(*errors)
629              if not genitems:
630                  items = rep.result
631              else:
632                  if rep.passed:
633                      for node in rep.result:
634                          self.items.extend(self.genitems(node))
635  
636              self.config.pluginmanager.check_pending()
637              hook.pytest_collection_modifyitems(
638                  session=self, config=self.config, items=items
639              )
640          finally:
641              hook.pytest_collection_finish(session=self)
642  
643          self.testscollected = len(items)
644          return items

646      def collect(self) -> Iterator[Union[nodes.Item, nodes.Collector]]:
647          from _pytest.python import Package
648  
649          # Keep track of any collected nodes in here, so we don't duplicate fixtures.
650          node_cache1: Dict[py.path.local, Sequence[nodes.Collector]] = {}
651          node_cache2: Dict[
652              Tuple[Type[nodes.Collector], py.path.local], nodes.Collector
653          ] = ({})
654  
655          # Keep track of any collected collectors in matchnodes paths, so they
656          # are not collected more than once.
657          matchnodes_cache: Dict[Tuple[Type[nodes.Collector], str], CollectReport] = ({})
658  
659          # Dirnames of pkgs with dunder-init files.
660          pkg_roots: Dict[str, Package] = {}
661  
662          for argpath, names in self._initial_parts:
663              self.trace("processing argument", (argpath, names))
664              self.trace.root.indent += 1
665  
666              # Start with a Session root, and delve to argpath item (dir or file)
667              # and stack all Packages found on the way.
668              # No point in finding packages when collecting doctests.
669              if not self.config.getoption("doctestmodules", False):
670                  pm = self.config.pluginmanager
671                  for parent in reversed(argpath.parts()):
672                      if pm._confcutdir and pm._confcutdir.relto(parent):
673                          break
674  
675                      if parent.isdir():
676                          pkginit = parent.join("__init__.py")
677                          if pkginit.isfile() and pkginit not in node_cache1:
678                              col = self._collectfile(pkginit, handle_dupes=False)
679                              if col:
680                                  if isinstance(col[0], Package):
681                                      pkg_roots[str(parent)] = col[0]
682                                  node_cache1[col[0].fspath] = [col[0]]
683  
684              # If it's a directory argument, recurse and look for any Subpackages.
685              # Let the Package collector deal with subnodes, don't collect here.
686              if argpath.check(dir=1):
687                  assert not names, "invalid arg {!r}".format((argpath, names))
688  
689                  seen_dirs: Set[py.path.local] = set()
690                  for direntry in visit(str(argpath), self._recurse):
691                      if not direntry.is_file():
692                          continue
693  
694                      path = py.path.local(direntry.path)
695                      dirpath = path.dirpath()
696  
697                      if dirpath not in seen_dirs:
698                          # Collect packages first.
699                          seen_dirs.add(dirpath)
700                          pkginit = dirpath.join("__init__.py")
701                          if pkginit.exists():
702                              for x in self._collectfile(pkginit):
703                                  yield x
704                                  if isinstance(x, Package):
705                                      pkg_roots[str(dirpath)] = x
706                      if str(dirpath) in pkg_roots:
707                          # Do not collect packages here.
708                          continue
709  
710                      for x in self._collectfile(path):
711                          key = (type(x), x.fspath)
712                          if key in node_cache2:
713                              yield node_cache2[key]
714                          else:
715                              node_cache2[key] = x
716                              yield x
717              else:
718                  assert argpath.check(file=1)
719  
720                  if argpath in node_cache1:
721                      col = node_cache1[argpath]
722                  else:
723                      collect_root = pkg_roots.get(argpath.dirname, self)
724                      col = collect_root._collectfile(argpath, handle_dupes=False)
725                      if col:
726                          node_cache1[argpath] = col
727  
728                  matching = []
729                  work: List[
730                      Tuple[Sequence[Union[nodes.Item, nodes.Collector]], Sequence[str]]
731                  ] = [(col, names)]
732                  while work:
733                      self.trace("matchnodes", col, names)
734                      self.trace.root.indent += 1
735  
736                      matchnodes, matchnames = work.pop()
737                      for node in matchnodes:
738                          if not matchnames:
739                              matching.append(node)
740                              continue
741                          if not isinstance(node, nodes.Collector):
742                              continue
743                          key = (type(node), node.nodeid)
744                          if key in matchnodes_cache:
745                              rep = matchnodes_cache[key]
746                          else:
747                              rep = collect_one_node(node)
748                              matchnodes_cache[key] = rep
749                          if rep.passed:
750                              submatchnodes = []
751                              for r in rep.result:
752                                  # TODO: Remove parametrized workaround once collection structure contains
753                                  # parametrization.
754                                  if (
755                                      r.name == matchnames[0]
756                                      or r.name.split("[")[0] == matchnames[0]
757                                  ):
758                                      submatchnodes.append(r)
759                              if submatchnodes:
760                                  work.append((submatchnodes, matchnames[1:]))
761                              # XXX Accept IDs that don't have "()" for class instances.
762                              elif len(rep.result) == 1 and rep.result[0].name == "()":
763                                  work.append((rep.result, matchnames))
764                          else:
765                              # Report collection failures here to avoid failing to run some test
766                              # specified in the command line because the module could not be
767                              # imported (#134).
768                              node.ihook.pytest_collectreport(report=rep)
769  
770                      self.trace("matchnodes finished -> ", len(matching), "nodes")
771                      self.trace.root.indent -= 1
772  
773                  if not matching:
774                      report_arg = "::".join((str(argpath), *names))
775                      self._notfound.append((report_arg, col))
776                      continue
777  
778                  # If __init__.py was the only file requested, then the matched
779                  # node will be the corresponding Package (by default), and the
780                  # first yielded item will be the __init__ Module itself, so
781                  # just use that. If this special case isn't taken, then all the
782                  # files in the package will be yielded.
783                  if argpath.basename == "__init__.py" and isinstance(
784                      matching[0], Package
785                  ):
786                      try:
787                          yield next(iter(matching[0].collect()))
788                      except StopIteration:
789                          # The package collects nothing with only an __init__.py
790                          # file in it, which gets ignored by the default
791                          # "python_files" option.
792                          pass
793                      continue
794  
795                  yield from matching
796  
797              self.trace.root.indent -= 1
798  
799      def genitems(
800          self, node: Union[nodes.Item, nodes.Collector]
801      ) -> Iterator[nodes.Item]:
802          self.trace("genitems", node)
803          if isinstance(node, nodes.Item):
804              node.ihook.pytest_itemcollected(item=node)
805              yield node
806          else:
807              assert isinstance(node, nodes.Collector)
808              rep = collect_one_node(node)
809              if rep.passed:
810                  for subnode in rep.result:
811                      yield from self.genitems(subnode)
812              node.ihook.pytest_collectreport(report=rep)

815  def search_pypath(module_name: str) -> str:
816      """Search sys.path for the given a dotted module name, and return its file system path."""
817      try:
818          spec = importlib.util.find_spec(module_name)
819      # AttributeError: looks like package module, but actually filename
820      # ImportError: module does not exist
821      # ValueError: not a module name
822      except (AttributeError, ImportError, ValueError):
823          return module_name
824      if spec is None or spec.origin is None or spec.origin == "namespace":
825          return module_name
826      elif spec.submodule_search_locations:
827          return os.path.dirname(spec.origin)
828      else:
829          return spec.origin

832  def resolve_collection_argument(
833      invocation_path: Path, arg: str, *, as_pypath: bool = False
834  ) -> Tuple[py.path.local, List[str]]:
835      """Parse path arguments optionally containing selection parts and return (fspath, names). 
836   
837      Command-line arguments can point to files and/or directories, and optionally contain 
838      parts for specific tests selection, for example: 
839   
840          "pkg/tests/test_foo.py::TestClass::test_foo" 
841   
842      This function ensures the path exists, and returns a tuple: 
843   
844          (py.path.path("/full/path/to/pkg/tests/test_foo.py"), ["TestClass", "test_foo"]) 
845   
846      When as_pypath is True, expects that the command-line argument actually contains 
847      module paths instead of file-system paths: 
848   
849          "pkg.tests.test_foo::TestClass::test_foo" 
850   
851      In which case we search sys.path for a matching module, and then return the *path* to the 
852      found module. 
853   
854      If the path doesn't exist, raise UsageError. 
855      If the path is a directory and selection parts are present, raise UsageError. 
856      """
857      strpath, *parts = str(arg).split("::")
858      if as_pypath:
859          strpath = search_pypath(strpath)
860      fspath = invocation_path / strpath
861      fspath = absolutepath(fspath)
862      if not fspath.exists():
863          msg = (
864              "module or package not found: {arg} (missing __init__.py?)"
865              if as_pypath
866              else "file or directory not found: {arg}"
867          )
868          raise UsageError(msg.format(arg=arg))
869      if parts and fspath.is_dir():
870          msg = (
871              "package argument cannot contain :: selection parts: {arg}"
872              if as_pypath
873              else "directory argument cannot contain :: selection parts: {arg}"
874          )
875          raise UsageError(msg.format(arg=arg))
876      return py.path.local(str(fspath)), parts
