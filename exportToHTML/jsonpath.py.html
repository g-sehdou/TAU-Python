71   def normalize(x):
72       """normalize the path expression; outside jsonpath to allow testing"""
73       subx = []
74   
75       # replace index/filter expressions with placeholders
76       # Python anonymous functions (lambdas) are cryptic, hard to debug
77       def f1(m):
78           n = len(subx)   # before append
79           g1 = m.group(1)
80           subx.append(g1)
81           ret = "[#%d]" % n
82   #       print("f1:", g1, ret)
83           return ret
84       x = re.sub(r"[\['](\??\(.*?\))[\]']", f1, x)
85   
86       # added the negative lookbehind -krhodes
87       x = re.sub(r"'?(?<!@)\.'?|\['?", ";", x)
88   
89       x = re.sub(r";;;|;;", ";..;", x)
90   
91       x = re.sub(r";$|'?\]|'$", "", x)
92   
93       # put expressions back
94       def f2(m):
95           g1 = m.group(1)
96   #       print("f2:", g1)
97           return subx[int(g1)]
98   
99       x = re.sub(r"#([0-9]+)", f2, x)
100  
101      return x

103  def jsonpath(obj, expr, result_type='VALUE', debug=0, use_eval=True):
104      """traverse JSON object using jsonpath expr, returning values or paths"""
105  
106      def s(x,y):
107          """concatenate path elements"""
108          return str(x) + ';' + str(y)
109  
110      def isint(x):
111          """check if argument represents a decimal integer"""
112          return x.isdigit()
113  
114      def as_path(path):
115          """convert internal path representation to 
116             "full bracket notation" for PATH output"""
117          p = '$'
118          for piece in path.split(';')[1:]:
119              # make a guess on how to index
120              # XXX need to apply \ quoting on '!!
121              if isint(piece):
122                  p += "[%s]" % piece
123              else:
124                  p += "['%s']" % piece
125          return p
126  
127      def store(path, object):
128          if result_type == 'VALUE':
129              result.append(object)
130          elif result_type == 'IPATH': # Index format path (Python ext)
131              # return list of list of indices -- can be used w/o "eval" or split
132              result.append(path.split(';')[1:])
133          else: # PATH
134              result.append(as_path(path))
135          return path
136  
137      def trace(expr, obj, path):
138          if debug: print("trace", expr, "/", path)
139          if expr:
140              x = expr.split(';')
141              loc = x[0]
142              x = ';'.join(x[1:])
143              if debug: print("\t", loc, type(obj))
144              if loc == "*":
145                  def f03(key, loc, expr, obj, path):
146                      if debug > 1: print("\tf03", key, loc, expr, path)
147                      trace(s(key, expr), obj, path)
148                  walk(loc, x, obj, path, f03)
149              elif loc == "..":
150                  trace(x, obj, path)
151                  def f04(key, loc, expr, obj, path):
152                      if debug > 1: print("\tf04", key, loc, expr, path)
153                      if isinstance(obj, dict):
154                          if key in obj:
155                              trace(s('..', expr), obj[key], s(path, key))
156                      else:
157                          if key < len(obj):
158                              trace(s('..', expr), obj[key], s(path, key))
159                  walk(loc, x, obj, path, f04)
160              elif loc == "!":
161                  # Perl jsonpath extension: return keys
162                  def f06(key, loc, expr, obj, path):
163                      if isinstance(obj, dict):
164                          trace(expr, key, path)
165                  walk(loc, x, obj, path, f06)
166              elif isinstance(obj, dict) and loc in obj:
167                  trace(x, obj[loc], s(path, loc))
168              elif isinstance(obj, list) and isint(loc):
169                  iloc = int(loc)
170                  if debug: print("----->", iloc, len(obj))
171                  if len(obj) > iloc:
172                      trace(x, obj[iloc], s(path, loc))
173              else:
174                  # [(index_expression)]
175                  if loc.startswith("(") and loc.endswith(")"):
176                      if debug > 1: print("index", loc)
177                      e = evalx(loc, obj)
178                      trace(s(e,x), obj, path)
179                      return
180  
181                  # ?(filter_expression)
182                  if loc.startswith("?(") and loc.endswith(")"):
183                      if debug > 1: print("filter", loc)
184                      def f05(key, loc, expr, obj, path):
185                          if debug > 1: print("f05", key, loc, expr, path)
186                          if isinstance(obj, dict):
187                              eval_result = evalx(loc, obj[key])
188                          else:
189                              eval_result = evalx(loc, obj[int(key)])
190                          if eval_result:
191                              trace(s(key, expr), obj, path)
192  
193                      loc = loc[2:-1]
194                      walk(loc, x, obj, path, f05)
195                      return
196  
197                  m = re.match(r'(-?[0-9]*):(-?[0-9]*):?(-?[0-9]*)$', loc)
198                  if m:
199                      if isinstance(obj, (dict, list)):
200                          def max(x,y):
201                              if x > y:
202                                  return x
203                              return y
204  
205                          def min(x,y):
206                              if x < y:
207                                  return x
208                              return y
209  
210                          objlen = len(obj)
211                          s0 = m.group(1)
212                          s1 = m.group(2)
213                          s2 = m.group(3)
214  
215                          # XXX int("badstr") raises exception
216                          start = int(s0) if s0 else 0
217                          end = int(s1) if s1 else objlen
218                          step = int(s2) if s2 else 1
219  
220                          if start < 0:
221                              start = max(0, start+objlen)
222                          else:
223                              start = min(objlen, start)
224                          if end < 0:
225                              end = max(0, end+objlen)
226                          else:
227                              end = min(objlen, end)
228  
229                          for i in xrange(start, end, step):
230                              trace(s(i, x), obj, path)
231                      return
232  
233                  # after (expr) & ?(expr)
234                  if loc.find(",") >= 0:
235                      # [index,index....]
236                      for piece in re.split(r"'?,'?", loc):
237                          if debug > 1: print("piece", piece)
238                          trace(s(piece, x), obj, path)
239          else:
240              store(path, obj)
241  
242      def walk(loc, expr, obj, path, funct):
243          if isinstance(obj, list):
244              for i in xrange(0, len(obj)):
245                  funct(i, loc, expr, obj, path)
246          elif isinstance(obj, dict):
247              for key in obj:
248                  funct(key, loc, expr, obj, path)
249  
250      def evalx(loc, obj):
251          """eval expression"""
252  
253          if debug: print("evalx", loc)
254  
255          # a nod to JavaScript. doesn't work for @.name.name.length
256          # Write len(@.name.name) instead!!!
257          loc = loc.replace("@.length", "len(__obj)")
258  
259          loc = loc.replace("&&", " and ").replace("||", " or ")
260  
261          # replace !@.name with 'name' not in obj
262          # XXX handle !@.name.name.name....
263          def notvar(m):
264              return "'%s' not in __obj" % m.group(1)
265          loc = re.sub("!@\.([a-zA-Z@_0-9-]*)", notvar, loc)
266  
267          # replace @.name.... with __obj['name']....
268          # handle @.name[.name...].length
269          def varmatch(m):
270              def brackets(elts):
271                  ret = "__obj"
272                  for e in elts:
273                      if isint(e):
274                          ret += "[%s]" % e # ain't necessarily so
275                      else:
276                          ret += "['%s']" % e # XXX beware quotes!!!!
277                  return ret
278              g1 = m.group(1)
279              elts = g1.split('.')
280              if elts[-1] == "length":
281                  return "len(%s)" % brackets(elts[1:-1])
282              return brackets(elts[1:])
283  
284          loc = re.sub(r'(?<!\\)(@\.[a-zA-Z@_.0-9]+)', varmatch, loc)
285  
286          # removed = -> == translation
287          # causes problems if a string contains =
288  
289          # replace @  w/ "__obj", but \@ means a literal @
290          loc = re.sub(r'(?<!\\)@', "__obj", loc).replace(r'\@', '@')
291          if not use_eval:
292              if debug: print("eval disabled")
293              raise Exception("eval disabled")
294          if debug: print("eval", loc)
295          try:
296              # eval w/ caller globals, w/ local "__obj"!
297              v = eval(loc, caller_globals, {'__obj': obj})
298          except Exception as e:
299              if debug: print(repr(e))
300              return False
301  
302          if debug: print("->", v)
303          return v
304  
305      # body of jsonpath()
306  
307      # Get caller globals so eval can pick up user functions!!!
308      caller_globals = sys._getframe(1).f_globals
309      result = []
310      if expr and obj:
311          cleaned_expr = normalize(expr)
312          if cleaned_expr.startswith("$;"):
313              cleaned_expr = cleaned_expr[2:]
314  
315          # XXX wrap this in a try??
316          trace(cleaned_expr, obj, '$')
317  
318          if len(result) > 0:
319              return result
320      return False
