32   class Feature(TagAndStatusStatement, Replayable):
33       """A `feature`_ parsed from a *feature file*. 
34    
35       The attributes are: 
36    
37       .. attribute:: keyword 
38    
39          This is the keyword as seen in the *feature file*. In English this will 
40          be "Feature". 
41    
42       .. attribute:: name 
43    
44          The name of the feature (the text after "Feature".) 
45    
46       .. attribute:: description 
47    
48          The description of the feature as seen in the *feature file*. This is 
49          stored as a list of text lines. 
50    
51       .. attribute:: background 
52    
53          The :class:`~behave.model.Background` for this feature, if any. 
54    
55       .. attribute:: scenarios 
56    
57          A list of :class:`~behave.model.Scenario` making up this feature. 
58    
59       .. attribute:: tags 
60    
61          A list of @tags (as :class:`~behave.model.Tag` which are basically 
62          glorified strings) attached to the feature. 
63          See :ref:`controlling things with tags`. 
64    
65       .. attribute:: status 
66    
67          Read-Only. A summary status of the feature's run. If read before the 
68          feature is fully tested it will return "untested" otherwise it will 
69          return one of: 
70    
71          Status.untested 
72            The feature was has not been completely tested yet. 
73          Status.skipped 
74            One or more steps of this feature was passed over during testing. 
75          Status.passed 
76            The feature was tested successfully. 
77          Status.failed 
78            One or more steps of this feature failed. 
79    
80          .. versionchanged:: 1.2.6 
81               Use Status enum class (was: string). 
82    
83       .. attribute:: hook_failed 
84    
85           Indicates if a hook failure occured while running this feature. 
86    
87           .. versionadded:: 1.2.6 
88    
89       .. attribute:: duration 
90    
91          The time, in seconds, that it took to test this feature. If read before 
92          the feature is tested it will return 0.0. 
93    
94       .. attribute:: filename 
95    
96          The file name (or "<string>") of the *feature file* where the feature 
97          was found. 
98    
99       .. attribute:: line 
100   
101         The line number of the *feature file* where the feature was found. 
102   
103      .. attribute:: language 
104   
105         Indicates which spoken language (English, French, German, ..) was used 
106         for parsing the feature file and its keywords. The I18N language code 
107         indicates which language is used. This corresponds to the language tag 
108         at the beginning of the feature file. 
109   
110         .. versionadded:: 1.2.6 
111   
112      .. _`feature`: gherkin.html#features 
113      """
114  
115      type = "feature"
116  
117      def __init__(self, filename, line, keyword, name, tags=None,
118                   description=None, scenarios=None, background=None,
119                   language=None):
120          tags = tags or []
121          super(Feature, self).__init__(filename, line, keyword, name, tags)
122          self.description = description or []
123          self.scenarios = []
124          self.background = background
125          self.language = language
126          self.parser = None
127          self.hook_failed = False
128          if scenarios:
129              for scenario in scenarios:
130                  self.add_scenario(scenario)
131  
132      def reset(self):
133          """Reset to clean state before a test run."""
134          super(Feature, self).reset()
135          self.hook_failed = False
136          for scenario in self.scenarios:
137              scenario.reset()
138  
139      def __repr__(self):
140          return '<Feature "%s": %d scenario(s)>' % \
141              (self.name, len(self.scenarios))
142  
143      def __iter__(self):
144          return iter(self.scenarios)
145  
146      def add_scenario(self, scenario):
147          scenario.feature = self
148          scenario.background = self.background
149          self.scenarios.append(scenario)
150  
151      def compute_status(self):
152          """Compute the status of this feature based on its: 
153            * scenarios 
154            * scenario outlines 
155            * hook failures 
156   
157          :return: Computed status (as string-enum). 
158          """
159          if self.hook_failed:
160              return Status.failed
161  
162          skipped = True
163          passed_count = 0
164          for scenario in self.scenarios:
165              scenario_status = scenario.status
166              if scenario_status == Status.failed:
167                  return Status.failed
168              elif scenario_status == Status.untested:
169                  if passed_count > 0:
170                      return Status.failed  # ABORTED: Some passed, now untested.
171                  return Status.untested
172              if scenario_status != Status.skipped:
173                  skipped = False
174              if scenario_status == Status.passed:
175                  passed_count += 1
176  
177          if skipped:
178              return Status.skipped
179          else:
180              return Status.passed

183      @property
184      def duration(self):
185          # -- NEW: Background is executed N times, now part of scenarios.
186          feature_duration = 0.0
187          for scenario in self.scenarios:
188              feature_duration += scenario.duration
189          return feature_duration

191      def walk_scenarios(self, with_outlines=False):
192          """ 
193          Provides a flat list of all scenarios of this feature. 
194          A ScenarioOutline element adds its scenarios to this list. 
195          But the ScenarioOutline element itself is only added when specified. 
196   
197          A flat scenario list is useful when all scenarios of a features 
198          should be processed. 
199   
200          :param with_outlines: If ScenarioOutline items should be added, too. 
201          :return: List of all scenarios of this feature. 
202          """
203          all_scenarios = []
204          for scenario in self.scenarios:
205              if isinstance(scenario, ScenarioOutline):
206                  scenario_outline = scenario
207                  if with_outlines:
208                      all_scenarios.append(scenario_outline)
209                  all_scenarios.extend(scenario_outline.scenarios)
210              else:
211                  all_scenarios.append(scenario)
212          return all_scenarios
213  
214      def should_run(self, config=None):
215          """ 
216          Determines if this Feature (and its scenarios) should run. 
217          Implements the run decision logic for a feature. 
218          The decision depends on: 
219   
220            * if the Feature is marked as skipped 
221            * if the config.tags (tag expression) enable/disable this feature 
222   
223          :param config:  Runner configuration to use (optional). 
224          :return: True, if scenario should run. False, otherwise. 
225          """
226          answer = not self.should_skip
227          if answer and config:
228              answer = self.should_run_with_tags(config.tags)
229          return answer
230  
231      def should_run_with_tags(self, tag_expression):
232          """Determines if this feature should run when the tag expression is used. 
233          A feature should run if: 
234            * it should run according to its tags 
235            * any of its scenarios should run according to its tags 
236   
237          :param tag_expression:  Runner/config environment tags to use. 
238          :return: True, if feature should run. False, otherwise (skip it). 
239          """
240          run_feature = tag_expression.check(self.tags)
241          if not run_feature:
242              for scenario in self:
243                  if scenario.should_run_with_tags(tag_expression):
244                      run_feature = True
245                      break
246          return run_feature
247  
248      def mark_skipped(self):
249          """Marks this feature (and all its scenarios and steps) as skipped. 
250          Note this function may be called before the feature is executed. 
251          """
252          self.skip(require_not_executed=True)
253          assert self.status == Status.skipped or self.hook_failed
254  
255      def skip(self, reason=None, require_not_executed=False):
256          """Skip executing this feature or the remaining parts of it. 
257          Note that this feature may be already partly executed 
258          when this function is called. 
259   
260          :param reason:  Optional reason why feature should be skipped (as string). 
261          :param require_not_executed: Optional, requires that feature is not 
262                          executed yet (default: false). 
263          """
264          if reason:
265              logger = logging.getLogger("behave")
266              logger.warning(u"SKIP FEATURE %s: %s", self.name, reason)
267  
268          self.clear_status()
269          self.should_skip = True
270          self.skip_reason = reason
271          for scenario in self.scenarios:
272              scenario.skip(reason, require_not_executed)
273          if not self.scenarios:
274              # -- SPECIAL CASE: Feature without scenarios
275              self.set_status(Status.skipped)
276          assert self.status in self.final_status #< skipped, failed or passed.
277  
278      def run(self, runner):
279          # pylint: disable=too-many-branches
280          # MAYBE: self.reset()
281          self.clear_status()
282          self.hook_failed = False
283  
284          runner.context._push(layer_name="feature")      # pylint: disable=protected-access
285          runner.context.feature = self
286          runner.context.tags = set(self.tags)
287  
288          skip_feature_untested = runner.aborted
289          run_feature = self.should_run(runner.config)
290          failed_count = 0
291          hooks_called = False
292          if not runner.config.dry_run and run_feature:
293              hooks_called = True
294              for tag in self.tags:
295                  runner.run_hook("before_tag", runner.context, tag)
296              runner.run_hook("before_feature", runner.context, self)
297              if self.hook_failed:
298                  failed_count += 1
299  
300              # -- RE-EVALUATE SHOULD-RUN STATE:
301              # Hook may call feature.mark_skipped() to exclude it.
302              skip_feature_untested = self.hook_failed or runner.aborted
303              run_feature = self.should_run()
304  
305          # run this feature if the tags say so or any one of its scenarios
306          if run_feature or runner.config.show_skipped:
307              for formatter in runner.formatters:
308                  formatter.feature(self)
309              if self.background:
310                  for formatter in runner.formatters:
311                      formatter.background(self.background)
312  
313          if not skip_feature_untested:
314              for scenario in self.scenarios:
315                  # -- OPTIONAL: Select scenario by name (regular expressions).
316                  if (runner.config.name and
317                          not scenario.should_run_with_name_select(runner.config)):
318                      scenario.mark_skipped()
319                      continue
320  
321                  failed = scenario.run(runner)
322                  if failed:
323                      failed_count += 1
324                      if runner.config.stop or runner.aborted:
325                          # -- FAIL-EARLY: Stop after first failure.
326                          break
327  
328          self.clear_status()  # -- ENFORCE: compute_status() after run.
329          if not self.scenarios and not run_feature:
330              # -- SPECIAL CASE: Feature without scenarios
331              self.set_status(Status.skipped)
332  
333          if hooks_called:
334              runner.run_hook("after_feature", runner.context, self)
335              for tag in self.tags:
336                  runner.run_hook("after_tag", runner.context, tag)
337              if self.hook_failed:
338                  failed_count += 1
339                  self.set_status(Status.failed)
340  
341          # -- PERFORM CONTEXT CLEANUP: May raise cleanup errors.
342          try:
343              runner.context._pop()       # pylint: disable=protected-access
344          except Exception:
345              # -- CLEANUP-ERROR:
346              self.set_status(Status.failed)
347  
348          if run_feature or runner.config.show_skipped:
349              for formatter in runner.formatters:
350                  formatter.eof()
351  
352          failed = (failed_count > 0)
353          return failed

410  class Scenario(TagAndStatusStatement, Replayable):
411      """A `scenario`_ parsed from a *feature file*. 
412   
413      The attributes are: 
414   
415      .. attribute:: keyword 
416   
417         This is the keyword as seen in the *feature file*. In English this will 
418         typically be "Scenario". 
419   
420      .. attribute:: name 
421   
422         The name of the scenario (the text after "Scenario:".) 
423   
424      .. attribute:: description 
425   
426         The description of the scenario as seen in the *feature file*. 
427         This is stored as a list of text lines. 
428   
429      .. attribute:: feature 
430   
431         The :class:`~behave.model.Feature` this scenario belongs to. 
432   
433      .. attribute:: steps 
434   
435         A list of :class:`~behave.model.Step` making up this scenario. 
436   
437      .. attribute:: tags 
438   
439         A list of @tags (as :class:`~behave.model.Tag` which are basically 
440         glorified strings) attached to the scenario. 
441         See :ref:`controlling things with tags`. 
442   
443      .. attribute:: status 
444   
445         Read-Only. A summary status of the scenario's run. If read before the 
446         scenario is fully tested it will return "untested" otherwise it will 
447         return one of: 
448   
449   
450         Status.untested 
451           The scenario was has not been completely tested yet. 
452         Status.skipped 
453           One or more steps of this scenario was passed over during testing. 
454         Status.passed 
455           The scenario was tested successfully. 
456         Status.failed 
457           One or more steps of this scenario failed. 
458   
459         .. versionchanged:: 1.2.6 
460              Use Status enum class (was: string) 
461   
462      .. attribute:: hook_failed 
463   
464          Indicates if a hook failure occured while running this scenario. 
465   
466          .. versionadded:: 1.2.6 
467   
468      .. attribute:: duration 
469   
470         The time, in seconds, that it took to test this scenario. If read before 
471         the scenario is tested it will return 0.0. 
472   
473      .. attribute:: filename 
474   
475         The file name (or "<string>") of the *feature file* where the scenario 
476         was found. 
477   
478      .. attribute:: line 
479   
480         The line number of the *feature file* where the scenario was found. 
481   
482   
483      .. _`scenario`: gherkin.html#scenarios 
484      """
485      # pylint: disable=too-many-instance-attributes
486      type = "scenario"
487      continue_after_failed_step = False
488  
489      def __init__(self, filename, line, keyword, name, tags=None, steps=None,
490                   description=None):
491          tags = tags or []
492          super(Scenario, self).__init__(filename, line, keyword, name, tags)
493          self.description = description or []
494          self.steps = steps or []
495          self.background = None
496          self.feature = None  # REFER-TO: owner=Feature
497          self.hook_failed = False
498          self._background_steps = None
499          self._row = None
500          self.was_dry_run = False
501  
502      def reset(self):
503          """Reset the internal data to reintroduce new-born state just after the 
504          ctor was called. 
505          """
506          super(Scenario, self).reset()
507          self.hook_failed = False
508          self._row = None
509          self.was_dry_run = False
510          for step in self.all_steps:
511              step.reset()
512  
513      @property
514      def background_steps(self):
515          """Provide background steps if feature has a background. 
516          Lazy init that copies the background steps. 
517   
518          Note that a copy of the background steps is needed to ensure 
519          that the background step status is specific to the scenario. 
520   
521          :return:  List of background steps or empty list 
522          """
523          if self._background_steps is None:
524              # -- LAZY-INIT (need copy of background.steps):
525              # Each scenario needs own background.steps.
526              # Otherwise, background step status of the last-run scenario is used.
527              steps = []
528              if self.background:
529                  steps = [copy.copy(step) for step in self.background.steps]
530              self._background_steps = steps
531          return self._background_steps
532  
533      @property
534      def all_steps(self):
535          """Returns iterator to all steps, including background steps if any."""
536          if self.background is not None:
537              return itertools.chain(self.background_steps, self.steps)
538          else:
539              return iter(self.steps)
540  
541      def __repr__(self):
542          return '<Scenario "%s">' % self.name
543  
544      def __iter__(self):
545          return self.all_steps
546  
547      def compute_status(self):
548          """Compute the status of the scenario from its steps 
549          (and hook failures). 
550   
551          :return: Computed status (as enum value). 
552          """
553          if self.hook_failed:
554              return Status.failed
555  
556          for step in self.all_steps:
557              if step.status == Status.undefined:
558                  if self.was_dry_run:
559                      # -- SPECIAL CASE: In dry-run with undefined-step discovery
560                      #    Undefined steps should not cause failed scenario.
561                      return Status.untested
562                  else:
563                      # -- NORMALLY: Undefined steps cause failed scenario.
564                      return Status.failed
565              elif step.status != Status.passed:
566                  # pylint: disable=line-too-long
567                  assert step.status in (Status.failed, Status.skipped, Status.untested)
568                  return step.status
569          return Status.passed
570  
571      @property
572      def duration(self):
573          # -- ORIG: for step in self.steps:  Background steps were excluded.
574          scenario_duration = 0
575          for step in self.all_steps:
576              scenario_duration += step.duration
577          return scenario_duration
578  
579      @property
580      def effective_tags(self):
581          """ 
582          Effective tags for this scenario: 
583            * own tags 
584            * tags inherited from its feature 
585          """
586          tags = self.tags
587          if self.feature:
588              tags = self.feature.tags + self.tags
589          return tags

591      def should_run(self, config=None):
592          """ 
593          Determines if this Scenario (or ScenarioOutline) should run. 
594          Implements the run decision logic for a scenario. 
595          The decision depends on: 
596   
597            * if the Scenario is marked as skipped 
598            * if the config.tags (tag expression) enable/disable this scenario 
599            * if the scenario is selected by name 
600   
601          :param config:  Runner configuration to use (optional). 
602          :return: True, if scenario should run. False, otherwise. 
603          """
604          answer = not self.should_skip
605          if answer and config:
606              answer = (self.should_run_with_tags(config.tags) and
607                        self.should_run_with_name_select(config))
608          return answer
609  
610      def should_run_with_tags(self, tag_expression):
611          """ 
612          Determines if this scenario should run when the tag expression is used. 
613   
614          :param tag_expression:  Runner/config environment tags to use. 
615          :return: True, if scenario should run. False, otherwise (skip it). 
616          """
617          return tag_expression.check(self.effective_tags)
618  
619      def should_run_with_name_select(self, config):
620          """Determines if this scenario should run when it is selected by name. 
621   
622          :param config:  Runner/config environment name regexp (if any). 
623          :return: True, if scenario should run. False, otherwise (skip it). 
624          """
625          # -- SELECT-ANY: If select by name is not specified (not config.name).
626          return not config.name or config.name_re.search(self.name)
627  
628      def mark_skipped(self):
629          """Marks this scenario (and all its steps) as skipped. 
630          Note that this method can be called before the scenario is executed. 
631          """
632          self.skip(require_not_executed=True)
633          assert self.status == Status.skipped or self.hook_failed, \
634                 "OOPS: scenario.status=%s" % self.status.name
635  
636      def skip(self, reason=None, require_not_executed=False):
637          """Skip from executing this scenario or the remaining parts of it. 
638          Note that the scenario may be already partly executed 
639          when this method is called. 
640   
641          :param reason:  Optional reason why it should be skipped (as string). 
642          """
643          if reason:
644              scenario_type = self.__class__.__name__
645              logger = logging.getLogger("behave")
646              logger.warning(u"SKIP %s %s: %s", scenario_type, self.name, reason)
647  
648          self.clear_status()
649          self.should_skip = True
650          self.skip_reason = reason
651          for step in self.all_steps:
652              not_executed = step.status in (Status.untested, Status.skipped)
653              if not_executed:
654                  step.status = Status.skipped
655              else:
656                  assert not require_not_executed, \
657                      "REQUIRE NOT-EXECUTED, but step is %s" % step.status
658  
659          scenario_without_steps = not self.steps and not self.background_steps
660          if scenario_without_steps:
661              self.set_status(Status.skipped)
662          assert self.status in self.final_status #< skipped, failed or passed
663  
664      def run(self, runner):
665          # pylint: disable=too-many-branches, too-many-statements
666          self.clear_status()
667          self.captured.reset()
668          self.hook_failed = False
669          failed = False
670          skip_scenario_untested = runner.aborted
671          run_scenario = self.should_run(runner.config)
672          run_steps = run_scenario and not runner.config.dry_run
673          dry_run_scenario = run_scenario and runner.config.dry_run
674          self.was_dry_run = dry_run_scenario
675  
676          runner.context._push(layer_name="scenario")      # pylint: disable=protected-access
677          runner.context.scenario = self
678          runner.context.tags = set(self.effective_tags)
679  
680          hooks_called = False
681          if not runner.config.dry_run and run_scenario:
682              hooks_called = True
683              for tag in self.tags:
684                  runner.run_hook("before_tag", runner.context, tag)
685              runner.run_hook("before_scenario", runner.context, self)
686              if self.hook_failed:
687                  # -- SKIP: Scenario steps and behave like dry_run_scenario
688                  failed = True
689  
690              # -- RE-EVALUATE SHOULD-RUN STATE:
691              # Hook may call scenario.mark_skipped() to exclude it.
692              skip_scenario_untested = self.hook_failed or runner.aborted
693              run_scenario = self.should_run()
694              run_steps = run_scenario and not runner.config.dry_run
695  
696          if run_scenario or runner.config.show_skipped:
697              for formatter in runner.formatters:
698                  formatter.scenario(self)
699  
700          # TODO: Reevaluate location => Move in front of hook-calls
701          runner.setup_capture()
702  
703          if run_scenario or runner.config.show_skipped:
704              for step in self:
705                  for formatter in runner.formatters:
706                      formatter.step(step)
707  
708          if not skip_scenario_untested:
709              for step in self.all_steps:
710                  if run_steps:
711                      if not step.run(runner):
712                          # -- CASE: Failed or undefined step
713                          #    Optionally continue_after_failed_step if enabled.
714                          #    But disable run_steps after undefined-step.
715                          run_steps = (self.continue_after_failed_step and
716                                       step.status == Status.failed)
717                          failed = True
718                          # pylint: disable=protected-access
719                          runner.context._set_root_attribute("failed", True)
720                          self.set_status(Status.failed)
721                      elif self.should_skip:
722                          # -- CASE: Step skipped remaining scenario.
723                          # assert self.status == Status.skipped
724                          run_steps = False
725                  elif failed or dry_run_scenario:
726                      # -- SKIP STEPS: After failure/undefined-step occurred.
727                      # BUT: Detect all remaining undefined steps.
728                      step.status = Status.skipped
729                      if dry_run_scenario:
730                          # pylint: disable=redefined-variable-type
731                          step.status = Status.untested
732                      found_step_match = runner.step_registry.find_match(step)
733                      if not found_step_match:
734                          step.status = Status.undefined
735                          runner.undefined_steps.append(step)
736                      elif dry_run_scenario:
737                          # -- BETTER DIAGNOSTICS: Provide step file location
738                          # (when --format=pretty is used).
739                          assert step.status == Status.untested
740                          for formatter in runner.formatters:
741                              # -- EMULATE: Step.run() protocol w/o step execution.
742                              formatter.match(found_step_match)
743                              formatter.result(step)
744                  else:
745                      # -- SKIP STEPS: For disabled scenario.
746                      # CASES:
747                      #   * Undefined steps are not detected (by intention).
748                      #   * Step skipped remaining scenario.
749                      step.status = Status.skipped
750  
751          self.clear_status()  # -- ENFORCE: compute_status() after run.
752          if not run_scenario and not self.steps:
753              # -- SPECIAL CASE: Scenario without steps.
754              self.set_status(Status.skipped)
755  
756  
757          if hooks_called:
758              runner.run_hook("after_scenario", runner.context, self)
759              for tag in self.tags:
760                  runner.run_hook("after_tag", runner.context, tag)
761              if self.hook_failed:
762                  self.set_status(Status.failed)
763                  failed = True
764  
765          # -- PERFORM CONTEXT-CLEANUP: May raise cleanup errors.
766          try:
767              runner.context._pop()       # pylint: disable=protected-access
768          except Exception:
769              self.set_status(Status.failed)
770              failed = True
771  
772          # -- CAPTURED-OUTPUT:
773          store_captured = (runner.config.junit or self.status == Status.failed)
774          if store_captured:
775              self.captured = runner.capture_controller.captured
776  
777          runner.teardown_capture()
778          return failed

781  class ScenarioOutlineBuilder(object):
782      """Helper class to use a ScenarioOutline as a template and 
783      build its scenarios (as template instances). 
784      """
785  
786      def __init__(self, annotation_schema):
787          self.annotation_schema = annotation_schema
788  
789      @staticmethod
790      def render_template(text, row=None, params=None):
791          """Render a text template with placeholders, ala "Hello <name>". 
792   
793          :param row:     As placeholder provider (dict-like). 
794          :param params:  As additional placeholder provider (as dict). 
795          :return: Rendered text, known placeholders are substituted w/ values. 
796          """
797          if not ("<" in text and ">" in text):
798              return text
799  
800          safe_values = False
801          for placeholders in (row, params):
802              if not placeholders:
803                  continue
804              for name, value in placeholders.items():
805                  if safe_values and ("<" in value and ">" in value):
806                      continue    # -- OOPS, value looks like placeholder.
807                  text = text.replace("<%s>" % name, value)
808          return text
809  
810      def make_scenario_name(self, outline_name, example, row, params=None):
811          """Build a scenario name for an example row of this scenario outline. 
812          Placeholders for row data are replaced by values. 
813   
814          SCHEMA: "{outline_name} -*- {examples.name}@{row.id}" 
815   
816          :param outline_name:    ScenarioOutline's name (as template). 
817          :param example:         Examples object. 
818          :param row:             Row of this example. 
819          :param params:          Additional placeholders for example/row. 
820          :return: Computed name for the scenario representing example/row. 
821          """
822          if params is None:
823              params = {}
824          params["examples.name"] = example.name or ""
825          params.setdefault("examples.index", example.index)
826          params.setdefault("row.index", row.index)
827          params.setdefault("row.id", row.id)
828  
829          # -- STEP: Replace placeholders in scenario/example name (if any).
830          examples_name = self.render_template(example.name, row, params)
831          params["examples.name"] = examples_name
832          scenario_name = self.render_template(outline_name, row, params)
833  
834          class Data(object):
835              def __init__(self, name, index):
836                  self.name = name
837                  self.index = index
838                  self.id = name      # pylint: disable=invalid-name
839  
840          example_data = Data(examples_name, example.index)
841          row_data = Data(row.id, row.index)
842          return self.annotation_schema.format(name=scenario_name,
843                                               examples=example_data, row=row_data)
844  
845      @classmethod
846      def make_row_tags(cls, outline_tags, row, params=None):
847          if not outline_tags:
848              return []
849  
850          tags = []
851          for tag in outline_tags:
852              if "<" in tag and ">" in tag:
853                  tag = cls.render_template(tag, row, params)
854              if "<" in tag or ">" in tag:
855                  # -- OOPS: Unknown placeholder, drop tag.
856                  continue
857              new_tag = Tag.make_name(tag, unescape=True)
858              tags.append(new_tag)
859          return tags
860  
861      @classmethod
862      def make_step_for_row(cls, outline_step, row, params=None):
863          # -- BASED-ON: new_step = outline_step.set_values(row)
864          new_step = copy.deepcopy(outline_step)
865          new_step.name = cls.render_template(new_step.name, row, params)
866          if new_step.text:
867              new_step.text = cls.render_template(new_step.text, row)
868          if new_step.table:
869              for name, value in row.items():
870                  for row in new_step.table:
871                      for i, cell in enumerate(row.cells):
872                          row.cells[i] = cell.replace("<%s>" % name, value)
873          return new_step
874  
875      def build_scenarios(self, scenario_outline):
876          """Build scenarios for a ScenarioOutline from its examples."""
877          # -- BUILD SCENARIOS (once): For this ScenarioOutline from examples.
878          params = {
879              "examples.name": None,
880              "examples.index": None,
881              "row.index": None,
882              "row.id": None,
883          }
884          scenarios = []
885          for example_index, example in enumerate(scenario_outline.examples):
886              example.index = example_index+1
887              params["examples.name"] = example.name
888              params["examples.index"] = _text(example.index)
889              for row_index, row in enumerate(example.table):
890                  row.index = row_index+1
891                  row.id = "%d.%d" % (example.index, row.index)
892                  params["row.id"] = row.id
893                  params["row.index"] = _text(row.index)
894                  scenario_name = self.make_scenario_name(scenario_outline.name,
895                                                          example, row, params)
896                  row_tags = self.make_row_tags(scenario_outline.tags, row, params)
897                  row_tags.extend(example.tags)
898                  new_steps = []
899                  for outline_step in scenario_outline.steps:
900                      new_step = self.make_step_for_row(outline_step, row, params)
901                      new_steps.append(new_step)
902  
903                  # -- STEP: Make Scenario name for this row.
904                  # scenario_line = example.line + 2 + row_index
905                  scenario_line = row.line
906                  scenario = Scenario(scenario_outline.filename, scenario_line,
907                                      scenario_outline.keyword,
908                                      scenario_name, row_tags, new_steps)
909                  scenario.feature = scenario_outline.feature
910                  scenario.background = scenario_outline.background
911                  scenario._row = row     # pylint: disable=protected-access
912                  scenarios.append(scenario)
913          return scenarios

1286     def run(self, runner, quiet=False, capture=True):
1287         # pylint: disable=too-many-branches, too-many-statements
1288         # -- RESET: Run-time information.
1289         # self.status = Status.untested
1290         # self.hook_failed = False
1291         self.reset()
1292 
1293         match = runner.step_registry.find_match(self)
1294         if match is None:
1295             runner.undefined_steps.append(self)
1296             if not quiet:
1297                 for formatter in runner.formatters:
1298                     formatter.match(NoMatch())
1299 
1300             self.status = Status.undefined
1301             if not quiet:
1302                 for formatter in runner.formatters:
1303                     formatter.result(self)
1304             return False
1305 
1306         keep_going = True
1307         error = u""
1308 
1309         if not quiet:
1310             for formatter in runner.formatters:
1311                 formatter.match(match)
1312 
1313         if capture:
1314             runner.start_capture()
1315 
1316         skip_step_untested = False
1317         runner.run_hook("before_step", runner.context, self)
1318         if self.hook_failed:
1319             skip_step_untested = True
1320 
1321         start = time.time()
1322         if not skip_step_untested:
1323             try:
1324                 # -- ENSURE:
1325                 #  * runner.context.text/.table attributes are reset (#66).
1326                 #  * Even EMPTY multiline text is available in context.
1327                 runner.context.text = self.text
1328                 runner.context.table = self.table
1329                 match.run(runner.context)
1330                 if self.status == Status.untested:
1331                     # -- NOTE: Executed step may have skipped scenario and itself.
1332                     # pylint: disable=redefined-variable-type
1333                     self.status = Status.passed
1334             except KeyboardInterrupt as e:
1335                 runner.aborted = True
1336                 error = u"ABORTED: By user (KeyboardInterrupt)."
1337                 self.status = Status.failed
1338                 self.store_exception_context(e)
1339             except AssertionError as e:
1340                 self.status = Status.failed
1341                 self.store_exception_context(e)
1342                 if e.args:
1343                     message = _text(e)
1344                     error = u"Assertion Failed: "+ message
1345                 else:
1346                     # no assertion text; format the exception
1347                     error = _text(traceback.format_exc())
1348             except Exception as e:      # pylint: disable=broad-except
1349                 self.status = Status.failed
1350                 error = _text(traceback.format_exc())
1351                 self.store_exception_context(e)
1352

1354         runner.run_hook("after_step", runner.context, self)
1355         if self.hook_failed:
1356             self.status = Status.failed
1357 
1358         if capture:
1359             runner.stop_capture()
1360

1362         store_captured_always = False   # PREPARED
1363         store_captured = self.status == Status.failed or store_captured_always
1364         if self.status == Status.failed:
1365             assert isinstance(error, six.text_type)
1366             if capture:
1367                 # -- CAPTURE-ONLY: Non-nested step failures.
1368                 self.captured = runner.capture_controller.captured
1369                 error2 = self.captured.make_report()
1370                 if error2:
1371                     error += "\n" + error2
1372             self.error_message = error
1373             keep_going = False
1374         elif store_captured and capture:
1375             self.captured = runner.capture_controller.captured
1376 
1377         if not quiet:
1378             for formatter in runner.formatters:
1379                 formatter.result(self)

1384 class Table(Replayable):
1385     """A `table`_ extracted from a *feature file*. 
1386  
1387     Table instance data is accessible using a number of methods: 
1388  
1389     **iteration** 
1390       Iterating over the Table will yield the :class:`~behave.model.Row` 
1391       instances from the .rows attribute. 
1392  
1393     **indexed access** 
1394       Individual rows may be accessed directly by index on the Table instance; 
1395       table[0] gives the first non-heading row and table[-1] gives the last 
1396       row. 
1397  
1398     The attributes are: 
1399  
1400     .. attribute:: headings 
1401  
1402        The headings of the table as a list of strings. 
1403  
1404     .. attribute:: rows 
1405  
1406        An list of instances of :class:`~behave.model.Row` that make up the body 
1407        of the table in the *feature file*. 
1408  
1409     Tables are also comparable, for what that's worth. Headings and row data 
1410     are compared. 
1411  
1412     .. _`table`: gherkin.html#table 
1413     """
1414     type = "table"
1415 
1416     def __init__(self, headings, line=None, rows=None):
1417         Replayable.__init__(self)
1418         self.headings = headings
1419         self.line = line
1420         self.rows = []
1421         if rows:
1422             for row in rows:
1423                 self.add_row(row, line)
1424 
1425     def add_row(self, row, line=None):
1426         self.rows.append(Row(self.headings, row, line))
1427 
1428     def add_column(self, column_name, values=None, default_value=u""):
1429         """Adds a new column to this table. 
1430         Uses :param:`default_value` for new cells (if :param:`values` are 
1431         not provided). param:`values` are extended with :param:`default_value` 
1432         if values list is smaller than the number of table rows. 
1433  
1434         :param column_name: Name of new column (as string). 
1435         :param values: Optional list of cell values in new column. 
1436         :param default_value: Default value for cell (if values not provided). 
1437         :returns: Index of new column (as number). 
1438         """
1439         # assert isinstance(column_name, unicode)
1440         assert not self.has_column(column_name)
1441         if values is None:
1442             values = [default_value] * len(self.rows)
1443         elif not isinstance(values, list):
1444             values = list(values)
1445         if len(values) < len(self.rows):
1446             more_size = len(self.rows) - len(values)
1447             more_values = [default_value] * more_size
1448             values.extend(more_values)
1449 
1450         new_column_index = len(self.headings)
1451         self.headings.append(column_name)
1452         for row, value in zip(self.rows, values):
1453             assert len(row.cells) == new_column_index
1454             row.cells.append(value)
1455         return new_column_index
1456 
1457     def remove_column(self, column_name):
1458         if not isinstance(column_name, int):
1459             try:
1460                 column_index = self.get_column_index(column_name)
1461             except ValueError:
1462                 raise KeyError("column=%s is unknown" % column_name)
1463 
1464         assert isinstance(column_index, int)
1465         assert column_index < len(self.headings)
1466         del self.headings[column_index]
1467         for row in self.rows:
1468             assert column_index < len(row.cells)
1469             del row.cells[column_index]
1470 
1471     def remove_columns(self, column_names):
1472         for column_name in column_names:
1473             self.remove_column(column_name)
1474 
1475     def has_column(self, column_name):
1476         return column_name in self.headings
1477 
1478     def get_column_index(self, column_name):
1479         return self.headings.index(column_name)
1480 
1481     def require_column(self, column_name):
1482         """Require that a column exists in the table. 
1483         Raise an AssertionError if the column does not exist. 
1484  
1485         :param column_name: Name of new column (as string). 
1486         :return: Index of column (as number) if it exists. 
1487         """
1488         if not self.has_column(column_name):
1489             columns = ", ".join(self.headings)
1490             msg = "REQUIRE COLUMN: %s (columns: %s)" % (column_name, columns)
1491             raise AssertionError(msg)
1492         return self.get_column_index(column_name)
1493 
1494     def require_columns(self, column_names):
1495         for column_name in column_names:
1496             self.require_column(column_name)
1497 
1498     def ensure_column_exists(self, column_name):
1499         """Ensures that a column with the given name exists. 
1500         If the column does not exist, the column is added. 
1501  
1502         :param column_name: Name of column (as string). 
1503         :return: Index of column (as number). 
1504         """
1505         if self.has_column(column_name):
1506             return self.get_column_index(column_name)
1507         else:
1508             return self.add_column(column_name)
1509 
1510     def __repr__(self):
1511         return "<Table: %dx%d>" % (len(self.headings), len(self.rows))
1512 
1513     def __eq__(self, other):
1514         if isinstance(other, Table):
1515             if self.headings != other.headings:
1516                 return False
1517             for my_row, their_row in zip(self.rows, other.rows):
1518                 if my_row != their_row:
1519                     return False
1520         else:
1521             # -- ASSUME: table <=> raw data comparison
1522             other_rows = other
1523             for my_row, their_row in zip(self.rows, other_rows):
1524                 if my_row != their_row:
1525                     return False
1526         return True
1527 
1528     def __ne__(self, other):
1529         return not self.__eq__(other)
1530 
1531     def __iter__(self):
1532         return iter(self.rows)
1533 
1534     def __getitem__(self, index):
1535         return self.rows[index]
1536 
1537     def assert_equals(self, data):
1538         """Assert that this table's cells are the same as the supplied "data". 
1539  
1540         The data passed in must be a list of lists giving: 
1541  
1542             [ 
1543                 [row 1], 
1544                 [row 2], 
1545                 [row 3], 
1546             ] 
1547  
1548         If the cells do not match then a useful AssertionError will be raised. 
1549         """
1550         assert self == data
1551         raise NotImplementedError

1554 class Row(object):
1555     """One row of a `table`_ parsed from a *feature file*. 
1556  
1557     Row data is accessible using a number of methods: 
1558  
1559     **iteration** 
1560       Iterating over the Row will yield the individual cells as strings. 
1561  
1562     **named access** 
1563       Individual cells may be accessed by heading name; row["name"] would give 
1564       the cell value for the column with heading "name". 
1565  
1566     **indexed access** 
1567       Individual cells may be accessed directly by index on the Row instance; 
1568       row[0] gives the first cell and row[-1] gives the last cell. 
1569  
1570     The attributes are: 
1571  
1572     .. attribute:: cells 
1573  
1574        The list of strings that form the cells of this row. 
1575  
1576     .. attribute:: headings 
1577  
1578        The headings of the table as a list of strings. 
1579  
1580     Rows are also comparable, for what that's worth. Only the cells are 
1581     compared. 
1582  
1583     .. _`table`: gherkin.html#table 
1584     """
1585     def __init__(self, headings, cells, line=None, comments=None):
1586         self.headings = headings
1587         self.comments = comments
1588         for c in cells:
1589             assert isinstance(c, six.text_type)
1590         self.cells = cells
1591         self.line = line
1592 
1593     def __getitem__(self, name):
1594         try:
1595             index = self.headings.index(name)
1596         except ValueError:
1597             if isinstance(name, int):
1598                 index = name
1599             else:
1600                 raise KeyError('"%s" is not a row heading' % name)
1601         return self.cells[index]
1602 
1603     def __repr__(self):
1604         return "<Row %r>" % (self.cells,)
1605 
1606     def __eq__(self, other):
1607         return self.cells == other.cells
1608 
1609     def __ne__(self, other):
1610         return not self.__eq__(other)
1611 
1612     def __len__(self):
1613         return len(self.cells)
1614 
1615     def __iter__(self):
1616         return iter(self.cells)
1617 
1618     def items(self):
1619         return zip(self.headings, self.cells)
1620 
1621     def get(self, key, default=None):
1622         try:
1623             return self[key]
1624         except KeyError:
1625             return default
1626 
1627     def as_dict(self):
1628         """Converts the row and its cell data into a dictionary. 
1629         :return: Row data as dictionary (without comments, line info). 
1630         """
1631         from behave.compat.collections import OrderedDict
1632         return OrderedDict(self.items())

1635 class Tag(six.text_type):
1636     """Tags appear may be associated with Features or Scenarios. 
1637  
1638     They're a subclass of regular strings (unicode pre-Python 3) with an 
1639     additional ``line`` number attribute (where the tag was seen in the source 
1640     feature file. 
1641  
1642     See :ref:`controlling things with tags`. 
1643     """
1644     allowed_chars = u"._-=:"    # In addition to aplha-numerical chars.
1645     quoting_chars = ("'", '"', "<", ">")
1646 
1647     def __new__(cls, name, line):
1648         o = six.text_type.__new__(cls, name)
1649         o.line = line
1650         return o
1651 
1652     @classmethod
1653     def make_name(cls, text, unescape=False, allowed_chars=None):
1654         """Translate text into a "valid tag" without whitespace, etc. 
1655         Translation rules are: 
1656           * alnum chars => same, kept 
1657           * space chars => "_" 
1658           * other chars => deleted 
1659  
1660         Preserve following characters (in addition to alnums, like: A-z, 0-9): 
1661           * dots        => "." (support: dotted-names, active-tag name schema) 
1662           * minus       => "-" (support: dashed-names) 
1663           * underscore  => "_" 
1664           * equal       => "=" (support: active-tag name schema) 
1665           * colon       => ":" (support: active-tag name schema or similar) 
1666  
1667         :param text: Unicode text as input for name. 
1668         :param unescape: Optional flag to unescape some chars (default: false) 
1669         :param allowed_chars: Optional string with additional preserved chars. 
1670         :return: Unicode name that can be used as tag. 
1671         """
1672         assert isinstance(text, six.text_type)
1673         if allowed_chars is None:
1674             allowed_chars = cls.allowed_chars
1675 
1676         if unescape:
1677             # -- UNESCAPE: Some escaped sequences
1678             text = text.replace("\\t", "\t").replace("\\n", "\n")
1679         chars = []
1680         for char in text:
1681             if char.isalnum() or (allowed_chars and char in allowed_chars):
1682                 chars.append(char)
1683             elif char.isspace():
1684                 chars.append(u"_")
1685             elif char in cls.quoting_chars:
1686                 pass    # -- NORMALIZE: Remove any quoting chars.
1687             # -- MAYBE:
1688             # else:
1689             #     # -- OTHERWISE: Accept gracefully any other character.
1690             #     chars.append(char)
1691         return u"".join(chars)

1694 class Text(six.text_type):
1695     """Store multiline text from a Step definition. 
1696  
1697     The attributes are: 
1698  
1699     .. attribute:: value 
1700  
1701        The actual text parsed from the *feature file*. 
1702  
1703     .. attribute:: content_type 
1704  
1705        Currently only "text/plain". 
1706     """
1707     def __new__(cls, value, content_type=u"text/plain", line=0):
1708         assert isinstance(value, six.text_type)
1709         assert isinstance(content_type, six.text_type)
1710         o = six.text_type.__new__(cls, value)
1711         o.content_type = content_type
1712         o.line = line
1713         return o
1714 
1715     def line_range(self):
1716         line_count = len(self.splitlines())
1717         return (self.line, self.line + line_count + 1)
1718 
1719     def replace(self, old, new, count=-1):
1720         return Text(super(Text, self).replace(old, new, count), self.content_type,
1721                     self.line)
1722 
1723     def assert_equals(self, expected):
1724         """Assert that my text is identical to the "expected" text. 
1725  
1726         A nice context diff will be displayed if they do not match. 
1727         """
1728         if self == expected:
1729             return True
1730         diff = []
1731         for line in difflib.unified_diff(self.splitlines(),
1732                                          expected.splitlines()):
1733             diff.append(line)
1734         # strip unnecessary diff prefix
1735         diff = ["Text does not match:"] + diff[3:]
1736         raise AssertionError("\n".join(diff))
