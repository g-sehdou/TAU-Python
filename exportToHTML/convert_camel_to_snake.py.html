1    import re
2    from .remove_non_alphanumeric import remove_non_alphanumeric as remove_non_alpha
3    
4    _first_cap_re = re.compile('(.)([A-Z][a-z]+)')
5    _all_cap_re = re.compile('([a-z0-9])([A-Z])')
6    
7    
8    def convert_camel_to_snake(string, remove_non_alphanumeric=True):
9    	""" 
10       converts CamelCase to snake_case 
11       :type string: str 
12       :rtype: str 
13       """
14   	if remove_non_alphanumeric:
15   		string = remove_non_alpha(string, replace_with='_', keep_underscore=True)
16   
17   	s1 = _first_cap_re.sub(r'\1_\2', string)
18   	result = _all_cap_re.sub(r'\1_\2', s1).lower()
19   	result = re.sub(pattern='\s*_+', repl="_", string=result)
20   	return result
