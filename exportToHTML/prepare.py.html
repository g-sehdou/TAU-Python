49   def _get_prepared_distribution(
50       req: InstallRequirement,
51       req_tracker: RequirementTracker,
52       finder: PackageFinder,
53       build_isolation: bool,
54   ) -> BaseDistribution:
55       """Prepare a distribution for installation."""
56       abstract_dist = make_distribution_for_install_requirement(req)
57       with req_tracker.track(req):
58           abstract_dist.prepare_distribution_metadata(finder, build_isolation)
59       return abstract_dist.get_metadata_distribution()

62   def unpack_vcs_link(link: Link, location: str) -> None:
63       vcs_backend = vcs.get_backend_for_scheme(link.scheme)
64       assert vcs_backend is not None
65       vcs_backend.unpack(location, url=hide_url(link.url))

68   class File:
69       def __init__(self, path: str, content_type: Optional[str]) -> None:
70           self.path = path
71           if content_type is None:
72               self.content_type = mimetypes.guess_type(path)[0]
73           else:
74               self.content_type = content_type

77   def get_http_url(
78       link: Link,
79       download: Downloader,
80       download_dir: Optional[str] = None,
81       hashes: Optional[Hashes] = None,
82   ) -> File:
83       temp_dir = TempDirectory(kind="unpack", globally_managed=True)
84       # If a download dir is specified, is the file already downloaded there?
85       already_downloaded_path = None
86       if download_dir:
87           already_downloaded_path = _check_download_dir(link, download_dir, hashes)
88   
89       if already_downloaded_path:
90           from_path = already_downloaded_path
91           content_type = None
92       else:
93           # let's download to a tmp dir
94           from_path, content_type = download(link, temp_dir.path)
95           if hashes:
96               hashes.check_against_path(from_path)
97   
98       return File(from_path, content_type)

101  def _copy2_ignoring_special_files(src: str, dest: str) -> None:
102      """Copying special files is not supported, but as a convenience to users 
103      we skip errors copying them. This supports tools that may create e.g. 
104      socket files in the project source directory. 
105      """
106      try:
107          copy2_fixed(src, dest)
108      except shutil.SpecialFileError as e:
109          # SpecialFileError may be raised due to either the source or
110          # destination. If the destination was the cause then we would actually
111          # care, but since the destination directory is deleted prior to
112          # copy we ignore all of them assuming it is caused by the source.
113          logger.warning(
114              "Ignoring special file error '%s' encountered copying %s to %s.",
115              str(e),
116              src,
117              dest,
118          )

121  def _copy_source_tree(source: str, target: str) -> None:
122      target_abspath = os.path.abspath(target)
123      target_basename = os.path.basename(target_abspath)
124      target_dirname = os.path.dirname(target_abspath)
125  
126      def ignore(d: str, names: List[str]) -> List[str]:
127          skipped: List[str] = []
128          if d == source:
129              # Pulling in those directories can potentially be very slow,
130              # exclude the following directories if they appear in the top
131              # level dir (and only it).
132              # See discussion at https://github.com/pypa/pip/pull/6770
133              skipped += [".tox", ".nox"]
134          if os.path.abspath(d) == target_dirname:
135              # Prevent an infinite recursion if the target is in source.
136              # This can happen when TMPDIR is set to ${PWD}/...
137              # and we copy PWD to TMPDIR.
138              skipped += [target_basename]
139          return skipped
140  
141      shutil.copytree(
142          source,
143          target,
144          ignore=ignore,
145          symlinks=True,
146          copy_function=_copy2_ignoring_special_files,
147      )

150  def get_file_url(
151      link: Link, download_dir: Optional[str] = None, hashes: Optional[Hashes] = None
152  ) -> File:
153      """Get file and optionally check its hash."""
154      # If a download dir is specified, is the file already there and valid?
155      already_downloaded_path = None
156      if download_dir:
157          already_downloaded_path = _check_download_dir(link, download_dir, hashes)
158  
159      if already_downloaded_path:
160          from_path = already_downloaded_path
161      else:
162          from_path = link.file_path
163  
164      # If --require-hashes is off, `hashes` is either empty, the
165      # link's embedded hash, or MissingHashes; it is required to
166      # match. If --require-hashes is on, we are satisfied by any
167      # hash in `hashes` matching: a URL-based or an option-based
168      # one; no internet-sourced hash will be in `hashes`.
169      if hashes:
170          hashes.check_against_path(from_path)
171      return File(from_path, None)

174  def unpack_url(
175      link: Link,
176      location: str,
177      download: Downloader,
178      download_dir: Optional[str] = None,
179      hashes: Optional[Hashes] = None,
180  ) -> Optional[File]:
181      """Unpack link into location, downloading if required. 
182   
183      :param hashes: A Hashes object, one of whose embedded hashes must match, 
184          or HashMismatch will be raised. If the Hashes is empty, no matches are 
185          required, and unhashable types of requirements (like VCS ones, which 
186          would ordinarily raise HashUnsupported) are allowed. 
187      """
188      # non-editable vcs urls
189      if link.is_vcs:
190          unpack_vcs_link(link, location)
191          return None
192  
193      # Once out-of-tree-builds are no longer supported, could potentially
194      # replace the below condition with `assert not link.is_existing_dir`
195      # - unpack_url does not need to be called for in-tree-builds.
196      #
197      # As further cleanup, _copy_source_tree and accompanying tests can
198      # be removed.
199      #
200      # TODO when use-deprecated=out-of-tree-build is removed
201      if link.is_existing_dir():
202          if os.path.isdir(location):
203              rmtree(location)
204          _copy_source_tree(link.file_path, location)
205          return None
206  
207      # file urls
208      if link.is_file:
209          file = get_file_url(link, download_dir, hashes=hashes)
210  
211      # http urls
212      else:
213          file = get_http_url(
214              link,
215              download,
216              download_dir,
217              hashes=hashes,
218          )
219  
220      # unpack the archive to the build dir location. even when only downloading
221      # archives, they have to be unpacked to parse dependencies, except wheels
222      if not link.is_wheel:
223          unpack_file(file.path, location, file.content_type)
224  
225      return file

228  def _check_download_dir(
229      link: Link, download_dir: str, hashes: Optional[Hashes]
230  ) -> Optional[str]:
231      """Check download_dir for previously downloaded file with correct hash 
232      If a correct file is found return its path else None 
233      """
234      download_path = os.path.join(download_dir, link.filename)
235  
236      if not os.path.exists(download_path):
237          return None
238  
239      # If already downloaded, does its hash match?
240      logger.info("File was already downloaded %s", download_path)
241      if hashes:
242          try:
243              hashes.check_against_path(download_path)
244          except HashMismatch:
245              logger.warning(
246                  "Previously-downloaded file %s has bad hash. Re-downloading.",
247                  download_path,
248              )
249              os.unlink(download_path)
250              return None
251      return download_path

257      def __init__(
258          self,
259          build_dir: str,
260          download_dir: Optional[str],
261          src_dir: str,
262          build_isolation: bool,
263          req_tracker: RequirementTracker,
264          session: PipSession,
265          progress_bar: str,
266          finder: PackageFinder,
267          require_hashes: bool,
268          use_user_site: bool,
269          lazy_wheel: bool,
270          in_tree_build: bool,
271      ) -> None:
272          super().__init__()
273  
274          self.src_dir = src_dir
275          self.build_dir = build_dir
276          self.req_tracker = req_tracker
277          self._session = session
278          self._download = Downloader(session, progress_bar)
279          self._batch_download = BatchDownloader(session, progress_bar)
280          self.finder = finder
281  
282          # Where still-packed archives should be written to. If None, they are
283          # not saved, and are deleted immediately after unpacking.
284          self.download_dir = download_dir
285  
286          # Is build isolation allowed?
287          self.build_isolation = build_isolation
288  
289          # Should hash-checking be required?
290          self.require_hashes = require_hashes
291  
292          # Should install in user site-packages?
293          self.use_user_site = use_user_site
294  
295          # Should wheels be downloaded lazily?
296          self.use_lazy_wheel = lazy_wheel
297  
298          # Should in-tree builds be used for local paths?
299          self.in_tree_build = in_tree_build
300  
301          # Memoized downloaded files, as mapping of url: path.
302          self._downloaded: Dict[str, str] = {}
303  
304          # Previous "header" printed for a link-based InstallRequirement
305          self._previous_requirement_header = ("", "")

307      def _log_preparing_link(self, req: InstallRequirement) -> None:
308          """Provide context for the requirement being prepared."""
309          if req.link.is_file and not req.original_link_is_in_wheel_cache:
310              message = "Processing %s"
311              information = str(display_path(req.link.file_path))
312          else:
313              message = "Collecting %s"
314              information = str(req.req or req)
315  
316          if (message, information) != self._previous_requirement_header:
317              self._previous_requirement_header = (message, information)
318              logger.info(message, information)
319  
320          if req.original_link_is_in_wheel_cache:
321              with indent_log():
322                  logger.info("Using cached %s", req.link.filename)

324      def _ensure_link_req_src_dir(
325          self, req: InstallRequirement, parallel_builds: bool
326      ) -> None:
327          """Ensure source_dir of a linked InstallRequirement."""
328          # Since source_dir is only set for editable requirements.
329          if req.link.is_wheel:
330              # We don't need to unpack wheels, so no need for a source
331              # directory.
332              return
333          assert req.source_dir is None
334          if req.link.is_existing_dir() and self.in_tree_build:
335              # build local directories in-tree
336              req.source_dir = req.link.file_path
337              return
338  
339          # We always delete unpacked sdists after pip runs.
340          req.ensure_has_source_dir(
341              self.build_dir,
342              autodelete=True,
343              parallel_builds=parallel_builds,
344          )
345  
346          # If a checkout exists, it's unwise to keep going.  version
347          # inconsistencies are logged later, but do not fail the
348          # installation.
349          # FIXME: this won't upgrade when there's an existing
350          # package unpacked in `req.source_dir`
351          # TODO: this check is now probably dead code
352          if is_installable_dir(req.source_dir):
353              raise PreviousBuildDirError(
354                  "pip can't proceed with requirements '{}' due to a"
355                  "pre-existing build directory ({}). This is likely "
356                  "due to a previous installation that failed . pip is "
357                  "being responsible and not assuming it can delete this. "
358                  "Please delete it and try again.".format(req, req.source_dir)
359              )

361      def _get_linked_req_hashes(self, req: InstallRequirement) -> Hashes:
362          # By the time this is called, the requirement's link should have
363          # been checked so we can tell what kind of requirements req is
364          # and raise some more informative errors than otherwise.
365          # (For example, we can raise VcsHashUnsupported for a VCS URL
366          # rather than HashMissing.)
367          if not self.require_hashes:
368              return req.hashes(trust_internet=True)
369  
370          # We could check these first 2 conditions inside unpack_url
371          # and save repetition of conditions, but then we would
372          # report less-useful error messages for unhashable
373          # requirements, complaining that there's no hash provided.
374          if req.link.is_vcs:
375              raise VcsHashUnsupported()
376          if req.link.is_existing_dir():
377              raise DirectoryUrlHashUnsupported()
378  
379          # Unpinned packages are asking for trouble when a new version
380          # is uploaded.  This isn't a security check, but it saves users
381          # a surprising hash mismatch in the future.
382          # file:/// URLs aren't pinnable, so don't complain about them
383          # not being pinned.
384          if req.original_link is None and not req.is_pinned:
385              raise HashUnpinned()
386  
387          # If known-good hashes are missing for this requirement,
388          # shim it with a facade object that will provoke hash
389          # computation and then raise a HashMissing exception
390          # showing the user what the hash should be.
391          return req.hashes(trust_internet=False) or MissingHashes()

393      def _fetch_metadata_using_lazy_wheel(
394          self,
395          link: Link,
396      ) -> Optional[BaseDistribution]:
397          """Fetch metadata using lazy wheel, if possible."""
398          if not self.use_lazy_wheel:
399              return None
400          if self.require_hashes:
401              logger.debug("Lazy wheel is not used as hash checking is required")
402              return None
403          if link.is_file or not link.is_wheel:
404              logger.debug(
405                  "Lazy wheel is not used as %r does not points to a remote wheel",
406                  link,
407              )
408              return None
409  
410          wheel = Wheel(link.filename)
411          name = canonicalize_name(wheel.name)
412          logger.info(
413              "Obtaining dependency information from %s %s",
414              name,
415              wheel.version,
416          )
417          url = link.url.split("#", 1)[0]
418          try:
419              return dist_from_wheel_url(name, url, self._session)
420          except HTTPRangeRequestUnsupported:
421              logger.debug("%s does not support range requests", url)
422              return None

424      def _complete_partial_requirements(
425          self,
426          partially_downloaded_reqs: Iterable[InstallRequirement],
427          parallel_builds: bool = False,
428      ) -> None:
429          """Download any requirements which were only fetched by metadata."""
430          # Download to a temporary directory. These will be copied over as
431          # needed for downstream 'download', 'wheel', and 'install' commands.
432          temp_dir = TempDirectory(kind="unpack", globally_managed=True).path
433  
434          # Map each link to the requirement that owns it. This allows us to set
435          # `req.local_file_path` on the appropriate requirement after passing
436          # all the links at once into BatchDownloader.
437          links_to_fully_download: Dict[Link, InstallRequirement] = {}
438          for req in partially_downloaded_reqs:
439              assert req.link
440              links_to_fully_download[req.link] = req
441  
442          batch_download = self._batch_download(
443              links_to_fully_download.keys(),
444              temp_dir,
445          )
446          for link, (filepath, _) in batch_download:
447              logger.debug("Downloading link %s to %s", link, filepath)
448              req = links_to_fully_download[link]
449              req.local_file_path = filepath
450  
451          # This step is necessary to ensure all lazy wheels are processed
452          # successfully by the 'download', 'wheel', and 'install' commands.
453          for req in partially_downloaded_reqs:
454              self._prepare_linked_requirement(req, parallel_builds)
455  
456      def prepare_linked_requirement(
457          self, req: InstallRequirement, parallel_builds: bool = False
458      ) -> BaseDistribution:
459          """Prepare a requirement to be obtained from req.link."""
460          assert req.link
461          link = req.link
462          self._log_preparing_link(req)
463          with indent_log():
464              # Check if the relevant file is already available
465              # in the download directory
466              file_path = None
467              if self.download_dir is not None and link.is_wheel:
468                  hashes = self._get_linked_req_hashes(req)
469                  file_path = _check_download_dir(req.link, self.download_dir, hashes)
470  
471              if file_path is not None:
472                  # The file is already available, so mark it as downloaded
473                  self._downloaded[req.link.url] = file_path
474              else:
475                  # The file is not available, attempt to fetch only metadata
476                  wheel_dist = self._fetch_metadata_using_lazy_wheel(link)
477                  if wheel_dist is not None:
478                      req.needs_more_preparation = True
479                      return wheel_dist
480  
481              # None of the optimizations worked, fully prepare the requirement
482              return self._prepare_linked_requirement(req, parallel_builds)
483  
484      def prepare_linked_requirements_more(
485          self, reqs: Iterable[InstallRequirement], parallel_builds: bool = False
486      ) -> None:
487          """Prepare linked requirements more, if needed."""
488          reqs = [req for req in reqs if req.needs_more_preparation]
489          for req in reqs:
490              # Determine if any of these requirements were already downloaded.
491              if self.download_dir is not None and req.link.is_wheel:
492                  hashes = self._get_linked_req_hashes(req)
493                  file_path = _check_download_dir(req.link, self.download_dir, hashes)
494                  if file_path is not None:
495                      self._downloaded[req.link.url] = file_path
496                      req.needs_more_preparation = False
497  
498          # Prepare requirements we found were already downloaded for some
499          # reason. The other downloads will be completed separately.
500          partially_downloaded_reqs: List[InstallRequirement] = []
501          for req in reqs:
502              if req.needs_more_preparation:
503                  partially_downloaded_reqs.append(req)
504              else:
505                  self._prepare_linked_requirement(req, parallel_builds)
506  
507          # TODO: separate this part out from RequirementPreparer when the v1
508          # resolver can be removed!
509          self._complete_partial_requirements(
510              partially_downloaded_reqs,
511              parallel_builds=parallel_builds,
512          )
513  
514      def _prepare_linked_requirement(
515          self, req: InstallRequirement, parallel_builds: bool
516      ) -> BaseDistribution:
517          assert req.link
518          link = req.link
519  
520          self._ensure_link_req_src_dir(req, parallel_builds)
521          hashes = self._get_linked_req_hashes(req)
522  
523          if link.is_existing_dir() and self.in_tree_build:
524              local_file = None
525          elif link.url not in self._downloaded:
526              try:
527                  local_file = unpack_url(
528                      link, req.source_dir, self._download, self.download_dir, hashes
529                  )
530              except NetworkConnectionError as exc:
531                  raise InstallationError(
532                      "Could not install requirement {} because of HTTP "
533                      "error {} for URL {}".format(req, exc, link)
534                  )
535          else:
536              file_path = self._downloaded[link.url]
537              if hashes:
538                  hashes.check_against_path(file_path)
539              local_file = File(file_path, content_type=None)
540  
541          # For use in later processing,
542          # preserve the file path on the requirement.
543          if local_file:
544              req.local_file_path = local_file.path
545  
546          dist = _get_prepared_distribution(
547              req,
548              self.req_tracker,
549              self.finder,
550              self.build_isolation,
551          )
552          return dist
553  
554      def save_linked_requirement(self, req: InstallRequirement) -> None:
555          assert self.download_dir is not None
556          assert req.link is not None
557          link = req.link
558          if link.is_vcs or (link.is_existing_dir() and req.editable):
559              # Make a .zip of the source_dir we already created.
560              req.archive(self.download_dir)
561              return
562  
563          if link.is_existing_dir():
564              logger.debug(
565                  "Not copying link to destination directory "
566                  "since it is a directory: %s",
567                  link,
568              )
569              return
570          if req.local_file_path is None:
571              # No distribution was downloaded for this requirement.
572              return
573  
574          download_location = os.path.join(self.download_dir, link.filename)
575          if not os.path.exists(download_location):
576              shutil.copy(req.local_file_path, download_location)
577              download_path = display_path(download_location)
578              logger.info("Saved %s", download_path)

580      def prepare_editable_requirement(
581          self,
582          req: InstallRequirement,
583      ) -> BaseDistribution:
584          """Prepare an editable requirement."""
585          assert req.editable, "cannot prepare a non-editable req as editable"
586  
587          logger.info("Obtaining %s", req)
588  
589          with indent_log():
590              if self.require_hashes:
591                  raise InstallationError(
592                      "The editable requirement {} cannot be installed when "
593                      "requiring hashes, because there is no single file to "
594                      "hash.".format(req)
595                  )
596              req.ensure_has_source_dir(self.src_dir)
597              req.update_editable()
598  
599              dist = _get_prepared_distribution(
600                  req,
601                  self.req_tracker,
602                  self.finder,
603                  self.build_isolation,
604              )
605  
606              req.check_if_exists(self.use_user_site)
607  
608          return dist

610      def prepare_installed_requirement(
611          self,
612          req: InstallRequirement,
613          skip_reason: str,
614      ) -> BaseDistribution:
615          """Prepare an already-installed requirement."""
616          assert req.satisfied_by, "req should have been satisfied but isn't"
617          assert skip_reason is not None, (
618              "did not get skip reason skipped but req.satisfied_by "
619              "is set to {}".format(req.satisfied_by)
620          )
621          logger.info(
622              "Requirement %s: %s (%s)", skip_reason, req, req.satisfied_by.version
623          )
624          with indent_log():
625              if self.require_hashes:
626                  logger.debug(
627                      "Since it is already installed, we are trusting this "
628                      "package without checking its hash. To ensure a "
629                      "completely repeatable environment, install into an "
630                      "empty virtualenv."
631                  )
632              return InstalledDistribution(req).get_metadata_distribution()
633
