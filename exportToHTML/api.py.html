35   def from_bytes(
36       sequences: bytes,
37       steps: int = 5,
38       chunk_size: int = 512,
39       threshold: float = 0.2,
40       cp_isolation: List[str] = None,
41       cp_exclusion: List[str] = None,
42       preemptive_behaviour: bool = True,
43       explain: bool = False,
44   ) -> CharsetMatches:
45       """ 
46       Given a raw bytes sequence, return the best possibles charset usable to render str objects. 
47       If there is no results, it is a strong indicator that the source is binary/not text. 
48       By default, the process will extract 5 blocs of 512o each to assess the mess and coherence of a given sequence. 
49       And will give up a particular code page after 20% of measured mess. Those criteria are customizable at will. 
50    
51       The preemptive behavior DOES NOT replace the traditional detection workflow, it prioritize a particular code page 
52       but never take it for granted. Can improve the performance. 
53    
54       You may want to focus your attention to some code page or/and not others, use cp_isolation and cp_exclusion for that 
55       purpose. 
56    
57       This function will strip the SIG in the payload/sequence every time except on UTF-16, UTF-32. 
58       By default the library does not setup any handler other than the NullHandler, if you choose to set the 'explain' 
59       toggle to True it will alter the logger configuration to add a StreamHandler that is suitable for debugging. 
60       Custom logging format and handler can be set manually. 
61       """
62   
63       if not isinstance(sequences, (bytearray, bytes)):
64           raise TypeError(
65               "Expected object of type bytes or bytearray, got: {0}".format(
66                   type(sequences)
67               )
68           )
69   
70       if explain:
71           previous_logger_level = logger.level  # type: int
72           logger.addHandler(explain_handler)
73           logger.setLevel(logging.DEBUG)
74   
75       length = len(sequences)  # type: int
76   
77       if length == 0:
78           logger.warning("Encoding detection on empty bytes, assuming utf_8 intention.")
79           if explain:
80               logger.removeHandler(explain_handler)
81               logger.setLevel(previous_logger_level or logging.WARNING)
82           return CharsetMatches([CharsetMatch(sequences, "utf_8", 0.0, False, [], "")])
83   
84       if cp_isolation is not None:
85           logger.debug(
86               "cp_isolation is set. use this flag for debugging purpose. "
87               "limited list of encoding allowed : %s.",
88               ", ".join(cp_isolation),
89           )
90           cp_isolation = [iana_name(cp, False) for cp in cp_isolation]
91       else:
92           cp_isolation = []
93   
94       if cp_exclusion is not None:
95           logger.debug(
96               "cp_exclusion is set. use this flag for debugging purpose. "
97               "limited list of encoding excluded : %s.",
98               ", ".join(cp_exclusion),
99           )
100          cp_exclusion = [iana_name(cp, False) for cp in cp_exclusion]
101      else:
102          cp_exclusion = []
103  
104      if length <= (chunk_size * steps):
105          logger.debug(
106              "override steps (%i) and chunk_size (%i) as content does not fit (%i byte(s) given) parameters.",
107              steps,
108              chunk_size,
109              length,
110          )
111          steps = 1
112          chunk_size = length
113  
114      if steps > 1 and length / steps < chunk_size:
115          chunk_size = int(length / steps)
116  
117      is_too_small_sequence = len(sequences) < TOO_SMALL_SEQUENCE  # type: bool
118      is_too_large_sequence = len(sequences) >= TOO_BIG_SEQUENCE  # type: bool
119  
120      if is_too_small_sequence:
121          logger.warning(
122              "Trying to detect encoding from a tiny portion of ({}) byte(s).".format(
123                  length
124              )
125          )
126      elif is_too_large_sequence:
127          logger.info(
128              "Using lazy str decoding because the payload is quite large, ({}) byte(s).".format(
129                  length
130              )
131          )
132  
133      prioritized_encodings = []  # type: List[str]
134  
135      specified_encoding = (
136          any_specified_encoding(sequences) if preemptive_behaviour else None
137      )  # type: Optional[str]
138  
139      if specified_encoding is not None:
140          prioritized_encodings.append(specified_encoding)
141          logger.info(
142              "Detected declarative mark in sequence. Priority +1 given for %s.",
143              specified_encoding,
144          )
145  
146      tested = set()  # type: Set[str]
147      tested_but_hard_failure = []  # type: List[str]
148      tested_but_soft_failure = []  # type: List[str]
149  
150      fallback_ascii = None  # type: Optional[CharsetMatch]
151      fallback_u8 = None  # type: Optional[CharsetMatch]
152      fallback_specified = None  # type: Optional[CharsetMatch]
153  
154      results = CharsetMatches()  # type: CharsetMatches
155  
156      sig_encoding, sig_payload = identify_sig_or_bom(sequences)
157  
158      if sig_encoding is not None:
159          prioritized_encodings.append(sig_encoding)
160          logger.info(
161              "Detected a SIG or BOM mark on first %i byte(s). Priority +1 given for %s.",
162              len(sig_payload),
163              sig_encoding,
164          )
165  
166      prioritized_encodings.append("ascii")
167  
168      if "utf_8" not in prioritized_encodings:
169          prioritized_encodings.append("utf_8")
170  
171      for encoding_iana in prioritized_encodings + IANA_SUPPORTED:
172  
173          if cp_isolation and encoding_iana not in cp_isolation:
174              continue
175  
176          if cp_exclusion and encoding_iana in cp_exclusion:
177              continue
178  
179          if encoding_iana in tested:
180              continue
181  
182          tested.add(encoding_iana)
183  
184          decoded_payload = None  # type: Optional[str]
185          bom_or_sig_available = sig_encoding == encoding_iana  # type: bool
186          strip_sig_or_bom = bom_or_sig_available and should_strip_sig_or_bom(
187              encoding_iana
188          )  # type: bool
189  
190          if encoding_iana in {"utf_16", "utf_32"} and not bom_or_sig_available:
191              logger.debug(
192                  "Encoding %s wont be tested as-is because it require a BOM. Will try some sub-encoder LE/BE.",
193                  encoding_iana,
194              )
195              continue
196  
197          try:
198              is_multi_byte_decoder = is_multi_byte_encoding(encoding_iana)  # type: bool
199          except (ModuleNotFoundError, ImportError):
200              logger.debug(
201                  "Encoding %s does not provide an IncrementalDecoder", encoding_iana
202              )
203              continue
204  
205          try:
206              if is_too_large_sequence and is_multi_byte_decoder is False:
207                  str(
208                      sequences[: int(50e4)]
209                      if strip_sig_or_bom is False
210                      else sequences[len(sig_payload) : int(50e4)],
211                      encoding=encoding_iana,
212                  )
213              else:
214                  decoded_payload = str(
215                      sequences
216                      if strip_sig_or_bom is False
217                      else sequences[len(sig_payload) :],
218                      encoding=encoding_iana,
219                  )
220          except (UnicodeDecodeError, LookupError) as e:
221              if not isinstance(e, LookupError):
222                  logger.debug(
223                      "Code page %s does not fit given bytes sequence at ALL. %s",
224                      encoding_iana,
225                      str(e),
226                  )
227              tested_but_hard_failure.append(encoding_iana)
228              continue
229  
230          similar_soft_failure_test = False  # type: bool
231  
232          for encoding_soft_failed in tested_but_soft_failure:
233              if is_cp_similar(encoding_iana, encoding_soft_failed):
234                  similar_soft_failure_test = True
235                  break
236  
237          if similar_soft_failure_test:
238              logger.info(
239                  "%s is deemed too similar to code page %s and was consider unsuited already. Continuing!",
240                  encoding_iana,
241                  encoding_soft_failed,
242              )
243              continue
244  
245          r_ = range(
246              0 if not bom_or_sig_available else len(sig_payload),
247              length,
248              int(length / steps),
249          )
250  
251          multi_byte_bonus = (
252              is_multi_byte_decoder
253              and decoded_payload is not None
254              and len(decoded_payload) < length
255          )  # type: bool
256  
257          if multi_byte_bonus:
258              logger.info(
259                  "Code page %s is a multi byte encoding table and it appear that at least one character "
260                  "was encoded using n-bytes.",
261                  encoding_iana,
262              )
263  
264          max_chunk_gave_up = int(len(r_) / 4)  # type: int
265  
266          max_chunk_gave_up = max(max_chunk_gave_up, 2)
267          early_stop_count = 0  # type: int
268  
269          md_chunks = []  # type: List[str]
270          md_ratios = []
271  
272          for i in r_:
273              if i + chunk_size > length + 8:
274                  continue
275  
276              cut_sequence = sequences[i : i + chunk_size]
277  
278              if bom_or_sig_available and strip_sig_or_bom is False:
279                  cut_sequence = sig_payload + cut_sequence
280  
281              try:
282                  chunk = cut_sequence.decode(
283                      encoding_iana,
284                      errors="ignore" if is_multi_byte_decoder else "strict",
285                  )  # type: str
286              except UnicodeDecodeError as e:  # Lazy str loading may have missed something there
287                  logger.warning(
288                      "LazyStr Loading: After MD chunk decode, code page %s does not fit given bytes sequence at ALL. %s",
289                      encoding_iana,
290                      str(e),
291                  )
292                  early_stop_count = max_chunk_gave_up
293                  break
294  
295              # multi-byte bad cutting detector and adjustment
296              # not the cleanest way to perform that fix but clever enough for now.
297              if is_multi_byte_decoder and i > 0 and sequences[i] >= 0x80:
298  
299                  chunk_partial_size_chk = min(chunk_size, 16)  # type: int
300  
301                  if (
302                      decoded_payload
303                      and chunk[:chunk_partial_size_chk] not in decoded_payload
304                  ):
305                      for j in range(i, i - 4, -1):
306                          cut_sequence = sequences[j : i + chunk_size]
307  
308                          if bom_or_sig_available and strip_sig_or_bom is False:
309                              cut_sequence = sig_payload + cut_sequence
310  
311                          chunk = cut_sequence.decode(encoding_iana, errors="ignore")
312  
313                          if chunk[:chunk_partial_size_chk] in decoded_payload:
314                              break
315  
316              md_chunks.append(chunk)
317  
318              md_ratios.append(mess_ratio(chunk, threshold))
319  
320              if md_ratios[-1] >= threshold:
321                  early_stop_count += 1
322  
323              if (early_stop_count >= max_chunk_gave_up) or (
324                  bom_or_sig_available and strip_sig_or_bom is False
325              ):
326                  break
327  
328          mean_mess_ratio = (
329              sum(md_ratios) / len(md_ratios) if md_ratios else 0.0
330          )  # type: float
331          if mean_mess_ratio >= threshold or early_stop_count >= max_chunk_gave_up:
332              tested_but_soft_failure.append(encoding_iana)
333              logger.info(
334                  "%s was excluded because of initial chaos probing. Gave up %i time(s). "
335                  "Computed mean chaos is %f %%.",
336                  encoding_iana,
337                  early_stop_count,
338                  round(mean_mess_ratio * 100, ndigits=3),
339              )
340              # Preparing those fallbacks in case we got nothing.
341              if encoding_iana in ["ascii", "utf_8", specified_encoding]:
342                  fallback_entry = CharsetMatch(
343                      sequences, encoding_iana, threshold, False, [], decoded_payload
344                  )
345                  if encoding_iana == specified_encoding:
346                      fallback_specified = fallback_entry
347                  elif encoding_iana == "ascii":
348                      fallback_ascii = fallback_entry
349                  else:
350                      fallback_u8 = fallback_entry
351              continue
352  
353          logger.info(
354              "%s passed initial chaos probing. Mean measured chaos is %f %%",
355              encoding_iana,
356              round(mean_mess_ratio * 100, ndigits=3),
357          )
358  
359          if not is_multi_byte_decoder:
360              target_languages = encoding_languages(encoding_iana)  # type: List[str]
361          else:
362              target_languages = mb_encoding_languages(encoding_iana)
363  
364          if target_languages:
365              logger.info(
366                  "{} should target any language(s) of {}".format(
367                      encoding_iana, str(target_languages)
368                  )
369              )
370  
371          cd_ratios = []
372  
373          for chunk in md_chunks:
374              chunk_languages = coherence_ratio(
375                  chunk, 0.1, ",".join(target_languages) if target_languages else None
376              )
377  
378              cd_ratios.append(chunk_languages)
379  
380          cd_ratios_merged = merge_coherence_ratios(cd_ratios)
381  
382          if cd_ratios_merged:
383              logger.info(
384                  "We detected language {} using {}".format(
385                      cd_ratios_merged, encoding_iana
386                  )
387              )
388  
389          # We might want to check the sequence again with the whole content
390          # Only if initial MD/CD tests passes
391          if is_too_large_sequence and not is_multi_byte_decoder:
392              try:
393                  sequences[int(50e3) :].decode(encoding_iana, errors="strict")
394              except UnicodeDecodeError as e:
395                  logger.warning(
396                      "LazyStr Loading: After final lookup, code page %s does not fit given bytes sequence at ALL. %s",
397                      encoding_iana,
398                      str(e),
399                  )
400                  tested_but_hard_failure.append(encoding_iana)
401                  continue
402  
403          results.append(
404              CharsetMatch(
405                  sequences,
406                  encoding_iana,
407                  mean_mess_ratio,
408                  bom_or_sig_available,
409                  cd_ratios_merged,
410                  decoded_payload,
411              )
412          )
413  
414          if (
415              encoding_iana in [specified_encoding, "ascii", "utf_8"]
416              and mean_mess_ratio < 0.1
417          ):
418              logger.info(
419                  "%s is most likely the one. Stopping the process.", encoding_iana
420              )
421              if explain:
422                  logger.removeHandler(explain_handler)
423                  logger.setLevel(previous_logger_level)
424              return CharsetMatches([results[encoding_iana]])
425  
426          if encoding_iana == sig_encoding:
427              logger.info(
428                  "%s is most likely the one as we detected a BOM or SIG within the beginning of the sequence.",
429                  encoding_iana,
430              )
431              if explain:
432                  logger.removeHandler(explain_handler)
433                  logger.setLevel(previous_logger_level)
434              return CharsetMatches([results[encoding_iana]])
435  
436      if len(results) == 0:
437          if fallback_u8 or fallback_ascii or fallback_specified:
438              logger.warning(
439                  "Nothing got out of the detection process. Using ASCII/UTF-8/Specified fallback."
440              )
441  
442          if fallback_specified:
443              logger.warning(
444                  "%s will be used as a fallback match", fallback_specified.encoding
445              )
446              results.append(fallback_specified)
447          elif (
448              (fallback_u8 and fallback_ascii is None)
449              or (
450                  fallback_u8
451                  and fallback_ascii
452                  and fallback_u8.fingerprint != fallback_ascii.fingerprint
453              )
454              or (fallback_u8 is not None)
455          ):
456              logger.warning("utf_8 will be used as a fallback match")
457              results.append(fallback_u8)
458          elif fallback_ascii:
459              logger.warning("ascii will be used as a fallback match")
460              results.append(fallback_ascii)
461  
462      if explain:
463          logger.removeHandler(explain_handler)
464          logger.setLevel(previous_logger_level)
465  
466      return results

469  def from_fp(
470      fp: BinaryIO,
471      steps: int = 5,
472      chunk_size: int = 512,
473      threshold: float = 0.20,
474      cp_isolation: List[str] = None,
475      cp_exclusion: List[str] = None,
476      preemptive_behaviour: bool = True,
477      explain: bool = False,
478  ) -> CharsetMatches:
479      """ 
480      Same thing than the function from_bytes but using a file pointer that is already ready. 
481      Will not close the file pointer. 
482      """
483      return from_bytes(
484          fp.read(),
485          steps,
486          chunk_size,
487          threshold,
488          cp_isolation,
489          cp_exclusion,
490          preemptive_behaviour,
491          explain,
492      )

495  def from_path(
496      path: PathLike,
497      steps: int = 5,
498      chunk_size: int = 512,
499      threshold: float = 0.20,
500      cp_isolation: List[str] = None,
501      cp_exclusion: List[str] = None,
502      preemptive_behaviour: bool = True,
503      explain: bool = False,
504  ) -> CharsetMatches:
505      """ 
506      Same thing than the function from_bytes but with one extra step. Opening and reading given file path in binary mode. 
507      Can raise IOError. 
508      """
509      with open(path, "rb") as fp:
510          return from_fp(
511              fp,
512              steps,
513              chunk_size,
514              threshold,
515              cp_isolation,
516              cp_exclusion,
517              preemptive_behaviour,
518              explain,
519          )

522  def normalize(
523      path: PathLike,
524      steps: int = 5,
525      chunk_size: int = 512,
526      threshold: float = 0.20,
527      cp_isolation: List[str] = None,
528      cp_exclusion: List[str] = None,
529      preemptive_behaviour: bool = True,
530  ) -> CharsetMatch:
531      """ 
532      Take a (text-based) file path and try to create another file next to it, this time using UTF-8. 
533      """
534      results = from_path(
535          path,
536          steps,
537          chunk_size,
538          threshold,
539          cp_isolation,
540          cp_exclusion,
541          preemptive_behaviour,
542      )
543  
544      filename = basename(path)
545      target_extensions = list(splitext(filename))
546  
547      if len(results) == 0:
548          raise IOError(
549              'Unable to normalize "{}", no encoding charset seems to fit.'.format(
550                  filename
551              )
552          )
553  
554      result = results.best()
555  
556      target_extensions[0] += "-" + result.encoding  # type: ignore
557  
558      with open(
559          "{}".format(str(path).replace(filename, "".join(target_extensions))), "wb"
560      ) as fp:
561          fp.write(result.output())  # type: ignore
562  
563      return result  # type: ignore
