51   def _check_link_requires_python(
52       link: Link,
53       version_info: Tuple[int, int, int],
54       ignore_requires_python: bool = False,
55   ) -> bool:
56       """ 
57       Return whether the given Python version is compatible with a link's 
58       "Requires-Python" value. 
59    
60       :param version_info: A 3-tuple of ints representing the Python 
61           major-minor-micro version to check. 
62       :param ignore_requires_python: Whether to ignore the "Requires-Python" 
63           value if the given Python version isn't compatible. 
64       """
65       try:
66           is_compatible = check_requires_python(
67               link.requires_python,
68               version_info=version_info,
69           )
70       except specifiers.InvalidSpecifier:
71           logger.debug(
72               "Ignoring invalid Requires-Python (%r) for link: %s",
73               link.requires_python,
74               link,
75           )
76       else:
77           if not is_compatible:
78               version = ".".join(map(str, version_info))
79               if not ignore_requires_python:
80                   logger.verbose(
81                       "Link requires a different Python (%s not in: %r): %s",
82                       version,
83                       link.requires_python,
84                       link,
85                   )
86                   return False
87   
88               logger.debug(
89                   "Ignoring failed Requires-Python check (%s not in: %r) for link: %s",
90                   version,
91                   link.requires_python,
92                   link,
93               )
94   
95       return True

100      """ 
101      Responsible for evaluating links for a particular project.

103  
104      _py_version_re = re.compile(r"-py([123]\.?[0-9]?)$")
105  
106      # Don't include an allow_yanked default value to make sure each call
107      # site considers whether yanked releases are allowed. This also causes
108      # that decision to be made explicit in the calling code, which helps
109      # people when reading the code.
110      def __init__(
111          self,
112          project_name: str,
113          canonical_name: str,
114          formats: FrozenSet[str],
115          target_python: TargetPython,
116          allow_yanked: bool,
117          ignore_requires_python: Optional[bool] = None,
118      ) -> None:
119          """ 
120          :param project_name: The user supplied package name. 
121          :param canonical_name: The canonical package name. 
122          :param formats: The formats allowed for this package. Should be a set 
123              with 'binary' or 'source' or both in it. 
124          :param target_python: The target Python interpreter to use when 
125              evaluating link compatibility. This is used, for example, to 
126              check wheel compatibility, as well as when checking the Python 
127              version, e.g. the Python version embedded in a link filename 
128              (or egg fragment) and against an HTML link's optional PEP 503 
129              "data-requires-python" attribute. 
130          :param allow_yanked: Whether files marked as yanked (in the sense 
131              of PEP 592) are permitted to be candidates for install. 
132          :param ignore_requires_python: Whether to ignore incompatible 
133              PEP 503 "data-requires-python" values in HTML links. Defaults 
134              to False. 
135          """
136          if ignore_requires_python is None:
137              ignore_requires_python = False
138  
139          self._allow_yanked = allow_yanked
140          self._canonical_name = canonical_name
141          self._ignore_requires_python = ignore_requires_python
142          self._formats = formats
143          self._target_python = target_python
144  
145          self.project_name = project_name
146  
147      def evaluate_link(self, link: Link) -> Tuple[bool, Optional[str]]:
148          """ 
149          Determine whether a link is a candidate for installation. 
150   
151          :return: A tuple (is_candidate, result), where `result` is (1) a 
152              version string if `is_candidate` is True, and (2) if 
153              `is_candidate` is False, an optional string to log the reason 
154              the link fails to qualify. 
155          """
156          version = None
157          if link.is_yanked and not self._allow_yanked:
158              reason = link.yanked_reason or "<none given>"
159              return (False, f"yanked for reason: {reason}")
160  
161          if link.egg_fragment:
162              egg_info = link.egg_fragment
163              ext = link.ext
164          else:
165              egg_info, ext = link.splitext()
166              if not ext:
167                  return (False, "not a file")
168              if ext not in SUPPORTED_EXTENSIONS:
169                  return (False, f"unsupported archive format: {ext}")
170              if "binary" not in self._formats and ext == WHEEL_EXTENSION:
171                  reason = "No binaries permitted for {}".format(self.project_name)
172                  return (False, reason)
173              if "macosx10" in link.path and ext == ".zip":
174                  return (False, "macosx10 one")
175              if ext == WHEEL_EXTENSION:
176                  try:
177                      wheel = Wheel(link.filename)
178                  except InvalidWheelFilename:
179                      return (False, "invalid wheel filename")
180                  if canonicalize_name(wheel.name) != self._canonical_name:
181                      reason = "wrong project name (not {})".format(self.project_name)
182                      return (False, reason)
183  
184                  supported_tags = self._target_python.get_tags()
185                  if not wheel.supported(supported_tags):
186                      # Include the wheel's tags in the reason string to
187                      # simplify troubleshooting compatibility issues.
188                      file_tags = wheel.get_formatted_file_tags()
189                      reason = (
190                          "none of the wheel's tags ({}) are compatible "
191                          "(run pip debug --verbose to show compatible tags)".format(
192                              ", ".join(file_tags)
193                          )
194                      )
195                      return (False, reason)
196  
197                  version = wheel.version
198  
199          # This should be up by the self.ok_binary check, but see issue 2700.
200          if "source" not in self._formats and ext != WHEEL_EXTENSION:
201              reason = f"No sources permitted for {self.project_name}"
202              return (False, reason)
203  
204          if not version:
205              version = _extract_version_from_fragment(
206                  egg_info,
207                  self._canonical_name,
208              )
209          if not version:
210              reason = f"Missing project version for {self.project_name}"
211              return (False, reason)
212  
213          match = self._py_version_re.search(version)
214          if match:
215              version = version[: match.start()]
216              py_version = match.group(1)
217              if py_version != self._target_python.py_version:
218                  return (False, "Python version is incorrect")
219  
220          supports_python = _check_link_requires_python(
221              link,
222              version_info=self._target_python.py_version_info,
223              ignore_requires_python=self._ignore_requires_python,
224          )
225          if not supports_python:
226              # Return None for the reason text to suppress calling
227              # _log_skipped_link().
228              return (False, None)
229  
230          logger.debug("Found link %s, version: %s", link, version)
231  
232          return (True, version)

235  def filter_unallowed_hashes(
236      candidates: List[InstallationCandidate],
237      hashes: Hashes,
238      project_name: str,
239  ) -> List[InstallationCandidate]:
240      """ 
241      Filter out candidates whose hashes aren't allowed, and return a new 
242      list of candidates. 
243   
244      If at least one candidate has an allowed hash, then all candidates with 
245      either an allowed hash or no hash specified are returned.  Otherwise, 
246      the given candidates are returned. 
247   
248      Including the candidates with no hash specified when there is a match 
249      allows a warning to be logged if there is a more preferred candidate 
250      with no hash specified.  Returning all candidates in the case of no 
251      matches lets pip report the hash of the candidate that would otherwise 
252      have been installed (e.g. permitting the user to more easily update 
253      their requirements file with the desired hash). 
254      """
255      if not hashes:
256          logger.debug(
257              "Given no hashes to check %s links for project %r: "
258              "discarding no candidates",
259              len(candidates),
260              project_name,
261          )
262          # Make sure we're not returning back the given value.
263          return list(candidates)
264  
265      matches_or_no_digest = []
266      # Collect the non-matches for logging purposes.
267      non_matches = []
268      match_count = 0
269      for candidate in candidates:
270          link = candidate.link
271          if not link.has_hash:
272              pass
273          elif link.is_hash_allowed(hashes=hashes):
274              match_count += 1
275          else:
276              non_matches.append(candidate)
277              continue
278  
279          matches_or_no_digest.append(candidate)
280  
281      if match_count:
282          filtered = matches_or_no_digest
283      else:
284          # Make sure we're not returning back the given value.
285          filtered = list(candidates)
286  
287      if len(filtered) == len(candidates):
288          discard_message = "discarding no candidates"
289      else:
290          discard_message = "discarding {} non-matches:\n  {}".format(
291              len(non_matches),
292              "\n  ".join(str(candidate.link) for candidate in non_matches),
293          )
294  
295      logger.debug(
296          "Checked %s links for project %r against %s hashes "
297          "(%s matches, %s no digest): %s",
298          len(candidates),
299          project_name,
300          hashes.digest_count,
301          match_count,
302          len(matches_or_no_digest) - match_count,
303          discard_message,
304      )
305  
306      return filtered

328  class BestCandidateResult:
329      """A collection of candidates, returned by `PackageFinder.find_best_candidate`. 
330   
331      This class is only intended to be instantiated by CandidateEvaluator's 
332      `compute_best_candidate()` method. 
333      """
334  
335      def __init__(
336          self,
337          candidates: List[InstallationCandidate],
338          applicable_candidates: List[InstallationCandidate],
339          best_candidate: Optional[InstallationCandidate],
340      ) -> None:
341          """ 
342          :param candidates: A sequence of all available candidates found. 
343          :param applicable_candidates: The applicable candidates. 
344          :param best_candidate: The most preferred candidate found, or None 
345              if no applicable candidates were found. 
346          """
347          assert set(applicable_candidates) <= set(candidates)
348  
349          if best_candidate is None:
350              assert not applicable_candidates
351          else:
352              assert best_candidate in applicable_candidates
353  
354          self._applicable_candidates = applicable_candidates
355          self._candidates = candidates
356  
357          self.best_candidate = best_candidate
358  
359      def iter_all(self) -> Iterable[InstallationCandidate]:
360          """Iterate through all candidates."""
361          return iter(self._candidates)
362  
363      def iter_applicable(self) -> Iterable[InstallationCandidate]:
364          """Iterate through the applicable candidates."""
365          return iter(self._applicable_candidates)

368  class CandidateEvaluator:
369  
370      """ 
371      Responsible for filtering and sorting candidates for installation based 
372      on what tags are valid. 
373      """
374  
375      @classmethod
376      def create(
377          cls,
378          project_name: str,
379          target_python: Optional[TargetPython] = None,
380          prefer_binary: bool = False,
381          allow_all_prereleases: bool = False,
382          specifier: Optional[specifiers.BaseSpecifier] = None,
383          hashes: Optional[Hashes] = None,
384      ) -> "CandidateEvaluator":
385          """Create a CandidateEvaluator object. 
386   
387          :param target_python: The target Python interpreter to use when 
388              checking compatibility. If None (the default), a TargetPython 
389              object will be constructed from the running Python. 
390          :param specifier: An optional object implementing `filter` 
391              (e.g. `packaging.specifiers.SpecifierSet`) to filter applicable 
392              versions. 
393          :param hashes: An optional collection of allowed hashes. 
394          """
395          if target_python is None:
396              target_python = TargetPython()
397          if specifier is None:
398              specifier = specifiers.SpecifierSet()
399  
400          supported_tags = target_python.get_tags()
401  
402          return cls(
403              project_name=project_name,
404              supported_tags=supported_tags,
405              specifier=specifier,
406              prefer_binary=prefer_binary,
407              allow_all_prereleases=allow_all_prereleases,
408              hashes=hashes,
409          )
410  
411      def __init__(
412          self,
413          project_name: str,
414          supported_tags: List[Tag],
415          specifier: specifiers.BaseSpecifier,
416          prefer_binary: bool = False,
417          allow_all_prereleases: bool = False,
418          hashes: Optional[Hashes] = None,
419      ) -> None:
420          """ 
421          :param supported_tags: The PEP 425 tags supported by the target 
422              Python in order of preference (most preferred first). 
423          """
424          self._allow_all_prereleases = allow_all_prereleases
425          self._hashes = hashes
426          self._prefer_binary = prefer_binary
427          self._project_name = project_name
428          self._specifier = specifier
429          self._supported_tags = supported_tags
430          # Since the index of the tag in the _supported_tags list is used
431          # as a priority, precompute a map from tag to index/priority to be
432          # used in wheel.find_most_preferred_tag.
433          self._wheel_tag_preferences = {
434              tag: idx for idx, tag in enumerate(supported_tags)
435          }
436  
437      def get_applicable_candidates(
438          self,
439          candidates: List[InstallationCandidate],
440      ) -> List[InstallationCandidate]:
441          """ 
442          Return the applicable candidates from a list of candidates. 
443          """
444          # Using None infers from the specifier instead.
445          allow_prereleases = self._allow_all_prereleases or None
446          specifier = self._specifier
447          versions = {
448              str(v)
449              for v in specifier.filter(
450                  # We turn the version object into a str here because otherwise
451                  # when we're debundled but setuptools isn't, Python will see
452                  # packaging.version.Version and
453                  # pkg_resources._vendor.packaging.version.Version as different
454                  # types. This way we'll use a str as a common data interchange
455                  # format. If we stop using the pkg_resources provided specifier
456                  # and start using our own, we can drop the cast to str().
457                  (str(c.version) for c in candidates),
458                  prereleases=allow_prereleases,
459              )
460          }
461  
462          # Again, converting version to str to deal with debundling.
463          applicable_candidates = [c for c in candidates if str(c.version) in versions]
464  
465          filtered_applicable_candidates = filter_unallowed_hashes(
466              candidates=applicable_candidates,
467              hashes=self._hashes,
468              project_name=self._project_name,
469          )
470  
471          return sorted(filtered_applicable_candidates, key=self._sort_key)
472  
473      def _sort_key(self, candidate: InstallationCandidate) -> CandidateSortingKey:
474          """ 
475          Function to pass as the `key` argument to a call to sorted() to sort 
476          InstallationCandidates by preference. 
477   
478          Returns a tuple such that tuples sorting as greater using Python's 
479          default comparison operator are more preferred. 
480   
481          The preference is as follows: 
482   
483          First and foremost, candidates with allowed (matching) hashes are 
484          always preferred over candidates without matching hashes. This is 
485          because e.g. if the only candidate with an allowed hash is yanked, 
486          we still want to use that candidate. 
487   
488          Second, excepting hash considerations, candidates that have been 
489          yanked (in the sense of PEP 592) are always less preferred than 
490          candidates that haven't been yanked. Then: 
491   
492          If not finding wheels, they are sorted by version only. 
493          If finding wheels, then the sort order is by version, then: 
494            1. existing installs 
495            2. wheels ordered via Wheel.support_index_min(self._supported_tags) 
496            3. source archives 
497          If prefer_binary was set, then all wheels are sorted above sources. 
498   
499          Note: it was considered to embed this logic into the Link 
500                comparison operators, but then different sdist links 
501                with the same version, would have to be considered equal 
502          """
503          valid_tags = self._supported_tags
504          support_num = len(valid_tags)
505          build_tag: BuildTag = ()
506          binary_preference = 0
507          link = candidate.link
508          if link.is_wheel:
509              # can raise InvalidWheelFilename
510              wheel = Wheel(link.filename)
511              try:
512                  pri = -(
513                      wheel.find_most_preferred_tag(
514                          valid_tags, self._wheel_tag_preferences
515                      )
516                  )
517              except ValueError:
518                  raise UnsupportedWheel(
519                      "{} is not a supported wheel for this platform. It "
520                      "can't be sorted.".format(wheel.filename)
521                  )
522              if self._prefer_binary:
523                  binary_preference = 1
524              if wheel.build_tag is not None:
525                  match = re.match(r"^(\d+)(.*)$", wheel.build_tag)
526                  build_tag_groups = match.groups()
527                  build_tag = (int(build_tag_groups[0]), build_tag_groups[1])
528          else:  # sdist
529              pri = -(support_num)
530          has_allowed_hash = int(link.is_hash_allowed(self._hashes))
531          yank_value = -1 * int(link.is_yanked)  # -1 for yanked.
532          return (
533              has_allowed_hash,
534              yank_value,
535              binary_preference,
536              candidate.version,
537              pri,
538              build_tag,
539          )
540  
541      def sort_best_candidate(
542          self,
543          candidates: List[InstallationCandidate],
544      ) -> Optional[InstallationCandidate]:
545          """ 
546          Return the best candidate per the instance's sort order, or None if 
547          no candidate is acceptable. 
548          """
549          if not candidates:
550              return None
551          best_candidate = max(candidates, key=self._sort_key)
552          return best_candidate
553  
554      def compute_best_candidate(
555          self,
556          candidates: List[InstallationCandidate],
557      ) -> BestCandidateResult:
558          """ 
559          Compute and return a `BestCandidateResult` instance. 
560          """
561          applicable_candidates = self.get_applicable_candidates(candidates)
562  
563          best_candidate = self.sort_best_candidate(applicable_candidates)
564  
565          return BestCandidateResult(
566              candidates,
567              applicable_candidates=applicable_candidates,
568              best_candidate=best_candidate,
569          )

572  class PackageFinder:
573      """This finds packages. 
574   
575      This is meant to match easy_install's technique for looking for 
576      packages, by reading pages and looking for appropriate links. 
577      """
578  
579      def __init__(
580          self,
581          link_collector: LinkCollector,
582          target_python: TargetPython,
583          allow_yanked: bool,
584          format_control: Optional[FormatControl] = None,
585          candidate_prefs: Optional[CandidatePreferences] = None,
586          ignore_requires_python: Optional[bool] = None,
587      ) -> None:
588          """ 
589          This constructor is primarily meant to be used by the create() class 
590          method and from tests. 
591   
592          :param format_control: A FormatControl object, used to control 
593              the selection of source packages / binary packages when consulting 
594              the index and links. 
595          :param candidate_prefs: Options to use when creating a 
596              CandidateEvaluator object. 
597          """
598          if candidate_prefs is None:
599              candidate_prefs = CandidatePreferences()
600  
601          format_control = format_control or FormatControl(set(), set())
602  
603          self._allow_yanked = allow_yanked
604          self._candidate_prefs = candidate_prefs
605          self._ignore_requires_python = ignore_requires_python
606          self._link_collector = link_collector
607          self._target_python = target_python
608  
609          self.format_control = format_control
610  
611          # These are boring links that have already been logged somehow.
612          self._logged_links: Set[Link] = set()
613  
614      # Don't include an allow_yanked default value to make sure each call
615      # site considers whether yanked releases are allowed. This also causes
616      # that decision to be made explicit in the calling code, which helps
617      # people when reading the code.
618      @classmethod
619      def create(
620          cls,
621          link_collector: LinkCollector,
622          selection_prefs: SelectionPreferences,
623          target_python: Optional[TargetPython] = None,
624      ) -> "PackageFinder":
625          """Create a PackageFinder. 
626   
627          :param selection_prefs: The candidate selection preferences, as a 
628              SelectionPreferences object. 
629          :param target_python: The target Python interpreter to use when 
630              checking compatibility. If None (the default), a TargetPython 
631              object will be constructed from the running Python. 
632          """
633          if target_python is None:
634              target_python = TargetPython()
635  
636          candidate_prefs = CandidatePreferences(
637              prefer_binary=selection_prefs.prefer_binary,
638              allow_all_prereleases=selection_prefs.allow_all_prereleases,
639          )
640  
641          return cls(
642              candidate_prefs=candidate_prefs,
643              link_collector=link_collector,
644              target_python=target_python,
645              allow_yanked=selection_prefs.allow_yanked,
646              format_control=selection_prefs.format_control,
647              ignore_requires_python=selection_prefs.ignore_requires_python,
648          )

689      def make_link_evaluator(self, project_name: str) -> LinkEvaluator:
690          canonical_name = canonicalize_name(project_name)
691          formats = self.format_control.get_allowed_formats(canonical_name)
692  
693          return LinkEvaluator(
694              project_name=project_name,
695              canonical_name=canonical_name,
696              formats=formats,
697              target_python=self._target_python,
698              allow_yanked=self._allow_yanked,
699              ignore_requires_python=self._ignore_requires_python,
700          )

702      def _sort_links(self, links: Iterable[Link]) -> List[Link]:
703          """ 
704          Returns elements of links in order, non-egg links first, egg links 
705          second, while eliminating duplicates 
706          """
707          eggs, no_eggs = [], []
708          seen: Set[Link] = set()
709          for link in links:
710              if link not in seen:
711                  seen.add(link)
712                  if link.egg_fragment:
713                      eggs.append(link)
714                  else:
715                      no_eggs.append(link)
716          return no_eggs + eggs
717  
718      def _log_skipped_link(self, link: Link, reason: str) -> None:
719          if link not in self._logged_links:
720              # Put the link at the end so the reason is more visible and because
721              # the link string is usually very long.
722              logger.debug("Skipping link: %s: %s", reason, link)
723              self._logged_links.add(link)

725      def get_install_candidate(
726          self, link_evaluator: LinkEvaluator, link: Link
727      ) -> Optional[InstallationCandidate]:
728          """ 
729          If the link is a candidate for install, convert it to an 
730          InstallationCandidate and return it. Otherwise, return None. 
731          """
732          is_candidate, result = link_evaluator.evaluate_link(link)
733          if not is_candidate:
734              if result:
735                  self._log_skipped_link(link, reason=result)
736              return None
737  
738          return InstallationCandidate(
739              name=link_evaluator.project_name,
740              link=link,
741              version=result,
742          )
743  
744      def evaluate_links(
745          self, link_evaluator: LinkEvaluator, links: Iterable[Link]
746      ) -> List[InstallationCandidate]:
747          """ 
748          Convert links that are candidates to InstallationCandidate objects. 
749          """
750          candidates = []
751          for link in self._sort_links(links):
752              candidate = self.get_install_candidate(link_evaluator, link)
753              if candidate is not None:
754                  candidates.append(candidate)
755  
756          return candidates

758      def process_project_url(
759          self, project_url: Link, link_evaluator: LinkEvaluator
760      ) -> List[InstallationCandidate]:
761          logger.debug(
762              "Fetching project page and analyzing links: %s",
763              project_url,
764          )
765          html_page = self._link_collector.fetch_page(project_url)
766          if html_page is None:
767              return []
768  
769          page_links = list(parse_links(html_page))
770  
771          with indent_log():
772              package_links = self.evaluate_links(
773                  link_evaluator,
774                  links=page_links,
775              )
776  
777          return package_links

779      @functools.lru_cache(maxsize=None)
780      def find_all_candidates(self, project_name: str) -> List[InstallationCandidate]:
781          """Find all available InstallationCandidate for project_name 
782   
783          This checks index_urls and find_links. 
784          All versions found are returned as an InstallationCandidate list. 
785   
786          See LinkEvaluator.evaluate_link() for details on which files 
787          are accepted. 
788          """
789          link_evaluator = self.make_link_evaluator(project_name)
790  
791          collected_sources = self._link_collector.collect_sources(
792              project_name=project_name,
793              candidates_from_page=functools.partial(
794                  self.process_project_url,
795                  link_evaluator=link_evaluator,
796              ),
797          )
798  
799          page_candidates_it = itertools.chain.from_iterable(
800              source.page_candidates()
801              for sources in collected_sources
802              for source in sources
803              if source is not None
804          )
805          page_candidates = list(page_candidates_it)
806  
807          file_links_it = itertools.chain.from_iterable(
808              source.file_links()
809              for sources in collected_sources
810              for source in sources
811              if source is not None
812          )
813          file_candidates = self.evaluate_links(
814              link_evaluator,
815              sorted(file_links_it, reverse=True),
816          )
817  
818          if logger.isEnabledFor(logging.DEBUG) and file_candidates:
819              paths = [url_to_path(c.link.url) for c in file_candidates]
820              logger.debug("Local files found: %s", ", ".join(paths))
821  
822          # This is an intentional priority ordering
823          return file_candidates + page_candidates

829          hashes: Optional[Hashes] = None,

836              prefer_binary=candidate_prefs.prefer_binary,

842      @functools.lru_cache(maxsize=None)
843      def find_best_candidate(
844          self,
845          project_name: str,
846          specifier: Optional[specifiers.BaseSpecifier] = None,
847          hashes: Optional[Hashes] = None,
848      ) -> BestCandidateResult:
849          """Find matches for the given project and specifier. 
850   
851          :param specifier: An optional object implementing `filter` 
852              (e.g. `packaging.specifiers.SpecifierSet`) to filter applicable 
853              versions. 
854   
855          :return: A `BestCandidateResult` instance. 
856          """
857          candidates = self.find_all_candidates(project_name)
858          candidate_evaluator = self.make_candidate_evaluator(
859              project_name=project_name,
860              specifier=specifier,
861              hashes=hashes,
862          )
863          return candidate_evaluator.compute_best_candidate(candidates)

865      def find_requirement(
866          self, req: InstallRequirement, upgrade: bool
867      ) -> Optional[InstallationCandidate]:
868          """Try to find a Link matching req 
869   
870          Expects req, an InstallRequirement and upgrade, a boolean 
871          Returns a InstallationCandidate if found, 
872          Raises DistributionNotFound or BestVersionAlreadyInstalled otherwise 
873          """
874          hashes = req.hashes(trust_internet=False)
875          best_candidate_result = self.find_best_candidate(
876              req.name,
877              specifier=req.specifier,
878              hashes=hashes,
879          )
880          best_candidate = best_candidate_result.best_candidate
881  
882          installed_version: Optional[_BaseVersion] = None
883          if req.satisfied_by is not None:
884              installed_version = parse_version(req.satisfied_by.version)
885  
886          def _format_versions(cand_iter: Iterable[InstallationCandidate]) -> str:
887              # This repeated parse_version and str() conversion is needed to
888              # handle different vendoring sources from pip and pkg_resources.
889              # If we stop using the pkg_resources provided specifier and start
890              # using our own, we can drop the cast to str().
891              return (
892                  ", ".join(
893                      sorted(
894                          {str(c.version) for c in cand_iter},
895                          key=parse_version,
896                      )
897                  )
898                  or "none"
899              )
900  
901          if installed_version is None and best_candidate is None:
902              logger.critical(
903                  "Could not find a version that satisfies the requirement %s "
904                  "(from versions: %s)",
905                  req,
906                  _format_versions(best_candidate_result.iter_all()),
907              )
908  
909              raise DistributionNotFound(
910                  "No matching distribution found for {}".format(req)
911              )
912  
913          best_installed = False
914          if installed_version and (
915              best_candidate is None or best_candidate.version <= installed_version
916          ):
917              best_installed = True
918  
919          if not upgrade and installed_version is not None:
920              if best_installed:
921                  logger.debug(
922                      "Existing installed version (%s) is most up-to-date and "
923                      "satisfies requirement",
924                      installed_version,
925                  )
926              else:
927                  logger.debug(
928                      "Existing installed version (%s) satisfies requirement "
929                      "(most up-to-date version is %s)",
930                      installed_version,
931                      best_candidate.version,
932                  )
933              return None
934  
935          if best_installed:
936              # We have an existing version, and its the best version
937              logger.debug(
938                  "Installed version (%s) is most up-to-date (past versions: %s)",
939                  installed_version,
940                  _format_versions(best_candidate_result.iter_applicable()),
941              )
942              raise BestVersionAlreadyInstalled
943  
944          logger.debug(
945              "Using version %s (newest of versions: %s)",
946              best_candidate.version,
947              _format_versions(best_candidate_result.iter_applicable()),
948          )
949          return best_candidate

952  def _find_name_version_sep(fragment: str, canonical_name: str) -> int:
953      """Find the separator's index based on the package's canonical name. 
954   
955      :param fragment: A <package>+<version> filename "fragment" (stem) or 
956          egg fragment. 
957      :param canonical_name: The package's canonical name. 
958   
959      This function is needed since the canonicalized name does not necessarily 
960      have the same length as the egg info's name part. An example:: 
961   
962      >>> fragment = 'foo__bar-1.0' 
963      >>> canonical_name = 'foo-bar' 
964      >>> _find_name_version_sep(fragment, canonical_name) 
965      8 
966      """
967      # Project name and version must be separated by one single dash. Find all
968      # occurrences of dashes; if the string in front of it matches the canonical
969      # name, this is the one separating the name and version parts.
970      for i, c in enumerate(fragment):
971          if c != "-":
972              continue
973          if canonicalize_name(fragment[:i]) == canonical_name:
974              return i
975      raise ValueError(f"{fragment} does not match {canonical_name}")
976  
977  
978  def _extract_version_from_fragment(fragment: str, canonical_name: str) -> Optional[str]:
979      """Parse the version string from a <package>+<version> filename 
980      "fragment" (stem) or egg fragment. 
981   
982      :param fragment: The string to parse. E.g. foo-2.1 
983      :param canonical_name: The canonicalized name of the package this 
984          belongs to. 
985      """
986      try:
987          version_start = _find_name_version_sep(fragment, canonical_name) + 1
988      except ValueError:
989          return None
990      version = fragment[version_start:]
991      if not version:
992          return None
993      return version
