14   r'''Parse strings using a specification based on the Python format() syntax. 
15    
16      ``parse()`` is the opposite of ``format()`` 
17    
18   The module is set up to only export ``parse()``, ``search()``, ``findall()``, 
19   and ``with_pattern()`` when ``import \*`` is used: 
20    
21   >>> from parse import * 
22    
23   From there it's a simple thing to parse a string: 
24    
25   >>> parse("It's {}, I love it!", "It's spam, I love it!") 
26   <Result ('spam',) {}> 
27   >>> _[0] 
28   'spam' 
29    
30   Or to search a string for some pattern: 
31    
32   >>> search('Age: {:d}\n', 'Name: Rufus\nAge: 42\nColor: red\n') 
33   <Result (42,) {}> 
34    
35   Or find all the occurrences of some pattern in a string: 
36    
37   >>> ''.join(r.fixed[0] for r in findall(">{}<", "<p>the <b>bold</b> text</p>")) 
38   'the bold text' 
39    
40   If you're going to use the same pattern to match lots of strings you can 
41   compile it once: 
42    
43   >>> from parse import compile 
44   >>> p = compile("It's {}, I love it!") 
45   >>> print(p) 
46   <Parser "It's {}, I love it!"> 
47   >>> p.parse("It's spam, I love it!") 
48   <Result ('spam',) {}> 
49    
50   ("compile" is not exported for ``import *`` usage as it would override the 
51   built-in ``compile()`` function) 
52    
53   The default behaviour is to match strings case insensitively. You may match with 
54   case by specifying `case_sensitive=True`: 
55    
56   >>> parse('SPAM', 'spam', case_sensitive=True) is None 
57   True 
58    
59    
60   Format Syntax 
61   ------------- 
62    
63   A basic version of the `Format String Syntax`_ is supported with anonymous 
64   (fixed-position), named and formatted fields:: 
65    
66      {[field name]:[format spec]} 
67    
68   Field names must be a valid Python identifiers, including dotted names; 
69   element indexes imply dictionaries (see below for example). 
70    
71   Numbered fields are also not supported: the result of parsing will include 
72   the parsed fields in the order they are parsed. 
73    
74   The conversion of fields to types other than strings is done based on the 
75   type in the format specification, which mirrors the ``format()`` behaviour. 
76   There are no "!" field conversions like ``format()`` has. 
77    
78   Some simple parse() format string examples: 
79    
80   >>> parse("Bring me a {}", "Bring me a shrubbery") 
81   <Result ('shrubbery',) {}> 
82   >>> r = parse("The {} who say {}", "The knights who say Ni!") 
83   >>> print(r) 
84   <Result ('knights', 'Ni!') {}> 
85   >>> print(r.fixed) 
86   ('knights', 'Ni!') 
87   >>> r = parse("Bring out the holy {item}", "Bring out the holy hand grenade") 
88   >>> print(r) 
89   <Result () {'item': 'hand grenade'}> 
90   >>> print(r.named) 
91   {'item': 'hand grenade'} 
92   >>> print(r['item']) 
93   hand grenade 
94   >>> 'item' in r 
95   True 
96    
97   Note that `in` only works if you have named fields. Dotted names and indexes 
98   are possible though the application must make additional sense of the result: 
99    
100  >>> r = parse("Mmm, {food.type}, I love it!", "Mmm, spam, I love it!") 
101  >>> print(r) 
102  <Result () {'food.type': 'spam'}> 
103  >>> print(r.named) 
104  {'food.type': 'spam'} 
105  >>> print(r['food.type']) 
106  spam 
107  >>> r = parse("My quest is {quest[name]}", "My quest is to seek the holy grail!") 
108  >>> print(r) 
109  <Result () {'quest': {'name': 'to seek the holy grail!'}}> 
110  >>> print(r['quest']) 
111  {'name': 'to seek the holy grail!'} 
112  >>> print(r['quest']['name']) 
113  to seek the holy grail! 
114   
115  If the text you're matching has braces in it you can match those by including 
116  a double-brace ``{{`` or ``}}`` in your format string, just like format() does. 
117   
118   
119  Format Specification 
120  -------------------- 
121   
122  Most often a straight format-less ``{}`` will suffice where a more complex 
123  format specification might have been used. 
124   
125  Most of `format()`'s `Format Specification Mini-Language`_ is supported: 
126   
127     [[fill]align][0][width][.precision][type] 
128   
129  The differences between `parse()` and `format()` are: 
130   
131  - The align operators will cause spaces (or specified fill character) to be 
132    stripped from the parsed value. The width is not enforced; it just indicates 
133    there may be whitespace or "0"s to strip. 
134  - Numeric parsing will automatically handle a "0b", "0o" or "0x" prefix. 
135    That is, the "#" format character is handled automatically by d, b, o 
136    and x formats. For "d" any will be accepted, but for the others the correct 
137    prefix must be present if at all. 
138  - Numeric sign is handled automatically. 
139  - The thousands separator is handled automatically if the "n" type is used. 
140  - The types supported are a slightly different mix to the format() types.  Some 
141    format() types come directly over: "d", "n", "%", "f", "e", "b", "o" and "x". 
142    In addition some regular expression character group types "D", "w", "W", "s" 
143    and "S" are also available. 
144  - The "e" and "g" types are case-insensitive so there is not need for 
145    the "E" or "G" types. 
146   
147  ===== =========================================== ======== 
148  Type  Characters Matched                          Output 
149  ===== =========================================== ======== 
150  l     Letters (ASCII)                             str 
151  w     Letters, numbers and underscore             str 
152  W     Not letters, numbers and underscore         str 
153  s     Whitespace                                  str 
154  S     Non-whitespace                              str 
155  d     Digits (effectively integer numbers)        int 
156  D     Non-digit                                   str 
157  n     Numbers with thousands separators (, or .)  int 
158  %     Percentage (converted to value/100.0)       float 
159  f     Fixed-point numbers                         float 
160  F     Decimal numbers                             Decimal 
161  e     Floating-point numbers with exponent        float 
162        e.g. 1.1e-10, NAN (all case insensitive) 
163  g     General number format (either d, f or e)    float 
164  b     Binary numbers                              int 
165  o     Octal numbers                               int 
166  x     Hexadecimal numbers (lower and upper case)  int 
167  ti    ISO 8601 format date/time                   datetime 
168        e.g. 1972-01-20T10:21:36Z ("T" and "Z" 
169        optional) 
170  te    RFC2822 e-mail format date/time             datetime 
171        e.g. Mon, 20 Jan 1972 10:21:36 +1000 
172  tg    Global (day/month) format date/time         datetime 
173        e.g. 20/1/1972 10:21:36 AM +1:00 
174  ta    US (month/day) format date/time             datetime 
175        e.g. 1/20/1972 10:21:36 PM +10:30 
176  tc    ctime() format date/time                    datetime 
177        e.g. Sun Sep 16 01:03:52 1973 
178  th    HTTP log format date/time                   datetime 
179        e.g. 21/Nov/2011:00:07:11 +0000 
180  ts    Linux system log format date/time           datetime 
181        e.g. Nov  9 03:37:44 
182  tt    Time                                        time 
183        e.g. 10:21:36 PM -5:30 
184  ===== =========================================== ======== 
185   
186  Some examples of typed parsing with ``None`` returned if the typing 
187  does not match: 
188   
189  >>> parse('Our {:d} {:w} are...', 'Our 3 weapons are...') 
190  <Result (3, 'weapons') {}> 
191  >>> parse('Our {:d} {:w} are...', 'Our three weapons are...') 
192  >>> parse('Meet at {:tg}', 'Meet at 1/2/2011 11:00 PM') 
193  <Result (datetime.datetime(2011, 2, 1, 23, 0),) {}> 
194   
195  And messing about with alignment: 
196   
197  >>> parse('with {:>} herring', 'with     a herring') 
198  <Result ('a',) {}> 
199  >>> parse('spam {:^} spam', 'spam    lovely     spam') 
200  <Result ('lovely',) {}> 
201   
202  Note that the "center" alignment does not test to make sure the value is 
203  centered - it just strips leading and trailing whitespace. 
204   
205  Width and precision may be used to restrict the size of matched text 
206  from the input. Width specifies a minimum size and precision specifies 
207  a maximum. For example: 
208   
209  >>> parse('{:.2}{:.2}', 'look')           # specifying precision 
210  <Result ('lo', 'ok') {}> 
211  >>> parse('{:4}{:4}', 'look at that')     # specifying width 
212  <Result ('look', 'at that') {}> 
213  >>> parse('{:4}{:.4}', 'look at that')    # specifying both 
214  <Result ('look at ', 'that') {}> 
215  >>> parse('{:2d}{:2d}', '0440')           # parsing two contiguous numbers 
216  <Result (4, 40) {}> 
217   
218  Some notes for the date and time types: 
219   
220  - the presence of the time part is optional (including ISO 8601, starting 
221    at the "T"). A full datetime object will always be returned; the time 
222    will be set to 00:00:00. You may also specify a time without seconds. 
223  - when a seconds amount is present in the input fractions will be parsed 
224    to give microseconds. 
225  - except in ISO 8601 the day and month digits may be 0-padded. 
226  - the date separator for the tg and ta formats may be "-" or "/". 
227  - named months (abbreviations or full names) may be used in the ta and tg 
228    formats in place of numeric months. 
229  - as per RFC 2822 the e-mail format may omit the day (and comma), and the 
230    seconds but nothing else. 
231  - hours greater than 12 will be happily accepted. 
232  - the AM/PM are optional, and if PM is found then 12 hours will be added 
233    to the datetime object's hours amount - even if the hour is greater 
234    than 12 (for consistency.) 
235  - in ISO 8601 the "Z" (UTC) timezone part may be a numeric offset 
236  - timezones are specified as "+HH:MM" or "-HH:MM". The hour may be one or two 
237    digits (0-padded is OK.) Also, the ":" is optional. 
238  - the timezone is optional in all except the e-mail format (it defaults to 
239    UTC.) 
240  - named timezones are not handled yet. 
241   
242  Note: attempting to match too many datetime fields in a single parse() will 
243  currently result in a resource allocation issue. A TooManyFields exception 
244  will be raised in this instance. The current limit is about 15. It is hoped 
245  that this limit will be removed one day.

248    http://docs.python.org/library/string.html#format-string-syntax 
249  .. _`Format Specification Mini-Language`: 
250    http://docs.python.org/library/string.html#format-specification-mini-language

253  Result and Match Objects 
254  ------------------------ 
255   
256  The result of a ``parse()`` and ``search()`` operation is either ``None`` (no match), a 
257  ``Result`` instance or a ``Match`` instance if ``evaluate_result`` is False. 
258   
259  The ``Result`` instance has three attributes: 
260   
261  fixed 
262     A tuple of the fixed-position, anonymous fields extracted from the input. 
263  named 
264     A dictionary of the named fields extracted from the input. 
265  spans 
266     A dictionary mapping the names and fixed position indices matched to a 
267     2-tuple slice range of where the match occurred in the input. 
268     The span does not include any stripped padding (alignment or width). 
269   
270  The ``Match`` instance has one method: 
271   
272  evaluate_result() 
273     Generates and returns a ``Result`` instance for this ``Match`` object.

277  Custom Type Conversions 
278  ----------------------- 
279   
280  If you wish to have matched fields automatically converted to your own type you 
281  may pass in a dictionary of type conversion information to ``parse()`` and 
282  ``compile()``. 
283   
284  The converter will be passed the field string matched. Whatever it returns 
285  will be substituted in the ``Result`` instance for that field. 
286   
287  Your custom type conversions may override the builtin types if you supply one 
288  with the same identifier. 
289   
290  >>> def shouty(string): 
291  ...    return string.upper() 
292  ... 
293  >>> parse('{:shouty} world', 'hello world', dict(shouty=shouty)) 
294  <Result ('HELLO',) {}> 
295   
296  If the type converter has the optional ``pattern`` attribute, it is used as 
297  regular expression for better pattern matching (instead of the default one). 
298   
299  >>> def parse_number(text): 
300  ...    return int(text) 
301  >>> parse_number.pattern = r'\d+' 
302  >>> parse('Answer: {number:Number}', 'Answer: 42', dict(Number=parse_number)) 
303  <Result () {'number': 42}> 
304  >>> _ = parse('Answer: {:Number}', 'Answer: Alice', dict(Number=parse_number)) 
305  >>> assert _ is None, "MISMATCH" 
306   
307  You can also use the ``with_pattern(pattern)`` decorator to add this 
308  information to a type converter function: 
309   
310  >>> from parse import with_pattern 
311  >>> @with_pattern(r'\d+') 
312  ... def parse_number(text): 
313  ...    return int(text) 
314  >>> parse('Answer: {number:Number}', 'Answer: 42', dict(Number=parse_number)) 
315  <Result () {'number': 42}> 
316   
317  A more complete example of a custom type might be: 
318   
319  >>> yesno_mapping = { 
320  ...     "yes":  True,   "no":    False, 
321  ...     "on":   True,   "off":   False, 
322  ...     "true": True,   "false": False, 
323  ... } 
324  >>> @with_pattern(r"|".join(yesno_mapping)) 
325  ... def parse_yesno(text): 
326  ...     return yesno_mapping[text.lower()] 
327   
328   
329  If the type converter ``pattern`` uses regex-grouping (with parenthesis), 
330  you should indicate this by using the optional ``regex_group_count`` parameter 
331  in the ``with_pattern()`` decorator: 
332   
333  >>> @with_pattern(r'((\d+))', regex_group_count=2) 
334  ... def parse_number2(text): 
335  ...    return int(text) 
336  >>> parse('Answer: {:Number2} {:Number2}', 'Answer: 42 43', dict(Number2=parse_number2)) 
337  <Result (42, 43) {}> 
338   
339  Otherwise, this may cause parsing problems with unnamed/fixed parameters. 
340   
341   
342  Potential Gotchas 
343  ----------------- 
344   
345  `parse()` will always match the shortest text necessary (from left to right) 
346  to fulfil the parse pattern, so for example: 
347   
348  >>> pattern = '{dir1}/{dir2}' 
349  >>> data = 'root/parent/subdir' 
350  >>> sorted(parse(pattern, data).named.items()) 
351  [('dir1', 'root'), ('dir2', 'parent/subdir')] 
352   
353  So, even though `{'dir1': 'root/parent', 'dir2': 'subdir'}` would also fit 
354  the pattern, the actual match represents the shortest successful match for 
355  `dir1`.

447  def with_pattern(pattern, regex_group_count=None):
448      r"""Attach a regular expression pattern matcher to a custom type converter 
449      function. 
450   
451      This annotates the type converter with the :attr:`pattern` attribute. 
452   
453      EXAMPLE: 
454          >>> import parse 
455          >>> @parse.with_pattern(r"\d+") 
456          ... def parse_number(text): 
457          ...     return int(text) 
458   
459      is equivalent to: 
460   
461          >>> def parse_number(text): 
462          ...     return int(text) 
463          >>> parse_number.pattern = r"\d+" 
464   
465      :param pattern: regular expression pattern (as text) 
466      :param regex_group_count: Indicates how many regex-groups are in pattern. 
467      :return: wrapped function 
468      """
469      def decorator(func):
470          func.pattern = pattern
471          func.regex_group_count = regex_group_count
472          return func
473      return decorator

476  def int_convert(base):
477      '''Convert a string to an integer. 
478   
479      The string may start with a sign. 
480   
481      It may be of a base other than 10. 
482   
483      If may start with a base indicator, 0#nnnn, which we assume should 
484      override the specified base. 
485   
486      It may also have other non-numeric characters that we can ignore. 
487      '''
488      CHARS = '0123456789abcdefghijklmnopqrstuvwxyz'
489  
490      def f(string, match, base=base):
491          if string[0] == '-':
492              sign = -1
493          else:
494              sign = 1
495  
496          if string[0] == '0' and len(string) > 2:
497              if string[1] in 'bB':
498                  base = 2
499              elif string[1] in 'oO':
500                  base = 8
501              elif string[1] in 'xX':
502                  base = 16
503              else:
504                  # just go with the base specifed
505                  pass
506  
507          chars = CHARS[:base]
508          string = re.sub('[^%s]' % chars, '', string.lower())
509          return sign * int(string, base)
510      return f

517  class FixedTzOffset(tzinfo):
518      """Fixed offset in minutes east from UTC. 
519      """
520      ZERO = timedelta(0)
521  
522      def __init__(self, offset, name):
523          self._offset = timedelta(minutes=offset)
524          self._name = name
525  
526      def __repr__(self):
527          return '<%s %s %s>' % (self.__class__.__name__, self._name,
528              self._offset)
529  
530      def utcoffset(self, dt):
531          return self._offset
532  
533      def tzname(self, dt):
534          return self._name
535  
536      def dst(self, dt):
537          return self.ZERO
538  
539      def __eq__(self, other):
540          return self._name == other._name and self._offset == other._offset

565  def date_convert(string, match, ymd=None, mdy=None, dmy=None,
566          d_m_y=None, hms=None, am=None, tz=None, mm=None, dd=None):
567      '''Convert the incoming string containing some date / time info into a 
568      datetime instance. 
569      '''
570      groups = match.groups()
571      time_only = False
572      if mm and dd:
573          y=datetime.today().year
574          m=groups[mm]
575          d=groups[dd]
576      elif ymd is not None:
577          y, m, d = re.split(r'[-/\s]', groups[ymd])
578      elif mdy is not None:
579          m, d, y = re.split(r'[-/\s]', groups[mdy])
580      elif dmy is not None:
581          d, m, y = re.split(r'[-/\s]', groups[dmy])
582      elif d_m_y is not None:
583          d, m, y = d_m_y
584          d = groups[d]
585          m = groups[m]
586          y = groups[y]
587      else:
588          time_only = True
589  
590      H = M = S = u = 0
591      if hms is not None and groups[hms]:
592          t = groups[hms].split(':')
593          if len(t) == 2:
594              H, M = t
595          else:
596              H, M, S = t
597              if '.' in S:
598                  S, u = S.split('.')
599                  u = int(float('.' + u) * 1000000)
600              S = int(S)
601          H = int(H)
602          M = int(M)
603  
604      if am is not None:
605          am = groups[am]
606          if am:
607              am = am.strip()
608          if am == 'AM' and H == 12:
609              # correction for "12" hour functioning as "0" hour: 12:15 AM = 00:15 by 24 hr clock
610              H -= 12
611          elif am == 'PM' and H == 12:
612              # no correction needed: 12PM is midday, 12:00 by 24 hour clock
613              pass
614          elif am == 'PM':
615              H += 12
616  
617      if tz is not None:
618          tz = groups[tz]
619      if tz == 'Z':
620          tz = FixedTzOffset(0, 'UTC')
621      elif tz:
622          tz = tz.strip()
623          if tz.isupper():
624              # TODO use the awesome python TZ module?
625              pass
626          else:
627              sign = tz[0]
628              if ':' in tz:
629                  tzh, tzm = tz[1:].split(':')
630              elif len(tz) == 4:  # 'snnn'
631                  tzh, tzm = tz[1], tz[2:4]
632              else:
633                  tzh, tzm = tz[1:3], tz[3:5]
634              offset = int(tzm) + int(tzh) * 60
635              if sign == '-':
636                  offset = -offset
637              tz = FixedTzOffset(offset, tz)
638  
639      if time_only:
640          d = time(H, M, S, u, tzinfo=tz)
641      else:
642          y = int(y)
643          if m.isdigit():
644              m = int(m)
645          else:
646              m = MONTHS_MAP[m]
647          d = int(d)
648          d = datetime(y, m, d, H, M, S, u, tzinfo=tz)
649  
650      return d

662  # note: I don't use r'' here because Sublime Text 2 syntax highlight has a fit
663  REGEX_SAFETY = re.compile(r'([?\\\\.[\]()*+\^$!\|])')

667      ['t' + c for c in 'ieahgcts'])

670  def extract_format(format, extra_types):
671      '''Pull apart the format [[fill]align][0][width][.precision][type] 
672      '''
673      fill = align = None
674      if format[0] in '<>=^':
675          align = format[0]
676          format = format[1:]
677      elif len(format) > 1 and format[1] in '<>=^':
678          fill = format[0]
679          align = format[1]
680          format = format[2:]
681  
682      zero = False
683      if format and format[0] == '0':
684          zero = True
685          format = format[1:]
686  
687      width = ''
688      while format:
689          if not format[0].isdigit():
690              break
691          width += format[0]
692          format = format[1:]
693  
694      if format.startswith('.'):
695          # Precision isn't needed but we need to capture it so that
696          # the ValueError isn't raised.
697          format = format[1:]  # drop the '.'
698          precision = ''
699          while format:
700              if not format[0].isdigit():
701                  break
702              precision += format[0]
703              format = format[1:]
704  
705      # the rest is the type, if present
706      type = format
707      if type and type not in ALLOWED_TYPES and type not in extra_types:
708          raise ValueError('format spec %r not recognised' % type)
709  
710      return locals()

713  PARSE_RE = re.compile(r"""({{|}}|{\w*(?:(?:\.\w+)|(?:\[[^\]]+\]))*(?::[^}]+)?})""")

716  class Parser(object):
717      '''Encapsulate a format string that may be used to parse other strings. 
718      '''
719      def __init__(self, format, extra_types=None, case_sensitive=False):
720          # a mapping of a name as in {hello.world} to a regex-group compatible
721          # name, like hello__world Its used to prevent the transformation of
722          # name-to-group and group to name to fail subtly, such as in:
723          # hello_.world-> hello___world->hello._world
724          self._group_to_name_map = {}
725          # also store the original field name to group name mapping to allow
726          # multiple instances of a name in the format string
727          self._name_to_group_map = {}
728          # and to sanity check the repeated instances store away the first
729          # field type specification for the named field
730          self._name_types = {}
731  
732          self._format = format
733          if extra_types is None:
734              extra_types = {}
735          self._extra_types = extra_types
736          if case_sensitive:
737              self._re_flags = re.DOTALL
738          else:
739              self._re_flags = re.IGNORECASE | re.DOTALL
740          self._fixed_fields = []
741          self._named_fields = []
742          self._group_index = 0
743          self._type_conversions = {}
744          self._expression = self._generate_expression()
745          self.__search_re = None
746          self.__match_re = None
747  
748          log.debug('format %r -> %r', format, self._expression)
749  
750      def __repr__(self):
751          if len(self._format) > 20:
752              return '<%s %r>' % (self.__class__.__name__,
753                  self._format[:17] + '...')
754          return '<%s %r>' % (self.__class__.__name__, self._format)
755  
756      @property
757      def _search_re(self):
758          if self.__search_re is None:
759              try:
760                  self.__search_re = re.compile(self._expression, self._re_flags)
761              except AssertionError:
762                  # access error through sys to keep py3k and backward compat
763                  e = str(sys.exc_info()[1])
764                  if e.endswith('this version only supports 100 named groups'):
765                      raise TooManyFields('sorry, you are attempting to parse '
766                          'too many complex fields')
767          return self.__search_re
768  
769      @property
770      def _match_re(self):
771          if self.__match_re is None:
772              expression = r'^%s$' % self._expression
773              try:
774                  self.__match_re = re.compile(expression, self._re_flags)
775              except AssertionError:
776                  # access error through sys to keep py3k and backward compat
777                  e = str(sys.exc_info()[1])
778                  if e.endswith('this version only supports 100 named groups'):
779                      raise TooManyFields('sorry, you are attempting to parse '
780                          'too many complex fields')
781              except re.error:
782                  raise NotImplementedError("Group names (e.g. (?P<name>) can "
783                      "cause failure, as they are not escaped properly: '%s'" %
784                      expression)
785          return self.__match_re

787      def parse(self, string, evaluate_result=True):
788          '''Match my format to the string exactly. 
789   
790          Return a Result or Match instance or None if there's no match. 
791          '''
792          m = self._match_re.match(string)
793          if m is None:
794              return None
795  
796          if evaluate_result:
797              return self.evaluate_result(m)
798          else:
799              return Match(self, m)

801      def search(self, string, pos=0, endpos=None, evaluate_result=True):
802          '''Search the string for my format. 
803   
804          Optionally start the search at "pos" character index and limit the 
805          search to a maximum index of endpos - equivalent to 
806          search(string[:endpos]). 
807   
808          If the ``evaluate_result`` argument is set to ``False`` a 
809          Match instance is returned instead of the actual Result instance. 
810   
811          Return either a Result instance or None if there's no match. 
812          '''
813          if endpos is None:
814              endpos = len(string)
815          m = self._search_re.search(string, pos, endpos)
816          if m is None:
817              return None
818  
819          if evaluate_result:
820              return self.evaluate_result(m)
821          else:
822              return Match(self, m)
823  
824      def findall(self, string, pos=0, endpos=None, extra_types=None, evaluate_result=True):
825          '''Search "string" for all occurrences of "format". 
826   
827          Optionally start the search at "pos" character index and limit the 
828          search to a maximum index of endpos - equivalent to 
829          search(string[:endpos]). 
830   
831          Returns an iterator that holds Result or Match instances for each format match 
832          found. 
833          '''
834          if endpos is None:
835              endpos = len(string)
836          return ResultIterator(self, string, pos, endpos, evaluate_result=evaluate_result)

838      def _expand_named_fields(self, named_fields):
839          result = {}
840          for field, value in named_fields.items():
841              # split 'aaa[bbb][ccc]...' into 'aaa' and '[bbb][ccc]...'
842              basename, subkeys = re.match(r'([^\[]+)(.*)', field).groups()
843  
844              # create nested dictionaries {'aaa': {'bbb': {'ccc': ...}}}
845              d = result
846              k = basename
847  
848              if subkeys:
849                  for subkey in re.findall(r'\[[^\]]+\]', subkeys):
850                      d = d.setdefault(k,{})
851                      k = subkey[1:-1]
852  
853              # assign the value to the last key
854              d[k] = value
855  
856          return result
857  
858      def evaluate_result(self, m):
859          '''Generate a Result instance for the given regex match object'''
860          # ok, figure the fixed fields we've pulled out and type convert them
861          fixed_fields = list(m.groups())
862          for n in self._fixed_fields:
863              if n in self._type_conversions:
864                  fixed_fields[n] = self._type_conversions[n](fixed_fields[n], m)
865          fixed_fields = tuple(fixed_fields[n] for n in self._fixed_fields)
866  
867          # grab the named fields, converting where requested
868          groupdict = m.groupdict()
869          named_fields = {}
870          name_map = {}
871          for k in self._named_fields:
872              korig = self._group_to_name_map[k]
873              name_map[korig] = k
874              if k in self._type_conversions:
875                  value = self._type_conversions[k](groupdict[k], m)
876              else:
877                  value = groupdict[k]
878  
879              named_fields[korig] = value
880  
881          # now figure the match spans
882          spans = dict((n, m.span(name_map[n])) for n in named_fields)
883          spans.update((i, m.span(n + 1))
884              for i, n in enumerate(self._fixed_fields))
885  
886          # and that's our result
887          return Result(fixed_fields, self._expand_named_fields(named_fields), spans)
888  
889      def _regex_replace(self, match):
890          return '\\' + match.group(1)
891  
892      def _generate_expression(self):
893          # turn my _format attribute into the _expression attribute
894          e = []
895          for part in PARSE_RE.split(self._format):
896              if not part:
897                  continue
898              elif part == '{{':
899                  e.append(r'\{')
900              elif part == '}}':
901                  e.append(r'\}')
902              elif part[0] == '{' and part[-1] == '}':
903                  # this will be a braces-delimited field to handle
904                  e.append(self._handle_field(part))
905              else:
906                  # just some text to match
907                  e.append(REGEX_SAFETY.sub(self._regex_replace, part))
908          return ''.join(e)
909  
910      def _to_group_name(self, field):
911          # return a version of field which can be used as capture group, even
912          # though it might contain '.'
913          group = field.replace('.', '_').replace('[', '_').replace(']', '_')
914  
915          # make sure we don't collide ("a.b" colliding with "a_b")
916          n = 1
917          while group in self._group_to_name_map:
918              n += 1
919              if '.' in field:
920                  group = field.replace('.', '_' * n)
921              elif '_' in field:
922                  group = field.replace('_', '_' * n)
923              else:
924                  raise KeyError('duplicated group name %r' % (field,))
925  
926          # save off the mapping
927          self._group_to_name_map[group] = field
928          self._name_to_group_map[field] = group
929          return group
930  
931      def _handle_field(self, field):
932          # first: lose the braces
933          field = field[1:-1]
934  
935          # now figure whether this is an anonymous or named field, and whether
936          # there's any format specification
937          format = ''
938          if field and field[0].isalpha():
939              if ':' in field:
940                  name, format = field.split(':')
941              else:
942                  name = field
943              if name in self._name_to_group_map:
944                  if self._name_types[name] != format:
945                      raise RepeatedNameError('field type %r for field "%s" '
946                          'does not match previous seen type %r' % (format,
947                          name, self._name_types[name]))
948                  group = self._name_to_group_map[name]
949                  # match previously-seen value
950                  return r'(?P=%s)' % group
951              else:
952                  group = self._to_group_name(name)
953                  self._name_types[name] = format
954              self._named_fields.append(group)
955              # this will become a group, which must not contain dots
956              wrap = r'(?P<%s>%%s)' % group
957          else:
958              self._fixed_fields.append(self._group_index)
959              wrap = r'(%s)'
960              if ':' in field:
961                  format = field[1:]
962              group = self._group_index
963  
964          # simplest case: no type specifier ({} or {name})
965          if not format:
966              self._group_index += 1
967              return wrap % r'.+?'
968  
969          # decode the format specification
970          format = extract_format(format, self._extra_types)
971  
972          # figure type conversions, if any
973          type = format['type']
974          is_numeric = type and type in 'n%fegdobh'
975          if type in self._extra_types:
976              type_converter = self._extra_types[type]
977              s = getattr(type_converter, 'pattern', r'.+?')
978              regex_group_count = getattr(type_converter, 'regex_group_count', 0)
979              if regex_group_count is None:
980                  regex_group_count = 0
981              self._group_index += regex_group_count
982  
983              def f(string, m):
984                  return type_converter(string)
985              self._type_conversions[group] = f
986          elif type == 'n':
987              s = r'\d{1,3}([,.]\d{3})*'
988              self._group_index += 1
989              self._type_conversions[group] = int_convert(10)
990          elif type == 'b':
991              s = r'(0[bB])?[01]+'
992              self._type_conversions[group] = int_convert(2)
993              self._group_index += 1
994          elif type == 'o':
995              s = r'(0[oO])?[0-7]+'
996              self._type_conversions[group] = int_convert(8)
997              self._group_index += 1
998          elif type == 'x':
999              s = r'(0[xX])?[0-9a-fA-F]+'
1000             self._type_conversions[group] = int_convert(16)
1001             self._group_index += 1
1002         elif type == '%':
1003             s = r'\d+(\.\d+)?%'
1004             self._group_index += 1
1005             self._type_conversions[group] = percentage
1006         elif type == 'f':
1007             s = r'\d+\.\d+'
1008             self._type_conversions[group] = lambda s, m: float(s)
1009         elif type == 'F':
1010             s = r'\d+\.\d+'
1011             self._type_conversions[group] = lambda s, m: Decimal(s)
1012         elif type == 'e':
1013             s = r'\d+\.\d+[eE][-+]?\d+|nan|NAN|[-+]?inf|[-+]?INF'
1014             self._type_conversions[group] = lambda s, m: float(s)
1015         elif type == 'g':
1016             s = r'\d+(\.\d+)?([eE][-+]?\d+)?|nan|NAN|[-+]?inf|[-+]?INF'
1017             self._group_index += 2
1018             self._type_conversions[group] = lambda s, m: float(s)
1019         elif type == 'd':
1020             if format.get('width'):
1021                 width = r'{1,%s}' % int(format['width'])
1022             else:
1023                 width = '+'
1024             s = r'\d{w}|0[xX][0-9a-fA-F]{w}|0[bB][01]{w}|0[oO][0-7]{w}'.format(w=width)
1025             self._type_conversions[group] = int_convert(10)
1026         elif type == 'ti':
1027             s = r'(\d{4}-\d\d-\d\d)((\s+|T)%s)?(Z|\s*[-+]\d\d:?\d\d)?' % \
1028                 TIME_PAT
1029             n = self._group_index
1030             self._type_conversions[group] = partial(date_convert, ymd=n + 1,
1031                 hms=n + 4, tz=n + 7)
1032             self._group_index += 7
1033         elif type == 'tg':
1034             s = r'(\d{1,2}[-/](\d{1,2}|%s)[-/]\d{4})(\s+%s)?%s?%s?' % (
1035                 ALL_MONTHS_PAT, TIME_PAT, AM_PAT, TZ_PAT)
1036             n = self._group_index
1037             self._type_conversions[group] = partial(date_convert, dmy=n + 1,
1038                 hms=n + 5, am=n + 8, tz=n + 9)
1039             self._group_index += 9
1040         elif type == 'ta':
1041             s = r'((\d{1,2}|%s)[-/]\d{1,2}[-/]\d{4})(\s+%s)?%s?%s?' % (
1042                 ALL_MONTHS_PAT, TIME_PAT, AM_PAT, TZ_PAT)
1043             n = self._group_index
1044             self._type_conversions[group] = partial(date_convert, mdy=n + 1,
1045                 hms=n + 5, am=n + 8, tz=n + 9)
1046             self._group_index += 9
1047         elif type == 'te':
1048             # this will allow microseconds through if they're present, but meh
1049             s = r'(%s,\s+)?(\d{1,2}\s+%s\s+\d{4})\s+%s%s' % (DAYS_PAT,
1050                 MONTHS_PAT, TIME_PAT, TZ_PAT)
1051             n = self._group_index
1052             self._type_conversions[group] = partial(date_convert, dmy=n + 3,
1053                 hms=n + 5, tz=n + 8)
1054             self._group_index += 8
1055         elif type == 'th':
1056             # slight flexibility here from the stock Apache format
1057             s = r'(\d{1,2}[-/]%s[-/]\d{4}):%s%s' % (MONTHS_PAT, TIME_PAT,
1058                 TZ_PAT)
1059             n = self._group_index
1060             self._type_conversions[group] = partial(date_convert, dmy=n + 1,
1061                 hms=n + 3, tz=n + 6)
1062             self._group_index += 6
1063         elif type == 'tc':
1064             s = r'(%s)\s+%s\s+(\d{1,2})\s+%s\s+(\d{4})' % (
1065                 DAYS_PAT, MONTHS_PAT, TIME_PAT)
1066             n = self._group_index
1067             self._type_conversions[group] = partial(date_convert,
1068                 d_m_y=(n + 4, n + 3, n + 8), hms=n + 5)
1069             self._group_index += 8
1070         elif type == 'tt':
1071             s = r'%s?%s?%s?' % (TIME_PAT, AM_PAT, TZ_PAT)
1072             n = self._group_index
1073             self._type_conversions[group] = partial(date_convert, hms=n + 1,
1074                 am=n + 4, tz=n + 5)
1075             self._group_index += 5
1076         elif type == 'ts':
1077             s = r'%s(\s+)(\d+)(\s+)(\d{1,2}:\d{1,2}:\d{1,2})?' % MONTHS_PAT
1078             n = self._group_index
1079             self._type_conversions[group] = partial(date_convert, mm=n+1, dd=n+3,
1080                 hms=n + 5)
1081             self._group_index += 5
1082         elif type == 'l':
1083             s = r'[A-Za-z]+'
1084         elif type:
1085             s = r'\%s+' % type
1086         elif format.get('precision'):
1087             if format.get('width'):
1088                 s = r'.{%s,%s}?' % (format['width'], format['precision'])
1089             else:
1090                 s = r'.{1,%s}?' % format['precision']
1091         elif format.get('width'):
1092             s = r'.{%s,}?' % format['width']
1093         else:
1094             s = r'.+?'
1095 
1096         align = format['align']
1097         fill = format['fill']
1098 
1099         # handle some numeric-specific things like fill and sign
1100         if is_numeric:
1101             # prefix with something (align "=" trumps zero)
1102             if align == '=':
1103                 # special case - align "=" acts like the zero above but with
1104                 # configurable fill defaulting to "0"
1105                 if not fill:
1106                     fill = '0'
1107                 s = r'%s*' % fill + s
1108 
1109             # allow numbers to be prefixed with a sign
1110             s = r'[-+ ]?' + s
1111 
1112         if not fill:
1113             fill = ' '
1114 
1115         # Place into a group now - this captures the value we want to keep.
1116         # Everything else from now is just padding to be stripped off
1117         if wrap:
1118             s = wrap % s
1119             self._group_index += 1
1120 
1121         if format['width']:
1122             # all we really care about is that if the format originally
1123             # specified a width then there will probably be padding - without
1124             # an explicit alignment that'll mean right alignment with spaces
1125             # padding
1126             if not align:
1127                 align = '>'
1128 
1129         if fill in r'.\+?*[](){}^$':
1130             fill = '\\' + fill
1131 
1132         # align "=" has been handled
1133         if align == '<':
1134             s = '%s%s*' % (s, fill)
1135         elif align == '>':
1136             s = '%s*%s' % (fill, s)
1137         elif align == '^':
1138             s = '%s*%s%s*' % (fill, s, fill)
1139 
1140         return s

1143 class Result(object):
1144     '''The result of a parse() or search(). 
1145  
1146     Fixed results may be looked up using `result[index]`. 
1147  
1148     Named results may be looked up using `result['name']`. 
1149  
1150     Named results may be tested for existence using `'name' in result`. 
1151     '''
1152     def __init__(self, fixed, named, spans):
1153         self.fixed = fixed
1154         self.named = named
1155         self.spans = spans
1156 
1157     def __getitem__(self, item):
1158         if isinstance(item, int):
1159             return self.fixed[item]
1160         return self.named[item]
1161 
1162     def __repr__(self):
1163         return '<%s %r %r>' % (self.__class__.__name__, self.fixed,
1164             self.named)
1165 
1166     def __contains__(self, name):
1167         return name in self.named

1177         self.parser = parser

1187

1189     '''

1215 def parse(format, string, extra_types=None, evaluate_result=True, case_sensitive=False):
1216     '''Using "format" attempt to pull values from "string". 
1217  
1218     The format must match the string contents exactly. If the value 
1219     you're looking for is instead just a part of the string use 
1220     search(). 
1221
