1    """A PEP 517 interface to setuptools 
2     
3    Previously, when a user or a command line tool (let's call it a "frontend") 
4    needed to make a request of setuptools to take a certain action, for 
5    example, generating a list of installation requirements, the frontend would 
6    would call "setup.py egg_info" or "setup.py bdist_wheel" on the command line. 
7     
8    PEP 517 defines a different method of interfacing with setuptools. Rather 
9    than calling "setup.py" directly, the frontend should: 
10    
11     1. Set the current directory to the directory with a setup.py file 
12     2. Import this module into a safe python interpreter (one in which 
13        setuptools can potentially set global variables or crash hard). 
14     3. Call one of the functions defined in PEP 517. 
15    
16   What each function does is defined in PEP 517. However, here is a "casual" 
17   definition of the functions (this definition should not be relied on for 
18   bug reports or API stability): 
19    
20     - `build_wheel`: build a wheel in the folder and return the basename 
21     - `get_requires_for_build_wheel`: get the `setup_requires` to build 
22     - `prepare_metadata_for_build_wheel`: get the `install_requires` 
23     - `build_sdist`: build an sdist in the folder and return the basename 
24     - `get_requires_for_build_sdist`: get the `setup_requires` to build 
25    
26   Again, this is not a formal definition! Just a "taste" of the module. 
27   """

51   class SetupRequirementsError(BaseException):
52       def __init__(self, specifiers):
53           self.specifiers = specifiers
54   
55   
56   class Distribution(setuptools.dist.Distribution):
57       def fetch_build_eggs(self, specifiers):
58           specifier_list = list(map(str, parse_requirements(specifiers)))
59   
60           raise SetupRequirementsError(specifier_list)
61   
62       @classmethod
63       @contextlib.contextmanager
64       def patch(cls):
65           """ 
66           Replace 
67           distutils.dist.Distribution with this class 
68           for the duration of this context. 
69           """
70           orig = distutils.core.Distribution
71           distutils.core.Distribution = cls
72           try:
73               yield
74           finally:
75               distutils.core.Distribution = orig
76   
77   
78   @contextlib.contextmanager
79   def no_install_setup_requires():
80       """Temporarily disable installing setup_requires 
81    
82       Under PEP 517, the backend reports build dependencies to the frontend, 
83       and the frontend is responsible for ensuring they're installed. 
84       So setuptools (acting as a backend) should not try to install them. 
85       """
86       orig = setuptools._install_setup_requires
87       setuptools._install_setup_requires = lambda attrs: None
88       try:
89           yield
90       finally:
91           setuptools._install_setup_requires = orig
92

94   def _get_immediate_subdirectories(a_dir):
95       return [name for name in os.listdir(a_dir)
96               if os.path.isdir(os.path.join(a_dir, name))]
97   
98   
99   def _file_with_extension(directory, extension):
100      matching = (
101          f for f in os.listdir(directory)
102          if f.endswith(extension)
103      )
104      try:
105          file, = matching
106      except ValueError:
107          raise ValueError(
108              'No distribution was found. Ensure that `setup.py` '
109              'is not empty and that it calls `setup()`.')
110      return file
111  
112  
113  def _open_setup_script(setup_script):
114      if not os.path.exists(setup_script):
115          # Supply a default setup.py
116          return io.StringIO(u"from setuptools import setup; setup()")
117  
118      return getattr(tokenize, 'open', open)(setup_script)

121  class _BuildMetaBackend(object):
122  
123      def _fix_config(self, config_settings):
124          config_settings = config_settings or {}
125          config_settings.setdefault('--global-option', [])
126          return config_settings
127  
128      def _get_build_requires(self, config_settings, requirements):
129          config_settings = self._fix_config(config_settings)
130  
131          sys.argv = sys.argv[:1] + ['egg_info'] + \
132              config_settings["--global-option"]
133          try:
134              with Distribution.patch():
135                  self.run_setup()
136          except SetupRequirementsError as e:
137              requirements += e.specifiers
138  
139          return requirements
140  
141      def run_setup(self, setup_script='setup.py'):
142          # Note that we can reuse our build directory between calls
143          # Correctness comes first, then optimization later
144          __file__ = setup_script
145          __name__ = '__main__'
146  
147          with _open_setup_script(__file__) as f:
148              code = f.read().replace(r'\r\n', r'\n')
149  
150          exec(compile(code, __file__, 'exec'), locals())
151  
152      def get_requires_for_build_wheel(self, config_settings=None):
153          config_settings = self._fix_config(config_settings)
154          return self._get_build_requires(
155              config_settings, requirements=['wheel'])
156  
157      def get_requires_for_build_sdist(self, config_settings=None):
158          config_settings = self._fix_config(config_settings)
159          return self._get_build_requires(config_settings, requirements=[])
160  
161      def prepare_metadata_for_build_wheel(self, metadata_directory,
162                                           config_settings=None):
163          sys.argv = sys.argv[:1] + [
164              'dist_info', '--egg-base', metadata_directory]
165          with no_install_setup_requires():
166              self.run_setup()
167  
168          dist_info_directory = metadata_directory
169          while True:
170              dist_infos = [f for f in os.listdir(dist_info_directory)
171                            if f.endswith('.dist-info')]
172  
173              if (
174                  len(dist_infos) == 0 and
175                  len(_get_immediate_subdirectories(dist_info_directory)) == 1
176              ):
177  
178                  dist_info_directory = os.path.join(
179                      dist_info_directory, os.listdir(dist_info_directory)[0])
180                  continue
181  
182              assert len(dist_infos) == 1
183              break
184  
185          # PEP 517 requires that the .dist-info directory be placed in the
186          # metadata_directory. To comply, we MUST copy the directory to the root
187          if dist_info_directory != metadata_directory:
188              shutil.move(
189                  os.path.join(dist_info_directory, dist_infos[0]),
190                  metadata_directory)
191              shutil.rmtree(dist_info_directory, ignore_errors=True)
192  
193          return dist_infos[0]
194  
195      def _build_with_temp_dir(self, setup_command, result_extension,
196                               result_directory, config_settings):
197          config_settings = self._fix_config(config_settings)
198          result_directory = os.path.abspath(result_directory)
199  
200          # Build in a temporary directory, then copy to the target.
201          os.makedirs(result_directory, exist_ok=True)
202          with tempfile.TemporaryDirectory(dir=result_directory) as tmp_dist_dir:
203              sys.argv = (sys.argv[:1] + setup_command +
204                          ['--dist-dir', tmp_dist_dir] +
205                          config_settings["--global-option"])
206              with no_install_setup_requires():
207                  self.run_setup()
208  
209              result_basename = _file_with_extension(
210                  tmp_dist_dir, result_extension)
211              result_path = os.path.join(result_directory, result_basename)
212              if os.path.exists(result_path):
213                  # os.rename will fail overwriting on non-Unix.
214                  os.remove(result_path)
215              os.rename(os.path.join(tmp_dist_dir, result_basename), result_path)
216  
217          return result_basename
218  
219      def build_wheel(self, wheel_directory, config_settings=None,
220                      metadata_directory=None):
221          return self._build_with_temp_dir(['bdist_wheel'], '.whl',
222                                           wheel_directory, config_settings)
223  
224      def build_sdist(self, sdist_directory, config_settings=None):
225          return self._build_with_temp_dir(['sdist', '--formats', 'gztar'],
226                                           '.tar.gz', sdist_directory,
227                                           config_settings)

230  class _BuildMetaLegacyBackend(_BuildMetaBackend):
231      """Compatibility backend for setuptools 
232   
233      This is a version of setuptools.build_meta that endeavors 
234      to maintain backwards 
235      compatibility with pre-PEP 517 modes of invocation. It 
236      exists as a temporary 
237      bridge between the old packaging mechanism and the new 
238      packaging mechanism, 
239      and will eventually be removed. 
240      """
241      def run_setup(self, setup_script='setup.py'):
242          # In order to maintain compatibility with scripts assuming that
243          # the setup.py script is in a directory on the PYTHONPATH, inject
244          # '' into sys.path. (pypa/setuptools#1642)
245          sys_path = list(sys.path)           # Save the original path
246  
247          script_dir = os.path.dirname(os.path.abspath(setup_script))
248          if script_dir not in sys.path:
249              sys.path.insert(0, script_dir)
250  
251          # Some setup.py scripts (e.g. in pygame and numpy) use sys.argv[0] to
252          # get the directory of the source code. They expect it to refer to the
253          # setup.py script.
254          sys_argv_0 = sys.argv[0]
255          sys.argv[0] = setup_script
256  
257          try:
258              super(_BuildMetaLegacyBackend,
259                    self).run_setup(setup_script=setup_script)
260          finally:
261              # While PEP 517 frontends should be calling each hook in a fresh
262              # subprocess according to the standard (and thus it should not be
263              # strictly necessary to restore the old sys.path), we'll restore
264              # the original path so that the path manipulation does not persist
265              # within the hook after run_setup is called.
266              sys.path[:] = sys_path
267              sys.argv[0] = sys_argv_0
