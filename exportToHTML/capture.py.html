71   def _readline_workaround() -> None:
72       """Ensure readline is imported so that it attaches to the correct stdio 
73       handles on Windows. 
74    
75       Pdb uses readline support where available--when not running from the Python 
76       prompt, the readline module is not imported until running the pdb REPL.  If 
77       running pytest with the --pdb option this means the readline module is not 
78       imported until after I/O capture has been started. 
79    
80       This is a problem for pyreadline, which is often used to implement readline 
81       support on Windows, as it does not attach to the correct handles for stdout 
82       and/or stdin if they have been redirected by the FDCapture mechanism.  This 
83       workaround ensures that readline is imported before I/O capture is setup so 
84       that it can attach to the actual stdin/out for the console. 
85    
86       See https://github.com/pytest-dev/pytest/pull/1281. 
87       """
88       if sys.platform.startswith("win32"):
89           try:
90               import readline  # noqa: F401
91           except ImportError:
92               pass

95   def _py36_windowsconsoleio_workaround(stream: TextIO) -> None:
96       """Workaround for Windows Unicode console handling on Python>=3.6. 
97    
98       Python 3.6 implemented Unicode console handling for Windows. This works 
99       by reading/writing to the raw console handle using 
100      ``{Read,Write}ConsoleW``. 
101   
102      The problem is that we are going to ``dup2`` over the stdio file 
103      descriptors when doing ``FDCapture`` and this will ``CloseHandle`` the 
104      handles used by Python to write to the console. Though there is still some 
105      weirdness and the console handle seems to only be closed randomly and not 
106      on the first call to ``CloseHandle``, or maybe it gets reopened with the 
107      same handle value when we suspend capturing. 
108   
109      The workaround in this case will reopen stdio with a different fd which 
110      also means a different handle by replicating the logic in 
111      "Py_lifecycle.c:initstdio/create_stdio". 
112   
113      :param stream: 
114          In practice ``sys.stdout`` or ``sys.stderr``, but given 
115          here as parameter for unittesting purposes. 
116   
117      See https://github.com/pytest-dev/py/issues/103. 
118      """
119      if not sys.platform.startswith("win32") or hasattr(sys, "pypy_version_info"):
120          return
121  
122      # Bail out if ``stream`` doesn't seem like a proper ``io`` stream (#2666).
123      if not hasattr(stream, "buffer"):  # type: ignore[unreachable]
124          return
125  
126      buffered = hasattr(stream.buffer, "raw")
127      raw_stdout = stream.buffer.raw if buffered else stream.buffer  # type: ignore[attr-defined]
128  
129      if not isinstance(raw_stdout, io._WindowsConsoleIO):  # type: ignore[attr-defined]
130          return
131  
132      def _reopen_stdio(f, mode):
133          if not buffered and mode[0] == "w":
134              buffering = 0
135          else:
136              buffering = -1
137  
138          return io.TextIOWrapper(
139              open(os.dup(f.fileno()), mode, buffering),  # type: ignore[arg-type]
140              f.encoding,
141              f.errors,
142              f.newlines,
143              f.line_buffering,
144          )
145  
146      sys.stdin = _reopen_stdio(sys.stdin, "rb")
147      sys.stdout = _reopen_stdio(sys.stdout, "wb")
148      sys.stderr = _reopen_stdio(sys.stderr, "wb")

151  @hookimpl(hookwrapper=True)
152  def pytest_load_initial_conftests(early_config: Config):
153      ns = early_config.known_args_namespace
154      if ns.capture == "fd":
155          _py36_windowsconsoleio_workaround(sys.stdout)
156      _colorama_workaround()
157      _readline_workaround()
158      pluginmanager = early_config.pluginmanager
159      capman = CaptureManager(ns.capture)
160      pluginmanager.register(capman, "capturemanager")
161  
162      # Make sure that capturemanager is properly reset at final shutdown.
163      early_config.add_cleanup(capman.stop_global_capturing)
164  
165      # Finally trigger conftest loading but while capturing (issue #93).
166      capman.start_global_capturing()
167      outcome = yield
168      capman.suspend_global_capture()
169      if outcome.excinfo is not None:
170          out, err = capman.read_global_capture()
171          sys.stdout.write(out)
172          sys.stderr.write(err)

253  class SysCaptureBinary:
254  
255      EMPTY_BUFFER = b""
256  
257      def __init__(self, fd: int, tmpfile=None, *, tee: bool = False) -> None:
258          name = patchsysdict[fd]
259          self._old = getattr(sys, name)
260          self.name = name
261          if tmpfile is None:
262              if name == "stdin":
263                  tmpfile = DontReadFromInput()
264              else:
265                  tmpfile = CaptureIO() if not tee else TeeCaptureIO(self._old)
266          self.tmpfile = tmpfile
267          self._state = "initialized"
268  
269      def repr(self, class_name: str) -> str:
270          return "<{} {} _old={} _state={!r} tmpfile={!r}>".format(
271              class_name,
272              self.name,
273              hasattr(self, "_old") and repr(self._old) or "<UNSET>",
274              self._state,
275              self.tmpfile,
276          )
277  
278      def __repr__(self) -> str:
279          return "<{} {} _old={} _state={!r} tmpfile={!r}>".format(
280              self.__class__.__name__,
281              self.name,
282              hasattr(self, "_old") and repr(self._old) or "<UNSET>",
283              self._state,
284              self.tmpfile,
285          )
286  
287      def _assert_state(self, op: str, states: Tuple[str, ...]) -> None:
288          assert (
289              self._state in states
290          ), "cannot {} in state {!r}: expected one of {}".format(
291              op, self._state, ", ".join(states)
292          )
293  
294      def start(self) -> None:
295          self._assert_state("start", ("initialized",))
296          setattr(sys, self.name, self.tmpfile)
297          self._state = "started"
298  
299      def snap(self):
300          self._assert_state("snap", ("started", "suspended"))
301          self.tmpfile.seek(0)
302          res = self.tmpfile.buffer.read()
303          self.tmpfile.seek(0)
304          self.tmpfile.truncate()
305          return res
306  
307      def done(self) -> None:
308          self._assert_state("done", ("initialized", "started", "suspended", "done"))
309          if self._state == "done":
310              return
311          setattr(sys, self.name, self._old)
312          del self._old
313          self.tmpfile.close()
314          self._state = "done"
315  
316      def suspend(self) -> None:
317          self._assert_state("suspend", ("started", "suspended"))
318          setattr(sys, self.name, self._old)
319          self._state = "suspended"
320  
321      def resume(self) -> None:
322          self._assert_state("resume", ("started", "suspended"))
323          if self._state == "started":
324              return
325          setattr(sys, self.name, self.tmpfile)
326          self._state = "started"
327  
328      def writeorg(self, data) -> None:
329          self._assert_state("writeorg", ("started", "suspended"))
330          self._old.flush()
331          self._old.buffer.write(data)
332          self._old.buffer.flush()

350  class FDCaptureBinary:
351      """Capture IO to/from a given OS-level file descriptor. 
352   
353      snap() produces `bytes`. 
354      """
355  
356      EMPTY_BUFFER = b""
357  
358      def __init__(self, targetfd: int) -> None:
359          self.targetfd = targetfd
360  
361          try:
362              os.fstat(targetfd)
363          except OSError:
364              # FD capturing is conceptually simple -- create a temporary file,
365              # redirect the FD to it, redirect back when done. But when the
366              # target FD is invalid it throws a wrench into this loveley scheme.
367              #
368              # Tests themselves shouldn't care if the FD is valid, FD capturing
369              # should work regardless of external circumstances. So falling back
370              # to just sys capturing is not a good option.
371              #
372              # Further complications are the need to support suspend() and the
373              # possibility of FD reuse (e.g. the tmpfile getting the very same
374              # target FD). The following approach is robust, I believe.
375              self.targetfd_invalid: Optional[int] = os.open(os.devnull, os.O_RDWR)
376              os.dup2(self.targetfd_invalid, targetfd)
377          else:
378              self.targetfd_invalid = None
379          self.targetfd_save = os.dup(targetfd)
380  
381          if targetfd == 0:
382              self.tmpfile = open(os.devnull)
383              self.syscapture = SysCapture(targetfd)
384          else:
385              self.tmpfile = EncodedFile(
386                  TemporaryFile(buffering=0),
387                  encoding="utf-8",
388                  errors="replace",
389                  newline="",
390                  write_through=True,
391              )
392              if targetfd in patchsysdict:
393                  self.syscapture = SysCapture(targetfd, self.tmpfile)
394              else:
395                  self.syscapture = NoCapture()
396  
397          self._state = "initialized"
398  
399      def __repr__(self) -> str:
400          return "<{} {} oldfd={} _state={!r} tmpfile={!r}>".format(
401              self.__class__.__name__,
402              self.targetfd,
403              self.targetfd_save,
404              self._state,
405              self.tmpfile,
406          )
407  
408      def _assert_state(self, op: str, states: Tuple[str, ...]) -> None:
409          assert (
410              self._state in states
411          ), "cannot {} in state {!r}: expected one of {}".format(
412              op, self._state, ", ".join(states)
413          )
414  
415      def start(self) -> None:
416          """Start capturing on targetfd using memorized tmpfile."""
417          self._assert_state("start", ("initialized",))
418          os.dup2(self.tmpfile.fileno(), self.targetfd)
419          self.syscapture.start()
420          self._state = "started"
421  
422      def snap(self):
423          self._assert_state("snap", ("started", "suspended"))
424          self.tmpfile.seek(0)
425          res = self.tmpfile.buffer.read()
426          self.tmpfile.seek(0)
427          self.tmpfile.truncate()
428          return res
429  
430      def done(self) -> None:
431          """Stop capturing, restore streams, return original capture file, 
432          seeked to position zero."""
433          self._assert_state("done", ("initialized", "started", "suspended", "done"))
434          if self._state == "done":
435              return
436          os.dup2(self.targetfd_save, self.targetfd)
437          os.close(self.targetfd_save)
438          if self.targetfd_invalid is not None:
439              if self.targetfd_invalid != self.targetfd:
440                  os.close(self.targetfd)
441              os.close(self.targetfd_invalid)
442          self.syscapture.done()
443          self.tmpfile.close()
444          self._state = "done"
445  
446      def suspend(self) -> None:
447          self._assert_state("suspend", ("started", "suspended"))
448          if self._state == "suspended":
449              return
450          self.syscapture.suspend()
451          os.dup2(self.targetfd_save, self.targetfd)
452          self._state = "suspended"
453  
454      def resume(self) -> None:
455          self._assert_state("resume", ("started", "suspended"))
456          if self._state == "started":
457              return
458          self.syscapture.resume()
459          os.dup2(self.tmpfile.fileno(), self.targetfd)
460          self._state = "started"
461  
462      def writeorg(self, data):
463          """Write to original file descriptor."""
464          self._assert_state("writeorg", ("started", "suspended"))
465          os.write(self.targetfd_save, data)

548  class MultiCapture(Generic[AnyStr]):
549      _state = None
550      _in_suspended = False
551  
552      def __init__(self, in_, out, err) -> None:
553          self.in_ = in_
554          self.out = out
555          self.err = err
556  
557      def __repr__(self) -> str:
558          return "<MultiCapture out={!r} err={!r} in_={!r} _state={!r} _in_suspended={!r}>".format(
559              self.out, self.err, self.in_, self._state, self._in_suspended,
560          )
561  
562      def start_capturing(self) -> None:
563          self._state = "started"
564          if self.in_:
565              self.in_.start()
566          if self.out:
567              self.out.start()
568          if self.err:
569              self.err.start()
570  
571      def pop_outerr_to_orig(self) -> Tuple[AnyStr, AnyStr]:
572          """Pop current snapshot out/err capture and flush to orig streams."""
573          out, err = self.readouterr()
574          if out:
575              self.out.writeorg(out)
576          if err:
577              self.err.writeorg(err)
578          return out, err
579  
580      def suspend_capturing(self, in_: bool = False) -> None:
581          self._state = "suspended"
582          if self.out:
583              self.out.suspend()
584          if self.err:
585              self.err.suspend()
586          if in_ and self.in_:
587              self.in_.suspend()
588              self._in_suspended = True
589  
590      def resume_capturing(self) -> None:
591          self._state = "started"
592          if self.out:
593              self.out.resume()
594          if self.err:
595              self.err.resume()
596          if self._in_suspended:
597              self.in_.resume()
598              self._in_suspended = False
599  
600      def stop_capturing(self) -> None:
601          """Stop capturing and reset capturing streams."""
602          if self._state == "stopped":
603              raise ValueError("was already stopped")
604          self._state = "stopped"
605          if self.out:
606              self.out.done()
607          if self.err:
608              self.err.done()
609          if self.in_:
610              self.in_.done()
611  
612      def is_started(self) -> bool:
613          """Whether actively capturing -- not suspended or stopped."""
614          return self._state == "started"
615  
616      def readouterr(self) -> CaptureResult[AnyStr]:
617          if self.out:
618              out = self.out.snap()
619          else:
620              out = ""
621          if self.err:
622              err = self.err.snap()
623          else:
624              err = ""
625          return CaptureResult(out, err)
626

628  def _get_multicapture(method: "_CaptureMethod") -> MultiCapture[str]:
629      if method == "fd":
630          return MultiCapture(in_=FDCapture(0), out=FDCapture(1), err=FDCapture(2))
631      elif method == "sys":
632          return MultiCapture(in_=SysCapture(0), out=SysCapture(1), err=SysCapture(2))
633      elif method == "no":
634          return MultiCapture(in_=None, out=None, err=None)
635      elif method == "tee-sys":
636          return MultiCapture(
637              in_=None, out=SysCapture(1, tee=True), err=SysCapture(2, tee=True)
638          )
639      raise ValueError(f"unknown capturing method: {method!r}")

645  class CaptureManager:
646      """The capture plugin. 
647   
648      Manages that the appropriate capture method is enabled/disabled during 
649      collection and each test phase (setup, call, teardown). After each of 
650      those points, the captured output is obtained and attached to the 
651      collection/runtest report. 
652   
653      There are two levels of capture: 
654   
655      * global: enabled by default and can be suppressed by the ``-s`` 
656        option. This is always enabled/disabled during collection and each test 
657        phase. 
658   
659      * fixture: when a test function or one of its fixture depend on the 
660        ``capsys`` or ``capfd`` fixtures. In this case special handling is 
661        needed to ensure the fixtures take precedence over the global capture. 
662      """
663  
664      def __init__(self, method: "_CaptureMethod") -> None:
665          self._method = method
666          self._global_capturing: Optional[MultiCapture[str]] = None
667          self._capture_fixture: Optional[CaptureFixture[Any]] = None
668  
669      def __repr__(self) -> str:
670          return "<CaptureManager _method={!r} _global_capturing={!r} _capture_fixture={!r}>".format(
671              self._method, self._global_capturing, self._capture_fixture
672          )
673  
674      def is_capturing(self) -> Union[str, bool]:
675          if self.is_globally_capturing():
676              return "global"
677          if self._capture_fixture:
678              return "fixture %s" % self._capture_fixture.request.fixturename
679          return False
680  
681      # Global capturing control
682  
683      def is_globally_capturing(self) -> bool:
684          return self._method != "no"
685  
686      def start_global_capturing(self) -> None:
687          assert self._global_capturing is None
688          self._global_capturing = _get_multicapture(self._method)
689          self._global_capturing.start_capturing()
690  
691      def stop_global_capturing(self) -> None:
692          if self._global_capturing is not None:
693              self._global_capturing.pop_outerr_to_orig()
694              self._global_capturing.stop_capturing()
695              self._global_capturing = None
696  
697      def resume_global_capture(self) -> None:
698          # During teardown of the python process, and on rare occasions, capture
699          # attributes can be `None` while trying to resume global capture.
700          if self._global_capturing is not None:
701              self._global_capturing.resume_capturing()
702  
703      def suspend_global_capture(self, in_: bool = False) -> None:
704          if self._global_capturing is not None:
705              self._global_capturing.suspend_capturing(in_=in_)
706  
707      def suspend(self, in_: bool = False) -> None:
708          # Need to undo local capsys-et-al if it exists before disabling global capture.
709          self.suspend_fixture()
710          self.suspend_global_capture(in_)
711  
712      def resume(self) -> None:
713          self.resume_global_capture()
714          self.resume_fixture()
715  
716      def read_global_capture(self) -> CaptureResult[str]:
717          assert self._global_capturing is not None
718          return self._global_capturing.readouterr()

720      # Fixture Control
721  
722      def set_fixture(self, capture_fixture: "CaptureFixture[Any]") -> None:
723          if self._capture_fixture:
724              current_fixture = self._capture_fixture.request.fixturename
725              requested_fixture = capture_fixture.request.fixturename
726              capture_fixture.request.raiseerror(
727                  "cannot use {} and {} at the same time".format(
728                      requested_fixture, current_fixture
729                  )
730              )
731          self._capture_fixture = capture_fixture
732  
733      def unset_fixture(self) -> None:
734          self._capture_fixture = None
735  
736      def activate_fixture(self) -> None:
737          """If the current item is using ``capsys`` or ``capfd``, activate 
738          them so they take precedence over the global capture."""
739          if self._capture_fixture:
740              self._capture_fixture._start()
741  
742      def deactivate_fixture(self) -> None:
743          """Deactivate the ``capsys`` or ``capfd`` fixture of this item, if any."""
744          if self._capture_fixture:
745              self._capture_fixture.close()
746  
747      def suspend_fixture(self) -> None:
748          if self._capture_fixture:
749              self._capture_fixture._suspend()
750  
751      def resume_fixture(self) -> None:
752          if self._capture_fixture:
753              self._capture_fixture._resume()

755      # Helper context managers
756  
757      @contextlib.contextmanager
758      def global_and_fixture_disabled(self) -> Generator[None, None, None]:
759          """Context manager to temporarily disable global and current fixture capturing."""
760          do_fixture = self._capture_fixture and self._capture_fixture._is_started()
761          if do_fixture:
762              self.suspend_fixture()
763          do_global = self._global_capturing and self._global_capturing.is_started()
764          if do_global:
765              self.suspend_global_capture()
766          try:
767              yield
768          finally:
769              if do_global:
770                  self.resume_global_capture()
771              if do_fixture:
772                  self.resume_fixture()
773  
774      @contextlib.contextmanager
775      def item_capture(self, when: str, item: Item) -> Generator[None, None, None]:
776          self.resume_global_capture()
777          self.activate_fixture()
778          try:
779              yield
780          finally:
781              self.deactivate_fixture()
782              self.suspend_global_capture(in_=False)
783  
784          out, err = self.read_global_capture()
785          item.add_report_section(when, "stdout", out)
786          item.add_report_section(when, "stderr", err)

829  class CaptureFixture(Generic[AnyStr]):
830      """Object returned by the :fixture:`capsys`, :fixture:`capsysbinary`, 
831      :fixture:`capfd` and :fixture:`capfdbinary` fixtures."""
832  
833      def __init__(
834          self, captureclass, request: SubRequest, *, _ispytest: bool = False
835      ) -> None:
836          check_ispytest(_ispytest)
837          self.captureclass = captureclass
838          self.request = request
839          self._capture: Optional[MultiCapture[AnyStr]] = None
840          self._captured_out = self.captureclass.EMPTY_BUFFER
841          self._captured_err = self.captureclass.EMPTY_BUFFER
842  
843      def _start(self) -> None:
844          if self._capture is None:
845              self._capture = MultiCapture(
846                  in_=None, out=self.captureclass(1), err=self.captureclass(2),
847              )
848              self._capture.start_capturing()
849  
850      def close(self) -> None:
851          if self._capture is not None:
852              out, err = self._capture.pop_outerr_to_orig()
853              self._captured_out += out
854              self._captured_err += err
855              self._capture.stop_capturing()
856              self._capture = None
857  
858      def readouterr(self) -> CaptureResult[AnyStr]:
859          """Read and return the captured output so far, resetting the internal 
860          buffer. 
861   
862          :returns: 
863              The captured content as a namedtuple with ``out`` and ``err`` 
864              string attributes. 
865          """
866          captured_out, captured_err = self._captured_out, self._captured_err
867          if self._capture is not None:
868              out, err = self._capture.readouterr()
869              captured_out += out
870              captured_err += err
871          self._captured_out = self.captureclass.EMPTY_BUFFER
872          self._captured_err = self.captureclass.EMPTY_BUFFER
873          return CaptureResult(captured_out, captured_err)
874  
875      def _suspend(self) -> None:
876          """Suspend this fixture's own capturing temporarily."""
877          if self._capture is not None:
878              self._capture.suspend_capturing()
879  
880      def _resume(self) -> None:
881          """Resume this fixture's own capturing temporarily."""
882          if self._capture is not None:
883              self._capture.resume_capturing()
884  
885      def _is_started(self) -> bool:
886          """Whether actively capturing -- not disabled or closed."""
887          if self._capture is not None:
888              return self._capture.is_started()
889          return False

892      def disabled(self) -> Generator[None, None, None]:
893          """Temporarily disable capturing while inside the ``with`` block."""
894          capmanager = self.request.config.pluginmanager.getplugin("capturemanager")
895          with capmanager.global_and_fixture_disabled():
