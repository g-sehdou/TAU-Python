81       """Register argparse-style options and ini-style config values, 
82       called once at the beginning of a test run. 
83    
84       .. note:: 
85    
86           This function should be implemented only in plugins or ``conftest.py`` 
87           files situated at the tests root directory due to how pytest 
88           :ref:`discovers plugins during startup <pluginorder>`. 
89    
90       :param _pytest.config.argparsing.Parser parser: 
91           To add command line options, call 
92           :py:func:`parser.addoption(...) <_pytest.config.argparsing.Parser.addoption>`. 
93           To add ini-file values call :py:func:`parser.addini(...) 
94           <_pytest.config.argparsing.Parser.addini>`. 
95    
96       :param _pytest.config.PytestPluginManager pluginmanager: 
97           pytest plugin manager, which can be used to install :py:func:`hookspec`'s 
98           or :py:func:`hookimpl`'s and allow one plugin to call another plugin's hooks 
99           to change how command line options are added. 
100   
101      Options can later be accessed through the 
102      :py:class:`config <_pytest.config.Config>` object, respectively: 
103   
104      - :py:func:`config.getoption(name) <_pytest.config.Config.getoption>` to 
105        retrieve the value of a command line option. 
106   
107      - :py:func:`config.getini(name) <_pytest.config.Config.getini>` to retrieve 
108        a value read from an ini-style file. 
109   
110      The config object is passed around on many internal objects via the ``.config`` 
111      attribute or can be retrieved as the ``pytestconfig`` fixture. 
112   
113      .. note:: 
114          This hook is incompatible with ``hookwrapper=True``. 
115      """

207  @hookspec(firstresult=True)
208  def pytest_collection(session: "Session") -> Optional[object]:
209      """Perform the collection phase for the given session. 
210   
211      Stops at first non-None result, see :ref:`firstresult`. 
212      The return value is not used, but only stops further processing. 
213   
214      The default collection phase is this (see individual hooks for full details): 
215   
216      1. Starting from ``session`` as the initial collector: 
217   
218        1. ``pytest_collectstart(collector)`` 
219        2. ``report = pytest_make_collect_report(collector)`` 
220        3. ``pytest_exception_interact(collector, call, report)`` if an interactive exception occurred 
221        4. For each collected node: 
222   
223          1. If an item, ``pytest_itemcollected(item)`` 
224          2. If a collector, recurse into it. 
225   
226        5. ``pytest_collectreport(report)`` 
227   
228      2. ``pytest_collection_modifyitems(session, config, items)`` 
229   
230        1. ``pytest_deselected(items)`` for any deselected items (may be called multiple times) 
231   
232      3. ``pytest_collection_finish(session)`` 
233      4. Set ``session.items`` to the list of collected items 
234      5. Set ``session.testscollected`` to the number of collected items 
235   
236      You can implement this hook to only perform some action before collection, 
237      for example the terminal plugin uses it to start displaying the collection 
238      counter (and returns `None`). 
239   
240      :param pytest.Session session: The pytest session object. 
241      """

382  @hookspec(firstresult=True)
383  def pytest_runtestloop(session: "Session") -> Optional[object]:
384      """Perform the main runtest loop (after collection finished). 
385   
386      The default hook implementation performs the runtest protocol for all items 
387      collected in the session (``session.items``), unless the collection failed 
388      or the ``collectonly`` pytest option is set. 
389   
390      If at any point :py:func:`pytest.exit` is called, the loop is 
391      terminated immediately. 
392   
393      If at any point ``session.shouldfail`` or ``session.shouldstop`` are set, the 
394      loop is terminated after the runtest protocol for the current item is finished. 
395   
396      :param pytest.Session session: The pytest session object. 
397   
398      Stops at first non-None result, see :ref:`firstresult`. 
399      The return value is not used, but only stops further processing. 
400      """

403  @hookspec(firstresult=True)
404  def pytest_runtest_protocol(
405      item: "Item", nextitem: "Optional[Item]"
406  ) -> Optional[object]:
407      """Perform the runtest protocol for a single test item. 
408   
409      The default runtest protocol is this (see individual hooks for full details): 
410   
411      - ``pytest_runtest_logstart(nodeid, location)`` 
412   
413      - Setup phase: 
414          - ``call = pytest_runtest_setup(item)`` (wrapped in ``CallInfo(when="setup")``) 
415          - ``report = pytest_runtest_makereport(item, call)`` 
416          - ``pytest_runtest_logreport(report)`` 
417          - ``pytest_exception_interact(call, report)`` if an interactive exception occurred 
418   
419      - Call phase, if the the setup passed and the ``setuponly`` pytest option is not set: 
420          - ``call = pytest_runtest_call(item)`` (wrapped in ``CallInfo(when="call")``) 
421          - ``report = pytest_runtest_makereport(item, call)`` 
422          - ``pytest_runtest_logreport(report)`` 
423          - ``pytest_exception_interact(call, report)`` if an interactive exception occurred 
424   
425      - Teardown phase: 
426          - ``call = pytest_runtest_teardown(item, nextitem)`` (wrapped in ``CallInfo(when="teardown")``) 
427          - ``report = pytest_runtest_makereport(item, call)`` 
428          - ``pytest_runtest_logreport(report)`` 
429          - ``pytest_exception_interact(call, report)`` if an interactive exception occurred 
430   
431      - ``pytest_runtest_logfinish(nodeid, location)`` 
432   
433      :param item: Test item for which the runtest protocol is performed. 
434      :param nextitem: The scheduled-to-be-next test item (or None if this is the end my friend). 
435   
436      Stops at first non-None result, see :ref:`firstresult`. 
437      The return value is not used, but only stops further processing. 
438      """
439

636      :param pytest.Item item: pytest item object of current test. 
637      :param int lineno: Line number of the assert statement. 
638      :param str orig: String with the original assertion. 
639      :param str expl: String with the assert explanation. 
640   
641      .. note:: 
642   
643          This hook is **experimental**, so its parameters or even the hook itself might 
644          be changed/removed without warning in any future pytest release.
