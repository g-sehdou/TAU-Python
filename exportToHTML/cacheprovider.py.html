56   @final
57   @attr.s(init=False)
58   class Cache:
59       _cachedir = attr.ib(type=Path, repr=False)
60       _config = attr.ib(type=Config, repr=False)
61   
62       # sub-directory under cache-dir for directories created by "makedir"
63       _CACHE_PREFIX_DIRS = "d"
64   
65       # sub-directory under cache-dir for values created by "set"
66       _CACHE_PREFIX_VALUES = "v"
67   
68       def __init__(
69           self, cachedir: Path, config: Config, *, _ispytest: bool = False
70       ) -> None:
71           check_ispytest(_ispytest)
72           self._cachedir = cachedir
73           self._config = config
74   
75       @classmethod
76       def for_config(cls, config: Config, *, _ispytest: bool = False) -> "Cache":
77           """Create the Cache instance for a Config. 
78    
79           :meta private: 
80           """
81           check_ispytest(_ispytest)
82           cachedir = cls.cache_dir_from_config(config, _ispytest=True)
83           if config.getoption("cacheclear") and cachedir.is_dir():
84               cls.clear_cache(cachedir, _ispytest=True)
85           return cls(cachedir, config, _ispytest=True)
86   
87       @classmethod
88       def clear_cache(cls, cachedir: Path, _ispytest: bool = False) -> None:
89           """Clear the sub-directories used to hold cached directories and values. 
90    
91           :meta private: 
92           """
93           check_ispytest(_ispytest)
94           for prefix in (cls._CACHE_PREFIX_DIRS, cls._CACHE_PREFIX_VALUES):
95               d = cachedir / prefix
96               if d.is_dir():
97                   rm_rf(d)
98   
99       @staticmethod
100      def cache_dir_from_config(config: Config, *, _ispytest: bool = False) -> Path:
101          """Get the path to the cache directory for a Config. 
102   
103          :meta private: 
104          """
105          check_ispytest(_ispytest)
106          return resolve_from_str(config.getini("cache_dir"), config.rootpath)
107  
108      def warn(self, fmt: str, *, _ispytest: bool = False, **args: object) -> None:
109          """Issue a cache warning. 
110   
111          :meta private: 
112          """
113          check_ispytest(_ispytest)
114          import warnings
115          from _pytest.warning_types import PytestCacheWarning
116  
117          warnings.warn(
118              PytestCacheWarning(fmt.format(**args) if args else fmt),
119              self._config.hook,
120              stacklevel=3,
121          )

123      def makedir(self, name: str) -> py.path.local:
124          """Return a directory path object with the given name. 
125   
126          If the directory does not yet exist, it will be created. You can use 
127          it to manage files to e.g. store/retrieve database dumps across test 
128          sessions. 
129   
130          :param name: 
131              Must be a string not containing a ``/`` separator. 
132              Make sure the name contains your plugin or application 
133              identifiers to prevent clashes with other cache users. 
134          """
135          path = Path(name)
136          if len(path.parts) > 1:
137              raise ValueError("name is not allowed to contain path separators")
138          res = self._cachedir.joinpath(self._CACHE_PREFIX_DIRS, path)
139          res.mkdir(exist_ok=True, parents=True)
140          return py.path.local(res)
141  
142      def _getvaluepath(self, key: str) -> Path:
143          return self._cachedir.joinpath(self._CACHE_PREFIX_VALUES, Path(key))
144  
145      def get(self, key: str, default):
146          """Return the cached value for the given key. 
147   
148          If no value was yet cached or the value cannot be read, the specified 
149          default is returned. 
150   
151          :param key: 
152              Must be a ``/`` separated value. Usually the first 
153              name is the name of your plugin or your application. 
154          :param default: 
155              The value to return in case of a cache-miss or invalid cache value. 
156          """
157          path = self._getvaluepath(key)
158          try:
159              with path.open("r") as f:
160                  return json.load(f)
161          except (ValueError, OSError):
162              return default
163  
164      def set(self, key: str, value: object) -> None:
165          """Save value for the given key. 
166   
167          :param key: 
168              Must be a ``/`` separated value. Usually the first 
169              name is the name of your plugin or your application. 
170          :param value: 
171              Must be of any combination of basic python types, 
172              including nested types like lists of dictionaries. 
173          """
174          path = self._getvaluepath(key)
175          try:
176              if path.parent.is_dir():
177                  cache_dir_exists_already = True
178              else:
179                  cache_dir_exists_already = self._cachedir.exists()
180                  path.parent.mkdir(exist_ok=True, parents=True)
181          except OSError:
182              self.warn("could not create cache path {path}", path=path, _ispytest=True)
183              return
184          if not cache_dir_exists_already:
185              self._ensure_supporting_files()
186          data = json.dumps(value, indent=2, sort_keys=True)
187          try:
188              f = path.open("w")
189          except OSError:
190              self.warn("cache could not write path {path}", path=path, _ispytest=True)
191          else:
192              with f:
193                  f.write(data)
194  
195      def _ensure_supporting_files(self) -> None:
196          """Create supporting files in the cache dir that are not really part of the cache."""
197          readme_path = self._cachedir / "README.md"
198          readme_path.write_text(README_CONTENT)
199  
200          gitignore_path = self._cachedir.joinpath(".gitignore")
201          msg = "# Created by pytest automatically.\n*\n"
202          gitignore_path.write_text(msg, encoding="UTF-8")
203  
204          cachedir_tag_path = self._cachedir.joinpath("CACHEDIR.TAG")
205          cachedir_tag_path.write_bytes(CACHEDIR_TAG_CONTENT)

208  class LFPluginCollWrapper:
209      def __init__(self, lfplugin: "LFPlugin") -> None:
210          self.lfplugin = lfplugin
211          self._collected_at_least_one_failure = False
212  
213      @hookimpl(hookwrapper=True)
214      def pytest_make_collect_report(self, collector: nodes.Collector):
215          if isinstance(collector, Session):
216              out = yield
217              res: CollectReport = out.get_result()
218  
219              # Sort any lf-paths to the beginning.
220              lf_paths = self.lfplugin._last_failed_paths
221              res.result = sorted(
222                  res.result, key=lambda x: 0 if Path(str(x.fspath)) in lf_paths else 1,
223              )
224              return
225  
226          elif isinstance(collector, Module):
227              if Path(str(collector.fspath)) in self.lfplugin._last_failed_paths:
228                  out = yield
229                  res = out.get_result()
230                  result = res.result
231                  lastfailed = self.lfplugin.lastfailed
232  
233                  # Only filter with known failures.
234                  if not self._collected_at_least_one_failure:
235                      if not any(x.nodeid in lastfailed for x in result):
236                          return
237                      self.lfplugin.config.pluginmanager.register(
238                          LFPluginCollSkipfiles(self.lfplugin), "lfplugin-collskip"
239                      )
240                      self._collected_at_least_one_failure = True
241  
242                  session = collector.session
243                  result[:] = [
244                      x
245                      for x in result
246                      if x.nodeid in lastfailed
247                      # Include any passed arguments (not trivial to filter).
248                      or session.isinitpath(x.fspath)
249                      # Keep all sub-collectors.
250                      or isinstance(x, nodes.Collector)
251                  ]
252                  return
253          yield

256  class LFPluginCollSkipfiles:
257      def __init__(self, lfplugin: "LFPlugin") -> None:
258          self.lfplugin = lfplugin
259  
260      @hookimpl
261      def pytest_make_collect_report(
262          self, collector: nodes.Collector
263      ) -> Optional[CollectReport]:
264          # Packages are Modules, but _last_failed_paths only contains
265          # test-bearing paths and doesn't try to include the paths of their
266          # packages, so don't filter them.
267          if isinstance(collector, Module) and not isinstance(collector, Package):
268              if Path(str(collector.fspath)) not in self.lfplugin._last_failed_paths:
269                  self.lfplugin._skipped_files += 1
270  
271                  return CollectReport(
272                      collector.nodeid, "passed", longrepr=None, result=[]
273                  )
274          return None

277  class LFPlugin:
278      """Plugin which implements the --lf (run last-failing) option."""
279  
280      def __init__(self, config: Config) -> None:
281          self.config = config
282          active_keys = "lf", "failedfirst"
283          self.active = any(config.getoption(key) for key in active_keys)
284          assert config.cache
285          self.lastfailed: Dict[str, bool] = config.cache.get("cache/lastfailed", {})
286          self._previously_failed_count: Optional[int] = None
287          self._report_status: Optional[str] = None
288          self._skipped_files = 0  # count skipped files during collection due to --lf
289  
290          if config.getoption("lf"):
291              self._last_failed_paths = self.get_last_failed_paths()
292              config.pluginmanager.register(
293                  LFPluginCollWrapper(self), "lfplugin-collwrapper"
294              )
295  
296      def get_last_failed_paths(self) -> Set[Path]:
297          """Return a set with all Paths()s of the previously failed nodeids."""
298          rootpath = self.config.rootpath
299          result = {rootpath / nodeid.split("::")[0] for nodeid in self.lastfailed}
300          return {x for x in result if x.exists()}
301  
302      def pytest_report_collectionfinish(self) -> Optional[str]:
303          if self.active and self.config.getoption("verbose") >= 0:
304              return "run-last-failure: %s" % self._report_status
305          return None
306  
307      def pytest_runtest_logreport(self, report: TestReport) -> None:
308          if (report.when == "call" and report.passed) or report.skipped:
309              self.lastfailed.pop(report.nodeid, None)
310          elif report.failed:
311              self.lastfailed[report.nodeid] = True
312  
313      def pytest_collectreport(self, report: CollectReport) -> None:
314          passed = report.outcome in ("passed", "skipped")
315          if passed:
316              if report.nodeid in self.lastfailed:
317                  self.lastfailed.pop(report.nodeid)
318                  self.lastfailed.update((item.nodeid, True) for item in report.result)
319          else:
320              self.lastfailed[report.nodeid] = True
321  
322      @hookimpl(hookwrapper=True, tryfirst=True)
323      def pytest_collection_modifyitems(
324          self, config: Config, items: List[nodes.Item]
325      ) -> Generator[None, None, None]:
326          yield
327  
328          if not self.active:
329              return
330  
331          if self.lastfailed:
332              previously_failed = []
333              previously_passed = []
334              for item in items:
335                  if item.nodeid in self.lastfailed:
336                      previously_failed.append(item)
337                  else:
338                      previously_passed.append(item)
339              self._previously_failed_count = len(previously_failed)
340  
341              if not previously_failed:
342                  # Running a subset of all tests with recorded failures
343                  # only outside of it.
344                  self._report_status = "%d known failures not in selected tests" % (
345                      len(self.lastfailed),
346                  )
347              else:
348                  if self.config.getoption("lf"):
349                      items[:] = previously_failed
350                      config.hook.pytest_deselected(items=previously_passed)
351                  else:  # --failedfirst
352                      items[:] = previously_failed + previously_passed
353  
354                  noun = "failure" if self._previously_failed_count == 1 else "failures"
355                  suffix = " first" if self.config.getoption("failedfirst") else ""
356                  self._report_status = "rerun previous {count} {noun}{suffix}".format(
357                      count=self._previously_failed_count, suffix=suffix, noun=noun
358                  )
359  
360              if self._skipped_files > 0:
361                  files_noun = "file" if self._skipped_files == 1 else "files"
362                  self._report_status += " (skipped {files} {files_noun})".format(
363                      files=self._skipped_files, files_noun=files_noun
364                  )
365          else:
366              self._report_status = "no previously failed tests, "
367              if self.config.getoption("last_failed_no_failures") == "none":
368                  self._report_status += "deselecting all items."
369                  config.hook.pytest_deselected(items=items[:])
370                  items[:] = []
371              else:
372                  self._report_status += "not deselecting items."
373  
374      def pytest_sessionfinish(self, session: Session) -> None:
375          config = self.config
376          if config.getoption("cacheshow") or hasattr(config, "workerinput"):
377              return
378  
379          assert config.cache is not None
380          saved_lastfailed = config.cache.get("cache/lastfailed", {})
381          if saved_lastfailed != self.lastfailed:
382              config.cache.set("cache/lastfailed", self.lastfailed)

385  class NFPlugin:
386      """Plugin which implements the --nf (run new-first) option."""
387  
388      def __init__(self, config: Config) -> None:
389          self.config = config
390          self.active = config.option.newfirst
391          assert config.cache is not None
392          self.cached_nodeids = set(config.cache.get("cache/nodeids", []))
393  
394      @hookimpl(hookwrapper=True, tryfirst=True)
395      def pytest_collection_modifyitems(
396          self, items: List[nodes.Item]
397      ) -> Generator[None, None, None]:
398          yield
399  
400          if self.active:
401              new_items: Dict[str, nodes.Item] = {}
402              other_items: Dict[str, nodes.Item] = {}
403              for item in items:
404                  if item.nodeid not in self.cached_nodeids:
405                      new_items[item.nodeid] = item
406                  else:
407                      other_items[item.nodeid] = item
408  
409              items[:] = self._get_increasing_order(
410                  new_items.values()
411              ) + self._get_increasing_order(other_items.values())
412              self.cached_nodeids.update(new_items)
413          else:
414              self.cached_nodeids.update(item.nodeid for item in items)
415  
416      def _get_increasing_order(self, items: Iterable[nodes.Item]) -> List[nodes.Item]:
417          return sorted(items, key=lambda item: item.fspath.mtime(), reverse=True)  # type: ignore[no-any-return]
418  
419      def pytest_sessionfinish(self) -> None:
420          config = self.config
421          if config.getoption("cacheshow") or hasattr(config, "workerinput"):
422              return
423  
424          if config.getoption("collectonly"):
425              return
426  
427          assert config.cache is not None
428          config.cache.set("cache/nodeids", sorted(self.cached_nodeids))

431  def pytest_addoption(parser: Parser) -> None:
432      group = parser.getgroup("general")
433      group.addoption(
434          "--lf",
435          "--last-failed",
436          action="store_true",
437          dest="lf",
438          help="rerun only the tests that failed "
439          "at the last run (or all if none failed)",
440      )
441      group.addoption(
442          "--ff",
443          "--failed-first",
444          action="store_true",
445          dest="failedfirst",
446          help="run all tests, but run the last failures first.\n"
447          "This may re-order tests and thus lead to "
448          "repeated fixture setup/teardown.",
449      )
450      group.addoption(
451          "--nf",
452          "--new-first",
453          action="store_true",
454          dest="newfirst",
455          help="run tests from new files first, then the rest of the tests "
456          "sorted by file mtime",
457      )
458      group.addoption(
459          "--cache-show",
460          action="append",
461          nargs="?",
462          dest="cacheshow",
463          help=(
464              "show cache contents, don't perform collection or tests. "
465              "Optional argument: glob (default: '*')."
466          ),
467      )
468      group.addoption(
469          "--cache-clear",
470          action="store_true",
471          dest="cacheclear",
472          help="remove all cache contents at start of test run.",
473      )
474      cache_dir_default = ".pytest_cache"
475      if "TOX_ENV_DIR" in os.environ:
476          cache_dir_default = os.path.join(os.environ["TOX_ENV_DIR"], cache_dir_default)
477      parser.addini("cache_dir", default=cache_dir_default, help="cache directory path.")
478      group.addoption(
479          "--lfnf",
480          "--last-failed-no-failures",
481          action="store",
482          dest="last_failed_no_failures",
483          choices=("all", "none"),
484          default="all",
485          help="which tests to run with no previously (known) failures.",
486      )

521      """Display cachedir with --cache-show and if non-default."""
522      if config.option.verbose > 0 or config.getini("cache_dir") != ".pytest_cache":
523          assert config.cache is not None
524          cachedir = config.cache._cachedir
525          # TODO: evaluate generating upward relative paths
526          # starting with .., ../.. if sensible
527  
528          try:
529              displaypath = cachedir.relative_to(config.rootpath)
530          except ValueError:
531              displaypath = cachedir
532          return f"cachedir: {displaypath}"
533      return None

536  def cacheshow(config: Config, session: Session) -> int:
537      from pprint import pformat
538  
539      assert config.cache is not None
540  
541      tw = TerminalWriter()
542      tw.line("cachedir: " + str(config.cache._cachedir))
543      if not config.cache._cachedir.is_dir():
544          tw.line("cache is empty")
545          return 0
546  
547      glob = config.option.cacheshow[0]
548      if glob is None:
549          glob = "*"
550  
551      dummy = object()
552      basedir = config.cache._cachedir
553      vdir = basedir / Cache._CACHE_PREFIX_VALUES
554      tw.sep("-", "cache values for %r" % glob)
555      for valpath in sorted(x for x in vdir.rglob(glob) if x.is_file()):
556          key = str(valpath.relative_to(vdir))
557          val = config.cache.get(key, dummy)
558          if val is dummy:
559              tw.line("%s contains unreadable content, will be ignored" % key)
560          else:
561              tw.line("%s contains:" % key)
562              for line in pformat(val).splitlines():
563                  tw.line("  " + line)
564  
565      ddir = basedir / Cache._CACHE_PREFIX_DIRS
566      if ddir.is_dir():
567          contents = sorted(ddir.rglob(glob))
568          tw.sep("-", "cache directories for %r" % glob)
569          for p in contents:
570              # if p.check(dir=1):
571              #    print("%s/" % p.relto(basedir))
572              if p.is_file():
573                  key = str(p.relative_to(basedir))
574                  tw.line(f"{key} is a file of length {p.stat().st_size:d}")
575      return 0
