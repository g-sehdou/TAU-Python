29   @fixture
30   def monkeypatch() -> Generator["MonkeyPatch", None, None]:
31       """A convenient fixture for monkey-patching. 
32    
33       The fixture provides these methods to modify objects, dictionaries or 
34       os.environ:: 
35    
36           monkeypatch.setattr(obj, name, value, raising=True) 
37           monkeypatch.delattr(obj, name, raising=True) 
38           monkeypatch.setitem(mapping, name, value) 
39           monkeypatch.delitem(obj, name, raising=True) 
40           monkeypatch.setenv(name, value, prepend=False) 
41           monkeypatch.delenv(name, raising=True) 
42           monkeypatch.syspath_prepend(path) 
43           monkeypatch.chdir(path) 
44    
45       All modifications will be undone after the requesting test function or 
46       fixture has finished. The ``raising`` parameter determines if a KeyError 
47       or AttributeError will be raised if the set/deletion operation has no target. 
48       """
49       mpatch = MonkeyPatch()
50       yield mpatch
51       mpatch.undo()

54   def resolve(name: str) -> object:
55       # Simplified from zope.dottedname.
56       parts = name.split(".")
57   
58       used = parts.pop(0)
59       found = __import__(used)
60       for part in parts:
61           used += "." + part
62           try:
63               found = getattr(found, part)
64           except AttributeError:
65               pass
66           else:
67               continue
68           # We use explicit un-nesting of the handling block in order
69           # to avoid nested exceptions.
70           try:
71               __import__(used)
72           except ImportError as ex:
73               expected = str(ex).split()[-1]
74               if expected == used:
75                   raise
76               else:
77                   raise ImportError(f"import error in {used}: {ex}") from ex
78           found = annotated_getattr(found, part, used)
79       return found

82   def annotated_getattr(obj: object, name: str, ann: str) -> object:
83       try:
84           obj = getattr(obj, name)
85       except AttributeError as e:
86           raise AttributeError(
87               "{!r} object at {} has no attribute {!r}".format(
88                   type(obj).__name__, ann, name
89               )
90           ) from e
91       return obj

94   def derive_importpath(import_path: str, raising: bool) -> Tuple[str, object]:
95       if not isinstance(import_path, str) or "." not in import_path:  # type: ignore[unreachable]
96           raise TypeError(f"must be absolute import path string, not {import_path!r}")
97       module, attr = import_path.rsplit(".", 1)
98       target = resolve(module)
99       if raising:
100          annotated_getattr(target, attr, ann=module)
101      return attr, target

112  @final
113  class MonkeyPatch:
114      """Helper to conveniently monkeypatch attributes/items/environment 
115      variables/syspath. 
116   
117      Returned by the :fixture:`monkeypatch` fixture. 
118   
119      :versionchanged:: 6.2 
120          Can now also be used directly as `pytest.MonkeyPatch()`, for when 
121          the fixture is not available. In this case, use 
122          :meth:`with MonkeyPatch.context() as mp: <context>` or remember to call 
123          :meth:`undo` explicitly. 
124      """
125  
126      def __init__(self) -> None:
127          self._setattr: List[Tuple[object, str, object]] = []
128          self._setitem: List[Tuple[MutableMapping[Any, Any], object, object]] = ([])
129          self._cwd: Optional[str] = None
130          self._savesyspath: Optional[List[str]] = None
131  
132      @classmethod
133      @contextmanager
134      def context(cls) -> Generator["MonkeyPatch", None, None]:
135          """Context manager that returns a new :class:`MonkeyPatch` object 
136          which undoes any patching done inside the ``with`` block upon exit. 
137   
138          Example: 
139   
140          .. code-block:: python 
141   
142              import functools 
143   
144   
145              def test_partial(monkeypatch): 
146                  with monkeypatch.context() as m: 
147                      m.setattr(functools, "partial", 3) 
148   
149          Useful in situations where it is desired to undo some patches before the test ends, 
150          such as mocking ``stdlib`` functions that might break pytest itself if mocked (for examples 
151          of this see `#3290 <https://github.com/pytest-dev/pytest/issues/3290>`_. 
152          """
153          m = cls()
154          try:
155              yield m
156          finally:
157              m.undo()

171      def setattr(
172          self,
173          target: Union[str, object],
174          name: Union[object, str],
175          value: object = notset,
176          raising: bool = True,
177      ) -> None:
178          """Set attribute value on target, memorizing the old value. 
179   
180          For convenience you can specify a string as ``target`` which 
181          will be interpreted as a dotted import path, with the last part 
182          being the attribute name. For example, 
183          ``monkeypatch.setattr("os.getcwd", lambda: "/")`` 
184          would set the ``getcwd`` function of the ``os`` module. 
185   
186          Raises AttributeError if the attribute does not exist, unless 
187          ``raising`` is set to False. 
188          """
189          __tracebackhide__ = True
190          import inspect
191  
192          if isinstance(value, Notset):
193              if not isinstance(target, str):
194                  raise TypeError(
195                      "use setattr(target, name, value) or "
196                      "setattr(target, value) with target being a dotted "
197                      "import string"
198                  )
199              value = name
200              name, target = derive_importpath(target, raising)
201          else:
202              if not isinstance(name, str):
203                  raise TypeError(
204                      "use setattr(target, name, value) with name being a string or "
205                      "setattr(target, value) with target being a dotted "
206                      "import string"
207                  )
208  
209          oldval = getattr(target, name, notset)
210          if raising and oldval is notset:
211              raise AttributeError(f"{target!r} has no attribute {name!r}")
212  
213          # avoid class descriptors like staticmethod/classmethod
214          if inspect.isclass(target):
215              oldval = target.__dict__.get(name, notset)
216          self._setattr.append((target, name, oldval))
217          setattr(target, name, value)
218  
219      def delattr(
220          self,
221          target: Union[object, str],
222          name: Union[str, Notset] = notset,
223          raising: bool = True,
224      ) -> None:
225          """Delete attribute ``name`` from ``target``. 
226   
227          If no ``name`` is specified and ``target`` is a string 
228          it will be interpreted as a dotted import path with the 
229          last part being the attribute name. 
230   
231          Raises AttributeError it the attribute does not exist, unless 
232          ``raising`` is set to False. 
233          """
234          __tracebackhide__ = True
235          import inspect
236  
237          if isinstance(name, Notset):
238              if not isinstance(target, str):
239                  raise TypeError(
240                      "use delattr(target, name) or "
241                      "delattr(target) with target being a dotted "
242                      "import string"
243                  )
244              name, target = derive_importpath(target, raising)
245  
246          if not hasattr(target, name):
247              if raising:
248                  raise AttributeError(name)
249          else:
250              oldval = getattr(target, name, notset)
251              # Avoid class descriptors like staticmethod/classmethod.
252              if inspect.isclass(target):
253                  oldval = target.__dict__.get(name, notset)
254              self._setattr.append((target, name, oldval))
255              delattr(target, name)

257      def setitem(self, dic: MutableMapping[K, V], name: K, value: V) -> None:
258          """Set dictionary entry ``name`` to value."""
259          self._setitem.append((dic, name, dic.get(name, notset)))
260          dic[name] = value
261  
262      def delitem(self, dic: MutableMapping[K, V], name: K, raising: bool = True) -> None:
263          """Delete ``name`` from dict. 
264   
265          Raises ``KeyError`` if it doesn't exist, unless ``raising`` is set to 
266          False. 
267          """
268          if name not in dic:
269              if raising:
270                  raise KeyError(name)
271          else:
272              self._setitem.append((dic, name, dic.get(name, notset)))
273              del dic[name]

275      def setenv(self, name: str, value: str, prepend: Optional[str] = None) -> None:
276          """Set environment variable ``name`` to ``value``. 
277   
278          If ``prepend`` is a character, read the current environment variable 
279          value and prepend the ``value`` adjoined with the ``prepend`` 
280          character. 
281          """
282          if not isinstance(value, str):
283              warnings.warn(  # type: ignore[unreachable]
284                  PytestWarning(
285                      "Value of environment variable {name} type should be str, but got "
286                      "{value!r} (type: {type}); converted to str implicitly".format(
287                          name=name, value=value, type=type(value).__name__
288                      )
289                  ),
290                  stacklevel=2,
291              )
292              value = str(value)
293          if prepend and name in os.environ:
294              value = value + prepend + os.environ[name]
295          self.setitem(os.environ, name, value)
296  
297      def delenv(self, name: str, raising: bool = True) -> None:
298          """Delete ``name`` from the environment. 
299   
300          Raises ``KeyError`` if it does not exist, unless ``raising`` is set to 
301          False. 
302          """
303          environ: MutableMapping[str, str] = os.environ
304          self.delitem(environ, name, raising=raising)

306      def syspath_prepend(self, path) -> None:
307          """Prepend ``path`` to ``sys.path`` list of import locations."""
308          from pkg_resources import fixup_namespace_packages
309  
310          if self._savesyspath is None:
311              self._savesyspath = sys.path[:]
312          sys.path.insert(0, str(path))
313  
314          # https://github.com/pypa/setuptools/blob/d8b901bc/docs/pkg_resources.txt#L162-L171
315          fixup_namespace_packages(str(path))
316  
317          # A call to syspathinsert() usually means that the caller wants to
318          # import some dynamically created files, thus with python3 we
319          # invalidate its import caches.
320          # This is especially important when any namespace package is in use,
321          # since then the mtime based FileFinder cache (that gets created in
322          # this case already) gets not invalidated when writing the new files
323          # quickly afterwards.
324          from importlib import invalidate_caches
325  
326          invalidate_caches()

328      def chdir(self, path) -> None:
329          """Change the current working directory to the specified path. 
330   
331          Path can be a string or a py.path.local object. 
332          """
333          if self._cwd is None:
334              self._cwd = os.getcwd()
335          if hasattr(path, "chdir"):
336              path.chdir()
337          elif isinstance(path, Path):
338              # Modern python uses the fspath protocol here LEGACY
339              os.chdir(str(path))
340          else:
341              os.chdir(path)

343      def undo(self) -> None:
344          """Undo previous changes. 
345   
346          This call consumes the undo stack. Calling it a second time has no 
347          effect unless you do more monkeypatching after the undo call. 
348   
349          There is generally no need to call `undo()`, since it is 
350          called automatically during tear-down. 
351   
352          Note that the same `monkeypatch` fixture is used across a 
353          single test function invocation. If `monkeypatch` is used both by 
354          the test function itself and one of the test fixtures, 
355          calling `undo()` will undo all of the changes made in 
356          both functions. 
357          """
358          for obj, name, value in reversed(self._setattr):
359              if value is not notset:
360                  setattr(obj, name, value)
361              else:
362                  delattr(obj, name)
363          self._setattr[:] = []
364          for dictionary, key, value in reversed(self._setitem):
365              if value is notset:
366                  try:
367                      del dictionary[key]
368                  except KeyError:
369                      pass  # Was already deleted, so we have the desired state.
370              else:
371                  dictionary[key] = value
372          self._setitem[:] = []
373          if self._savesyspath is not None:
374              sys.path[:] = self._savesyspath
375              self._savesyspath = None
376  
377          if self._cwd is not None:
378              os.chdir(self._cwd)
379              self._cwd = None
