36   def as_base_candidate(candidate: Candidate) -> Optional[BaseCandidate]:
37       """The runtime version of BaseCandidate."""
38       base_candidate_classes = (
39           AlreadyInstalledCandidate,
40           EditableCandidate,
41           LinkCandidate,
42       )
43       if isinstance(candidate, base_candidate_classes):
44           return candidate
45       return None

48   def make_install_req_from_link(
49       link: Link, template: InstallRequirement
50   ) -> InstallRequirement:
51       assert not template.editable, "template is editable"
52       if template.req:
53           line = str(template.req)
54       else:
55           line = link.url
56       ireq = install_req_from_line(
57           line,
58           user_supplied=template.user_supplied,
59           comes_from=template.comes_from,
60           use_pep517=template.use_pep517,
61           isolated=template.isolated,
62           constraint=template.constraint,
63           options=dict(
64               install_options=template.install_options,
65               global_options=template.global_options,
66               hashes=template.hash_options,
67           ),
68       )
69       ireq.original_link = template.original_link
70       ireq.link = link
71       return ireq

94   def _make_install_req_from_dist(
95       dist: BaseDistribution, template: InstallRequirement
96   ) -> InstallRequirement:
97       from pip._internal.metadata.pkg_resources import Distribution as _Dist
98   
99       if template.req:
100          line = str(template.req)
101      elif template.link:
102          line = f"{dist.canonical_name} @ {template.link.url}"
103      else:
104          line = f"{dist.canonical_name}=={dist.version}"
105      ireq = install_req_from_line(
106          line,
107          user_supplied=template.user_supplied,
108          comes_from=template.comes_from,
109          use_pep517=template.use_pep517,
110          isolated=template.isolated,
111          constraint=template.constraint,
112          options=dict(
113              install_options=template.install_options,
114              global_options=template.global_options,
115              hashes=template.hash_options,
116          ),
117      )
118      ireq.satisfied_by = cast(_Dist, dist)._dist
119      return ireq

122  class _InstallRequirementBackedCandidate(Candidate):
123      """A candidate backed by an ``InstallRequirement``. 
124   
125      This represents a package request with the target not being already 
126      in the environment, and needs to be fetched and installed. The backing 
127      ``InstallRequirement`` is responsible for most of the leg work; this 
128      class exposes appropriate information to the resolver. 
129   
130      :param link: The link passed to the ``InstallRequirement``. The backing 
131          ``InstallRequirement`` will use this link to fetch the distribution. 
132      :param source_link: The link this candidate "originates" from. This is 
133          different from ``link`` when the link is found in the wheel cache. 
134          ``link`` would point to the wheel cache, while this points to the 
135          found remote link (e.g. from pypi.org). 
136      """
137  
138      dist: BaseDistribution
139      is_installed = False
140  
141      def __init__(
142          self,
143          link: Link,
144          source_link: Link,
145          ireq: InstallRequirement,
146          factory: "Factory",
147          name: Optional[NormalizedName] = None,
148          version: Optional[CandidateVersion] = None,
149      ) -> None:
150          self._link = link
151          self._source_link = source_link
152          self._factory = factory
153          self._ireq = ireq
154          self._name = name
155          self._version = version
156          self.dist = self._prepare()
157  
158      def __str__(self) -> str:
159          return f"{self.name} {self.version}"
160  
161      def __repr__(self) -> str:
162          return "{class_name}({link!r})".format(
163              class_name=self.__class__.__name__,
164              link=str(self._link),
165          )
166  
167      def __hash__(self) -> int:
168          return hash((self.__class__, self._link))
169  
170      def __eq__(self, other: Any) -> bool:
171          if isinstance(other, self.__class__):
172              return links_equivalent(self._link, other._link)
173          return False
174  
175      @property
176      def source_link(self) -> Optional[Link]:
177          return self._source_link
178  
179      @property
180      def project_name(self) -> NormalizedName:
181          """The normalised name of the project the candidate refers to"""
182          if self._name is None:
183              self._name = self.dist.canonical_name
184          return self._name
185  
186      @property
187      def name(self) -> str:
188          return self.project_name
189  
190      @property
191      def version(self) -> CandidateVersion:
192          if self._version is None:
193              self._version = self.dist.version
194          return self._version
195  
196      def format_for_error(self) -> str:
197          return "{} {} (from {})".format(
198              self.name,
199              self.version,
200              self._link.file_path if self._link.is_file else self._link,
201          )
202  
203      def _prepare_distribution(self) -> BaseDistribution:
204          raise NotImplementedError("Override in subclass")
205  
206      def _check_metadata_consistency(self, dist: BaseDistribution) -> None:
207          """Check for consistency of project name and version of dist."""
208          if self._name is not None and self._name != dist.canonical_name:
209              raise MetadataInconsistent(
210                  self._ireq,
211                  "name",
212                  self._name,
213                  dist.canonical_name,
214              )
215          if self._version is not None and self._version != dist.version:
216              raise MetadataInconsistent(
217                  self._ireq,
218                  "version",
219                  str(self._version),
220                  str(dist.version),
221              )
222  
223      def _prepare(self) -> BaseDistribution:
224          try:
225              dist = self._prepare_distribution()
226          except HashError as e:
227              # Provide HashError the underlying ireq that caused it. This
228              # provides context for the resulting error message to show the
229              # offending line to the user.
230              e.req = self._ireq
231              raise
232          self._check_metadata_consistency(dist)
233          return dist
234  
235      def iter_dependencies(self, with_requires: bool) -> Iterable[Optional[Requirement]]:
236          requires = self.dist.iter_dependencies() if with_requires else ()
237          for r in requires:
238              yield self._factory.make_requirement_from_spec(str(r), self._ireq)
239          yield self._factory.make_requires_python_requirement(self.dist.requires_python)
240  
241      def get_install_requirement(self) -> Optional[InstallRequirement]:
242          return self._ireq

245  class LinkCandidate(_InstallRequirementBackedCandidate):
246      is_editable = False
247  
248      def __init__(
249          self,
250          link: Link,
251          template: InstallRequirement,
252          factory: "Factory",
253          name: Optional[NormalizedName] = None,
254          version: Optional[CandidateVersion] = None,
255      ) -> None:
256          source_link = link
257          cache_entry = factory.get_wheel_cache_entry(link, name)
258          if cache_entry is not None:
259              logger.debug("Using cached wheel link: %s", cache_entry.link)
260              link = cache_entry.link
261          ireq = make_install_req_from_link(link, template)
262          assert ireq.link == link
263          if ireq.link.is_wheel and not ireq.link.is_file:
264              wheel = Wheel(ireq.link.filename)
265              wheel_name = canonicalize_name(wheel.name)
266              assert name == wheel_name, f"{name!r} != {wheel_name!r} for wheel"
267              # Version may not be present for PEP 508 direct URLs
268              if version is not None:
269                  wheel_version = Version(wheel.version)
270                  assert version == wheel_version, "{!r} != {!r} for wheel {}".format(
271                      version, wheel_version, name
272                  )
273  
274          if (
275              cache_entry is not None
276              and cache_entry.persistent
277              and template.link is template.original_link
278          ):
279              ireq.original_link_is_in_wheel_cache = True
280  
281          super().__init__(
282              link=link,
283              source_link=source_link,
284              ireq=ireq,
285              factory=factory,
286              name=name,
287              version=version,
288          )
289  
290      def _prepare_distribution(self) -> BaseDistribution:
291          preparer = self._factory.preparer
292          return preparer.prepare_linked_requirement(self._ireq, parallel_builds=True)

323      def __init__(
324          self,
325          dist: BaseDistribution,
326          template: InstallRequirement,

328      ) -> None:
329          self.dist = dist
330          self._ireq = _make_install_req_from_dist(dist, template)

333          # This is just logging some messages, so we can do it eagerly.

347          )

350          return hash((self.__class__, self.name, self.version))
351  
352      def __eq__(self, other: Any) -> bool:
353          if isinstance(other, self.__class__):
354              return self.name == other.name and self.version == other.version
355          return False

386  class ExtrasCandidate(Candidate):
387      """A candidate that has 'extras', indicating additional dependencies. 
388   
389      Requirements can be for a project with dependencies, something like 
390      foo[extra].  The extras don't affect the project/version being installed 
391      directly, but indicate that we need additional dependencies. We model that 
392      by having an artificial ExtrasCandidate that wraps the "base" candidate. 
393   
394      The ExtrasCandidate differs from the base in the following ways: 
395   
396      1. It has a unique name, of the form foo[extra]. This causes the resolver 
397         to treat it as a separate node in the dependency graph. 
398      2. When we're getting the candidate's dependencies, 
399         a) We specify that we want the extra dependencies as well. 
400         b) We add a dependency on the base candidate. 
401            See below for why this is needed. 
402      3. We return None for the underlying InstallRequirement, as the base 
403         candidate will provide it, and we don't want to end up with duplicates. 
404   
405      The dependency on the base candidate is needed so that the resolver can't 
406      decide that it should recommend foo[extra1] version 1.0 and foo[extra2] 
407      version 2.0. Having those candidates depend on foo=1.0 and foo=2.0 
408      respectively forces the resolver to recognise that this is a conflict. 
409      """
410  
411      def __init__(
412          self,
413          base: BaseCandidate,
414          extras: FrozenSet[str],
415      ) -> None:
416          self.base = base
417          self.extras = extras
418  
419      def __str__(self) -> str:
420          name, rest = str(self.base).split(" ", 1)
421          return "{}[{}] {}".format(name, ",".join(self.extras), rest)
422  
423      def __repr__(self) -> str:
424          return "{class_name}(base={base!r}, extras={extras!r})".format(
425              class_name=self.__class__.__name__,
426              base=self.base,
427              extras=self.extras,
428          )
429  
430      def __hash__(self) -> int:
431          return hash((self.base, self.extras))
432  
433      def __eq__(self, other: Any) -> bool:
434          if isinstance(other, self.__class__):
435              return self.base == other.base and self.extras == other.extras
436          return False
437  
438      @property
439      def project_name(self) -> NormalizedName:
440          return self.base.project_name
441  
442      @property
443      def name(self) -> str:
444          """The normalised name of the project the candidate refers to"""
445          return format_name(self.base.project_name, self.extras)
446  
447      @property
448      def version(self) -> CandidateVersion:
449          return self.base.version
450  
451      def format_for_error(self) -> str:
452          return "{} [{}]".format(
453              self.base.format_for_error(), ", ".join(sorted(self.extras))
454          )
455  
456      @property
457      def is_installed(self) -> bool:
458          return self.base.is_installed
459  
460      @property
461      def is_editable(self) -> bool:
462          return self.base.is_editable
463  
464      @property
465      def source_link(self) -> Optional[Link]:
466          return self.base.source_link
467  
468      def iter_dependencies(self, with_requires: bool) -> Iterable[Optional[Requirement]]:
469          factory = self.base._factory
470  
471          # Add a dependency on the exact base
472          # (See note 2b in the class docstring)
473          yield factory.make_requirement_from_candidate(self.base)
474          if not with_requires:
475              return
476  
477          # The user may have specified extras that the candidate doesn't
478          # support. We ignore any unsupported extras here.
479          valid_extras = self.extras.intersection(self.base.dist.iter_provided_extras())
480          invalid_extras = self.extras.difference(self.base.dist.iter_provided_extras())
481          for extra in sorted(invalid_extras):
482              logger.warning(
483                  "%s %s does not provide the extra '%s'",
484                  self.base.name,
485                  self.version,
486                  extra,
487              )
488  
489          for r in self.base.dist.iter_dependencies(valid_extras):
490              requirement = factory.make_requirement_from_spec(
491                  str(r), self.base._ireq, valid_extras
492              )
493              if requirement:
494                  yield requirement
495  
496      def get_install_requirement(self) -> Optional[InstallRequirement]:
497          # We don't return anything here, because we always
498          # depend on the base candidate, and we'll get the
499          # install requirement from that.
500          return None

503  class RequiresPythonCandidate(Candidate):
504      is_installed = False
505      source_link = None
506  
507      def __init__(self, py_version_info: Optional[Tuple[int, ...]]) -> None:
508          if py_version_info is not None:
509              version_info = normalize_version_info(py_version_info)
510          else:
511              version_info = sys.version_info[:3]
512          self._version = Version(".".join(str(c) for c in version_info))
513  
514      # We don't need to implement __eq__() and __ne__() since there is always
515      # only one RequiresPythonCandidate in a resolution, i.e. the host Python.
516      # The built-in object.__eq__() and object.__ne__() do exactly what we want.
517  
518      def __str__(self) -> str:
519          return f"Python {self._version}"
520  
521      @property
522      def project_name(self) -> NormalizedName:
523          return REQUIRES_PYTHON_IDENTIFIER
524  
525      @property
526      def name(self) -> str:
527          return REQUIRES_PYTHON_IDENTIFIER
528  
529      @property
530      def version(self) -> CandidateVersion:
531          return self._version
532  
533      def format_for_error(self) -> str:
534          return f"Python {self.version}"
535  
536      def iter_dependencies(self, with_requires: bool) -> Iterable[Optional[Requirement]]:
537          return ()

539      def get_install_requirement(self) -> Optional[InstallRequirement]:
540          return None
