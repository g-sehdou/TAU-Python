3    """ 
4    A **fixture** provides a concept to simplify test support functionality 
5    that needs a setup/cleanup cycle per scenario, feature or test-run. 
6    A fixture is provided as fixture-function that contains the setup part and 
7    cleanup part similar to :func:`contextlib.contextmanager` or `pytest.fixture`_. 
8     
9    .. _pytest.fixture: https://docs.pytest.org/en/latest/fixture.html 
10    
11   A fixture is used when: 
12    
13   * the (registered) fixture tag is used for a scenario or feature 
14   * the :func:`.use_fixture()` is called in the environment file (normally) 
15    
16   .. sourcecode:: python 
17    
18       # -- FILE: behave4my_project/fixtures.py (or: features/environment.py) 
19       from behave import fixture 
20       from somewhere.browser.firefox import FirefoxBrowser 
21    
22       @fixture 
23       def browser_firefox(context, timeout=30, **kwargs): 
24           # -- SETUP-FIXTURE PART: 
25           context.browser = FirefoxBrowser(timeout, *args, **kwargs) 
26           yield context.browser 
27           # -- CLEANUP-FIXTURE PART: 
28           context.browser.shutdown() 
29    
30   .. sourcecode:: gherkin 
31    
32       # -- FILE: features/use_fixture.feature 
33       Feature: Use Fixture in Scenario 
34    
35           @fixture.browser.firefox 
36           Scenario: Use browser=firefox 
37             Given I use the browser 
38             ... 
39           # -- AFTER-SCENEARIO: Cleanup fixture.browser.firefox 
40    
41   .. sourcecode:: python 
42    
43       # -- FILE: features/environment.py 
44       from behave import use_fixture 
45       from behave4my_project.fixtures import browser_firefox 
46    
47       def before_tag(context, tag): 
48           if tag == "fixture.browser.firefox": 
49               # -- Performs fixture setup and registers fixture cleanup 
50               use_fixture(browser_firefox, context, timeout=10) 
51    
52   .. hidden: 
53    
54       BEHAVIORAL DECISIONS: 
55    
56       * Should scenario/feature be executed when fixture-setup fails 
57         (similar to before-hook failures) ? 
58         NO, scope is skipped, but after-hooks and cleanups are executed. 
59    
60       * Should remaining fixture-setups be performed after first fixture fails? 
61         NO, first setup-error aborts the setup and execution of the scope. 
62    
63       * Should remaining fixture-cleanups be performed when first cleanup-error 
64         occurs? 
65         YES, try to perform all fixture-cleanups and then reraise the 
66         first cleanup-error. 
67    
68    
69       OPEN ISSUES: 
70    
71       * AUTO_CALL_REGISTERED_FIXTURE (planned in future): 
72           Run fixture setup before or after before-hooks? 
73    
74       IDEAS: 
75    
76       * Fixture registers itself in fixture registry (runtime context). 
77       * Code in before_tag() will either be replaced w/ fixture processing function 
78         or will be automatically be executed (AUTO_CALL_REGISTERED_FIXTURE) 
79       * Support fixture tags w/ parameters that are automatically parsed and 
80         passed to fixture function, like: 
81         @fixture(name="foo", pattern="{name}={browser}") 
82   """

90   def iscoroutinefunction(func):
91       """Checks if a function is a coroutine-function, like: 
92    
93        * ``async def f(): ...`` (since Python 3.5) 
94        * ``@asyncio.coroutine def f(): ...`` (since Python3) 
95    
96       .. note:: Compatibility helper 
97    
98           Avoids to import :mod:`asyncio` module directly (since Python3), 
99           which in turns initializes the :mod:`logging` module as side-effect. 
100   
101      :param func:  Function to check. 
102      :return: True, if function is a coroutine function. 
103               False, otherwise. 
104      """
105      # -- NOTE: inspect.iscoroutinefunction() is available since Python 3.5
106      #    Checks also if @asyncio.coroutine decorator is not used.
107      # pylint: disable=no-member
108      return (getattr(func, "_is_coroutine", False) or
109              (hasattr(inspect, "iscoroutinefunction") and
110               inspect.iscoroutinefunction(func)))

113  def is_context_manager(func):
114      """Checks if a fixture function provides context-manager functionality, 
115      similar to :func`contextlib.contextmanager()` function decorator. 
116   
117      .. code-block:: python 
118   
119          @fixture 
120          def foo(context, *args, **kwargs): 
121              context.foo = setup_foo() 
122              yield context.foo 
123              cleanup_foo() 
124   
125          @fixture 
126          def bar(context, *args, **kwargs): 
127              context.bar = setup_bar() 
128              return context.bar 
129   
130          assert is_context_manager(foo) is True      # Generator-function 
131          assert is_context_manager(bar) is False     # Normal function 
132   
133      :param func:    Function to check. 
134      :return: True, if function is a generator/context-manager function. 
135               False, otherwise. 
136      """
137      genfunc = inspect.isgeneratorfunction(func)
138      return genfunc and not iscoroutinefunction(func)

144  class InvalidFixtureError(RuntimeError):
145      """Raised when a fixture is invalid. 
146      This occurs when a generator-function with more than one yield statement 
147      is used as fixture-function. 
148      """

154  def _setup_fixture(fixture_func, context, *fixture_args, **fixture_kwargs):
155      """Provides core functionality to setup a fixture and registers its 
156      cleanup part (if needed). 
157      """
158      if is_context_manager(fixture_func):
159          # -- CASE: Fixture function is a two-step generator (setup, cleanup).
160          def cleanup_fixture():
161              try:
162                  next(func_it)           # CLEANUP-FIXTURE PART
163                  # -- USE func_it: From outer scope (here).
164              except StopIteration:
165                  return False
166              # -- NOT-NEEDED:
167              # except Exception:
168              #    raise   # -- CLEANUP-FIXTURE PART raised an error.
169              else:
170                  message = "Has more than one yield: %r" % fixture_func
171                  raise InvalidFixtureError(message)
172  
173          # -- GENERATOR: Get instance via call and register cleanup.
174          #  Then perform setup_fixture to ensure that cleanup_fixture()
175          #  is called even if setup-error occurs.
176          #  NOTE: cleanup_fixture() is called when context layer is removed.
177          func_it = fixture_func(context, *fixture_args, **fixture_kwargs)
178          context.add_cleanup(cleanup_fixture)
179          setup_result = next(func_it) # SETUP-FIXTURE PART (may raise error)
180      else:
181          # -- CASE: Fixture is a simple function (setup-only)
182          # NOTE: No cleanup is registered (not needed by intention of user)
183          setup_result = fixture_func(context, *fixture_args, **fixture_kwargs)
184      return setup_result

187  def use_fixture(fixture_func, context, *fixture_args, **fixture_kwargs):
188      """Use fixture (function) and call it to perform its setup-part. 
189   
190      The fixture-function is similar to a :func:`contextlib.contextmanager` 
191      (and contains a yield-statement to seperate setup and cleanup part). 
192      If it contains a yield-statement, it registers a context-cleanup function 
193      to the context object to perform the fixture-cleanup at the end of the 
194      current scoped when the context layer is removed 
195      (and all context-cleanup functions are called). 
196   
197      Therefore, fixture-cleanup is performed after scenario, feature or test-run 
198      (depending when its fixture-setup is performed). 
199   
200      .. code-block:: python 
201   
202          # -- FILE: behave4my_project/fixtures.py (or: features/environment.py) 
203          from behave import fixture 
204          from somewhere.browser import FirefoxBrowser 
205   
206          @fixture(name="fixture.browser.firefox") 
207          def browser_firefox(context, *args, **kwargs): 
208              # -- SETUP-FIXTURE PART: 
209              context.browser = FirefoxBrowser(*args, **kwargs) 
210              yield context.browser 
211              # -- CLEANUP-FIXTURE PART: 
212              context.browser.shutdown() 
213   
214      .. code-block:: python 
215   
216          # -- FILE: features/environment.py 
217          from behave import use_fixture 
218          from behave4my_project.fixtures import browser_firefox 
219   
220          def before_tag(context, tag): 
221              if tag == "fixture.browser.firefox": 
222                  use_fixture(browser_firefox, context, timeout=10) 
223   
224   
225      :param fixture_func: Fixture function to use. 
226      :param context: Context object to use 
227      :param fixture_kwargs: Positional args, passed to the fixture function. 
228      :param fixture_kwargs: Additional kwargs, passed to the fixture function. 
229      :return: Setup result object (may be None). 
230      """
231      return _setup_fixture(fixture_func, context, *fixture_args, **fixture_kwargs)

234  def use_fixture_by_tag(tag, context, fixture_registry):
235      """Process any fixture-tag to perform :func:`use_fixture()` for its fixture. 
236      If the fixture-tag is known, the fixture data is retrieved from the 
237      fixture registry. 
238   
239      .. code-block:: python 
240   
241          # -- FILE: features/environment.py 
242          from behave.fixture import use_fixture_by_tag 
243          from behave4my_project.fixtures import browser_firefox, browser_chrome 
244   
245          # -- SCHEMA 1: fixture_func 
246          fixture_registry1 = { 
247              "fixture.browser.firefox": browser_firefox, 
248              "fixture.browser.chrome":  browser_chrome, 
249          } 
250          # -- SCHEMA 2: fixture_func, fixture_args, fixture_kwargs 
251          fixture_registry2 = { 
252              "fixture.browser.firefox": (browser_firefox, (), dict(timeout=10)), 
253              "fixture.browser.chrome":  (browser_chrome,  (), dict(timeout=12)), 
254          } 
255   
256          def before_tag(context, tag): 
257              if tag.startswith("fixture."): 
258                  return use_fixture_by_tag(tag, context, fixture_registry1): 
259              # -- MORE: Tag processing steps ... 
260   
261   
262      :param tag:     Fixture tag to process. 
263      :param context: Runtime context object, used for :func:`use_fixture()`. 
264      :param fixture_registry:  Registry maps fixture-tag to fixture data. 
265      :return: Fixture-setup result (same as: use_fixture()) 
266      :raises LookupError: If fixture-tag/fixture is unknown. 
267      :raises ValueError: If fixture data type is not supported. 
268      """
269      fixture_data = fixture_registry.get(tag, None)
270      if fixture_data is None:
271          raise LookupError("Unknown fixture-tag: %s" % tag)
272  
273      if callable(fixture_data):
274          fixture_func = fixture_data
275          use_fixture(fixture_func, context)
276      elif isinstance(fixture_data, (tuple, list)):
277          assert len(fixture_data) == 3
278          fixture_func, fixture_args, fixture_kwargs = fixture_data
279          return use_fixture(fixture_func, context, *fixture_args, **fixture_kwargs)
280      else:
281          message = "fixture_data: Expected tuple or fixture-func, but is: %r"
282          raise ValueError(message % fixture_data)

286      # -- SEE: use_composite_fixture_with()
287      return (fixture_func, args, kwargs)

290  def use_composite_fixture_with(context, fixture_funcs_with_params):
291      """Helper function when complex fixtures should be created and 
292      safe-cleanup is needed even if an setup-fixture-error occurs. 
293   
294      This function ensures that fixture-cleanup is performed 
295      for every fixture that was setup before the setup-error occured. 
296   
297      .. code-block:: python 
298   
299          # -- BAD-EXAMPLE: Simplistic composite-fixture 
300          # NOTE: Created fixtures (fixture1) are not cleaned up. 
301          @fixture 
302          def foo_and_bad0(context, *args, **kwargs): 
303              the_fixture1 = setup_fixture_foo(*args, **kwargs) 
304              the_fixture2 = setup_fixture_bar_with_error("OOPS-HERE") 
305              yield (the_fixture1, the_fixture2)  # NOT_REACHED. 
306              # -- NOT_REACHED: Due to fixture2-setup-error. 
307              the_fixture1.cleanup()  # NOT-CALLED (SAD). 
308              the_fixture2.cleanup()  # OOPS, the_fixture2 is None (if called). 
309   
310      .. code-block:: python 
311   
312          # -- GOOD-EXAMPLE: Sane composite-fixture 
313          # NOTE: Fixture foo.cleanup() occurs even after fixture2-setup-error. 
314          @fixture 
315          def foo(context, *args, **kwargs): 
316              the_fixture = setup_fixture_foo(*args, **kwargs) 
317              yield the_fixture 
318              cleanup_fixture_foo(the_fixture) 
319   
320          @fixture 
321          def bad_with_setup_error(context, *args, **kwargs): 
322              raise RuntimeError("BAD-FIXTURE-SETUP") 
323   
324          # -- SOLUTION 1: With use_fixture() 
325          @fixture 
326          def foo_and_bad1(context, *args, **kwargs): 
327              the_fixture1 = use_fixture(foo, context, *args, **kwargs) 
328              the_fixture2 = use_fixture(bad_with_setup_error, context, "OOPS") 
329              return (the_fixture1, the_fixture2) # NOT_REACHED 
330   
331          # -- SOLUTION 2: With use_composite_fixture_with() 
332          @fixture 
333          def foo_and_bad2(context, *args, **kwargs): 
334              the_fixture = use_composite_fixture_with(context, [ 
335                  fixture_call_params(foo, *args, **kwargs), 
336                  fixture_call_params(bad_with_setup_error, "OOPS") 
337               ]) 
338              return the_fixture 
339   
340      :param context:     Runtime context object, used for all fixtures. 
341      :param fixture_funcs_with_params: List of fixture functions with params. 
342      :return: List of created fixture objects. 
343      """
344      composite_fixture = []
345      for fixture_func, args, kwargs in fixture_funcs_with_params:
346          the_fixture = use_fixture(fixture_func, context, *args, **kwargs)
347          composite_fixture.append(the_fixture)
348      return composite_fixture

355  def fixture(func=None, name=None, pattern=None):
356      """Fixture decorator (currently mostly syntactic sugar). 
357   
358      .. code-block:: python 
359   
360          # -- FILE: features/environment.py 
361          # CASE FIXTURE-GENERATOR-FUNCTION (like @contextlib.contextmanager): 
362          @fixture 
363          def foo(context, *args, **kwargs): 
364              the_fixture = setup_fixture_foo(*args, **kwargs) 
365              context.foo = the_fixture 
366              yield the_fixture 
367              cleanup_fixture_foo(the_fixture) 
368   
369          # CASE FIXTURE-FUNCTION: No cleanup or cleanup via context-cleanup. 
370          @fixture(name="fixture.bar") 
371          def bar(context, *args, **kwargs): 
372              the_fixture = setup_fixture_bar(*args, **kwargs) 
373              context.bar = the_fixture 
374              context.add_cleanup(cleanup_fixture_bar, the_fixture.cleanup) 
375              return the_fixture 
376   
377      :param name:    Specifies the fixture tag name (as string). 
378   
379      .. seealso:: 
380   
381          * :func:`contextlib.contextmanager` decorator 
382          * `@pytest.fixture`_ 
383      """
384      def mark_as_fixture(func, name=None, pattern=None):
385          func.name = name
386          func.pattern = pattern
387          func.behave_fixture = True
388          return func
389  
390      if func is None:
391          # CASE: @fixture()
392          # CASE: @fixture(name="foo")
393          def decorator(func):
394              return mark_as_fixture(func, name, pattern=pattern)
395          return decorator
396      elif callable(func):
397          # CASE: @fixture
398          return mark_as_fixture(func, name, pattern=pattern)
399      else:
400          # -- OOPS: Should be callable-object or None
401          message = "Invalid func: func=%r, name=%r" % (func, name)
402          raise TypeError(message)
