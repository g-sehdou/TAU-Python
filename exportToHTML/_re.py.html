34   def match_to_datetime(match: "Match") -> Union[datetime, date]:
35       """Convert a `RE_DATETIME` match to `datetime.datetime` or `datetime.date`. 
36    
37       Raises ValueError if the match does not correspond to a valid date 
38       or datetime. 
39       """
40       (
41           year_str,
42           month_str,
43           day_str,
44           hour_str,
45           minute_str,
46           sec_str,
47           micros_str,
48           zulu_time,
49           offset_dir_str,
50           offset_hour_str,
51           offset_minute_str,
52       ) = match.groups()
53       year, month, day = int(year_str), int(month_str), int(day_str)
54       if hour_str is None:
55           return date(year, month, day)
56       hour, minute, sec = int(hour_str), int(minute_str), int(sec_str)
57       micros = int(micros_str[1:].ljust(6, "0")[:6]) if micros_str else 0
58       if offset_dir_str:
59           offset_dir = 1 if offset_dir_str == "+" else -1
60           tz: Optional[tzinfo] = timezone(
61               timedelta(
62                   hours=offset_dir * int(offset_hour_str),
63                   minutes=offset_dir * int(offset_minute_str),
64               )
65           )
66       elif zulu_time:
67           tz = timezone.utc
68       else:  # local date-time
69           tz = None
70       return datetime(year, month, day, hour, minute, sec, micros, tzinfo=tz)

73   def match_to_localtime(match: "Match") -> time:
74       hour_str, minute_str, sec_str, micros_str = match.groups()
75       micros = int(micros_str[1:].ljust(6, "0")[:6]) if micros_str else 0
76       return time(int(hour_str), int(minute_str), int(sec_str), micros)

79   def match_to_number(match: "Match", parse_float: "ParseFloat") -> Any:
80       match_str = match.group()
81       if "." in match_str or "e" in match_str or "E" in match_str:
82           return parse_float(match_str)
83       return int(match_str)
