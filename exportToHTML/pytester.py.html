119  class LsofFdLeakChecker:
120      def get_open_files(self) -> List[Tuple[str, str]]:
121          out = subprocess.run(
122              ("lsof", "-Ffn0", "-p", str(os.getpid())),
123              stdout=subprocess.PIPE,
124              stderr=subprocess.DEVNULL,
125              check=True,
126              universal_newlines=True,
127          ).stdout
128  
129          def isopen(line: str) -> bool:
130              return line.startswith("f") and (
131                  "deleted" not in line
132                  and "mem" not in line
133                  and "txt" not in line
134                  and "cwd" not in line
135              )
136  
137          open_files = []
138  
139          for line in out.split("\n"):
140              if isopen(line):
141                  fields = line.split("\0")
142                  fd = fields[0][1:]
143                  filename = fields[1][1:]
144                  if filename in IGNORE_PAM:
145                      continue
146                  if filename.startswith("/"):
147                      open_files.append((fd, filename))
148  
149          return open_files
150  
151      def matching_platform(self) -> bool:
152          try:
153              subprocess.run(("lsof", "-v"), check=True)
154          except (OSError, subprocess.CalledProcessError):
155              return False
156          else:
157              return True
158  
159      @hookimpl(hookwrapper=True, tryfirst=True)
160      def pytest_runtest_protocol(self, item: Item) -> Generator[None, None, None]:
161          lines1 = self.get_open_files()
162          yield
163          if hasattr(sys, "pypy_version_info"):
164              gc.collect()
165          lines2 = self.get_open_files()
166  
167          new_fds = {t[0] for t in lines2} - {t[0] for t in lines1}
168          leaked_files = [t for t in lines2 if t[0] in new_fds]
169          if leaked_files:
170              error = [
171                  "***** %s FD leakage detected" % len(leaked_files),
172                  *(str(f) for f in leaked_files),
173                  "*** Before:",
174                  *(str(f) for f in lines1),
175                  "*** After:",
176                  *(str(f) for f in lines2),
177                  "***** %s FD leakage detected" % len(leaked_files),
178                  "*** function %s:%s: %s " % item.location,
179                  "See issue #2366",
180              ]
181              item.warn(PytestWarning("\n".join(error)))

226  class HookRecorder:
227      """Record all hooks called in a plugin manager. 
228   
229      This wraps all the hook calls in the plugin manager, recording each call 
230      before propagating the normal calls. 
231      """
232  
233      def __init__(self, pluginmanager: PytestPluginManager) -> None:
234          self._pluginmanager = pluginmanager
235          self.calls: List[ParsedCall] = []
236          self.ret: Optional[Union[int, ExitCode]] = None
237  
238          def before(hook_name: str, hook_impls, kwargs) -> None:
239              self.calls.append(ParsedCall(hook_name, kwargs))
240  
241          def after(outcome, hook_name: str, hook_impls, kwargs) -> None:
242              pass
243  
244          self._undo_wrapping = pluginmanager.add_hookcall_monitoring(before, after)
245  
246      def finish_recording(self) -> None:
247          self._undo_wrapping()
248  
249      def getcalls(self, names: Union[str, Iterable[str]]) -> List[ParsedCall]:
250          if isinstance(names, str):
251              names = names.split()
252          return [call for call in self.calls if call._name in names]
253  
254      def assert_contains(self, entries: Sequence[Tuple[str, str]]) -> None:
255          __tracebackhide__ = True
256          i = 0
257          entries = list(entries)
258          backlocals = sys._getframe(1).f_locals
259          while entries:
260              name, check = entries.pop(0)
261              for ind, call in enumerate(self.calls[i:]):
262                  if call._name == name:
263                      print("NAMEMATCH", name, call)
264                      if eval(check, backlocals, call.__dict__):
265                          print("CHECKERMATCH", repr(check), "->", call)
266                      else:
267                          print("NOCHECKERMATCH", repr(check), "-", call)
268                          continue
269                      i += ind + 1
270                      break
271                  print("NONAMEMATCH", name, "with", call)
272              else:
273                  fail(f"could not find {name!r} check {check!r}")
274  
275      def popcall(self, name: str) -> ParsedCall:
276          __tracebackhide__ = True
277          for i, call in enumerate(self.calls):
278              if call._name == name:
279                  del self.calls[i]
280                  return call
281          lines = [f"could not find call {name!r}, in:"]
282          lines.extend(["  %s" % x for x in self.calls])
283          fail("\n".join(lines))
284  
285      def getcall(self, name: str) -> ParsedCall:
286          values = self.getcalls(name)
287          assert len(values) == 1, (name, values)
288          return values[0]

323      def matchreport(
324          self,
325          inamepart: str = "",
326          names: Union[str, Iterable[str]] = (
327              "pytest_runtest_logreport",
328              "pytest_collectreport",
329          ),
330          when: Optional[str] = None,
331      ) -> Union[CollectReport, TestReport]:
332          """Return a testreport whose dotted import path matches."""
333          values = []
334          for rep in self.getreports(names=names):
335              if not when and rep.when != "call" and rep.passed:
336                  # setup/teardown passing reports - let's ignore those
337                  continue
338              if when and rep.when != when:
339                  continue
340              if not inamepart or inamepart in rep.nodeid.split("::"):
341                  values.append(rep)
342          if not values:
343              raise ValueError(
344                  "could not find test report matching %r: "
345                  "no test reports at all!" % (inamepart,)
346              )
347          if len(values) > 1:
348              raise ValueError(
349                  "found 2 or more testreports matching {!r}: {}".format(
350                      inamepart, values
351                  )
352              )
353          return values[0]

497      """The result of running a command."""
498  
499      def __init__(
500          self,
501          ret: Union[int, ExitCode],
502          outlines: List[str],
503          errlines: List[str],
504          duration: float,
505      ) -> None:
506          try:
507              self.ret: Union[int, ExitCode] = ExitCode(ret)
508              """The return value."""
509          except ValueError:
510              self.ret = ret
511          self.outlines = outlines
512          """List of lines captured from stdout."""

519          :func:`stdout.fnmatch_lines() <LineMatcher.fnmatch_lines()>` method. 
520          """

523          self.duration = duration
524          """Duration in seconds."""

527          return (
528              "<RunResult ret=%s len(stdout.lines)=%d len(stderr.lines)=%d duration=%.2fs>"
529              % (self.ret, len(self.stdout.lines), len(self.stderr.lines), self.duration)

532      def parseoutcomes(self) -> Dict[str, int]:
533          """Return a dictionary of outcome noun -> count from parsing the terminal 
534          output that the test process produced. 
535   
536          The returned nouns will always be in plural form:: 
537   
538              ======= 1 failed, 1 passed, 1 warning, 1 error in 0.13s ==== 
539   
540          Will return ``{"failed": 1, "passed": 1, "warnings": 1, "errors": 1}``. 
541          """
542          return self.parse_summary_nouns(self.outlines)
543  
544      @classmethod
545      def parse_summary_nouns(cls, lines) -> Dict[str, int]:
546          """Extract the nouns from a pytest terminal summary line. 
547   
548          It always returns the plural noun for consistency:: 
549   
550              ======= 1 failed, 1 passed, 1 warning, 1 error in 0.13s ==== 
551   
552          Will return ``{"failed": 1, "passed": 1, "warnings": 1, "errors": 1}``. 
553          """
554          for line in reversed(lines):
555              if rex_session_duration.search(line):
556                  outcomes = rex_outcome.findall(line)
557                  ret = {noun: int(count) for (count, noun) in outcomes}
558                  break
559          else:
560              raise ValueError("Pytest terminal summary report not found")
561  
562          to_plural = {
563              "warning": "warnings",
564              "error": "errors",
565          }
566          return {to_plural.get(k, k): v for k, v in ret.items()}

624  @final
625  class Pytester:
626      """ 
627      Facilities to write tests/configuration files, execute pytest in isolation, and match 
628      against expected output, perfect for black-box testing of pytest plugins. 
629   
630      It attempts to isolate the test run from external factors as much as possible, modifying 
631      the current working directory to ``path`` and environment variables during initialization. 
632   
633      Attributes: 
634   
635      :ivar Path path: temporary directory path used to create files/run tests from, etc. 
636   
637      :ivar plugins: 
638         A list of plugins to use with :py:meth:`parseconfig` and 
639         :py:meth:`runpytest`.  Initially this is an empty list but plugins can 
640         be added to the list.  The type of items to add to the list depends on 
641         the method using them so refer to them for details. 
642      """
643  
644      __test__ = False
645  
646      CLOSE_STDIN = object
647  
648      class TimeoutExpired(Exception):
649          pass
650  
651      def __init__(
652          self,
653          request: FixtureRequest,
654          tmp_path_factory: TempPathFactory,
655          *,
656          _ispytest: bool = False,
657      ) -> None:
658          check_ispytest(_ispytest)
659          self._request = request
660          self._mod_collections: WeakKeyDictionary[
661              Collector, List[Union[Item, Collector]]
662          ] = (WeakKeyDictionary())
663          if request.function:
664              name: str = request.function.__name__
665          else:
666              name = request.node.name
667          self._name = name
668          self._path: Path = tmp_path_factory.mktemp(name, numbered=True)
669          self.plugins: List[Union[str, _PluggyPlugin]] = []
670          self._cwd_snapshot = CwdSnapshot()
671          self._sys_path_snapshot = SysPathsSnapshot()
672          self._sys_modules_snapshot = self.__take_sys_modules_snapshot()
673          self.chdir()
674          self._request.addfinalizer(self._finalize)
675          self._method = self._request.config.getoption("--runpytest")
676          self._test_tmproot = tmp_path_factory.mktemp(f"tmp-{name}", numbered=True)
677  
678          self._monkeypatch = mp = MonkeyPatch()
679          mp.setenv("PYTEST_DEBUG_TEMPROOT", str(self._test_tmproot))
680          # Ensure no unexpected caching via tox.
681          mp.delenv("TOX_ENV_DIR", raising=False)
682          # Discard outer pytest options.
683          mp.delenv("PYTEST_ADDOPTS", raising=False)
684          # Ensure no user config is used.
685          tmphome = str(self.path)
686          mp.setenv("HOME", tmphome)
687          mp.setenv("USERPROFILE", tmphome)
688          # Do not use colors for inner runs by default.
689          mp.setenv("PY_COLORS", "0")
690  
691      @property
692      def path(self) -> Path:
693          """Temporary directory where files are created and pytest is executed."""
694          return self._path
695  
696      def __repr__(self) -> str:
697          return f"<Pytester {self.path!r}>"
698  
699      def _finalize(self) -> None:
700          """ 
701          Clean up global state artifacts. 
702   
703          Some methods modify the global interpreter state and this tries to 
704          clean this up. It does not remove the temporary directory however so 
705          it can be looked at after the test run has finished. 
706          """

737      def _makefile(
738          self,
739          ext: str,
740          lines: Sequence[Union[Any, bytes]],
741          files: Dict[str, str],
742          encoding: str = "utf-8",
743      ) -> Path:
744          items = list(files.items())
745  
746          def to_text(s: Union[Any, bytes]) -> str:
747              return s.decode(encoding) if isinstance(s, bytes) else str(s)
748  
749          if lines:
750              source = "\n".join(to_text(x) for x in lines)
751              basename = self._name
752              items.insert(0, (basename, source))
753  
754          ret = None
755          for basename, value in items:
756              p = self.path.joinpath(basename).with_suffix(ext)
757              p.parent.mkdir(parents=True, exist_ok=True)
758              source_ = Source(value)
759              source = "\n".join(to_text(line) for line in source_.lines)
760              p.write_text(source.strip(), encoding=encoding)
761              if ret is None:
762                  ret = p
763          assert ret is not None
764          return ret

1020     def inline_run(
1021         self,
1022         *args: Union[str, "os.PathLike[str]"],
1023         plugins=(),
1024         no_reraise_ctrlc: bool = False,
1025     ) -> HookRecorder:
1026         """Run ``pytest.main()`` in-process, returning a HookRecorder. 
1027  
1028         Runs the :py:func:`pytest.main` function to run all of pytest inside 
1029         the test process itself.  This means it can return a 
1030         :py:class:`HookRecorder` instance which gives more detailed results 
1031         from that run than can be done by matching stdout/stderr from 
1032         :py:meth:`runpytest`. 
1033  
1034         :param args: 
1035             Command line arguments to pass to :py:func:`pytest.main`. 
1036         :param plugins: 
1037             Extra plugin instances the ``pytest.main()`` instance should use. 
1038         :param no_reraise_ctrlc: 
1039             Typically we reraise keyboard interrupts from the child run. If 
1040             True, the KeyboardInterrupt exception is captured. 
1041  
1042         :returns: A :py:class:`HookRecorder` instance. 
1043         """
1044         # (maybe a cpython bug?) the importlib cache sometimes isn't updated
1045         # properly between file creation and inline_run (especially if imports
1046         # are interspersed with file creation)
1047         importlib.invalidate_caches()
1048 
1049         plugins = list(plugins)
1050         finalizers = []
1051         try:
1052             # Any sys.module or sys.path changes done while running pytest
1053             # inline should be reverted after the test run completes to avoid
1054             # clashing with later inline tests run within the same pytest test,
1055             # e.g. just because they use matching test module names.
1056             finalizers.append(self.__take_sys_modules_snapshot().restore)
1057             finalizers.append(SysPathsSnapshot().restore)
1058 
1059             # Important note:
1060             # - our tests should not leave any other references/registrations
1061             #   laying around other than possibly loaded test modules
1062             #   referenced from sys.modules, as nothing will clean those up
1063             #   automatically
1064 
1065             rec = []
1066 
1067             class Collect:
1068                 def pytest_configure(x, config: Config) -> None:
1069                     rec.append(self.make_hook_recorder(config.pluginmanager))
1070 
1071             plugins.append(Collect())
1072             ret = main([str(x) for x in args], plugins=plugins)
1073             if len(rec) == 1:
1074                 reprec = rec.pop()
1075             else:
1076 
1077                 class reprec:  # type: ignore
1078                     pass
1079 
1080             reprec.ret = ret  # type: ignore
1081 
1082             # Typically we reraise keyboard interrupts from the child run
1083             # because it's our user requesting interruption of the testing.
1084             if ret == ExitCode.INTERRUPTED and not no_reraise_ctrlc:
1085                 calls = reprec.getcalls("pytest_keyboard_interrupt")
1086                 if calls and calls[-1].excinfo.type == KeyboardInterrupt:
1087                     raise KeyboardInterrupt()
1088             return reprec
1089         finally:
1090             for finalizer in finalizers:
1091                 finalizer()
1092 
1093     def runpytest_inprocess(
1094         self, *args: Union[str, "os.PathLike[str]"], **kwargs: Any
1095     ) -> RunResult:
1096         """Return result of running pytest in-process, providing a similar 
1097         interface to what self.runpytest() provides."""
1098         syspathinsert = kwargs.pop("syspathinsert", False)
1099 
1100         if syspathinsert:
1101             self.syspathinsert()
1102         now = timing.time()
1103         capture = _get_multicapture("sys")
1104         capture.start_capturing()
1105         try:
1106             try:
1107                 reprec = self.inline_run(*args, **kwargs)
1108             except SystemExit as e:
1109                 ret = e.args[0]
1110                 try:
1111                     ret = ExitCode(e.args[0])
1112                 except ValueError:
1113                     pass
1114 
1115                 class reprec:  # type: ignore
1116                     ret = ret
1117 
1118             except Exception:
1119                 traceback.print_exc()
1120 
1121                 class reprec:  # type: ignore
1122                     ret = ExitCode(3)
1123 
1124         finally:
1125             out, err = capture.readouterr()
1126             capture.stop_capturing()
1127             sys.stdout.write(out)
1128             sys.stderr.write(err)
1129 
1130         assert reprec.ret is not None
1131         res = RunResult(
1132             reprec.ret, out.splitlines(), err.splitlines(), timing.time() - now
1133         )
1134         res.reprec = reprec  # type: ignore
1135         return res

1270     def popen(
1271         self,
1272         cmdargs,
1273         stdout: Union[int, TextIO] = subprocess.PIPE,

1807         self,
1808         lines2: Sequence[str],
1809         match_func: Callable[[str, str], bool],
1810         match_nickname: str,
1811         *,
1812         consecutive: bool = False,
1813     ) -> None:
1814         """Underlying implementation of ``fnmatch_lines`` and ``re_match_lines``. 
1815  
1816         :param Sequence[str] lines2: 
1817             List of string patterns to match. The actual format depends on 
1818             ``match_func``. 
1819         :param match_func: 
1820             A callable ``match_func(line, pattern)`` where line is the 
1821             captured line from stdout/stderr and pattern is the matching 
1822             pattern. 
1823         :param str match_nickname: 
1824             The nickname for the match function that will be logged to stdout 
1825             when a match occurs. 
1826         :param consecutive: 
1827             Match lines consecutively? 
1828         """
1829         if not isinstance(lines2, collections.abc.Sequence):
1830             raise TypeError("invalid type for lines2: {}".format(type(lines2).__name__))
1831         lines2 = self._getlines(lines2)
1832         lines1 = self.lines[:]
1833         extralines = []
1834         __tracebackhide__ = True
1835         wnick = len(match_nickname) + 1
1836         started = False
1837         for line in lines2:
1838             nomatchprinted = False
1839             while lines1:
1840                 nextline = lines1.pop(0)
1841                 if line == nextline:
1842                     self._log("exact match:", repr(line))
1843                     started = True
1844                     break
1845                 elif match_func(nextline, line):
1846                     self._log("%s:" % match_nickname, repr(line))
1847                     self._log(
1848                         "{:>{width}}".format("with:", width=wnick), repr(nextline)
1849                     )
1850                     started = True
1851                     break
1852                 else:
1853                     if consecutive and started:
1854                         msg = f"no consecutive match: {line!r}"
1855                         self._log(msg)
1856                         self._log(
1857                             "{:>{width}}".format("with:", width=wnick), repr(nextline)
1858                         )
1859                         self._fail(msg)
1860                     if not nomatchprinted:
1861                         self._log(
1862                             "{:>{width}}".format("nomatch:", width=wnick), repr(line)
1863                         )
1864                         nomatchprinted = True
1865                     self._log("{:>{width}}".format("and:", width=wnick), repr(nextline))
1866                 extralines.append(nextline)
1867             else:
1868                 msg = f"remains unmatched: {line!r}"
1869                 self._log(msg)
1870                 self._fail(msg)
