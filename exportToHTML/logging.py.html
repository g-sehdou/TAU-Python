49   class ColoredLevelFormatter(logging.Formatter):
50       """A logging formatter which colorizes the %(levelname)..s part of the 
51       log format passed to __init__."""
52   
53       LOGLEVEL_COLOROPTS: Mapping[int, AbstractSet[str]] = {
54           logging.CRITICAL: {"red"},
55           logging.ERROR: {"red", "bold"},
56           logging.WARNING: {"yellow"},
57           logging.WARN: {"yellow"},
58           logging.INFO: {"green"},
59           logging.DEBUG: {"purple"},
60           logging.NOTSET: set(),
61       }
62       LEVELNAME_FMT_REGEX = re.compile(r"%\(levelname\)([+-.]?\d*s)")
63   
64       def __init__(self, terminalwriter: TerminalWriter, *args, **kwargs) -> None:
65           super().__init__(*args, **kwargs)
66           self._original_fmt = self._style._fmt
67           self._level_to_fmt_mapping: Dict[int, str] = {}
68   
69           assert self._fmt is not None
70           levelname_fmt_match = self.LEVELNAME_FMT_REGEX.search(self._fmt)
71           if not levelname_fmt_match:
72               return
73           levelname_fmt = levelname_fmt_match.group()
74   
75           for level, color_opts in self.LOGLEVEL_COLOROPTS.items():
76               formatted_levelname = levelname_fmt % {
77                   "levelname": logging.getLevelName(level)
78               }
79   
80               # add ANSI escape sequences around the formatted levelname
81               color_kwargs = {name: True for name in color_opts}
82               colorized_formatted_levelname = terminalwriter.markup(
83                   formatted_levelname, **color_kwargs
84               )
85               self._level_to_fmt_mapping[level] = self.LEVELNAME_FMT_REGEX.sub(
86                   colorized_formatted_levelname, self._fmt
87               )
88   
89       def format(self, record: logging.LogRecord) -> str:
90           fmt = self._level_to_fmt_mapping.get(record.levelno, self._original_fmt)
91           self._style._fmt = fmt
92           return super().format(record)

95   class PercentStyleMultiline(logging.PercentStyle):
96       """A logging style with special support for multiline messages. 
97    
98       If the message of a record consists of multiple lines, this style 
99       formats the message as if each line were logged separately. 
100      """
101  
102      def __init__(self, fmt: str, auto_indent: Union[int, str, bool, None]) -> None:
103          super().__init__(fmt)
104          self._auto_indent = self._get_auto_indent(auto_indent)
105  
106      @staticmethod
107      def _update_message(
108          record_dict: Dict[str, object], message: str
109      ) -> Dict[str, object]:
110          tmp = record_dict.copy()
111          tmp["message"] = message
112          return tmp
113  
114      @staticmethod
115      def _get_auto_indent(auto_indent_option: Union[int, str, bool, None]) -> int:
116          """Determine the current auto indentation setting. 
117   
118          Specify auto indent behavior (on/off/fixed) by passing in 
119          extra={"auto_indent": [value]} to the call to logging.log() or 
120          using a --log-auto-indent [value] command line or the 
121          log_auto_indent [value] config option. 
122   
123          Default behavior is auto-indent off. 
124   
125          Using the string "True" or "on" or the boolean True as the value 
126          turns auto indent on, using the string "False" or "off" or the 
127          boolean False or the int 0 turns it off, and specifying a 
128          positive integer fixes the indentation position to the value 
129          specified. 
130   
131          Any other values for the option are invalid, and will silently be 
132          converted to the default. 
133   
134          :param None|bool|int|str auto_indent_option: 
135              User specified option for indentation from command line, config 
136              or extra kwarg. Accepts int, bool or str. str option accepts the 
137              same range of values as boolean config options, as well as 
138              positive integers represented in str form. 
139   
140          :returns: 
141              Indentation value, which can be 
142              -1 (automatically determine indentation) or 
143              0 (auto-indent turned off) or 
144              >0 (explicitly set indentation position). 
145          """
146  
147          if auto_indent_option is None:
148              return 0
149          elif isinstance(auto_indent_option, bool):
150              if auto_indent_option:
151                  return -1
152              else:
153                  return 0
154          elif isinstance(auto_indent_option, int):
155              return int(auto_indent_option)
156          elif isinstance(auto_indent_option, str):
157              try:
158                  return int(auto_indent_option)
159              except ValueError:
160                  pass
161              try:
162                  if _strtobool(auto_indent_option):
163                      return -1
164              except ValueError:
165                  return 0
166  
167          return 0
168  
169      def format(self, record: logging.LogRecord) -> str:
170          if "\n" in record.message:
171              if hasattr(record, "auto_indent"):
172                  # Passed in from the "extra={}" kwarg on the call to logging.log().
173                  auto_indent = self._get_auto_indent(record.auto_indent)  # type: ignore[attr-defined]
174              else:
175                  auto_indent = self._auto_indent
176  
177              if auto_indent:
178                  lines = record.message.splitlines()
179                  formatted = self._fmt % self._update_message(record.__dict__, lines[0])
180  
181                  if auto_indent < 0:
182                      indentation = _remove_ansi_escape_sequences(formatted).find(
183                          lines[0]
184                      )
185                  else:
186                      # Optimizes logging by allowing a fixed indentation.
187                      indentation = auto_indent
188                  lines[0] = formatted
189                  return ("\n" + " " * indentation).join(lines)
190          return self._fmt % record.__dict__

291  # Not using @contextmanager for performance reasons.
292  class catching_logs:
293      """Context manager that prepares the whole logging machinery properly."""
294  
295      __slots__ = ("handler", "level", "orig_level")
296  
297      def __init__(self, handler: _HandlerType, level: Optional[int] = None) -> None:
298          self.handler = handler
299          self.level = level
300  
301      def __enter__(self):
302          root_logger = logging.getLogger()
303          if self.level is not None:
304              self.handler.setLevel(self.level)
305          root_logger.addHandler(self.handler)
306          if self.level is not None:
307              self.orig_level = root_logger.level
308              root_logger.setLevel(min(self.orig_level, self.level))
309          return self.handler
310  
311      def __exit__(self, type, value, traceback):
312          root_logger = logging.getLogger()
313          if self.level is not None:
314              root_logger.setLevel(self.orig_level)
315          root_logger.removeHandler(self.handler)

318  class LogCaptureHandler(logging.StreamHandler):
319      """A logging handler that stores log records and the log text."""
320  
321      stream: StringIO
322  
323      def __init__(self) -> None:
324          """Create a new log handler."""
325          super().__init__(StringIO())
326          self.records: List[logging.LogRecord] = []
327  
328      def emit(self, record: logging.LogRecord) -> None:
329          """Keep the log records in a list in addition to the log text."""
330          self.records.append(record)
331          super().emit(record)
332  
333      def reset(self) -> None:
334          self.records = []
335          self.stream = StringIO()
336  
337      def handleError(self, record: logging.LogRecord) -> None:
338          if logging.raiseExceptions:
339              # Fail the test if the log message is bad (emit failed).
340              # The default behavior of logging is to print "Logging error"
341              # to stderr with the call stack and some extra details.
342              # pytest wants to make such mistakes visible during testing.
343              raise

390      @property
391      def text(self) -> str:
392          """The formatted log text."""
393          return _remove_ansi_escape_sequences(self.handler.stream.getvalue())
394  
395      @property
396      def records(self) -> List[logging.LogRecord]:
397          """The list of log records."""
398          return self.handler.records

400      @property
401      def record_tuples(self) -> List[Tuple[str, int, str]]:
402          """A list of a stripped down version of log records intended 
403          for use in assertion comparison. 
404   
405          The format of the tuple is: 
406   
407              (logger_name, log_level, message) 
408          """
409          return [(r.name, r.levelno, r.getMessage()) for r in self.records]

411      @property
412      def messages(self) -> List[str]:
413          """A list of format-interpolated log messages. 
414   
415          Unlike 'records', which contains the format string and parameters for 
416          interpolation, log messages in this list are all interpolated. 
417   
418          Unlike 'text', which contains the output from the handler, log 
419          messages in this list are unadorned with levels, timestamps, etc, 
420          making exact comparisons more reliable. 
421   
422          Note that traceback or stack info (from :func:`logging.exception` or 
423          the `exc_info` or `stack_info` arguments to the logging functions) is 
424          not included, as this is added by the formatter in the handler. 
425   
426          .. versionadded:: 3.7 
427          """
428          return [r.getMessage() for r in self.records]

430      def clear(self) -> None:
431          """Reset the list of log records and the captured log text."""
432          self.handler.reset()

434      def set_level(self, level: Union[int, str], logger: Optional[str] = None) -> None:
435          """Set the level of a logger for the duration of a test. 
436   
437          .. versionchanged:: 3.4 
438              The levels of the loggers changed by this function will be 
439              restored to their initial values at the end of the test. 
440   
441          :param int level: The level. 
442          :param str logger: The logger to update. If not given, the root logger. 
443          """
444          logger_obj = logging.getLogger(logger)
445          # Save the original log-level to restore it during teardown.
446          self._initial_logger_levels.setdefault(logger, logger_obj.level)
447          logger_obj.setLevel(level)
448          if self._initial_handler_level is None:
449              self._initial_handler_level = self.handler.level
450          self.handler.setLevel(level)
451  
452      @contextmanager
453      def at_level(
454          self, level: int, logger: Optional[str] = None
455      ) -> Generator[None, None, None]:
456          """Context manager that sets the level for capturing of logs. After 
457          the end of the 'with' statement the level is restored to its original 
458          value. 
459   
460          :param int level: The level. 
461          :param str logger: The logger to update. If not given, the root logger. 
462          """
463          logger_obj = logging.getLogger(logger)
464          orig_level = logger_obj.level
465          logger_obj.setLevel(level)
466          handler_orig_level = self.handler.level
467          self.handler.setLevel(level)
468          try:
469              yield
470          finally:
471              logger_obj.setLevel(orig_level)
472              self.handler.setLevel(handler_orig_level)

492  def get_log_level_for_setting(config: Config, *setting_names: str) -> Optional[int]:
493      for setting_name in setting_names:
494          log_level = config.getoption(setting_name)
495          if log_level is None:
496              log_level = config.getini(setting_name)
497          if log_level:
498              break
499      else:
500          return None
501  
502      if isinstance(log_level, str):
503          log_level = log_level.upper()
504      try:
505          return int(getattr(logging, log_level, log_level))
506      except ValueError as e:
507          # Python logging does not recognise this as a logging level
508          raise UsageError(
509              "'{}' is not recognized as a logging level name for "
510              "'{}'. Please consider passing the "
511              "logging level num instead.".format(log_level, setting_name)
512          ) from e

521  class LoggingPlugin:
522      """Attaches to the logging module and captures log messages for each test."""
523  
524      def __init__(self, config: Config) -> None:
525          """Create a new plugin to capture log messages. 
526   
527          The formatter can be safely shared across all handlers so 
528          create a single one for the entire test session here. 
529          """
530          self._config = config
531  
532          # Report logging.
533          self.formatter = self._create_formatter(
534              get_option_ini(config, "log_format"),
535              get_option_ini(config, "log_date_format"),
536              get_option_ini(config, "log_auto_indent"),
537          )
538          self.log_level = get_log_level_for_setting(config, "log_level")
539          self.caplog_handler = LogCaptureHandler()
540          self.caplog_handler.setFormatter(self.formatter)
541          self.report_handler = LogCaptureHandler()
542          self.report_handler.setFormatter(self.formatter)
543  
544          # File logging.
545          self.log_file_level = get_log_level_for_setting(config, "log_file_level")
546          log_file = get_option_ini(config, "log_file") or os.devnull
547          if log_file != os.devnull:
548              directory = os.path.dirname(os.path.abspath(log_file))
549              if not os.path.isdir(directory):
550                  os.makedirs(directory)
551  
552          self.log_file_handler = _FileHandler(log_file, mode="w", encoding="UTF-8")
553          log_file_format = get_option_ini(config, "log_file_format", "log_format")
554          log_file_date_format = get_option_ini(
555              config, "log_file_date_format", "log_date_format"
556          )
557  
558          log_file_formatter = logging.Formatter(
559              log_file_format, datefmt=log_file_date_format
560          )
561          self.log_file_handler.setFormatter(log_file_formatter)
562  
563          # CLI/live logging.
564          self.log_cli_level = get_log_level_for_setting(
565              config, "log_cli_level", "log_level"
566          )
567          if self._log_cli_enabled():
568              terminal_reporter = config.pluginmanager.get_plugin("terminalreporter")
569              capture_manager = config.pluginmanager.get_plugin("capturemanager")
570              # if capturemanager plugin is disabled, live logging still works.
571              self.log_cli_handler: Union[
572                  _LiveLoggingStreamHandler, _LiveLoggingNullHandler
573              ] = _LiveLoggingStreamHandler(terminal_reporter, capture_manager)
574          else:
575              self.log_cli_handler = _LiveLoggingNullHandler()
576          log_cli_formatter = self._create_formatter(
577              get_option_ini(config, "log_cli_format", "log_format"),
578              get_option_ini(config, "log_cli_date_format", "log_date_format"),
579              get_option_ini(config, "log_auto_indent"),
580          )
581          self.log_cli_handler.setFormatter(log_cli_formatter)
582  
583      def _create_formatter(self, log_format, log_date_format, auto_indent):
584          # Color option doesn't exist if terminal plugin is disabled.
585          color = getattr(self._config.option, "color", "no")
586          if color != "no" and ColoredLevelFormatter.LEVELNAME_FMT_REGEX.search(
587              log_format
588          ):
589              formatter: logging.Formatter = ColoredLevelFormatter(
590                  create_terminal_writer(self._config), log_format, log_date_format
591              )
592          else:
593              formatter = logging.Formatter(log_format, log_date_format)
594  
595          formatter._style = PercentStyleMultiline(
596              formatter._style._fmt, auto_indent=auto_indent
597          )
598  
599          return formatter

601      def set_log_path(self, fname: str) -> None:
602          """Set the filename parameter for Logging.FileHandler(). 
603   
604          Creates parent directory if it does not exist. 
605   
606          .. warning:: 
607              This is an experimental API. 
608          """
609          fpath = Path(fname)
610  
611          if not fpath.is_absolute():
612              fpath = self._config.rootpath / fpath
613  
614          if not fpath.parent.exists():
615              fpath.parent.mkdir(exist_ok=True, parents=True)
616  
617          stream = fpath.open(mode="w", encoding="UTF-8")
618          if sys.version_info >= (3, 7):
619              old_stream = self.log_file_handler.setStream(stream)
620          else:
621              old_stream = self.log_file_handler.stream
622              self.log_file_handler.acquire()
623              try:
624                  self.log_file_handler.flush()
625                  self.log_file_handler.stream = stream
626              finally:
627                  self.log_file_handler.release()
628          if old_stream:
629              old_stream.close()

631      def _log_cli_enabled(self):
632          """Return whether live logging is enabled."""
633          enabled = self._config.getoption(
634              "--log-cli-level"
635          ) is not None or self._config.getini("log_cli")
636          if not enabled:
637              return False
638  
639          terminal_reporter = self._config.pluginmanager.get_plugin("terminalreporter")
640          if terminal_reporter is None:
641              # terminal reporter is disabled e.g. by pytest-xdist.
642              return False
643  
644          return True

663      def pytest_runtestloop(self, session: Session) -> Generator[None, None, None]:
664          if session.config.option.collectonly:
665              yield
666              return
667  
668          if self._log_cli_enabled() and self._config.getoption("verbose") < 1:
669              # The verbose flag is needed to avoid messy test progress output.
670              self._config.option.verbose = 1
671  
672          with catching_logs(self.log_cli_handler, level=self.log_cli_level):
673              with catching_logs(self.log_file_handler, level=self.log_file_level):
674                  yield  # Run all the tests.
675  
676      @hookimpl
677      def pytest_runtest_logstart(self) -> None:
678          self.log_cli_handler.reset()
679          self.log_cli_handler.set_when("start")
680  
681      @hookimpl
682      def pytest_runtest_logreport(self) -> None:
683          self.log_cli_handler.set_when("logreport")
684  
685      def _runtest_for(self, item: nodes.Item, when: str) -> Generator[None, None, None]:
686          """Implement the internals of the pytest_runtest_xxx() hooks."""
687          with catching_logs(
688              self.caplog_handler, level=self.log_level,
689          ) as caplog_handler, catching_logs(
690              self.report_handler, level=self.log_level,
691          ) as report_handler:
692              caplog_handler.reset()
693              report_handler.reset()
694              item._store[caplog_records_key][when] = caplog_handler.records
695              item._store[caplog_handler_key] = caplog_handler
696  
697              yield
698  
699              log = report_handler.stream.getvalue().strip()
700              item.add_report_section(when, "log", log)
701  
702      @hookimpl(hookwrapper=True)
703      def pytest_runtest_setup(self, item: nodes.Item) -> Generator[None, None, None]:
704          self.log_cli_handler.set_when("setup")
705  
706          empty: Dict[str, List[logging.LogRecord]] = {}
707          item._store[caplog_records_key] = empty
708          yield from self._runtest_for(item, "setup")
709  
710      @hookimpl(hookwrapper=True)
711      def pytest_runtest_call(self, item: nodes.Item) -> Generator[None, None, None]:
712          self.log_cli_handler.set_when("call")
713  
714          yield from self._runtest_for(item, "call")
715  
716      @hookimpl(hookwrapper=True)
717      def pytest_runtest_teardown(self, item: nodes.Item) -> Generator[None, None, None]:
718          self.log_cli_handler.set_when("teardown")
719  
720          yield from self._runtest_for(item, "teardown")
721          del item._store[caplog_records_key]
722          del item._store[caplog_handler_key]
723  
724      @hookimpl
725      def pytest_runtest_logfinish(self) -> None:
726          self.log_cli_handler.set_when("finish")
727  
728      @hookimpl(hookwrapper=True, tryfirst=True)
729      def pytest_sessionfinish(self) -> Generator[None, None, None]:
730          self.log_cli_handler.set_when("sessionfinish")
731  
732          with catching_logs(self.log_cli_handler, level=self.log_cli_level):
733              with catching_logs(self.log_file_handler, level=self.log_file_level):
734                  yield
735  
736      @hookimpl
737      def pytest_unconfigure(self) -> None:
738          # Close the FileHandler explicitly.
739          # (logging.shutdown might have lost the weakref?!)
740          self.log_file_handler.close()

751  class _LiveLoggingStreamHandler(logging.StreamHandler):
752      """A logging StreamHandler used by the live logging feature: it will 
753      write a newline before the first log message in each test. 
754   
755      During live logging we must also explicitly disable stdout/stderr 
756      capturing otherwise it will get captured and won't appear in the 
757      terminal. 
758      """
759  
760      # Officially stream needs to be a IO[str], but TerminalReporter
761      # isn't. So force it.
762      stream: TerminalReporter = None  # type: ignore
763  
764      def __init__(
765          self,
766          terminal_reporter: TerminalReporter,
767          capture_manager: Optional[CaptureManager],
768      ) -> None:
769          logging.StreamHandler.__init__(self, stream=terminal_reporter)  # type: ignore[arg-type]
770          self.capture_manager = capture_manager
771          self.reset()
772          self.set_when(None)
773          self._test_outcome_written = False
774  
775      def reset(self) -> None:
776          """Reset the handler; should be called before the start of each test."""
777          self._first_record_emitted = False
778  
779      def set_when(self, when: Optional[str]) -> None:
780          """Prepare for the given test phase (setup/call/teardown)."""
781          self._when = when
782          self._section_name_shown = False
783          if when == "start":
784              self._test_outcome_written = False
785  
786      def emit(self, record: logging.LogRecord) -> None:
787          ctx_manager = (
788              self.capture_manager.global_and_fixture_disabled()
789              if self.capture_manager
790              else nullcontext()
791          )
792          with ctx_manager:
793              if not self._first_record_emitted:
794                  self.stream.write("\n")
795                  self._first_record_emitted = True
796              elif self._when in ("teardown", "finish"):
797                  if not self._test_outcome_written:
798                      self._test_outcome_written = True
799                      self.stream.write("\n")
800              if not self._section_name_shown and self._when:
801                  self.stream.section("live log " + self._when, sep="-", bold=True)
802                  self._section_name_shown = True
803              super().emit(record)
804  
805      def handleError(self, record: logging.LogRecord) -> None:
806          # Handled by LogCaptureHandler.
807          pass
