81   def normalize_version_info(py_version_info: Tuple[int, ...]) -> Tuple[int, int, int]:
82       """ 
83       Convert a tuple of ints representing a Python version to one of length 
84       three. 
85    
86       :param py_version_info: a tuple of ints representing a Python version, 
87           or None to specify no version. The tuple can have any length. 
88    
89       :return: a tuple of length three if `py_version_info` is non-None. 
90           Otherwise, return `py_version_info` unchanged (i.e. None). 
91       """
92       if len(py_version_info) < 3:
93           py_version_info += (3 - len(py_version_info)) * (0,)
94       elif len(py_version_info) > 3:
95           py_version_info = py_version_info[:3]
96   
97       return cast("VersionInfo", py_version_info)

100  def ensure_dir(path: str) -> None:
101      """os.path.makedirs without EEXIST."""
102      try:
103          os.makedirs(path)
104      except OSError as e:
105          # Windows can raise spurious ENOTEMPTY errors. See #6426.
106          if e.errno != errno.EEXIST and e.errno != errno.ENOTEMPTY:
107              raise

129  def rmtree_errorhandler(func: Callable[..., Any], path: str, exc_info: ExcInfo) -> None:
130      """On Windows, the files in .svn are read-only, so when rmtree() tries to 
131      remove them, an exception is thrown.  We catch that here, remove the 
132      read-only attribute, and hopefully continue without problems."""
133      try:
134          has_attr_readonly = not (os.stat(path).st_mode & stat.S_IWRITE)
135      except OSError:
136          # it's equivalent to os.path.exists
137          return
138  
139      if has_attr_readonly:
140          # convert to read/write
141          os.chmod(path, stat.S_IWRITE)
142          # use the original function to repeat the operation
143          func(path)
144          return
145      else:
146          raise

149  def display_path(path: str) -> str:
150      """Gives the display value for a given path, making it relative to cwd 
151      if possible."""
152      path = os.path.normcase(os.path.abspath(path))
153      if path.startswith(os.getcwd() + os.path.sep):
154          path = "." + path[len(os.getcwd()) :]
155      return path

158  def backup_dir(dir: str, ext: str = ".bak") -> str:
159      """Figure out the name of a directory to back up the given dir to 
160      (adding .bak, .bak2, etc)"""
161      n = 1
162      extension = ext
163      while os.path.exists(dir + extension):
164          n += 1
165          extension = ext + str(n)
166      return dir + extension

184  def ask(message: str, options: Iterable[str]) -> str:
185      """Ask the message interactively, with the given possible responses"""
186      while 1:
187          _check_no_input(message)
188          response = input(message)
189          response = response.strip().lower()
190          if response not in options:
191              print(
192                  "Your response ({!r}) was not one of the expected responses: "
193                  "{}".format(response, ", ".join(options))
194              )
195          else:
196              return response

211  def strtobool(val: str) -> int:
212      """Convert a string representation of truth to true (1) or false (0). 
213   
214      True values are 'y', 'yes', 't', 'true', 'on', and '1'; false values 
215      are 'n', 'no', 'f', 'false', 'off', and '0'.  Raises ValueError if 
216      'val' is anything else. 
217      """
218      val = val.lower()
219      if val in ("y", "yes", "t", "true", "on", "1"):
220          return 1
221      elif val in ("n", "no", "f", "false", "off", "0"):
222          return 0
223      else:
224          raise ValueError(f"invalid truth value {val!r}")

238  def tabulate(rows: Iterable[Iterable[Any]]) -> Tuple[List[str], List[int]]:
239      """Return a list of formatted rows and a list of column sizes. 
240   
241      For example:: 
242   
243      >>> tabulate([['foobar', 2000], [0xdeadbeef]]) 
244      (['foobar     2000', '3735928559'], [10, 4]) 
245      """
246      rows = [tuple(map(str, row)) for row in rows]
247      sizes = [max(map(len, col)) for col in zip_longest(*rows, fillvalue="")]
248      table = [" ".join(map(str.ljust, row, sizes)).rstrip() for row in rows]
249      return table, sizes

252  def is_installable_dir(path: str) -> bool:
253      """Is path is a directory containing pyproject.toml or setup.py? 
254   
255      If pyproject.toml exists, this is a PEP 517 project. Otherwise we look for 
256      a legacy setuptools layout by identifying setup.py. We don't check for the 
257      setup.cfg because using it without setup.py is only available for PEP 517 
258      projects, which are already covered by the pyproject.toml check. 
259      """
260      if not os.path.isdir(path):
261          return False
262      if os.path.isfile(os.path.join(path, "pyproject.toml")):
263          return True
264      if os.path.isfile(os.path.join(path, "setup.py")):
265          return True
266      return False

269  def read_chunks(file: BinaryIO, size: int = io.DEFAULT_BUFFER_SIZE) -> Iterator[bytes]:
270      """Yield pieces of data from a file-like object until EOF."""
271      while True:
272          chunk = file.read(size)
273          if not chunk:
274              break
275          yield chunk

278  def normalize_path(path: str, resolve_symlinks: bool = True) -> str:
279      """ 
280      Convert a path to its canonical, case-normalized, absolute version. 
281   
282      """
283      path = os.path.expanduser(path)
284      if resolve_symlinks:
285          path = os.path.realpath(path)
286      else:
287          path = os.path.abspath(path)
288      return os.path.normcase(path)

291  def splitext(path: str) -> Tuple[str, str]:
292      """Like os.path.splitext, but take off .tar too"""
293      base, ext = posixpath.splitext(path)
294      if base.lower().endswith(".tar"):
295          ext = base[-4:] + ext
296          base = base[:-4]
297      return base, ext

300  def renames(old: str, new: str) -> None:
301      """Like os.renames(), but handles renaming across devices."""
302      # Implementation borrowed from os.renames().
303      head, tail = os.path.split(new)
304      if head and tail and not os.path.exists(head):
305          os.makedirs(head)
306  
307      shutil.move(old, new)
308  
309      head, tail = os.path.split(old)
310      if head and tail:
311          try:
312              os.removedirs(head)
313          except OSError:
314              pass

393  class StreamWrapper(StringIO):
394      orig_stream: TextIO = None
395  
396      @classmethod
397      def from_stream(cls, orig_stream: TextIO) -> "StreamWrapper":
398          cls.orig_stream = orig_stream

403      @property

405          return self.orig_stream.encoding

408  @contextlib.contextmanager
409  def captured_output(stream_name: str) -> Iterator[StreamWrapper]:
410      """Return a context manager used by captured_stdout/stdin/stderr 
411      that temporarily replaces the sys stream *stream_name* with a StringIO. 
412   
413      Taken from Lib/support/__init__.py in the CPython repo. 
414      """
415      orig_stdout = getattr(sys, stream_name)
416      setattr(sys, stream_name, StreamWrapper.from_stream(orig_stdout))
417      try:
418          yield getattr(sys, stream_name)
419      finally:
420          setattr(sys, stream_name, orig_stdout)

450  def build_netloc(host: str, port: Optional[int]) -> str:
451      """ 
452      Build a netloc from a host-port pair 
453      """
454      if port is None:
455          return host
456      if ":" in host:
457          # Only wrap host with square brackets when it is IPv6
458          host = f"[{host}]"
459      return f"{host}:{port}"

462  def build_url_from_netloc(netloc: str, scheme: str = "https") -> str:
463      """ 
464      Build a full URL from a netloc. 
465      """
466      if netloc.count(":") >= 2 and "@" not in netloc and "[" not in netloc:
467          # It must be a bare IPv6 address, so wrap it with brackets.
468          netloc = f"[{netloc}]"
469      return f"{scheme}://{netloc}"

472  def parse_netloc(netloc: str) -> Tuple[str, Optional[int]]:
473      """ 
474      Return the host-port pair from a netloc. 
475      """
476      url = build_url_from_netloc(netloc)
477      parsed = urllib.parse.urlparse(url)
478      return parsed.hostname, parsed.port

481  def split_auth_from_netloc(netloc: str) -> NetlocTuple:
482      """ 
483      Parse out and remove the auth information from a netloc. 
484   
485      Returns: (netloc, (username, password)). 
486      """
487      if "@" not in netloc:
488          return netloc, (None, None)
489  
490      # Split from the right because that's how urllib.parse.urlsplit()
491      # behaves if more than one @ is present (which can be checked using
492      # the password attribute of urlsplit()'s return value).
493      auth, netloc = netloc.rsplit("@", 1)
494      pw: Optional[str] = None
495      if ":" in auth:
496          # Split from the left because that's how urllib.parse.urlsplit()
497          # behaves if more than one : is present (which again can be checked
498          # using the password attribute of the return value)
499          user, pw = auth.split(":", 1)
500      else:
501          user, pw = auth, None
502  
503      user = urllib.parse.unquote(user)
504      if pw is not None:
505          pw = urllib.parse.unquote(pw)
506  
507      return netloc, (user, pw)

510  def redact_netloc(netloc: str) -> str:
511      """ 
512      Replace the sensitive data in a netloc with "****", if it exists. 
513   
514      For example: 
515          - "user:pass@example.com" returns "user:****@example.com" 
516          - "accesstoken@example.com" returns "****@example.com" 
517      """
518      netloc, (user, password) = split_auth_from_netloc(netloc)
519      if user is None:
520          return netloc
521      if password is None:
522          user = "****"
523          password = ""
524      else:
525          user = urllib.parse.quote(user)
526          password = ":****"
527      return "{user}{password}@{netloc}".format(
528          user=user, password=password, netloc=netloc
529      )

532  def _transform_url(
533      url: str, transform_netloc: Callable[[str], Tuple[Any, ...]]
534  ) -> Tuple[str, NetlocTuple]:
535      """Transform and replace netloc in a url. 
536   
537      transform_netloc is a function taking the netloc and returning a 
538      tuple. The first element of this tuple is the new netloc. The 
539      entire tuple is returned. 
540   
541      Returns a tuple containing the transformed url as item 0 and the 
542      original tuple returned by transform_netloc as item 1. 
543      """
544      purl = urllib.parse.urlsplit(url)
545      netloc_tuple = transform_netloc(purl.netloc)
546      # stripped url
547      url_pieces = (purl.scheme, netloc_tuple[0], purl.path, purl.query, purl.fragment)
548      surl = urllib.parse.urlunsplit(url_pieces)
549      return surl, cast("NetlocTuple", netloc_tuple)

582  class HiddenText:
583      def __init__(self, secret: str, redacted: str) -> None:
584          self.secret = secret
585          self.redacted = redacted
586  
587      def __repr__(self) -> str:
588          return "<HiddenText {!r}>".format(str(self))
589  
590      def __str__(self) -> str:
591          return self.redacted
592  
593      # This is useful for testing.
594      def __eq__(self, other: Any) -> bool:
595          if type(self) != type(other):
596              return False
597  
598          # The string being used for redaction doesn't also have to match,
599          # just the raw, original string.
600          return self.secret == other.secret
601  
602  
603  def hide_value(value: str) -> HiddenText:
604      return HiddenText(value, redacted="****")

607  def hide_url(url: str) -> HiddenText:
608      redacted = redact_auth_from_url(url)
609      return HiddenText(url, redacted=redacted)

612  def protect_pip_from_modification_on_windows(modifying_pip: bool) -> None:
613      """Protection of pip.exe from modification on Windows 
614   
615      On Windows, any operation modifying pip should be run as: 
616          python -m pip ... 
617      """
618      pip_names = [
619          "pip.exe",
620          "pip{}.exe".format(sys.version_info[0]),
621          "pip{}.{}.exe".format(*sys.version_info[:2]),
622      ]
623  
624      # See https://github.com/pypa/pip/issues/1299 for more discussion
625      should_show_use_python_msg = (
626          modifying_pip and WINDOWS and os.path.basename(sys.argv[0]) in pip_names
627      )
628  
629      if should_show_use_python_msg:
630          new_command = [sys.executable, "-m", "pip"] + sys.argv[1:]
631          raise CommandError(
632              "To modify pip, please run the following command:\n{}".format(
633                  " ".join(new_command)
634              )
635          )

643  def hash_file(path: str, blocksize: int = 1 << 20) -> Tuple[Any, int]:
644      """Return (hash, length) for path using hashlib.sha256()"""
645  
646      h = hashlib.sha256()
647      length = 0
648      with open(path, "rb") as f:
649          for block in read_chunks(f, size=blocksize):
650              length += len(block)
651              h.update(block)
652      return h, length

678  def partition(
679      pred: Callable[[T], bool],
680      iterable: Iterable[T],
681  ) -> Tuple[Iterable[T], Iterable[T]]:
682      """ 
683      Use a predicate to partition entries into false entries and true entries, 
684      like 
685   
686          partition(is_odd, range(10)) --> 0 2 4 6 8   and  1 3 5 7 9 
687      """
688      t1, t2 = tee(iterable)
689      return filterfalse(pred, t1), filter(pred, t2)
690
